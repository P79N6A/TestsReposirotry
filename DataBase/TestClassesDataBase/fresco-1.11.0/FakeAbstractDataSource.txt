private static class FakeAbstractDataSource extends AbstractDataSource<Value> {
  @Override public boolean setResult(  @Nullable Value value,  boolean isLast){
    return super.setResult(value,isLast);
  }
  @Override public boolean setFailure(  Throwable throwable){
    return super.setFailure(throwable);
  }
  @Override public boolean setProgress(  float progress){
    return super.setProgress(progress);
  }
  @Override public void closeResult(  Value result){
    result.close();
  }
}
