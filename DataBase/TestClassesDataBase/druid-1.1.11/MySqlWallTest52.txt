public class MySqlWallTest52 extends TestCase {
  public void test_true() throws Exception {
    WallProvider provider=new MySqlWallProvider();
    provider.getConfig().setLimitZeroAllow(true);
    Assert.assertTrue(provider.checkValid("select * from t limit 0"));
    Assert.assertEquals(1,provider.getTableStats().size());
  }
  public void test_false() throws Exception {
    WallProvider provider=new MySqlWallProvider();
    Assert.assertFalse(provider.checkValid("select * from t limit 0"));
    Assert.assertEquals(1,provider.getTableStats().size());
  }
}
