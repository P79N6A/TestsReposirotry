public class SQLServerSelectTest19 extends TestCase {
  public void test_simple() throws Exception {
    String sql="SELECT p.BusinessEntityID, FirstName, LastName, PhoneNumber AS Phone" + "\nFROM Person.Person AS p" + "\nJOIN Person.PersonPhone AS pph ON p.BusinessEntityID  = pph.BusinessEntityID"+ "\nWHERE LastName LIKE 'G%'"+ "\nORDER BY LastName, FirstName "+ "\nFOR XML AUTO, TYPE, XMLSCHEMA, ELEMENTS XSINIL;";
    String expect="SELECT p.BusinessEntityID, FirstName, LastName, PhoneNumber AS Phone" + "\nFROM Person.Person p" + "\n\tJOIN Person.PersonPhone pph ON p.BusinessEntityID = pph.BusinessEntityID"+ "\nWHERE LastName LIKE 'G%'"+ "\nORDER BY LastName, FirstName"+ "\nFOR XML , TYPE, XMLSCHEMA, ELEMENTS XSINIL;";
    SQLServerStatementParser parser=new SQLServerStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    String text=TestUtils.outputSqlServer(stmt);
    Assert.assertEquals(expect,text);
  }
}
