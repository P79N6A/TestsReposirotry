public class MySqlCreateTableTest50 extends MysqlTest {
  public void test_0() throws Exception {
    String sql="CREATE  TABLE `test`.`a` (" + "  `ida` INT NOT NULL COMMENT 'aaa' ," + "  PRIMARY KEY (`ida`) )"+ " COMMENT = 'bb';";
    MySqlStatementParser parser=new MySqlStatementParser(sql);
    SQLStatement stmt=parser.parseCreateTable();
    MySqlSchemaStatVisitor visitor=new MySqlSchemaStatVisitor();
    stmt.accept(visitor);
    Assert.assertEquals(1,visitor.getTables().size());
    Assert.assertEquals(1,visitor.getColumns().size());
    Assert.assertEquals(0,visitor.getConditions().size());
    Assert.assertTrue(visitor.getTables().containsKey(new TableStat.Name("test.a")));
    String output=SQLUtils.toMySqlString(stmt);
    Assert.assertEquals("CREATE TABLE `test`.`a` (" + "\n\t`ida` INT NOT NULL COMMENT 'aaa'," + "\n\tPRIMARY KEY (`ida`)"+ "\n) COMMENT 'bb'",output);
  }
}
