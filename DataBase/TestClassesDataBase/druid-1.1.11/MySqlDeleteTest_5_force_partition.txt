public class MySqlDeleteTest_5_force_partition extends MysqlTest {
  public void test_0() throws Exception {
    String sql="DELETE force partition 'pt001' car_tt FROM runoob_tbl WHERE runoob_id=3;";
    MySqlStatementParser parser=new MySqlStatementParser(sql);
    List<SQLStatement> statementList=parser.parseStatementList();
    SQLStatement stmt=statementList.get(0);
    Assert.assertEquals("DELETE FORCE PARTITION 'pt001' car_tt\n" + "FROM runoob_tbl\n" + "WHERE runoob_id = 3;",SQLUtils.toMySqlString(stmt));
    Assert.assertEquals("delete force partition 'pt001' car_tt\n" + "from runoob_tbl\n" + "where runoob_id = 3;",SQLUtils.toMySqlString(stmt,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
    Assert.assertEquals(1,statementList.size());
    System.out.println(stmt.toString());
    MySqlSchemaStatVisitor visitor=new MySqlSchemaStatVisitor();
    stmt.accept(visitor);
    System.out.println("Tables : " + visitor.getTables());
    System.out.println("fields : " + visitor.getColumns());
    Assert.assertEquals(2,visitor.getTables().size());
    Assert.assertEquals(1,visitor.getColumns().size());
    Assert.assertEquals(1,visitor.getConditions().size());
    Assert.assertTrue(visitor.getTables().containsKey(new TableStat.Name("runoob_tbl")));
    Assert.assertTrue(visitor.getColumns().contains(new Column("car_tt","runoob_id")));
  }
}
