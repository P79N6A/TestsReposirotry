public class SQLServerUpdateTest2 extends TestCase {
  public void test_update() throws Exception {
    String sql="UPDATE Cities " + "SET Location = CONVERT(Point, '12.3:46.2') " + "WHERE Name = 'Anchorage';";
    String expect="UPDATE Cities" + "\nSET Location = CONVERT(Point, '12.3:46.2')" + "\nWHERE Name = 'Anchorage';";
    SQLServerStatementParser parser=new SQLServerStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    String text=TestUtils.outputSqlServer(stmt);
    Assert.assertEquals(expect,text);
  }
}
