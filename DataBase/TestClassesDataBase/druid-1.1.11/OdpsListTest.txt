    String sql="list roles";
    OdpsStatementParser parser=new OdpsStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    parser.match(Token.EOF);
    String output=SQLUtils.toOdpsString(stmt);
