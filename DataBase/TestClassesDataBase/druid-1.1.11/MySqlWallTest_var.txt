public class MySqlWallTest_var extends TestCase {
  public void test_true() throws Exception {
    Assert.assertTrue(WallUtils.isValidateMySql("SELECT @@GLOBAL.sql_mode"));
  }
  public void test_true_1() throws Exception {
    Assert.assertTrue(WallUtils.isValidateMySql("SELECT @@SESSION.sql_mode;"));
  }
  public void test_false() throws Exception {
    Assert.assertFalse(WallUtils.isValidateMySql("SELECT * FROM T WHERE @@SESSION.sql_mode = 'ANSI'"));
  }
  public void test_true2() throws Exception {
    Assert.assertTrue(WallUtils.isValidateMySql("SELECT * FROM T WHERE @@sql_mode = 'ANSI'"));
  }
  public void test_false_1() throws Exception {
    Assert.assertFalse(WallUtils.isValidateMySql("SELECT * FROM T WHERE @@basedir = 'ANSI'"));
  }
}
