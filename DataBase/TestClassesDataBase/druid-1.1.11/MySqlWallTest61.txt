public class MySqlWallTest61 extends TestCase {
  public void test_true() throws Exception {
    WallProvider provider=new MySqlWallProvider();
    provider.getConfig().setSchemaCheck(true);
    Assert.assertTrue(provider.checkValid("SELECT * FROM `information_schema`.`columns` WHERE TABLE_NAME='aaaa' AND TABLE_SCHEMA='xxxx'"));
    Assert.assertEquals(1,provider.getTableStats().size());
  }
  public void test_false() throws Exception {
    WallProvider provider=new MySqlWallProvider();
    provider.getConfig().setSchemaCheck(true);
    Assert.assertFalse(provider.checkValid("SELECT * FROM T " + "UNION " + "SELECT * FROM `information_schema`.`columns` WHERE TABLE_NAME='aaaa' AND TABLE_SCHEMA='xxxx'"));
    Assert.assertEquals(2,provider.getTableStats().size());
  }
}
