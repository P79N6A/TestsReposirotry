public class SQLServerSelectTest9 extends TestCase {
  public void test_isEmpty() throws Exception {
    String sql="SELECT AVG(UnitPrice) AS 'Average Price' " + "FROM Sales.SalesOrderDetail;";
    String expect="SELECT AVG(UnitPrice) AS 'Average Price'" + "\nFROM Sales.SalesOrderDetail;";
    SQLServerStatementParser parser=new SQLServerStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    String text=TestUtils.outputSqlServer(stmt);
    Assert.assertEquals(expect,text);
  }
}
