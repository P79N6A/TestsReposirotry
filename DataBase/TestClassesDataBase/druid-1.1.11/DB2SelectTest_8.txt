public class DB2SelectTest_8 extends DB2Test {
  public void test_0() throws Exception {
    String sql="SELECT unique id FROM DSN81010.EMP where id = 1;";
    DB2StatementParser parser=new DB2StatementParser(sql);
    List<SQLStatement> statementList=parser.parseStatementList();
    SQLStatement stmt=statementList.get(0);
    Assert.assertEquals(1,statementList.size());
    DB2SchemaStatVisitor visitor=new DB2SchemaStatVisitor();
    stmt.accept(visitor);
    Assert.assertEquals(1,visitor.getTables().size());
    Assert.assertEquals(1,visitor.getColumns().size());
    Assert.assertEquals(1,visitor.getConditions().size());
    Assert.assertTrue(visitor.getTables().containsKey(new TableStat.Name("DSN81010.EMP")));
    Assert.assertEquals("SELECT UNIQUE id" + "\nFROM DSN81010.EMP" + "\nWHERE id = 1;",SQLUtils.toSQLString(stmt,JdbcConstants.DB2));
    Assert.assertEquals("select unique id" + "\nfrom DSN81010.EMP" + "\nwhere id = 1;",SQLUtils.toSQLString(stmt,JdbcConstants.DB2,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
  }
}
