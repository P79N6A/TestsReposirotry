public class SQLServerSelectTest18 extends TestCase {
  public void test_simple() throws Exception {
    String sql="SELECT T0.[Address], T0.[Address] " + "FROM [dbo].[CRD1] T0 " + "WHERE T0.[CardCode] = (@P1) AND T0.[AdresType] = (@P2) "+ "ORDER BY T0.[Address] "+ "FOR BROWSE";
    String expect="SELECT T0.[Address], T0.[Address]" + "\nFROM [dbo].[CRD1] T0" + "\nWHERE T0.[CardCode] = @P1"+ "\n\tAND T0.[AdresType] = @P2"+ "\nORDER BY T0.[Address]"+ "\nFOR BROWSE";
    SQLServerStatementParser parser=new SQLServerStatementParser(sql);
    SQLStatement stmt=parser.parseStatementList().get(0);
    String text=TestUtils.outputSqlServer(stmt);
    Assert.assertEquals(expect,text);
  }
}
