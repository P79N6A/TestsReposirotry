public class OdpsSelectTest15 extends TestCase {
  public void test_select() throws Exception {
    String sql="SELECT '\\n' FROM dual;";
    Assert.assertEquals("SELECT '\\n'" + "\nFROM dual;",SQLUtils.formatOdps(sql));
    Assert.assertEquals("select '\\n'" + "\nfrom dual;",SQLUtils.formatOdps(sql,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
    List<SQLStatement> statementList=SQLUtils.parseStatements(sql,JdbcConstants.ODPS);
    SQLStatement stmt=statementList.get(0);
    Assert.assertEquals(1,statementList.size());
    SchemaStatVisitor visitor=SQLUtils.createSchemaStatVisitor(JdbcConstants.ODPS);
    stmt.accept(visitor);
    Assert.assertEquals(0,visitor.getTables().size());
    Assert.assertEquals(0,visitor.getColumns().size());
    Assert.assertEquals(0,visitor.getConditions().size());
  }
}
