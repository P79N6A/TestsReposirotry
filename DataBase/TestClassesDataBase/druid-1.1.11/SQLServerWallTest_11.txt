public class SQLServerWallTest_11 extends TestCase {
  public void test_true() throws Exception {

    String sql="SELECT KL_ArticleContent,KL_ArticleTitle " + "FROM dbo.KL_Article " + "WHERE KL_ArticleId =-4731 "+ "UNION ALL "+ "SELECT (SELECT TOP 1 CHAR(58)+CHAR(108)+CHAR(107)+CHAR(103)+CHAR(58)+ISNULL(CAST(name AS NVARCHAR(4000)),CHAR(32))+CHAR(58)+CHAR(109)+CHAR(122)+CHAR(104)+CHAR(58) FROM sys.sql_logins WHERE ISNULL(name,CHAR(32)) NOT IN (SELECT TOP 0 ISNULL(name,CHAR(32)) FROM sys.sql_logins ORDER BY 1) ORDER BY 1),NULL-- ";
    Assert.assertFalse(provider.checkValid(sql));
  }
}
