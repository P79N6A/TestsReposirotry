public class MySqlSelectTest_91_hints extends MysqlTest {
  public void test_0() throws Exception {
    String sql="select id from a /*+ xxx */ ";
    MySqlStatementParser parser=new MySqlStatementParser(sql);
    List<SQLStatement> statementList=parser.parseStatementList();
    assertEquals(1,statementList.size());
    SQLSelectStatement stmt=(SQLSelectStatement)statementList.get(0);
    assertEquals("SELECT id\n" + "FROM a/*+ xxx */",stmt.toString());
    SQLCommentHint hint=(SQLCommentHint)stmt.getSelect().getHints().get(0);
    assertEquals("+ xxx ",hint.getText());
  }
}
