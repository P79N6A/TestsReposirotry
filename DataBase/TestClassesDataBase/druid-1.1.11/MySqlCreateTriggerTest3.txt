    String sql="-- Create trigger 1\n" + "-- delimiter //\n" + "create trigger trg_my1 before delete on test.t1 for each row begin insert into log_table values (\"delete row from test.t1\"); insert into t4 values (old.col1, old.col1 + 5, old.col1 + 7); end; -- //-- delimiter ;";
