public class OracleSelectTest_param extends OracleTest {
  private String dbType=JdbcConstants.ORACLE;
  public void test_0() throws Exception {
    String sql="select c1 from t1 t where t.c2=1 and (t.c3=1 or t.c3=5)";
    List<SQLStatement> statementList=SQLUtils.parseStatements(sql,dbType);
    assertEquals(1,statementList.size());
    SQLStatement stmt=statementList.get(0);
    assertEquals("SELECT c1\n" + "FROM t1 t\n" + "WHERE t.c2 = 1\n"+ "\tAND (t.c3 = 1\n"+ "\t\tOR t.c3 = 5)",SQLUtils.toSQLString(stmt,dbType));
    assertEquals("SELECT c1\n" + "FROM t1 t\n" + "WHERE t.c2 = ?\n"+ "\tAND (t.c3 = ?)",SQLUtils.toSQLString(stmt,dbType,new SQLUtils.FormatOption(true,true,true)));
  }
}
