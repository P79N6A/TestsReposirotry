private static class DummyTest {
  private int field;
  @Override public boolean equals(  Object obj){
    if (this == obj) {
      return true;
    }
    if (!(obj instanceof DummyTest)) {
      return false;
    }
    DummyTest dt=(DummyTest)obj;
    return field == dt.field;
  }
  @Override public int hashCode(){
    return field;
  }
  public int getField(){
    return field;
  }
}
