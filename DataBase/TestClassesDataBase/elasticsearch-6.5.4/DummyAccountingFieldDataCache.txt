private static class DummyAccountingFieldDataCache implements IndexFieldDataCache {
  private int leafLevelBuilds=0;
  private int topLevelBuilds=0;
  @Override public <FD extends AtomicFieldData,IFD extends IndexFieldData<FD>>FD load(  LeafReaderContext context,  IFD indexFieldData) throws Exception {
    leafLevelBuilds++;
    return indexFieldData.loadDirect(context);
  }
  @Override public <FD extends AtomicFieldData,IFD extends IndexFieldData.Global<FD>>IFD load(  DirectoryReader indexReader,  IFD indexFieldData) throws Exception {
    topLevelBuilds++;
    return (IFD)indexFieldData.localGlobalDirect(indexReader);
  }
  @Override public void clear(){
  }
  @Override public void clear(  String fieldName){
  }
}
