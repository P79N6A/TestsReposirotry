private static class Stats {
  private long minorCount;
  private long majorCount;
  private long criticalCount;
  private void report(  double score){
    if (score < 50.0) {
      minorCount++;
    }
 else     if (score < 75.0) {
      majorCount++;
    }
 else {
      criticalCount++;
    }
  }
  private long totalCount(){
    return minorCount + majorCount + criticalCount;
  }
}
