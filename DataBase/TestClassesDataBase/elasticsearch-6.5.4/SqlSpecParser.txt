private static class SqlSpecParser implements Parser {
  private final StringBuilder query=new StringBuilder();
  @Override public Object parse(  String line){
    String q=null;
    if (line.endsWith(";")) {
      query.append(line.substring(0,line.length() - 1));
      q=query.toString();
      query.setLength(0);
    }
 else {
      query.append(line);
      query.append("\r\n");
    }
    return q;
  }
}
