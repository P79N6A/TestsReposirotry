protected interface TestExecutor<T> {
  void execute(  List<T> tasks);
  default String describeTasks(  List<T> tasks){
    return tasks.stream().map(T::toString).reduce((s1,s2) -> {
      if (s1.isEmpty()) {
        return s2;
      }
 else       if (s2.isEmpty()) {
        return s1;
      }
 else {
        return s1 + ", " + s2;
      }
    }
).orElse("");
  }
}
