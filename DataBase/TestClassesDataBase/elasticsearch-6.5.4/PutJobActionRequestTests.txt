public class PutJobActionRequestTests extends AbstractStreamableXContentTestCase<Request> {
  private String jobId;
  @Before public void setupJobID(){
    jobId=randomAlphaOfLengthBetween(1,10);
  }
  @Override protected Request createTestInstance(){
    return new Request(ConfigTestHelpers.randomRollupJobConfig(random(),jobId));
  }
  @Override protected boolean supportsUnknownFields(){
    return false;
  }
  @Override protected Request createBlankInstance(){
    return new Request();
  }
  @Override protected Request doParseInstance(  final XContentParser parser) throws IOException {
    return Request.fromXContent(parser,jobId);
  }
}
