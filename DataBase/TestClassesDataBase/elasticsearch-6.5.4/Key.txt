class Key {
  private final int key;
  Key(  int key){
    this.key=key;
  }
  @Override public boolean equals(  Object o){
    if (this == o)     return true;
    if (o == null || getClass() != o.getClass())     return false;
    Key key1=(Key)o;
    return key == key1.key;
  }
  @Override public int hashCode(){
    return key % 2;
  }
}
