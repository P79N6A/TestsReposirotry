@RunWith(RandomizedRunner.class) @TestMethodProviders({JUnit3MethodProvider.class}) public abstract class PackagingTestCase {
  protected final Log logger=LogFactory.getLog(getClass());
  @Rule public final TestName testNameRule=new TestName();
  @Before public void logTestNameBefore(){
    logger.info("[" + testNameRule.getMethodName() + "]: before test");
  }
}
