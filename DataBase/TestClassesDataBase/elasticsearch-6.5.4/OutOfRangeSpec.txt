static class OutOfRangeSpec<V> {
  final NumberType type;
  final V value;
  final String message;
  static <V>OutOfRangeSpec<V> of(  NumberType t,  V v,  String m){
    return new OutOfRangeSpec<>(t,v,m);
  }
  OutOfRangeSpec(  NumberType t,  V v,  String m){
    type=t;
    value=v;
    message=m;
  }
}
