public class SimpleTypeRegistryTest {
  @Test public void shouldTestIfClassIsSimpleTypeAndReturnTrue(){
    assertTrue(SimpleTypeRegistry.isSimpleType(String.class));
  }
  @Test public void shouldTestIfClassIsSimpleTypeAndReturnFalse(){
    assertFalse(SimpleTypeRegistry.isSimpleType(RichType.class));
  }
  @Test public void shouldTestIfMapIsSimpleTypeAndReturnFalse(){
    assertFalse(SimpleTypeRegistry.isSimpleType(HashMap.class));
  }
}
