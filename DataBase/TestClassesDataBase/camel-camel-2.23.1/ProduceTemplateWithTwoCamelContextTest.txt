@ContextConfiguration public class ProduceTemplateWithTwoCamelContextTest extends SpringRunWithTestSupport {
  @Autowired protected ProducerTemplate producer;
  @EndpointInject(uri="mock:result") protected MockEndpoint result;
  @Test public void testProducerTemplate() throws Exception {
    result.expectedBodiesReceived("hello");
    producer.sendBody("direct:start","hello");
    result.assertIsSatisfied();
  }
}
