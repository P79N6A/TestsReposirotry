public class HttpProducerUnsafeCharsTest extends BaseJettyTest {
  @Test public void unsafeCharsInHttpURIHeader() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedMessageCount(1);
    template.sendBodyAndHeader("jetty:http://localhost:{{port}}/test?bridgeEndpoint=true","Hello World",Exchange.HTTP_URI,"/<>{}");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("jetty:http://localhost:{{port}}/test").to("mock:result");
      }
    }
;
  }
}
