@RunWith(Arquillian.class) @ImportResource("imported-context.xml") public class XmlThreadPoolTest {
  @Inject @Uri("direct:inbound") private ProducerTemplate inbound;
  @Deployment public static Archive<?> deployment(){
    return ShrinkWrap.create(JavaArchive.class).addPackage(CdiCamelExtension.class.getPackage()).addAsResource(Paths.get("src/test/resources/camel-context-threadPool.xml").toFile(),"imported-context.xml").addAsManifestResource(EmptyAsset.INSTANCE,"beans.xml");
  }
  @Test public void verifyThreadPool(  @Named("thread-pool") ExecutorService executor){
    assertThat("Thread pool is incorrect!",executor,is(notNullValue()));
  }
  @Test public void sendMessageToInbound(){
    String body=inbound.requestBody((Object)"message",String.class);
    assertThat("Body is incorrect!",body,matchesPattern("^Processed \\[message] with \\[Camel \\(test\\) thread #\\d+ - Thread Name]$"));
  }
}
