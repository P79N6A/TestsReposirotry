@RunWith(Arquillian.class) public class PropertiesMultipleLocationTest {
  @Deployment(name="multiple-locations") public static Archive<?> multipleLocationsDeployment(){
    return ShrinkWrap.create(JavaArchive.class).addPackage(CdiCamelExtension.class.getPackage()).addClass(MultipleLocations.class).addAsManifestResource(EmptyAsset.INSTANCE,"beans.xml");
  }
  @Test @OperateOnDeployment("multiple-locations") public void resolvePropertyFromLocations(  CamelContext context) throws Exception {
    assertThat("Property from classpath locations does not resolve!",context.resolvePropertyPlaceholders("{{foo.property}}"),is(equalTo("foo.value")));
    assertThat("Property from classpath locations does not resolve!",context.resolvePropertyPlaceholders("{{bar.property}}"),is(equalTo("bar.value")));
  }
}
