@ContextConfiguration public class MyServiceTest extends SpringRunWithTestSupport {
  @Autowired private MyServiceInvoker invoker;
  @Test public void testInvokeMyService() throws Exception {
    Double value=31.7D;
    Double actual=invoker.invokeService(value);
    Double expected=Math.sqrt(value);
    assertEquals("The result should be the square root",expected,actual);
  }
}
