abstract class BaseSpringPropertiesComponentTest extends SpringTestSupport {
  @Override protected AbstractXmlApplicationContext createApplicationContext(){
    return new ClassPathXmlApplicationContext("org/apache/camel/component/properties/SpringPropertiesComponentTest.xml");
  }
  @Test public void testSpringPropertiesComponentStart() throws Exception {
    getMockEndpoint("mock:result").expectedMessageCount(1);
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Test public void testSpringPropertiesComponentBar() throws Exception {
    getMockEndpoint("mock:bar").expectedMessageCount(1);
    template.sendBody("direct:bar","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Test public void testSpringPropertiesComponentStart2() throws Exception {
    getMockEndpoint("mock:result").expectedMessageCount(1);
    template.sendBody("direct:start2","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Test public void testSpringPropertiesComponentBar2() throws Exception {
    getMockEndpoint("mock:bar").expectedMessageCount(1);
    template.sendBody("direct:bar2","Hello World");
    assertMockEndpointsSatisfied();
  }
}
