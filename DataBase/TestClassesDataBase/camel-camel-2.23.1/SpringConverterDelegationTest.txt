@DirtiesContext @RunWith(SpringRunner.class) @EnableAutoConfiguration @SpringBootTest(properties="camel.springboot.typeConversion=true") public class SpringConverterDelegationTest extends Assert {
  @Autowired TypeConverter typeConverter;
  @Test public void shouldConvertUsingSpringConverter(){
    String result=typeConverter.convertTo(String.class,new Convertable());
    assertEquals("converted!",result);
  }
@Configuration public static class Config {
    @Bean ConvertableConverter convertableConverter(){
      return new ConvertableConverter();
    }
  }
public static class Convertable {
  }
public static class ConvertableConverter implements Converter<Convertable,String> {
    @Override public String convert(    Convertable source){
      return "converted!";
    }
  }
}
