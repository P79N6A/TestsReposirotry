public class DeadLetterChannelHandleNewExceptionTest extends ContextTestSupport {
  @Override public boolean isUseRouteBuilder(){
    return false;
  }
  @Test public void testDeadLetterChannelHandleNewException() throws Exception {
    context.addRoutes(new RouteBuilder(){
      @Override public void configure() throws Exception {
        errorHandler(deadLetterChannel("mock:dead").deadLetterHandleNewException(true));
        from("direct:start").log("Incoming ${body}").throwException(new IllegalArgumentException("Forced"));
      }
    }
);
    context.start();
    getMockEndpoint("mock:dead").expectedMessageCount(1);
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Test public void testDeadLetterChannelNotHandleNewException() throws Exception {
    context.addRoutes(new RouteBuilder(){
      @Override public void configure() throws Exception {
        errorHandler(deadLetterChannel("mock:dead").deadLetterHandleNewException(false));
        from("direct:start").log("Incoming ${body}").throwException(new IllegalArgumentException("Forced"));
      }
    }
);
    context.start();
    getMockEndpoint("mock:dead").expectedMessageCount(1);
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
  }
}
