@DirtiesContext @RunWith(SpringRunner.class) @SpringBootTest public class PlainTest {
  @Autowired ApplicationContext applicationContext;
  @Autowired(required=false) CamelContext camelContext;
@Configuration @EnableAutoConfiguration(exclude=CamelAutoConfiguration.class) public static class MyConfiguration {
  }
  @Test public void testPlain(){
    Assert.assertNull("Should not auto configure CamelContext",camelContext);
    Assert.assertEquals("Should not auto configure CamelAutoConfiguration",0,applicationContext.getBeanNamesForType(CamelAutoConfiguration.class).length);
  }
}
