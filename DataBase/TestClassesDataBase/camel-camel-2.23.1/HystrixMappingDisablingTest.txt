/** 
 * Testing that the hystrix mapping can be disabled.
 */
@RunWith(SpringRunner.class) @SpringBootApplication @DirtiesContext @ContextConfiguration(classes=HystrixMappingDisablingTest.class) @SpringBootTest(webEnvironment=SpringBootTest.WebEnvironment.RANDOM_PORT,properties={"camel.component.hystrix.mapping.enabled=false"}) public class HystrixMappingDisablingTest {
  @Autowired private TestRestTemplate restTemplate;
  @Test public void testHystrixServletMapping(){
    Assert.assertEquals(404,restTemplate.getForEntity("/hystrix.stream",String.class).getStatusCodeValue());
  }
}
