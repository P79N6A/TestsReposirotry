public class OnCompletionOnCompleteOnlyTest extends OnCompletionTest {
  @Test public void testSynchronizeFailure() throws Exception {
    getMockEndpoint("mock:sync").expectedMessageCount(0);
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedMessageCount(0);
    try {
      template.sendBody("direct:start","Kabom");
      fail("Should throw exception");
    }
 catch (    CamelExecutionException e) {
      assertEquals("Kabom",e.getCause().getMessage());
    }
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").onCompletion().onCompleteOnly().to("log:sync").to("mock:sync").end().process(new MyProcessor()).to("mock:result");
      }
    }
;
  }
}
