public class StreamRouteBuilderTest extends CamelTestSupport {
  @Test public void testStringContent(){
    template.sendBody("direct:start","this is text\n");
  }
  @Test public void testBinaryContent(){
    template.sendBody("direct:start","This is bytes\n".getBytes());
  }
  protected RouteBuilder createRouteBuilder(){
    return new RouteBuilder(){
      public void configure(){
        from("direct:start").setHeader("stream",constant(System.out)).to("stream:err","stream:out","stream:header");
      }
    }
;
  }
}
