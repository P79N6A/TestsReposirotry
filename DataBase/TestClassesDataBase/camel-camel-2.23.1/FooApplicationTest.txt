@RunWith(CamelSpringBootRunner.class) @SpringBootTest(classes=SampleCamelApplication.class,properties="greeting = Hello foo") @EnableRouteCoverage @Ignore public class FooApplicationTest {
  @Autowired private CamelContext camelContext;
  @Test public void shouldSayFoo() throws Exception {
    NotifyBuilder notify=new NotifyBuilder(camelContext).whenDone(1).create();
    assertTrue(notify.matches(10,TimeUnit.SECONDS));
  }
}
