@RunWith(CamelSpringBootRunner.class) @SpringBootTest(classes=SampleCamelApplication.class) @EnableRouteCoverage public class SampleCamelApplicationTest {
  @Autowired private CamelContext camelContext;
  @Test public void shouldProduceMessages() throws Exception {
    NotifyBuilder notify=new NotifyBuilder(camelContext).whenDone(1).create();
    assertTrue(notify.matches(10,TimeUnit.SECONDS));
  }
}
