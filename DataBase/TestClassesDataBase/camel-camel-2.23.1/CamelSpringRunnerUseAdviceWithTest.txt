@UseAdviceWith public class CamelSpringRunnerUseAdviceWithTest extends CamelSpringRunnerPlainTest {
  @Before public void testContextStarted() throws Exception {
    assertEquals(ServiceStatus.Stopped,camelContext.getStatus());
    assertEquals(ServiceStatus.Stopped,camelContext2.getStatus());
    camelContext.start();
    camelContext2.start();
    Thread.sleep(1000);
  }
  @Test public void testStopwatch(){
    StopWatch stopWatch=StopWatchTestExecutionListener.getStopWatch();
    assertNotNull(stopWatch);
    long taken=stopWatch.taken();
    assertTrue(taken + " > 0, but was: " + taken,taken > 0);
    assertTrue(taken + " < 3000, but was: " + taken,taken < 3000);
  }
}
