public class JettyOnExceptionHandledTest extends BaseJettyTest {
  @Test public void testJettyOnException() throws Exception {
    Exchange reply=template.request("http://localhost:{{port}}/myserver?throwExceptionOnFailure=false",null);
    assertNotNull(reply);
    assertEquals("Dude something went wrong",reply.getOut().getBody(String.class));
    assertEquals(500,reply.getOut().getHeader(Exchange.HTTP_RESPONSE_CODE));
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("jetty://http://localhost:{{port}}/myserver").onException(Exception.class).handled(true).transform(constant("Dude something went wrong")).setHeader(Exchange.HTTP_RESPONSE_CODE,constant(500)).end().throwException(new IllegalArgumentException("I cannot do this"));
      }
    }
;
  }
}
