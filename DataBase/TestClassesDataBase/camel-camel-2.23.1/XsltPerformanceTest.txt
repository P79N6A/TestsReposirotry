public class XsltPerformanceTest extends AbstractBasePerformanceTest {
  private final int count=10000;
  @Test public void testXslt() throws InterruptedException {
    template.setDefaultEndpointUri("direct:xslt");
    execute(1000);
    resetMock(count);
    StopWatch watch=new StopWatch();
    execute(count);
    assertMockEndpointsSatisfied();
    log.warn("Ran {} tests in {}ms",count,watch.taken());
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      public void configure() throws Exception {
        from("direct:xslt").to("xslt://META-INF/xslt/transform.xslt").to("mock:end");
      }
    }
;
  }
}
