@ContextConfiguration public class FluentProduceTemplateTest extends SpringRunWithTestSupport {
  @Autowired protected FluentProducerTemplate producer;
  @EndpointInject(uri="mock:result") protected MockEndpoint result;
  @Test public void testProducerTemplate() throws Exception {
    result.expectedBodiesReceived("hello");
    producer.withBody("hello").to("direct:start").send();
    result.assertIsSatisfied();
  }
}
