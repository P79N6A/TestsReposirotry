public class Jms1TopicDurableTest extends BaseJms2TestSupport {
  @Test public void testJms11DurableTopic() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedBodiesReceived("Hello World");
    MockEndpoint mock2=getMockEndpoint("mock:result2");
    mock2.expectedBodiesReceived("Hello World");
    Thread.sleep(500);
    template.sendBody("jms:topic:foo","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("jms:topic:foo?clientId=123&durableSubscriptionName=bar").to("mock:result");
        from("jms:topic:foo?clientId=456&durableSubscriptionName=bar").to("mock:result2");
      }
    }
;
  }
}
