public class AggregateCompleteAllOnStopTest extends ContextTestSupport {
  @Test public void testCompleteAllOnStop() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:aggregated");
    mock.expectedBodiesReceived("A+B","C");
    template.sendBodyAndHeader("seda:start","A","id","foo");
    template.sendBodyAndHeader("seda:start","B","id","foo");
    template.sendBodyAndHeader("seda:start","C","id","foo");
    context.stopRoute("foo");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("seda:start").routeId("foo").aggregate(header("id"),new BodyInAggregatingStrategy()).aggregationRepository(new MemoryAggregationRepository()).completionSize(2).completionTimeout(100).completeAllOnStop().completionTimeoutCheckerInterval(10).to("mock:aggregated");
      }
    }
;
  }
}
