@RunWith(Arquillian.class) public class NoCamelContextTest {
  @Any @Inject private Instance<CamelContext> contexts;
  @Deployment public static Archive<?> deployment(){
    return ShrinkWrap.create(JavaArchive.class).addPackage(CdiCamelExtension.class.getPackage()).addAsManifestResource(EmptyAsset.INSTANCE,"beans.xml");
  }
  @Test public void verifyDeployment(){
    assertThat("Camel context beans are deployed!",contexts.isUnsatisfied(),is(equalTo(true)));
  }
}
