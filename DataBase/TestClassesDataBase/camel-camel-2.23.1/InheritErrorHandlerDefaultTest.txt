public class InheritErrorHandlerDefaultTest extends ContextTestSupport {
  private static int counter;
  @Test public void testInheritErrorHandlerDefault() throws Exception {
    counter=0;
    getMockEndpoint("mock:result").expectedMessageCount(0);
    getMockEndpoint("mock:dead").expectedMessageCount(1);
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
    assertEquals(3,counter);
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        errorHandler(deadLetterChannel("mock:dead").maximumRedeliveries(2).redeliveryDelay(0));
        from("direct:start").process(new MyProcessor()).to("mock:result");
      }
    }
;
  }
public static class MyProcessor implements Processor {
    public void process(    Exchange exchange) throws Exception {
      counter++;
      exchange.setException(new IllegalArgumentException("Damn"));
    }
  }
}
