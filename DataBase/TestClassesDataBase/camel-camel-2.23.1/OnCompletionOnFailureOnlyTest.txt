public class OnCompletionOnFailureOnlyTest extends OnCompletionTest {
  @Test public void testSynchronizeComplete() throws Exception {
    getMockEndpoint("mock:sync").expectedMessageCount(0);
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedBodiesReceived("Bye World");
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").onCompletion().onFailureOnly().to("log:sync").to("mock:sync").end().process(new MyProcessor()).to("mock:result");
      }
    }
;
  }
}
