@SuppressWarnings("deprecation") public static class MyExpression extends org.apache.camel.impl.ExpressionAdapter {
  @Override public Object evaluate(  Exchange exchange){
    String body=exchange.getIn().getBody(String.class);
    if (body.contains("Camel")) {
      return "Yes Camel rocks";
    }
 else {
      return "Hello " + body;
    }
  }
}
