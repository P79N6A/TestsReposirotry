@RunWith(CamelSpringBootRunner.class) @SpringBootTest(classes=SampleAmqApplication.class) public class SampleAmqApplicationTests {
  @Autowired private CamelContext camelContext;

    assertTrue(notify.matches(10,TimeUnit.SECONDS));
  }
}
