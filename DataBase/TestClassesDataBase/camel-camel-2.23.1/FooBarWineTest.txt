public class FooBarWineTest extends CamelTestSupport {
  @Test public void testFooBar() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:bar");
    mock.expectedMinimumMessageCount(2);
    mock.setAssertPeriod(500);
    assertMockEndpointsSatisfied();
  }
  @Test public void testFooWine() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:wine");
    mock.expectedMinimumMessageCount(2);
    mock.setAssertPeriod(500);
    assertMockEndpointsSatisfied();
  }
  @Override protected RoutesBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("foo:hello?period=100").to("bar:Beer?amount=5").to("mock:bar");
        from("foo:hello2?period=50").to("wine:wine?amount=2").to("mock:wine");
      }
    }
;
  }
}
