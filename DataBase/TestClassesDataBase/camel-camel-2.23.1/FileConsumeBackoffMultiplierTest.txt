public class FileConsumeBackoffMultiplierTest extends ContextTestSupport {
  @Override @Before public void setUp() throws Exception {
    deleteDirectory("target/files");
    super.setUp();
    template.sendBodyAndHeader("file://target/files","Hello World",Exchange.FILE_NAME,"report.txt");
  }
  @Test public void testBackoffMultiplier() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedBodiesReceived("Hello World");
    assertMockEndpointsSatisfied();
    oneExchangeDone.matchesMockWaitTime();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      public void configure() throws Exception {
        from("file://target/files/?initialDelay=0&delay=10&delete=true&backoffMultiplier=4&backoffIdleThreshold=3").convertBodyTo(String.class).to("mock:result");
      }
    }
;
  }
}
