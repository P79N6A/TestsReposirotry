@RunWith(Arquillian.class) public class ConsumeMethodTest {
  @Inject @Uri("seda:inbound") private ProducerTemplate inbound;
  @Inject @Uri("mock:outbound") private MockEndpoint outbound;
  @Deployment public static Archive<?> deployment(){
    return ShrinkWrap.create(JavaArchive.class).addPackage(CdiCamelExtension.class.getPackage()).addClass(ConsumeMethodBean.class).addAsManifestResource(EmptyAsset.INSTANCE,"beans.xml");
  }
  @Test public void consumeAnnotation() throws InterruptedException {
    outbound.expectedMessageCount(1);
    outbound.expectedBodiesReceived("test");
    inbound.sendBody("test");
    assertIsSatisfied(2L,TimeUnit.SECONDS,outbound);
  }
}
