@ContextConfiguration public class FileToJmsTest extends AbstractJUnit4SpringContextTests {
  @Autowired protected ProducerTemplate template;
  @EndpointInject(uri="mock:result") protected MockEndpoint result;
  @Test public void testFileToJms() throws Exception {
    result.expectedBodiesReceived("Hello World");
    template.sendBodyAndHeader("file://target/jmsfile?fileExist=Override","Hello World",Exchange.FILE_NAME,"hello.txt");
    result.assertIsSatisfied();
  }
}
