public class RedeliveryErrorHandlerTwoXmlFilesIssueTest extends SpringTestSupport {
  @Override protected AbstractXmlApplicationContext createApplicationContext(){
    return new ClassPathXmlApplicationContext("org/apache/camel/spring/issues/RedeliveryErrorHandlerTwoXmlFilesIssueTest-1.xml","org/apache/camel/spring/issues/RedeliveryErrorHandlerTwoXmlFilesIssueTest-2.xml");
  }
  @Test public void testRedeliveryErrorHandlerTwoXmlFilesIssue() throws Exception {
    getMockEndpoint("mock:handled").expectedMessageCount(1);
    getMockEndpoint("mock:dead").expectedMessageCount(0);
    getMockEndpoint("mock:result").expectedMessageCount(0);
    template.sendBody("direct:start","Hello World");
    assertMockEndpointsSatisfied();
    OnRedeliveryCounter counter=context.getRegistry().lookupByNameAndType("counter",OnRedeliveryCounter.class);
    assertEquals(3,counter.getCounter());
  }
public static class OnRedeliveryCounter implements Processor {
    private int counter;
    @Override public void process(    Exchange exchange) throws Exception {
      counter++;
    }
    public int getCounter(){
      return counter;
    }
  }
}
