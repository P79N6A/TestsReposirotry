public class RestSwaggerHelperTest {
  @Test(expected=IllegalArgumentException.class) public void emptyHostParamsAreNotAllowed(){
    RestSwaggerHelper.isHostParam("");
  }
  @Test(expected=IllegalArgumentException.class) public void nonUriHostParametersAreNotAllowed(){
    RestSwaggerHelper.isHostParam("carrot");
  }
  @Test(expected=IllegalArgumentException.class) public void nullHostParamsAreNotAllowed(){
    RestSwaggerHelper.isHostParam(null);
  }
  @Test public void shouldNiceify(){
    assertThat(RestSwaggerHelper.isHostParam("http://api.example.com")).isEqualTo("http://api.example.com");
  }
  @Test public void shouldReturnUriParameter(){
    assertThat(RestSwaggerHelper.isHostParam("http://api.example.com")).isEqualTo("http://api.example.com");
  }
}
