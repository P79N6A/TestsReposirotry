public class EchoProduceTest extends CamelTestSupport {
  @Produce(uri="direct:start") private Echo echo;
  @Test public void testEchoProduce() throws Exception {
    String reply=echo.hello("Camel");
    assertEquals("Hello Camel",reply);
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").transform().simple("Hello ${body}");
      }
    }
;
  }
}
