public class Mina2TcpTextlineProtocolTest extends BaseMina2Test {
  @Test public void testMinaRoute() throws Exception {
    MockEndpoint endpoint=getMockEndpoint("mock:result");
    Object body="Hello there!";
    endpoint.expectedBodiesReceived(body);
    template.sendBodyAndHeader(String.format("mina2:tcp://localhost:%1$s?textline=true&sync=false",getPort()),body,"cheese",123);
    assertMockEndpointsSatisfied();
  }
  protected RouteBuilder createRouteBuilder(){
    return new RouteBuilder(){
      public void configure(){
        from(String.format("mina2:tcp://localhost:%1$s?textline=true&sync=false",getPort())).to("log:before?showAll=true").to("mock:result").to("log:after?showAll=true");
      }
    }
;
  }
}
