/** 
 * Unit test for wiki documentation (same as Mustache component)
 */
public class ChunkLetterTest extends CamelTestSupport {
  private Exchange createLetter(){
    Exchange exchange=context.getEndpoint("direct:a").createExchange();
    Message msg=exchange.getIn();
    msg.setHeader("firstName","Claus");
    msg.setHeader("lastName","Ibsen");
    msg.setHeader("item","Camel in Action");
    msg.setBody("PS: Next beer is on me, James");
    return exchange;
  }
  @Test public void testChunkLetter() throws Exception {
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedMessageCount(1);
    mock.expectedBodiesReceived("Dear Ibsen, Claus\n\nThanks for the order of Camel in Action." + "\n\nRegards Camel Riders Bookstore\nPS: Next beer is on me, James\n");
    template.send("direct:a",createLetter());
    mock.assertIsSatisfied();
  }
  protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      public void configure() throws Exception {
        from("direct:a").to("chunk:letter").to("mock:result");
      }
    }
;
  }
}
