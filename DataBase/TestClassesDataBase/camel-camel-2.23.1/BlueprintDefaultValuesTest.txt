/** 
 * A test showing that Blueprint XML property placeholders work correctly with placeholders containing colon in the key.
 */
public class BlueprintDefaultValuesTest extends CamelBlueprintTestSupport {
  @Override protected String getBlueprintDescriptor(){
    return "org/apache/camel/test/blueprint/default-values.xml";
  }
  @Test public void testPropertyResolutionFailure() throws Exception {
    getMockEndpoint("mock:result").expectedBodiesReceived("dot=my.value colon=my:value");
    template.sendBody("direct:start",null);
    assertMockEndpointsSatisfied();
  }
  @Override protected Properties useOverridePropertiesWithPropertiesComponent(){
    Properties extra=new Properties();
    extra.put("my:key","my:value");
    return extra;
  }
}
