public class AOPBeforeTest extends ContextTestSupport {
  @Test public void testAOPBefore() throws Exception {
    getMockEndpoint("mock:before").expectedBodiesReceived("Hello World");
    MockEndpoint mock=getMockEndpoint("mock:result");
    mock.expectedBodiesReceived("Bye World");
    String out=template.requestBody("direct:start","Hello World",String.class);
    assertEquals("Bye World",out);
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @SuppressWarnings("deprecation") @Override public void configure() throws Exception {
        from("direct:start").aop().before("mock:before").transform(constant("Bye World")).to("mock:result");
      }
    }
;
  }
}
