@XmlAccessorType(XmlAccessType.FIELD) @XmlType(name="Twit",propOrder={"text"}) @XmlRootElement(name="twit") class Twit implements Serializable {
  private static final long serialVersionUID=1L;
  @XmlElement(required=true) protected String text;
  public String getText(){
    return text;
  }
  public void setText(  String value){
    this.text=value;
  }
  @Override public String toString(){
    return text;
  }
}
