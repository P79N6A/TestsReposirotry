@RunWith(SpringRunner.class) @SpringBootApplication @DirtiesContext @ContextConfiguration(classes=DataSourceAutoConfigurationTest.class) @SpringBootTest(properties={"spring.datasource.url=jdbc:dummy://localhost/test","spring.datasource.username=dbuser","spring.datasource.password=dbpass","spring.datasource.driver-class-name=org.apache.camel.component.sql.support.DummyJDBCDriver"}) public class DataSourceAutoConfigurationTest {
  @Autowired private DataSource datasource;
  @Autowired private CamelContext context;
  @Test public void testInjectionWorks(){
    assertNotNull(datasource);
  }
  @Test public void testSqlComponentUsesTheConfiguredDatasource() throws Exception {
    SqlComponent component=(SqlComponent)context.getComponent("sql");
    SqlEndpoint endpoint=(SqlEndpoint)component.createEndpoint("sql:select * from table where id=#");
    assertEquals(datasource,endpoint.getJdbcTemplate().getDataSource());
  }
  @Test public void testSqlStoredComponentUsesTheConfiguredDatasource() throws Exception {
    SqlStoredComponent component=(SqlStoredComponent)context.getComponent("sql-stored");
    SqlStoredEndpoint endpoint=(SqlStoredEndpoint)component.createEndpoint("sql:select * from table where id=#");
    assertEquals(datasource,endpoint.getJdbcTemplate().getDataSource());
  }
}
