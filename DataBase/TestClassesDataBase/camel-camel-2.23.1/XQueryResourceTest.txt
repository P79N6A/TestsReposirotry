public class XQueryResourceTest extends CamelTestSupport {
  @Test public void testXPathResource() throws Exception {
    getMockEndpoint("mock:result").expectedBodiesReceived("London");
    template.sendBody("direct:start","<person name='James' city='London'/>");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").transform().xquery("resource:classpath:myxquery.txt",String.class).to("mock:result");
      }
    }
;
  }
}
