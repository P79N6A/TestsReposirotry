public class NettyTextlineInOnlyNullDelimiterTest extends BaseNettyTest {
  @Test public void testTextlineInOnlyNull() throws Exception {
    getMockEndpoint("mock:result").expectedBodiesReceived("Hello World");
    template.sendBody("netty4:tcp://localhost:{{port}}?textline=true&delimiter=NULL&sync=false","Hello World\u0000");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("netty4:tcp://localhost:{{port}}?textline=true&delimiter=NULL&sync=false").validate(body().isInstanceOf(String.class)).to("mock:result");
      }
    }
;
  }
}
