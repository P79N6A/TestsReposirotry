  @Test public void testShutdownAllTasks() throws Exception {
    context.startRoute("route1");
    MockEndpoint bar=getMockEndpoint("mock:bar");
    bar.expectedMinimumMessageCount(1);
    bar.setResultWaitTime(3000);
    assertMockEndpointsSatisfied();
    context.stop();
    Thread.sleep(1000);
    assertEquals("Should complete all messages",8,bar.getReceivedCounter());
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("ibatis:selectAllAccounts").noAutoStartup().routeId("route1").shutdownRunningTask(ShutdownRunningTask.CompleteAllTasks).delay(1000).to("seda:foo");
        from("seda:foo").routeId("route2").to("mock:bar");
      }
    }
;
  }
