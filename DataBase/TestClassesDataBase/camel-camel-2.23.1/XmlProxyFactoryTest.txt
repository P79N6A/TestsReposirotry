@RunWith(Arquillian.class) @ImportResource("imported-context.xml") public class XmlProxyFactoryTest {
  @Named @Inject private Service proxy;
  @Deployment public static Archive<?> deployment(){
    return ShrinkWrap.create(JavaArchive.class).addPackage(CdiCamelExtension.class.getPackage()).addAsResource(Paths.get("src/test/resources/camel-context-proxy.xml").toFile(),"imported-context.xml").addAsManifestResource(EmptyAsset.INSTANCE,"beans.xml");
  }
  @Test public void sendMessageToProxy(){
    String response=proxy.service("request");
    assertThat("Proxy response is incorrect!",response,is(equalTo("Service called with: [request]")));
  }
}
