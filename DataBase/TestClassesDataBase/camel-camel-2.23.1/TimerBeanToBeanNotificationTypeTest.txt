public class TimerBeanToBeanNotificationTypeTest extends CamelTestSupport {
  @Test public void testBeanToBean() throws Exception {
    getMockEndpoint("mock:notify").expectedMessageCount(1);
    getMockEndpoint("mock:notify").message(0).body().isInstanceOf(NotificationType.class);
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("timer:foo?delay=500&repeatCount=1").log("Timer triggered").bean(MyNotificationService.class,"createNotification").bean(MyNotificationService.class,"sendNotification");
      }
    }
;
  }
}
