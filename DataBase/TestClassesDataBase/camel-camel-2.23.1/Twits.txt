@XmlAccessorType(XmlAccessType.FIELD) @XmlType(name="",propOrder={"twits"}) @XmlRootElement(name="twits") class Twits implements Serializable {
  private static final long serialVersionUID=1L;
  @XmlElement(name="twit",required=true) protected List<Twit> twits;
  public List<Twit> getTwits(){
    if (twits == null) {
      twits=new ArrayList<>();
    }
    return this.twits;
  }
  @Override public String toString(){
    if (twits == null || twits.isEmpty()) {
      return super.toString();
    }
    return super.toString() + "[" + twits.get(0).toString()+ "]";
  }
}
