public class XPathResourceTest extends ContextTestSupport {
  @Test public void testXPathResource() throws Exception {
    getMockEndpoint("mock:result").expectedBodiesReceived("James");
    template.sendBody("direct:start","<person><name>James</name><city>London</city></person>");
    assertMockEndpointsSatisfied();
  }
  @Override protected RouteBuilder createRouteBuilder() throws Exception {
    return new RouteBuilder(){
      @Override public void configure() throws Exception {
        from("direct:start").transform().xpath("resource:classpath:myxpath.txt").to("mock:result");
      }
    }
;
  }
}
