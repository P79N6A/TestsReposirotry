public class EpollSocketRstTest extends SocketRstTest {
  @Override protected List<TestsuitePermutation.BootstrapComboFactory<ServerBootstrap,Bootstrap>> newFactories(){
    return EpollSocketTestPermutation.INSTANCE.socket();
  }
  @Override protected void assertRstOnCloseException(  IOException cause,  Channel clientChannel){
    if (!AbstractEpollChannel.class.isInstance(clientChannel)) {
      super.assertRstOnCloseException(cause,clientChannel);
      return;
    }
    assertTrue("actual [type, message]: [" + cause.getClass() + ", "+ cause.getMessage()+ "]",cause instanceof NativeIoException);
    assertEquals(Errors.ERRNO_ECONNRESET_NEGATIVE,((NativeIoException)cause).expectedErr());
  }
}
