/** 
 * Tests little-endian direct channel buffers
 */
public class PooledLittleEndianDirectByteBufTest extends AbstractPooledByteBufTest {
  @Override protected ByteBuf alloc(  int length,  int maxCapacity){
    ByteBuf buffer=PooledByteBufAllocator.DEFAULT.directBuffer(length,maxCapacity).order(ByteOrder.LITTLE_ENDIAN);
    assertSame(ByteOrder.LITTLE_ENDIAN,buffer.order());
    return buffer;
  }
}
