/** 
 * A test to validate that either order of initialization of the  {@link EmptyHttpHeaders#INSTANCE} and{@link HttpHeaders#EMPTY_HEADERS} field results in both fields being non-null.Since this is testing static initialization, the tests might not actually test anything, except when run in isolation.
 */
public class EmptyHttpHeadersInitializationTest {
  @Test public void testEmptyHttpHeadersFirst(){
    assertNotNull(EmptyHttpHeaders.INSTANCE);
    assertNotNull(HttpHeaders.EMPTY_HEADERS);
  }
  @Test public void testHttpHeadersFirst(){
    assertNotNull(HttpHeaders.EMPTY_HEADERS);
    assertNotNull(EmptyHttpHeaders.INSTANCE);
  }
}
