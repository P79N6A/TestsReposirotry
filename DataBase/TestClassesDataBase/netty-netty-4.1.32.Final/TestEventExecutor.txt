private static final class TestEventExecutor extends SingleThreadEventExecutor {
  TestEventExecutor(){
    super(null,Executors.defaultThreadFactory(),true);
  }
  @Override protected void run(){
    for (; ; ) {
      Runnable task=takeTask();
      if (task != null) {
        task.run();
        updateLastExecutionTime();
      }
      if (confirmShutdown()) {
        break;
      }
    }
  }
}
