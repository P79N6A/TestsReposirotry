public class InetSocketAddressResolverTest {
  @Test public void testCloseDelegates(){
    @SuppressWarnings("unchecked") NameResolver<InetAddress> nameResolver=mock(NameResolver.class);
    InetSocketAddressResolver resolver=new InetSocketAddressResolver(ImmediateEventExecutor.INSTANCE,nameResolver);
    resolver.close();
    verify(nameResolver,times(1)).close();
  }
}
