public static class UnOrderableRunnersAreHandledWithoutCrashing {
public static class UnOrderableRunner extends Runner {
    public UnOrderableRunner(    Class<?> klass){
    }
    @Override public Description getDescription(){
      return Description.EMPTY;
    }
    @Override public void run(    RunNotifier notifier){
    }
  }
@RunWith(UnOrderableRunner.class) public static class UnOrderable {
    @Test public void a(){
    }
  }
  @Test public void unOrderablesAreHandledWithoutCrashing(){
    Request unordered=Request.aClass(UnOrderable.class).orderWith(AlphanumericOrdering.INSTANCE);
    new JUnitCore().run(unordered);
  }
}
