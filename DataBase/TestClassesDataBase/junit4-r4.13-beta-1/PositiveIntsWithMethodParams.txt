@RunWith(Theories.class) public static class PositiveIntsWithMethodParams {
  @DataPoint public static final int ONE=1;
  private int x;
  public PositiveIntsWithMethodParams(  int x){
    assumeTrue(x > 0);
    this.x=x;
  }
  @Theory public void haveAPostiveSquare(  int y){
    assumeTrue(y > 0);
    assertTrue(x * y > 0);
  }
}
