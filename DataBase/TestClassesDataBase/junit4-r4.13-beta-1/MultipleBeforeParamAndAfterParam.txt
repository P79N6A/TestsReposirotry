@RunWith(Parameterized.class) @FixMethodOrder(MethodSorters.NAME_ASCENDING) public static class MultipleBeforeParamAndAfterParam {
  @Parameterized.BeforeParam public static void before1(){
    fLog+="before1() ";
  }
  @Parameterized.BeforeParam public static void before2(  String x){
    fLog+="before2(" + x + ") ";
  }
  @Parameterized.AfterParam public static void after2(){
    fLog+="after2() ";
  }
  @Parameterized.AfterParam public static void after1(  String x){
    fLog+="after1(" + x + ") ";
  }
  private final String x;
  public MultipleBeforeParamAndAfterParam(  String x){
    this.x=x;
  }
  @Parameters public static Collection<String> data(){
    return Arrays.asList("A","B");
  }
  @Test public void first(){
    fLog+="first(" + x + ") ";
  }
  @Test public void second(){
    fLog+="second(" + x + ") ";
  }
}
