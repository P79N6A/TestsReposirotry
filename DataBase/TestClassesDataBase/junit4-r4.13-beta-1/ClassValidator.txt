private static class ClassValidator extends AnnotatableValidator<TestClass> {
  @Override Iterable<TestClass> getAnnotatablesForTestClass(  TestClass testClass){
    return singletonList(testClass);
  }
  @Override List<Exception> validateAnnotatable(  AnnotationValidator validator,  TestClass testClass){
    return validator.validateAnnotatedClass(testClass);
  }
}
