public static class TestClassWithNamedDataPoints {
  @DataPoints({"field","named"}) public static String[] values=new String[]{"named field"};
  @DataPoints public static String[] otherValues=new String[]{"other"};
  @DataPoints({"method","named"}) public static String[] getValues(){
    return new String[]{"named method"};
  }
  @DataPoint({"single","named"}) public static String singleValue="named single value";
  @DataPoint public static String otherSingleValue="other value";
  @DataPoint({"singlemethod","named"}) public static String getSingleValue(){
    return "named single method value";
  }
  @DataPoint public static String getSingleOtherValue(){
    return "other single method value";
  }
  @DataPoints public static String[] getOtherValues(){
    return new String[]{"other method"};
  }
}
