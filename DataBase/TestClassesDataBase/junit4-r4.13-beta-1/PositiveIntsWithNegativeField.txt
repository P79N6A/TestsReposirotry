@RunWith(Theories.class) public static class PositiveIntsWithNegativeField {
  @DataPoint public static final int ONE=1;
  @DataPoint public static final int NEGONE=-1;
  private int x;
  public PositiveIntsWithNegativeField(  int x){
    assumeTrue(x > 0);
    this.x=x;
  }
  @Theory public void haveAPostiveSquare(){
    assertTrue(x > 0);
  }
}
