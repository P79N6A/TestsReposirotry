private static class FieldValidator extends AnnotatableValidator<FrameworkField> {
  @Override Iterable<FrameworkField> getAnnotatablesForTestClass(  TestClass testClass){
    return testClass.getAnnotatedFields();
  }
  @Override List<Exception> validateAnnotatable(  AnnotationValidator validator,  FrameworkField field){
    return validator.validateAnnotatedField(field);
  }
}
