/** 
 * Factory for creating  {@link Ordering} instances.<p>For a factory to be used with  {@code @OrderWith} it needs to have a public no-argconstructor.
 */
public interface Factory {
  /** 
 * Creates an Ordering instance using the given context. Implementations of this method that do not need to use the context can return the same instance every time.
 */
  Ordering create(  Context context);
}
