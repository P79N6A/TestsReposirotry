public class FixedDelayBackoffStrategyTest {
  @Test(expected=IllegalArgumentException.class) public void negativeBackoff_ThrowsException(){
    new FixedDelayBackoffStrategy(-1);
  }
  @Test(expected=IllegalArgumentException.class) public void zeroBackoff_ThrowsException(){
    new FixedDelayBackoffStrategy(0);
  }
  @Test public void positiveBackoff_ReturnsFixedBackoffOnDelay(){
    long delay=new FixedDelayBackoffStrategy(100).computeDelayBeforeNextRetry(RetryPolicyContexts.EMPTY);
    assertEquals(100,delay);
  }
}
