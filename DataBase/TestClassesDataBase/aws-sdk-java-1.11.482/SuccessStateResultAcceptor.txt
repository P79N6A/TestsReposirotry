class SuccessStateResultAcceptor extends WaiterAcceptor<MockDescribeResult> {
  int retryCount=0;
  public boolean matches(  MockDescribeResult result){
    return true;
  }
  public WaiterState getState(){
    if (retryCount <= 1) {
      retryCount++;
      return WaiterState.RETRY;
    }
    return WaiterState.SUCCESS;
  }
}
