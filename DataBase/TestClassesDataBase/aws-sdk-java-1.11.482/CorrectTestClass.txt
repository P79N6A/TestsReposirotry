@DynamoDBTable(tableName="nonexisting-test-tablename") public static class CorrectTestClass {
  private String hk;
  private S3Link s3;
  public CorrectTestClass(){
  }
  @DynamoDBHashKey public String getHk(){
    return hk;
  }
  public void setHk(  String hk){
    this.hk=hk;
  }
  public S3Link getS3(){
    return s3;
  }
  public void setS3(  S3Link s3){
    this.s3=s3;
  }
}
