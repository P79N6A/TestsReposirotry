public class ApacheConnectionManagerFactoryTest {
  private final ApacheConnectionManagerFactory factory=new ApacheConnectionManagerFactory();
  @Test public void validateAfterInactivityMillis_RespectedInConnectionManager(){
    final int validateAfterInactivity=1234;
    final HttpClientSettings httpClientSettings=HttpClientSettings.adapt(new ClientConfiguration().withValidateAfterInactivityMillis(validateAfterInactivity));
    final PoolingHttpClientConnectionManager connectionManager=(PoolingHttpClientConnectionManager)factory.create(httpClientSettings);
    assertEquals(validateAfterInactivity,connectionManager.getValidateAfterInactivity());
  }
}
