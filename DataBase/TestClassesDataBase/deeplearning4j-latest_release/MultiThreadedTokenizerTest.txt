public class MultiThreadedTokenizerTest {
  @Test public void testMultiThreadedBocchan() throws IOException, InterruptedException {
    assertMultiThreadedTokenizedStreamEquals(5,25,"deeplearning4j-nlp-japanese/bocchan-ipadic-features.txt","deeplearning4j-nlp-japanese/bocchan.txt",new Tokenizer());
  }
  @Test public void testMultiThreadedUserDictionary() throws IOException, InterruptedException {
    ClassPathResource cpr=new ClassPathResource("deeplearning4j-nlp-japanese/userdict.txt");
    assertMultiThreadedTokenizedStreamEquals(5,250,"deeplearning4j-nlp-japanese/jawikisentences-ipadic-features.txt","/jawikisentences.txt",new Tokenizer.Builder().userDictionary(cpr.getInputStream()).build());
  }
}
