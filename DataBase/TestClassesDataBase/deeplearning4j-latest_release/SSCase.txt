@Builder(builderClassName="Builder") @Data private static class SSCase {
  private int[] shape;
  private int[] begin;
  private int[] end;
  private int[] strides;
  private int beginMask;
  private int endMask;
  private int ellipsisMask;
  private int newAxisMask;
  private int shrinkAxisMask;
public static class Builder {
    public Builder shape(    int... shape){
      this.shape=shape;
      return this;
    }
    public Builder begin(    int... begin){
      this.begin=begin;
      return this;
    }
    public Builder end(    int... end){
      this.end=end;
      return this;
    }
    public Builder strides(    int... strides){
      this.strides=strides;
      return this;
    }
  }
}
