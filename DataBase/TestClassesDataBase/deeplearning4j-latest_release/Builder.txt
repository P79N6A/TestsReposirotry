public static class Builder {
  public Builder shape(  int... shape){
    this.shape=shape;
    return this;
  }
  public Builder begin(  int... begin){
    this.begin=begin;
    return this;
  }
  public Builder end(  int... end){
    this.end=end;
    return this;
  }
  public Builder strides(  int... strides){
    this.strides=strides;
    return this;
  }
}
