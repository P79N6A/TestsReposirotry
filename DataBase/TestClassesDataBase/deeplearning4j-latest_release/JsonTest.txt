public class JsonTest {
  @Test public void testJsonPreprocessors() throws Exception {
    InputPreProcessor[] pp=new InputPreProcessor[]{new KerasFlattenRnnPreprocessor(10,5),new PermutePreprocessor(new int[]{0,1,2}),new ReshapePreprocessor(new long[]{10,10},new long[]{100,1}),new TensorFlowCnnToFeedForwardPreProcessor()};
    for (    InputPreProcessor p : pp) {
      String s=NeuralNetConfiguration.mapper().writeValueAsString(p);
      InputPreProcessor p2=NeuralNetConfiguration.mapper().readValue(s,InputPreProcessor.class);
      assertEquals(p,p2);
    }
  }
}
