class UninstantiableThrowableTest {
  @RepeatedTest(2) void should_behave_consistently(  RepetitionInfo i){
    List mock=Mockito.mock(List.class);
    if (i.getCurrentRepetition() == 1) {
      Assertions.assertThatThrownBy(() -> Mockito.doThrow(UninstantiableException.class).when(mock).clear()).isInstanceOf(InstantiationError.class);
    }
    Mockito.doThrow(RuntimeException.class).when(mock).clear();
  }
abstract static class UninstantiableException extends RuntimeException {
  }
}
