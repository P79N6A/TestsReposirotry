public class AtLeastXVerificationTest extends TestBase {
  @Mock private List<String> mock;
  @Test public void shouldVerifyAtLeastXTimes() throws Exception {
    mock.clear();
    mock.clear();
    mock.clear();
    verify(mock,atLeast(2)).clear();
  }
  @Test public void shouldFailVerificationAtLeastXTimes() throws Exception {
    mock.add("one");
    verify(mock,atLeast(1)).add(anyString());
    try {
      verify(mock,atLeast(2)).add(anyString());
      fail();
    }
 catch (    MockitoAssertionError e) {
    }
  }
  @Test public void shouldAllowAtLeastZeroForTheSakeOfVerifyNoMoreInteractionsSometimes() throws Exception {
    mock.add("one");
    mock.clear();
    verify(mock,atLeast(0)).add("one");
    verify(mock,atLeast(0)).clear();
    verifyNoMoreInteractions(mock);
  }
}
