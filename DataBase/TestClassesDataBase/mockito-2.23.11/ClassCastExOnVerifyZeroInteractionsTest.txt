public class ClassCastExOnVerifyZeroInteractionsTest {
public interface TestMock {
    boolean m1();
  }
  @Test(expected=NoInteractionsWanted.class) public void should_not_throw_ClassCastException_when_mock_verification_fails(){
    TestMock test=mock(TestMock.class,new Answer<Object>(){
      public Object answer(      InvocationOnMock invocation) throws Throwable {
        return false;
      }
    }
);
    test.m1();
    verifyZeroInteractions(test);
  }
  @Test(expected=WrongTypeOfReturnValue.class) public void should_report_bogus_default_answer() throws Exception {
    TestMock test=mock(TestMock.class,new Answer<Object>(){
      public Object answer(      InvocationOnMock invocation) throws Throwable {
        return false;
      }
    }
);
    test.toString();
  }
}
