private static final class CustomSectionPayloadReader implements SectionPayloadReader {
  List<Integer> parsedTableIds;
  @Override public void init(  TimestampAdjuster timestampAdjuster,  ExtractorOutput extractorOutput,  TsPayloadReader.TrackIdGenerator idGenerator){
    parsedTableIds=new ArrayList<>();
  }
  @Override public void consume(  ParsableByteArray sectionData){
    parsedTableIds.add(sectionData.readUnsignedByte());
  }
}
