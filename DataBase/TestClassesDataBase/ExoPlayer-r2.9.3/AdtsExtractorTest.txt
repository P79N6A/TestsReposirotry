/** 
 * Unit test for  {@link AdtsExtractor}. 
 */
@RunWith(RobolectricTestRunner.class) public final class AdtsExtractorTest {
  @Test public void testSample() throws Exception {
    ExtractorAsserts.assertBehavior(AdtsExtractor::new,"ts/sample.adts");
  }
  @Test public void testSample_withSeeking() throws Exception {
    ExtractorAsserts.assertBehavior(() -> new AdtsExtractor(0,AdtsExtractor.FLAG_ENABLE_CONSTANT_BITRATE_SEEKING),"ts/sample_cbs.adts");
  }
}
