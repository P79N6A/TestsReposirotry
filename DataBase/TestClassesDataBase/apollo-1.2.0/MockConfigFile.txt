private static class MockConfigFile implements ConfigFile {
  private ConfigFileFormat m_configFileFormat;
  private String m_namespace;
  public MockConfigFile(  String namespace,  ConfigFileFormat configFileFormat){
    m_namespace=namespace;
    m_configFileFormat=configFileFormat;
  }
  @Override public String getContent(){
    return m_namespace + ":" + m_configFileFormat.getValue();
  }
  @Override public boolean hasContent(){
    return true;
  }
  @Override public String getNamespace(){
    return m_namespace;
  }
  @Override public ConfigFileFormat getConfigFileFormat(){
    return m_configFileFormat;
  }
  @Override public void addChangeListener(  ConfigFileChangeListener listener){
  }
  @Override public boolean removeChangeListener(  ConfigChangeListener listener){
    return false;
  }
  @Override public ConfigSourceType getSourceType(){
    return null;
  }
}
