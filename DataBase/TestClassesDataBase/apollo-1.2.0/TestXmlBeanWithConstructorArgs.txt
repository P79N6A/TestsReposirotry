public static class TestXmlBeanWithConstructorArgs {
  private final int timeout;
  private final int batch;
  public TestXmlBeanWithConstructorArgs(  int timeout,  int batch){
    this.timeout=timeout;
    this.batch=batch;
  }
  public int getTimeout(){
    return timeout;
  }
  public int getBatch(){
    return batch;
  }
}
