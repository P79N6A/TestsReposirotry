private static class MockConfig extends AbstractConfig {
  private final String m_namespace;
  public MockConfig(  String namespace){
    m_namespace=namespace;
  }
  @Override public String getProperty(  String key,  String defaultValue){
    if (key.equals("unknown")) {
      return null;
    }
    return m_namespace + ":" + key;
  }
  @Override public Set<String> getPropertyNames(){
    return null;
  }
  @Override public ConfigSourceType getSourceType(){
    return null;
  }
}
