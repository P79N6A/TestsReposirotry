public static class MockConfigRegistry implements ConfigRegistry {
  public static String NAMESPACE_REGISTERED="some-namespace-registered";
  public static ConfigFactory REGISTERED_CONFIGFACTORY=new ConfigFactory(){
    @Override public Config create(    String namespace){
      return null;
    }
    @Override public ConfigFile createConfigFile(    String namespace,    ConfigFileFormat configFileFormat){
      return null;
    }
  }
;
  @Override public void register(  String namespace,  ConfigFactory factory){
  }
  @Override public ConfigFactory getFactory(  String namespace){
    if (namespace.equals(NAMESPACE_REGISTERED)) {
      return REGISTERED_CONFIGFACTORY;
    }
    return null;
  }
}
