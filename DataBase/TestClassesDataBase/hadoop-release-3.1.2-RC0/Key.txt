private static class Key {
  private final String name;
  Key(  String name){
    this.name=name;
  }
  @Override public int hashCode(){
    throw new RuntimeException("should not be used!");
  }
  @Override public boolean equals(  Object o){
    if (!(o instanceof Key)) {
      return false;
    }
    Key other=(Key)o;
    return name.equals(other.name);
  }
}
