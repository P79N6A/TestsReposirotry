/** 
 * Tests for the HDFS implementation of  {@link GetGroups}
 */
public class TestGetGroups extends GetGroupsTestBase {
  private MiniDFSCluster cluster;
  @Before public void setUpNameNode() throws IOException {
    conf=new HdfsConfiguration();
    cluster=new MiniDFSCluster.Builder(conf).numDataNodes(0).build();
  }
  @After public void tearDownNameNode(){
    if (cluster != null) {
      cluster.shutdown();
      cluster=null;
    }
  }
  @Override protected Tool getTool(  PrintStream o){
    return new GetGroups(conf,o);
  }
}
