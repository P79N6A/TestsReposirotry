public abstract class RecordWriterForPush implements RecordWriter<BytesWritable,BytesWritable> {
  protected int count=0;
  RecordWriterForPush(){
  }
  @Override public abstract void write(  BytesWritable key,  BytesWritable value) throws IOException ;
  @Override public void close(  Reporter reporter) throws IOException {
  }
  public void reset(){
    count=0;
  }
}
