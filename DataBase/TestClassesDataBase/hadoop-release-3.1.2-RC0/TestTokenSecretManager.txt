public static class TestTokenSecretManager extends SecretManager<TestTokenIdentifier> {
  @Override public byte[] createPassword(  TestTokenIdentifier id){
    return id.getBytes();
  }
  @Override public byte[] retrievePassword(  TestTokenIdentifier id) throws InvalidToken {
    return id.getBytes();
  }
  @Override public TestTokenIdentifier createIdentifier(){
    return new TestTokenIdentifier();
  }
}
