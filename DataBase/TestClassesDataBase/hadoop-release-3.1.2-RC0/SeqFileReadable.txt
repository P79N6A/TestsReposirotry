static class SeqFileReadable implements KVReadable {
  private SequenceFile.Reader reader;
  private BytesWritable key;
  private BytesWritable value;
  public SeqFileReadable(  FileSystem fs,  Path path,  int osBufferSize) throws IOException {
    Configuration conf=new Configuration();
    conf.setInt("io.file.buffer.size",osBufferSize);
    reader=new SequenceFile.Reader(fs,path,conf);
    key=new BytesWritable();
    value=new BytesWritable();
  }
  @Override public byte[] getKey(){
    return key.getBytes();
  }
  @Override public int getKeyLength(){
    return key.getLength();
  }
  @Override public byte[] getValue(){
    return value.getBytes();
  }
  @Override public int getValueLength(){
    return value.getLength();
  }
  @Override public boolean next() throws IOException {
    return reader.next(key,value);
  }
  @Override public void close() throws IOException {
    reader.close();
  }
}
