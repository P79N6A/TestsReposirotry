@SuppressWarnings("serial") public static class JobEndServlet extends HttpServlet {
  public static volatile int calledTimes=0;
  public static URI requestUri;
  public static String baseUrl;
  public static String foundJobState;
  @Override public void doGet(  HttpServletRequest request,  HttpServletResponse response) throws ServletException, IOException {
    InputStreamReader in=new InputStreamReader(request.getInputStream());
    PrintStream out=new PrintStream(response.getOutputStream());
    calledTimes++;
    try {
      requestUri=new URI(null,null,request.getRequestURI(),request.getQueryString(),null);
      foundJobState=request.getParameter("status");
    }
 catch (    URISyntaxException e) {
    }
    in.close();
    out.close();
  }
}
