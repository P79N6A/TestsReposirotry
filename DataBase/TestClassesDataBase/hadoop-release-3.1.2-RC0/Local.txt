/** 
 * Container for the local exceptions, to help keep visible which specific class of exception.
 */
private static class Local {
  /** 
 * A local exception with a name to match the expected one.
 */
private static class ConnectTimeoutException extends InterruptedIOException {
    ConnectTimeoutException(    String s){
      super(s);
    }
  }
  /** 
 * A local exception whose name should not match.
 */
private static class NotAConnectTimeoutException extends InterruptedIOException {
  }
}
