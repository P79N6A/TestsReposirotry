/** 
 * Mock input stream based on a byte array so that it can be used by a FSDataInputStream.
 */
private static class MockInputStream extends ByteArrayInputStream implements Seekable, PositionedReadable {
  public MockInputStream(  byte[] buf){
    super(buf);
  }
  public int read(  long position,  byte[] buffer,  int offset,  int length){
    return -1;
  }
  public void readFully(  long position,  byte[] buffer,  int offset,  int length){
  }
  public void readFully(  long position,  byte[] buffer){
  }
  public void seek(  long position){
  }
  public long getPos(){
    return 0;
  }
  public boolean seekToNewSource(  long targetPos){
    return false;
  }
}
