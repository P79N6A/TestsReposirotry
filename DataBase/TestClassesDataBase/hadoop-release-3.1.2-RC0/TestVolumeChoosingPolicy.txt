/** 
 * A VolumeChoosingPolicy test stub used to verify that the storageId passed in is indeed in the list of volumes.
 * @param < V >
 */
private static class TestVolumeChoosingPolicy<V extends FsVolumeSpi> extends RoundRobinVolumeChoosingPolicy<V> {
  static String expectedStorageId;
  @Override public V chooseVolume(  List<V> volumes,  long replicaSize,  String storageId) throws IOException {
    assertEquals(expectedStorageId,storageId);
    return super.chooseVolume(volumes,replicaSize,storageId);
  }
}
