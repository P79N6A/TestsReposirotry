public static class TestTokenSelector implements TokenSelector<TestTokenIdentifier> {
  @SuppressWarnings("unchecked") @Override public Token<TestTokenIdentifier> selectToken(  Text service,  Collection<Token<? extends TokenIdentifier>> tokens){
    if (service == null) {
      return null;
    }
    for (    Token<? extends TokenIdentifier> token : tokens) {
      if (TestTokenIdentifier.KIND_NAME.equals(token.getKind()) && service.equals(token.getService())) {
        return (Token<TestTokenIdentifier>)token;
      }
    }
    return null;
  }
}
