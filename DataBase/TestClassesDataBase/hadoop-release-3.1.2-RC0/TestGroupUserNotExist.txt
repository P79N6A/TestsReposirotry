private class TestGroupUserNotExist extends ShellBasedUnixGroupsMapping {
  /** 
 * Create a ShellCommandExecutor object which returns exit code 1, emulating the case that the user does not exist.
 * @param userName not used
 * @return a mock ShellCommandExecutor object
 */
  @Override protected ShellCommandExecutor createGroupExecutor(  String userName){
    ShellCommandExecutor executor=mock(ShellCommandExecutor.class);
    try {
      doThrow(new ExitCodeException(1,"id: foobarusernotexist: No such user")).when(executor).execute();
      when(executor.getOutput()).thenReturn("");
    }
 catch (    IOException e) {
      TESTLOG.warn(e.getMessage());
    }
    return executor;
  }
}
