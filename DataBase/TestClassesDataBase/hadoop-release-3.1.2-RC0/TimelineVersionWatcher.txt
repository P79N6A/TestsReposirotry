@InterfaceAudience.Private @InterfaceStability.Unstable public class TimelineVersionWatcher extends TestWatcher {
  static final float DEFAULT_TIMELINE_VERSION=1.0f;
  private TimelineVersion version;
  @Override protected void starting(  Description description){
    version=description.getAnnotation(TimelineVersion.class);
  }
  /** 
 * @return the version number of timeline server for the current test (usingtimeline server v1.0 by default)
 */
  public float getTimelineVersion(){
    if (version == null) {
      return DEFAULT_TIMELINE_VERSION;
    }
    return version.value();
  }
}
