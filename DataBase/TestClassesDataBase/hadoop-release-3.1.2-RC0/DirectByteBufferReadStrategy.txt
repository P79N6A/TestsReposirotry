static class DirectByteBufferReadStrategy extends ReadStrategy {
  private DomainChannel ch=null;
  @Override public void init(  DomainSocket s) throws IOException {
    ch=s.getChannel();
  }
  @Override public int read(  byte b[],  int off,  int length) throws IOException {
    ByteBuffer buf=ByteBuffer.allocateDirect(b.length);
    int nread=ch.read(buf);
    if (nread < 0)     return nread;
    buf.flip();
    buf.get(b,off,nread);
    return nread;
  }
}
