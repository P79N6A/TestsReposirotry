class BlockingOperation extends MetadataOperation {
  private final CyclicBarrier ready;
  private final CyclicBarrier completed;
  BlockingOperation(  HiveSession parentSession,  OperationType opType,  CyclicBarrier ready,  CyclicBarrier completed){
    super(parentSession,opType);
    this.ready=ready;
    this.completed=completed;
  }
  @Override protected void runInternal() throws HiveSQLException {
    try {
      ready.await();
      completed.await();
    }
 catch (    InterruptedException|BrokenBarrierException e) {
      throw new RuntimeException(e);
    }
  }
  @Override public TableSchema getResultSetSchema() throws HiveSQLException {
    return null;
  }
  @Override public RowSet getNextRowSet(  FetchOrientation orientation,  long maxRows) throws HiveSQLException {
    return null;
  }
}
