public static class FastBytesHashMultiSetElement {
  private byte[] key;
  private int multiSetCount;
  public FastBytesHashMultiSetElement(  byte[] key){
    this.key=key;
    multiSetCount=1;
  }
  public byte[] getKey(){
    return key;
  }
  public int getMultiSetCount(){
    return multiSetCount;
  }
  public void incrementMultiSetCount(){
    multiSetCount++;
  }
}
