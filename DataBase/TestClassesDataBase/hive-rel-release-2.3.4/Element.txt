private static class Element {
  public Element(  int x){
    this.x=x;
  }
  int x;
  @Override public boolean equals(  Object o){
    if (this == o) {
      return true;
    }
    if (o == null || getClass() != o.getClass()) {
      return false;
    }
    Element element=(Element)o;
    return x == element.x;
  }
  @Override public int hashCode(){
    return x;
  }
}
