public static class FastLongHashMultiSetElement {
  private long key;
  private int multiSetCount;
  public FastLongHashMultiSetElement(  long key){
    this.key=key;
    multiSetCount=1;
  }
  public long getKey(){
    return key;
  }
  public int getMultiSetCount(){
    return multiSetCount;
  }
  public void incrementMultiSetCount(){
    multiSetCount++;
  }
}
