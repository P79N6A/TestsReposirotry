@UseReporter(JunitReporter.class) public class TestPrepPhase extends AbstractTestPhase {
  private PrepPhase phase;
  @Before public void setup() throws Exception {
    initialize(getClass().getSimpleName());
    createHostExecutor();
  }
  @After public void teardown(){
    FileUtils.deleteQuietly(baseDir);
  }
  @Test public void testExecute() throws Exception {
    phase=new PrepPhase(hostExecutors,localCommandFactory,templateDefaults,baseDir,(File)null,logger);
    phase.execute();
    Approvals.verify(getExecutedCommands());
  }
}
