static class MockInputStream extends FSInputStream {
  final MockFile file;
  int offset=0;
  public MockInputStream(  MockFile file) throws IOException {
    this.file=file;
  }
  @Override public void seek(  long offset) throws IOException {
    this.offset=(int)offset;
  }
  @Override public long getPos() throws IOException {
    return offset;
  }
  @Override public boolean seekToNewSource(  long l) throws IOException {
    return false;
  }
  @Override public int read() throws IOException {
    if (offset < file.length) {
      return file.content[offset++] & 0xff;
    }
    return -1;
  }
}
