/** 
 * Tests for  {@link MockMvcAutoConfiguration}.
 * @author Madhura Bhave
 */
public class MockMvcAutoConfigurationTests {
  private WebApplicationContextRunner contextRunner=new WebApplicationContextRunner().withConfiguration(AutoConfigurations.of(MockMvcAutoConfiguration.class));
  @Test public void registersDispatcherServletFromMockMvc(){
    this.contextRunner.run((context) -> {
      MockMvc mockMvc=context.getBean(MockMvc.class);
      assertThat(context).hasSingleBean(DispatcherServlet.class);
      assertThat(context.getBean(DispatcherServlet.class)).isEqualTo(mockMvc.getDispatcherServlet());
    }
);
  }
}
