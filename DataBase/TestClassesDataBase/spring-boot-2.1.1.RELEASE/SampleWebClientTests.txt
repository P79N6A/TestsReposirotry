/** 
 * Example integration test that uses  {@link TestRestTemplate}.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @SpringBootTest(webEnvironment=WebEnvironment.RANDOM_PORT) public class SampleWebClientTests {
  @Autowired private TestRestTemplate template;
  @Test public void testRequest(){
    HttpHeaders headers=this.template.getForEntity("/example",String.class).getHeaders();
    assertThat(headers.getLocation()).hasHost("other.example.com");
  }
@TestConfiguration static class Config {
    @Bean public RestTemplateBuilder restTemplateBuilder(){
      return new RestTemplateBuilder().setConnectTimeout(Duration.ofSeconds(1)).setReadTimeout(Duration.ofSeconds(1));
    }
  }
}
