/** 
 * Integration tests for  {@link JpaTagRepository}.
 * @author Andy Wilkinson
 */
@RunWith(SpringRunner.class) @SpringBootTest @Transactional public class JpaTagRepositoryIntegrationTests {
  @Autowired JpaTagRepository repository;
  @Test public void findsAllTags(){
    List<Tag> tags=this.repository.findAll();
    assertThat(tags).hasSize(3);
    for (    Tag tag : tags) {
      assertThat(tag.getNotes().size()).isGreaterThan(0);
    }
  }
}
