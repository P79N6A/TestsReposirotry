@Configuration static class TestTwoDataSourcesAndPrimaryConfiguration {
  @Bean @Primary public DataSource firstDataSource(){
    return createRandomDataSource();
  }
  @Bean public DataSource secondDataSource(){
    return createRandomDataSource();
  }
  private DataSource createRandomDataSource(){
    String url="jdbc:h2:mem:init-" + UUID.randomUUID();
    return DataSourceBuilder.create().url(url).build();
  }
}
