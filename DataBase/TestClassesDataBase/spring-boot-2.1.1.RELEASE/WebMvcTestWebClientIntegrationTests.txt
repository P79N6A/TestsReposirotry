/** 
 * Tests for  {@link WebMvcTest} with {@link WebClient}.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @WebMvcTest @WithMockUser public class WebMvcTestWebClientIntegrationTests {
  @Autowired private WebClient webClient;
  @Test public void shouldAutoConfigureWebClient() throws Exception {
    HtmlPage page=this.webClient.getPage("/html");
    assertThat(page.getBody().getTextContent()).isEqualTo("Hello");
  }
}
