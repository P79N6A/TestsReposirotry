@Configuration protected static class CustomDispatcherServletRegistration {
  @Bean(name=DispatcherServletAutoConfiguration.DEFAULT_DISPATCHER_SERVLET_REGISTRATION_BEAN_NAME) public ServletRegistrationBean<DispatcherServlet> dispatcherServletRegistration(  DispatcherServlet dispatcherServlet){
    ServletRegistrationBean<DispatcherServlet> registration=new ServletRegistrationBean<>(dispatcherServlet,"/foo");
    registration.setName("customDispatcher");
    return registration;
  }
}
