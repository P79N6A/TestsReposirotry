/** 
 * Sample tests for  {@link DataMongoTest} using reactive repositories.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @DataMongoTest public class DataMongoTestReactiveIntegrationTests {
  @Autowired private ReactiveMongoTemplate mongoTemplate;
  @Autowired private ExampleReactiveRepository exampleRepository;
  @Test public void testRepository(){
    ExampleDocument exampleDocument=new ExampleDocument();
    exampleDocument.setText("Look, new @DataMongoTest!");
    exampleDocument=this.exampleRepository.save(exampleDocument).block();
    assertThat(exampleDocument.getId()).isNotNull();
    assertThat(this.mongoTemplate.collectionExists("exampleDocuments").block()).isTrue();
  }
}
