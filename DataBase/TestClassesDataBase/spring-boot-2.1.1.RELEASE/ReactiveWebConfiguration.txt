@Configuration @EnableWebFlux @Controller static class ReactiveWebConfiguration {
  @Bean public DispatcherHandlersMappingDescriptionProvider dispatcherHandlersMappingDescriptionProvider(){
    return new DispatcherHandlersMappingDescriptionProvider();
  }
  @Bean public RouterFunction<ServerResponse> routerFunction(){
    return route(GET("/one"),(request) -> ServerResponse.ok().build()).andRoute(POST("/two"),(request) -> ServerResponse.ok().build());
  }
  @RequestMapping("/three") public void three(){
  }
}
