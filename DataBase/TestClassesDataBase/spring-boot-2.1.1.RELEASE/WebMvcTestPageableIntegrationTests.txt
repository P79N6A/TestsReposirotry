/** 
 * Integration tests for  {@link WebMvcTest} and Pageable support.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @WebMvcTest @WithMockUser public class WebMvcTestPageableIntegrationTests {
  @Autowired private MockMvc mvc;
  @Test public void shouldSupportPageable() throws Exception {
    this.mvc.perform(get("/paged").param("page","2").param("size","42")).andExpect(status().isOk()).andExpect(content().string("2:42"));
  }
}
