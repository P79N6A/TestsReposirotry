/** 
 * Test  {@link MockBean} on a configuration class can be used to replace existing beans.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) public class MockBeanOnConfigurationClassForExistingBeanIntegrationTests {
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking(){
    given(this.caller.getService().greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
@Configuration @MockBean(ExampleService.class) @Import({ExampleServiceCaller.class,FailingExampleService.class}) static class Config {
  }
}
