private static class Count<T> implements Supplier<T> {
  private final Supplier<T> source;
  private int count;
  Count(  Supplier<T> source){
    this.source=source;
  }
  @Override public T get(){
    this.count++;
    return this.source.get();
  }
  public int getCount(){
    return this.count;
  }
}
