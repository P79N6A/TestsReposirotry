@Configuration @ConditionalOnWebApplication(type=Type.REACTIVE) protected static class ReactiveWebApplicationConfiguration {
  @Bean public String reactive(){
    return "reactive";
  }
  @Bean public ReactiveWebServerFactory reactiveWebServerFactory(){
    return new MockReactiveWebServerFactory();
  }
  @Bean public HttpHandler httpHandler(){
    return (request,response) -> Mono.empty();
  }
}
