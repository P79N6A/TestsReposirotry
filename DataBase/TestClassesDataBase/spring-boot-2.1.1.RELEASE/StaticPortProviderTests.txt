/** 
 * Tests for  {@link StaticPortProvider}.
 * @author Phillip Webb
 */
public class StaticPortProviderTests {
  @Test public void portMustBePositive(){
    assertThatIllegalArgumentException().isThrownBy(() -> new StaticPortProvider(0)).withMessageContaining("Port must be positive");
  }
  @Test public void getPort(){
    StaticPortProvider provider=new StaticPortProvider(123);
    assertThat(provider.getPort()).isEqualTo(123);
  }
}
