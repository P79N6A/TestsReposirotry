@Configuration static class SecurityWebFilterChainConfig {
  @Bean SecurityWebFilterChain testSpringSecurityFilterChain(  ServerHttpSecurity http,  ReactiveJwtDecoder decoder){
    http.authorizeExchange().pathMatchers("/message/**").hasRole("ADMIN").anyExchange().authenticated().and().oauth2ResourceServer().jwt().jwtDecoder(decoder);
    return http.build();
  }
}
