/** 
 * Test  {@link MockBean} on a test class field can be used to replace existing beans.
 * @author Phillip Webb
 * @see MockBeanOnTestFieldForExistingBeanCacheIntegrationTests
 */
@RunWith(SpringRunner.class) @ContextConfiguration(classes=MockBeanOnTestFieldForExistingBeanConfig.class) public class MockBeanOnTestFieldForExistingBeanIntegrationTests {
  @MockBean private ExampleService exampleService;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking(){
    given(this.exampleService.greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
}
