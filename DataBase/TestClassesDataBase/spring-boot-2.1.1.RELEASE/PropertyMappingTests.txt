/** 
 * Integration tests for  {@link PropertyMapping @PropertyMapping} annotations.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @ExampleMapping(exampleProperty="abc") public class PropertyMappingTests {
  @Autowired private Environment environment;
  @Test public void hasProperty(){
    assertThat(this.environment.getProperty("example-property")).isEqualTo("abc");
  }
}
