/** 
 * Basic integration tests for demo application.
 * @author Ivan Sopov
 */
@RunWith(SpringRunner.class) @SpringBootTest(webEnvironment=WebEnvironment.RANDOM_PORT) public class SampleUndertowSslApplicationTests {
  @Autowired private TestRestTemplate restTemplate;
  @Autowired private AbstractConfigurableWebServerFactory webServerFactory;
  @Test public void testSsl(){
    assertThat(this.webServerFactory.getSsl().isEnabled()).isTrue();
  }
  @Test public void testHome(){
    ResponseEntity<String> entity=this.restTemplate.getForEntity("/",String.class);
    assertThat(entity.getStatusCode()).isEqualTo(HttpStatus.OK);
    assertThat(entity.getBody()).isEqualTo("Hello World");
  }
}
