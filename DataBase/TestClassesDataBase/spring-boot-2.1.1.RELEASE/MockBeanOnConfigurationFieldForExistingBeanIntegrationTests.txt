/** 
 * Test  {@link MockBean} on a field on a {@code @Configuration} class can be used toreplace existing beans.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) public class MockBeanOnConfigurationFieldForExistingBeanIntegrationTests {
  @Autowired private Config config;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking(){
    given(this.config.exampleService.greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
@Configuration @Import({ExampleServiceCaller.class,FailingExampleService.class}) static class Config {
    @MockBean private ExampleService exampleService;
  }
}
