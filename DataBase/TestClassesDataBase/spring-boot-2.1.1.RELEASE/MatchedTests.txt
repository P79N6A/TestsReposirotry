/** 
 * Tests for  {@link Matched}.
 * @author Phillip Webb
 */
public class MatchedTests {
  @Test public void byMatcherMatches(){
    assertThat("1234").is(Matched.by(startsWith("12")));
  }
  @Test public void byMatcherDoesNotMatch(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat("1234").is(Matched.by(startsWith("23")))).withMessageContaining("a string starting with \"23\"");
  }
  @Test public void whenMatcherMatches(){
    assertThat("1234").is(Matched.when(startsWith("12")));
  }
  @Test public void whenMatcherDoesNotMatch(){
    assertThatExceptionOfType(AssertionError.class).isThrownBy(() -> assertThat("1234").is(Matched.when(startsWith("23")))).withMessageContaining("a string starting with \"23\"");
  }
}
