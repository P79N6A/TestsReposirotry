/** 
 * Tests for  {@link WebMvcTest} when a specific controller is defined.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @WebMvcTest @WithMockUser @TestPropertySource(properties="spring.test.mockmvc.print=NONE") public class WebMvcTestPrintDefaultOverrideIntegrationTests {
  @Rule public OutputCapture output=new OutputCapture();
  @Autowired private MockMvc mvc;
  @Test public void shouldFindController1() throws Exception {
    this.mvc.perform(get("/one")).andExpect(content().string("one")).andExpect(status().isOk());
    assertThat(this.output.toString()).doesNotContain("Request URI = /one");
  }
}
