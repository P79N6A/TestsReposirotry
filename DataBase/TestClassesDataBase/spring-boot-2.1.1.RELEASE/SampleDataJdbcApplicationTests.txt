/** 
 * Integration tests for  {@link SampleDataJdbcApplication}.
 * @author Andy Wilkinson
 */
@RunWith(SpringRunner.class) @SpringBootTest public class SampleDataJdbcApplicationTests {
  @Autowired private WebApplicationContext context;
  private MockMvc mvc;
  @Before public void setUp(){
    this.mvc=MockMvcBuilders.webAppContextSetup(this.context).build();
  }
  @Test public void testCustomers() throws Exception {
    this.mvc.perform(get("/").param("name","merEDith")).andExpect(status().isOk()).andExpect(content().string(containsString("Meredith")));
  }
}
