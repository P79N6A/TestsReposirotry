/** 
 * Test  {@link SpyBean} on a field on a {@code @Configuration} class can be used to injectnew spy instances.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) public class SpyBeanOnConfigurationFieldForNewBeanIntegrationTests {
  @Autowired private Config config;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testSpying(){
    assertThat(this.caller.sayGreeting()).isEqualTo("I say simple");
    verify(this.config.exampleService).greeting();
  }
@Configuration @Import(ExampleServiceCaller.class) static class Config {
    @SpyBean private SimpleExampleService exampleService;
  }
}
