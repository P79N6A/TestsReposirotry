/** 
 * Test for  {@link BasicDataSourceExample}.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @SpringBootTest(properties="app.datasource.jdbcUrl=jdbc:h2:mem:basic;DB_CLOSE_DELAY=-1") @Import(BasicDataSourceExample.BasicDataSourceConfiguration.class) public class BasicDataSourceExampleTests {
  @Autowired private ApplicationContext context;
  @Test public void validateConfiguration() throws SQLException {
    assertThat(this.context.getBeansOfType(DataSource.class)).hasSize(1);
    DataSource dataSource=this.context.getBean(DataSource.class);
    assertThat(dataSource.getConnection().getMetaData().getURL()).isEqualTo("jdbc:h2:mem:basic");
  }
}
