/** 
 * Integration tests for using  {@link MockBean} with {@link DirtiesContext} and{@link ClassMode#BEFORE_EACH_TEST_METHOD}.
 * @author Andy Wilkinson
 */
@RunWith(SpringRunner.class) @DirtiesContext(classMode=ClassMode.BEFORE_EACH_TEST_METHOD) public class MockBeanWithDirtiesContextClassModeBeforeMethodIntegrationTests {
  @MockBean private ExampleService exampleService;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking() throws Exception {
    given(this.exampleService.greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
@Configuration @Import(ExampleServiceCaller.class) static class Config {
  }
}
