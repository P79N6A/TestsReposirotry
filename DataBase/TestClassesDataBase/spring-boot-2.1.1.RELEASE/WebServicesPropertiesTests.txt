/** 
 * Tests for  {@link WebServicesProperties}.
 * @author Madhura Bhave
 */
public class WebServicesPropertiesTests {
  private WebServicesProperties properties;
  @Test public void pathMustNotBeEmpty(){
    this.properties=new WebServicesProperties();
    assertThatIllegalArgumentException().isThrownBy(() -> this.properties.setPath("")).withMessageContaining("Path must have length greater than 1");
  }
  @Test public void pathMustHaveLengthGreaterThanOne(){
    this.properties=new WebServicesProperties();
    assertThatIllegalArgumentException().isThrownBy(() -> this.properties.setPath("/")).withMessageContaining("Path must have length greater than 1");
  }
  @Test public void customPathMustBeginWithASlash(){
    this.properties=new WebServicesProperties();
    assertThatIllegalArgumentException().isThrownBy(() -> this.properties.setPath("custom")).withMessageContaining("Path must start with '/'");
  }
}
