@Configuration @EnableWebMvc public static class WebServerWithEverythingUndertow {
  @Bean MultipartConfigElement multipartConfigElement(){
    return new MultipartConfigElement("");
  }
  @Bean UndertowServletWebServerFactory webServerFactory(){
    return new UndertowServletWebServerFactory();
  }
  @Bean WebController webController(){
    return new WebController();
  }
}
