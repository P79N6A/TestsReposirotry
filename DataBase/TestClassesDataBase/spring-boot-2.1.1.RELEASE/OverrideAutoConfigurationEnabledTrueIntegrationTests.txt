/** 
 * Integration tests for  {@link OverrideAutoConfiguration} when {@code enabled} is{@code true}.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @OverrideAutoConfiguration(enabled=true) @BootstrapWith(SpringBootTestContextBootstrapper.class) @ImportAutoConfiguration(ExampleTestConfig.class) public class OverrideAutoConfigurationEnabledTrueIntegrationTests {
  @Autowired private ApplicationContext context;
  @Test public void autoConfiguredContext(){
    ApplicationContext context=this.context;
    assertThat(context.getBean(OverrideAutoConfigurationSpringBootApplication.class)).isNotNull();
    assertThat(context.getBean(ConfigurationPropertiesBindingPostProcessor.class)).isNotNull();
  }
}
