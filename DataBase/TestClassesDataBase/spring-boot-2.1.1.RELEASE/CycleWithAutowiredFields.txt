@org.springframework.context.annotation.Configuration public static class CycleWithAutowiredFields {
  @Bean public BeanOne one(  BeanTwo two){
    return new BeanOne();
  }
public static class BeanTwoConfiguration {
    @SuppressWarnings("unused") @Autowired private BeanThree three;
    @Bean public BeanTwo two(){
      return new BeanTwo();
    }
  }
public static class BeanThreeConfiguration {
    @Bean public BeanThree three(    BeanOne one){
      return new BeanThree();
    }
  }
}
