/** 
 * Tests for the publishing documentation.
 * @author Andy Wilkinson
 * @author Jean-Baptiste Nizet
 */
@RunWith(GradleMultiDslSuite.class) public class PublishingDocumentationTests {
  @Rule public GradleBuild gradleBuild;
  @Test public void mavenUpload() throws IOException {
    assertThat(this.gradleBuild.script("src/main/gradle/publishing/maven").build("deployerRepository").getOutput()).contains("https://repo.example.com");
  }
  @Test public void mavenPublish() throws IOException {
    assertThat(this.gradleBuild.script("src/main/gradle/publishing/maven-publish").build("publishingConfiguration").getOutput()).contains("MavenPublication").contains("https://repo.example.com");
  }
}
