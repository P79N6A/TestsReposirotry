@org.springframework.context.annotation.Configuration static class CyclicBeanMethodsConfiguration {
  @Bean public BeanThree three(  BeanOne one){
    return new BeanThree();
  }
@org.springframework.context.annotation.Configuration static class InnerConfiguration {
    @Bean public BeanTwo two(    BeanThree three){
      return new BeanTwo();
    }
@Configuration static class InnerInnerConfiguration {
      @Bean public BeanOne one(      BeanTwo two){
        return new BeanOne();
      }
    }
  }
}
