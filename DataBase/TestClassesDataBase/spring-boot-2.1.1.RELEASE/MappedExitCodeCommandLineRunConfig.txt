@Configuration static class MappedExitCodeCommandLineRunConfig {
  @Bean public CommandLineRunner runner(){
    return (args) -> {
      throw new IllegalStateException();
    }
;
  }
  @Bean public ExitCodeExceptionMapper exceptionMapper(){
    return (exception) -> {
      if (exception instanceof IllegalStateException) {
        return 11;
      }
      return 0;
    }
;
  }
}
