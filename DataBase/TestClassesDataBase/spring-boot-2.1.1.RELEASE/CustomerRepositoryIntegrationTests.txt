/** 
 * Integration tests for  {@link CustomerRepository}.
 * @author Andy Wilkinson
 */
@RunWith(SpringRunner.class) @SpringBootTest @AutoConfigureTestDatabase public class CustomerRepositoryIntegrationTests {
  @Autowired private CustomerRepository repository;
  @Test public void findAllCustomers(){
    assertThat(this.repository.findAll()).hasSize(2);
  }
  @Test public void findByNameWithMatch(){
    assertThat(this.repository.findByName("joan")).hasSize(1);
  }
  @Test public void findByNameWithNoMatch(){
    assertThat(this.repository.findByName("hugh")).isEmpty();
  }
}
