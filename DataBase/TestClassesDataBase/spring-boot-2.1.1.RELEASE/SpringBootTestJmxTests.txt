/** 
 * Tests for disabling JMX by default
 * @author Dave Syer
 */
@RunWith(SpringRunner.class) @DirtiesContext @SpringBootTest public class SpringBootTestJmxTests {
  @Value("${spring.jmx.enabled}") private boolean jmx;
  @Test public void disabledByDefault(){
    assertThat(this.jmx).isFalse();
  }
@Configuration protected static class Config {
    @Bean public static PropertySourcesPlaceholderConfigurer propertySourcesPlaceholderConfigurer(){
      return new PropertySourcesPlaceholderConfigurer();
    }
  }
}
