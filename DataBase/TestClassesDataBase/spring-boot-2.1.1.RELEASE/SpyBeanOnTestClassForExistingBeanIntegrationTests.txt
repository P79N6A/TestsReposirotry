/** 
 * Test  {@link SpyBean} on a test class can be used to replace existing beans.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @SpyBean(SimpleExampleService.class) public class SpyBeanOnTestClassForExistingBeanIntegrationTests {
  @Autowired private ExampleServiceCaller caller;
  @Test public void testSpying(){
    assertThat(this.caller.sayGreeting()).isEqualTo("I say simple");
    verify(this.caller.getService()).greeting();
  }
@Configuration @Import({ExampleServiceCaller.class,SimpleExampleService.class}) static class Config {
  }
}
