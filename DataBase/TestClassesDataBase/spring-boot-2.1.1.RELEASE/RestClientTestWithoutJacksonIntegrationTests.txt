/** 
 * Tests for  {@link RestClientTest} without Jackson.
 * @author Andy Wilkinson
 */
@RunWith(ModifiedClassPathRunner.class) @ClassPathExclusions("jackson-*.jar") public class RestClientTestWithoutJacksonIntegrationTests {
  @Test public void restClientTestCanBeUsedWhenJacksonIsNotOnTheClassPath(){
    assertThat(ClassUtils.isPresent("com.fasterxml.jackson.databind.Module",getClass().getClassLoader())).isFalse();
    Result result=JUnitCore.runClasses(RestClientTestWithComponentIntegrationTests.class);
    assertThat(result.getFailureCount()).isEqualTo(0);
    assertThat(result.getRunCount()).isGreaterThan(0);
  }
}
