/** 
 * Tests for  {@link SpringBootTest} configured with a user-defined {@link RestTemplate}that is named  {@code testRestTemplate}.
 * @author Phillip Webb
 * @author Andy Wilkinson
 */
@RunWith(SpringRunner.class) @DirtiesContext @SpringBootTest(webEnvironment=WebEnvironment.RANDOM_PORT,properties={"value=123"}) public class SpringBootTestUserDefinedTestRestTemplateTests extends AbstractSpringBootTestWebServerWebEnvironmentTests {
  @Test public void restTemplateIsUserDefined(){
    assertThat(getContext().getBean("testRestTemplate")).isInstanceOf(RestTemplate.class);
  }
@Configuration @EnableWebMvc @RestController protected static class Config extends AbstractConfig {
    @Bean public RestTemplate testRestTemplate(){
      return new RestTemplate();
    }
  }
}
