/** 
 * Tests for  {@link AutoConfigurations}.
 * @author Phillip Webb
 */
public class AutoConfigurationsTests {
  @Test public void ofShouldCreateOrderedConfigurations(){
    Configurations configurations=AutoConfigurations.of(AutoConfigureA.class,AutoConfigureB.class);
    assertThat(Configurations.getClasses(configurations)).containsExactly(AutoConfigureB.class,AutoConfigureA.class);
  }
@AutoConfigureAfter(AutoConfigureB.class) public static class AutoConfigureA {
  }
public static class AutoConfigureB {
  }
}
