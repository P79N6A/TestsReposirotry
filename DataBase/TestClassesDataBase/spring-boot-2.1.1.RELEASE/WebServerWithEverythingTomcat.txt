@Configuration @EnableWebMvc public static class WebServerWithEverythingTomcat {
  @Bean MultipartConfigElement multipartConfigElement(){
    return new MultipartConfigElement("");
  }
  @Bean TomcatServletWebServerFactory webServerFactory(){
    return new TomcatServletWebServerFactory();
  }
  @Bean WebController webController(){
    return new WebController();
  }
}
