@Configuration protected static class CustomFlywayMigrationInitializer {
  @Bean public FlywayMigrationInitializer flywayMigrationInitializer(  Flyway flyway){
    FlywayMigrationInitializer initializer=new FlywayMigrationInitializer(flyway);
    initializer.setOrder(Ordered.HIGHEST_PRECEDENCE);
    return initializer;
  }
}
