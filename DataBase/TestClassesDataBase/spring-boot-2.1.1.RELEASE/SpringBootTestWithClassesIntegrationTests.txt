/** 
 * Tests for  {@link SpringBootTest} configured with specific classes.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @DirtiesContext @SpringBootTest(classes=SpringBootTestWithClassesIntegrationTests.Config.class) public class SpringBootTestWithClassesIntegrationTests {
  @Autowired private ApplicationContext context;
  @Test public void injectsOnlyConfig(){
    assertThat(this.context.getBean(Config.class)).isNotNull();
    assertThatExceptionOfType(NoSuchBeanDefinitionException.class).isThrownBy(() -> this.context.getBean(AdditionalConfig.class));
  }
@Configuration static class Config {
  }
@Configuration static class AdditionalConfig {
  }
}
