/** 
 * Tests for  {@link WebMvcTest} to validate converters are discovered.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @WebMvcTest(controllers=ExampleController2.class) @WithMockUser public class WebMvcTestConverterIntegrationTests {
  @Autowired private MockMvc mvc;
  @Test public void shouldFindConverter() throws Exception {
    String id=UUID.randomUUID().toString();
    this.mvc.perform(get("/two/" + id)).andExpect(content().string(id + "two")).andExpect(status().isOk());
  }
}
