/** 
 * Test  {@link MockBean} on a test class can be used to inject new mock instances.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @MockBean(ExampleService.class) public class MockBeanOnTestClassForNewBeanIntegrationTests {
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking(){
    given(this.caller.getService().greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
@Configuration @Import(ExampleServiceCaller.class) static class Config {
  }
}
