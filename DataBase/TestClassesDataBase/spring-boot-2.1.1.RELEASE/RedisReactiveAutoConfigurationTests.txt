/** 
 * Tests for  {@link RedisReactiveAutoConfiguration}.
 * @author Stephane Nicoll
 */
public class RedisReactiveAutoConfigurationTests {
  private ApplicationContextRunner contextRunner=new ApplicationContextRunner().withConfiguration(AutoConfigurations.of(RedisAutoConfiguration.class,RedisReactiveAutoConfiguration.class));
  @Test public void testDefaultRedisConfiguration(){
    this.contextRunner.run((context) -> {
      Map<String,?> beans=context.getBeansOfType(ReactiveRedisTemplate.class);
      assertThat(beans).containsOnlyKeys("reactiveRedisTemplate");
    }
);
  }
}
