/** 
 * Test  {@link MockBean} on a test class field can be used to inject new mock instances.
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) public class MockBeanOnTestFieldForNewBeanIntegrationTests {
  @MockBean private ExampleService exampleService;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testMocking(){
    given(this.exampleService.greeting()).willReturn("Boot");
    assertThat(this.caller.sayGreeting()).isEqualTo("I say Boot");
  }
@Configuration @Import(ExampleServiceCaller.class) static class Config {
  }
}
