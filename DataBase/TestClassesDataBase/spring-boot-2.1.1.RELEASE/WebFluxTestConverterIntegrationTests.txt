/** 
 * Tests for  {@link WebFluxTest} to validate converters are discovered.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @WithMockUser @WebFluxTest(controllers=ExampleController2.class) public class WebFluxTestConverterIntegrationTests {
  @Autowired private WebTestClient webClient;
  @Test public void shouldFindConverter(){
    UUID id=UUID.randomUUID();
    this.webClient.get().uri("/two/" + id).exchange().expectStatus().isOk().expectBody(String.class).isEqualTo(id + "two");
  }
}
