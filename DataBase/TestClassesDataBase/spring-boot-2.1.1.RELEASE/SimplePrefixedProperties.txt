@EnableConfigurationProperties @ConfigurationProperties(prefix="com.example",ignoreUnknownFields=false) static class SimplePrefixedProperties {
  private int foo;
  private String bar;
  public void setBar(  String bar){
    this.bar=bar;
  }
  public String getBar(){
    return this.bar;
  }
  public int getFoo(){
    return this.foo;
  }
  public void setFoo(  int foo){
    this.foo=foo;
  }
}
