@Configuration static class FooEndpointAndExtensionEnabledByDefaultTrueConfiguration {
  @Bean @ConditionalOnEnabledEndpoint public FooEndpointEnabledByDefaultTrue foo(){
    return new FooEndpointEnabledByDefaultTrue();
  }
  @Bean @ConditionalOnEnabledEndpoint public FooEndpointExtensionEnabledByDefaultTrue fooExt(){
    return new FooEndpointExtensionEnabledByDefaultTrue();
  }
}
