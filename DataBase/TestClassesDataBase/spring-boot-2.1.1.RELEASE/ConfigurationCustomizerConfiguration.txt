@Configuration static class ConfigurationCustomizerConfiguration {
  @Bean @Order(1) public FlywayConfigurationCustomizer customizerOne(){
    return (configuration) -> configuration.connectRetries(5).ignorePendingMigrations(true);
  }
  @Bean @Order(0) public FlywayConfigurationCustomizer customizerTwo(){
    return (configuration) -> configuration.connectRetries(10).ignoreMissingMigrations(true);
  }
}
