/** 
 * Tests for  {@link WebFluxTest} when no explicit controller is defined.
 * @author Stephane Nicoll
 */
@RunWith(SpringRunner.class) @WithMockUser @WebFluxTest public class WebFluxTestAllControllersIntegrationTests {
  @Autowired private WebTestClient webClient;
  @Test public void shouldFindController1(){
    this.webClient.get().uri("/one").exchange().expectStatus().isOk().expectBody(String.class).isEqualTo("one");
  }
  @Test public void shouldFindController2(){
    this.webClient.get().uri("/two").exchange().expectStatus().isOk().expectBody(String.class).isEqualTo("two");
  }
  @Test public void webExceptionHandling(){
    this.webClient.get().uri("/one/error").exchange().expectStatus().isBadRequest();
  }
  @Test public void shouldFindJsonController(){
    this.webClient.get().uri("/json").exchange().expectStatus().isOk();
  }
}
