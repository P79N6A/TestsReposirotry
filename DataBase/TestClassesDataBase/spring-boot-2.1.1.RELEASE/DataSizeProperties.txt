@EnableConfigurationProperties @ConfigurationProperties(prefix="test") static class DataSizeProperties {
  private DataSize size;
  @DataSizeUnit(DataUnit.KILOBYTES) private DataSize anotherSize;
  public DataSize getSize(){
    return this.size;
  }
  public void setSize(  DataSize size){
    this.size=size;
  }
  public DataSize getAnotherSize(){
    return this.anotherSize;
  }
  public void setAnotherSize(  DataSize anotherSize){
    this.anotherSize=anotherSize;
  }
}
