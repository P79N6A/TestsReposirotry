/** 
 * Tests for  {@link LocalServerPort}.
 * @author Anand Shah
 * @author Phillip Webb
 */
@RunWith(SpringRunner.class) @TestPropertySource(properties="local.server.port=8181") public class LocalServerPortTests {
  @Value("${local.server.port}") private String fromValue;
  @LocalServerPort private String fromAnnotation;
  @Test public void testLocalServerPortAnnotation(){
    assertThat(this.fromAnnotation).isNotNull().isEqualTo(this.fromValue);
  }
@Configuration static class Config {
  }
}
