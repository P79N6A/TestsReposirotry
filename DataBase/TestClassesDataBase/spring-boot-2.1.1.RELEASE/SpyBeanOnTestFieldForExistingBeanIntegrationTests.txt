/** 
 * Test  {@link SpyBean} on a test class field can be used to replace existing beans.
 * @author Phillip Webb
 * @see SpyBeanOnTestFieldForExistingBeanCacheIntegrationTests
 */
@RunWith(SpringRunner.class) @ContextConfiguration(classes=SpyBeanOnTestFieldForExistingBeanConfig.class) public class SpyBeanOnTestFieldForExistingBeanIntegrationTests {
  @SpyBean private ExampleService exampleService;
  @Autowired private ExampleServiceCaller caller;
  @Test public void testSpying(){
    assertThat(this.caller.sayGreeting()).isEqualTo("I say simple");
    verify(this.caller.getService()).greeting();
  }
}
