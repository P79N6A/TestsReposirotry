/** 
 * POJO with hashCode.
 */
public static class POJOWithHashCode extends POJOWithoutHashCode {
  public POJOWithHashCode(){
  }
  public POJOWithHashCode(  int[] id){
    super(id);
  }
  @Override public int hashCode(){
    int hash=31;
    for (    int i : getId()) {
      hash=37 * hash + i;
    }
    return hash;
  }
}
