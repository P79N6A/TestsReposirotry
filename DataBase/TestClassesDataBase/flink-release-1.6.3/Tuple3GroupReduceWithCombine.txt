private static class Tuple3GroupReduceWithCombine implements GroupReduceFunction<Tuple3<Integer,Long,String>,Tuple2<Integer,String>>, GroupCombineFunction<Tuple3<Integer,Long,String>,Tuple3<Integer,Long,String>> {
  private static final long serialVersionUID=1L;
  @Override public void combine(  Iterable<Tuple3<Integer,Long,String>> values,  Collector<Tuple3<Integer,Long,String>> out){
    Tuple3<Integer,Long,String> o=new Tuple3<>(0,0L,"");
    for (    Tuple3<Integer,Long,String> t : values) {
      o.f0+=t.f0;
      o.f1=t.f1;
      o.f2="test" + o.f1;
    }
    out.collect(o);
  }
  @Override public void reduce(  Iterable<Tuple3<Integer,Long,String>> values,  Collector<Tuple2<Integer,String>> out){
    int i=0;
    String s="";
    for (    Tuple3<Integer,Long,String> t : values) {
      i+=t.f0;
      s=t.f2;
    }
    out.collect(new Tuple2<>(i,s));
  }
}
