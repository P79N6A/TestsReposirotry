/** 
 * Another small POJO.
 */
@SuppressWarnings("unused") public static class SmallPojo2 {
  public SmallPojo2(){
  }
  public SmallPojo2(  String a,  int b,  double c,  String d,  Integer[] e){
    this.a=a;
    this.b=b;
    this.c=c;
    this.d=d;
    this.e=e;
  }
  public String a;
  public int b;
  public double c;
  public String d;
  public Integer[] e;
  @Override public String toString(){
    return a + "," + b+ ","+ c+ ","+ d+ ","+ Arrays.toString(e);
  }
}
