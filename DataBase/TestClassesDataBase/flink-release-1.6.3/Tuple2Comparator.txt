private static class Tuple2Comparator<T0 extends Comparable<T0>,T1 extends Comparable<T1>> implements Comparator<Tuple2<T0,T1>> {
  @Override public int compare(  Tuple2<T0,T1> o1,  Tuple2<T0,T1> o2){
    int cmp=o1.f0.compareTo(o2.f0);
    if (cmp != 0) {
      return cmp;
    }
    return o1.f1.compareTo(o2.f1);
  }
}
