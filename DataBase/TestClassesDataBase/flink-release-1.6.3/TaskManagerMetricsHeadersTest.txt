/** 
 * Tests for  {@link TaskManagerMetricsHeaders}.
 */
public class TaskManagerMetricsHeadersTest extends TestLogger {
  private final TaskManagerMetricsHeaders taskManagerMetricsHeaders=TaskManagerMetricsHeaders.getInstance();
  @Test public void testUrl(){
    assertThat(taskManagerMetricsHeaders.getTargetRestEndpointURL(),equalTo("/taskmanagers/:" + TaskManagerIdPathParameter.KEY + "/metrics"));
  }
  @Test public void testMessageParameters(){
    assertThat(taskManagerMetricsHeaders.getUnresolvedMessageParameters(),instanceOf(TaskManagerMetricsMessageParameters.class));
  }
}
