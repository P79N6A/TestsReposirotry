private static class Tuple2toTuple2GroupReduce implements KvGroupReduce<Long,Tuple2<Integer,Long>,Tuple2<Integer,Long>,Tuple2<Integer,Long>> {
  @Override public void combine(  Iterable<Tuple2<Long,Tuple2<Integer,Long>>> values,  Collector<Tuple2<Long,Tuple2<Integer,Long>>> out) throws Exception {
    int i=0;
    long l=0;
    long key=0;
    for (    Tuple2<Long,Tuple2<Integer,Long>> value : values) {
      key=value.f0;
      Tuple2<Integer,Long> extracted=value.f1;
      i+=extracted.f0;
      l+=extracted.f1;
    }
    Tuple2<Integer,Long> result=new Tuple2<>(i,l);
    out.collect(new Tuple2<>(key,result));
  }
  @Override public void reduce(  Iterable<Tuple2<Long,Tuple2<Integer,Long>>> values,  Collector<Tuple2<Long,Tuple2<Integer,Long>>> out) throws Exception {
    combine(values,out);
  }
}
