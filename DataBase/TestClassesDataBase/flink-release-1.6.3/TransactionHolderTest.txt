/** 
 * Unit tests  {@link TransactionHolder}.
 */
public class TransactionHolderTest {
  @Test public void testElapsedTime(){
    final long elapsedTime=new TransactionHolder<>(new Object(),0).elapsedTime(Clock.fixed(Instant.ofEpochMilli(1000),ZoneOffset.UTC));
    assertThat(elapsedTime,equalTo(1000L));
  }
}
