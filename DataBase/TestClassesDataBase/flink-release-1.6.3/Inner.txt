/** 
 * POJO for testing field access.
 */
public static class Inner {
  public long x;
  public boolean b;
  public Inner(){
  }
  public Inner(  long x){
    this.x=x;
  }
  public Inner(  long x,  boolean b){
    this.x=x;
    this.b=b;
  }
  @Override public String toString(){
    return ((Long)x).toString() + ", " + b;
  }
}
