/** 
 * Record type for the speed test. <p>NOTE: needs to be <tt>public</tt> to allow deserialization!
 */
public static class SpeedTestRecord implements IOReadableWritable {
  private static final int RECORD_SIZE=128;
  private final byte[] buf=new byte[RECORD_SIZE];
  public SpeedTestRecord(){
    for (int i=0; i < RECORD_SIZE; ++i) {
      this.buf[i]=(byte)(i % 128);
    }
  }
  @Override public void write(  DataOutputView out) throws IOException {
    out.write(this.buf);
  }
  @Override public void read(  DataInputView in) throws IOException {
    in.readFully(this.buf);
  }
}
