private static class ByteArrayGroupReduce implements GroupReduceFunction<Tuple2<byte[],Integer>,Integer> {
  @Override public void reduce(  Iterable<Tuple2<byte[],Integer>> values,  Collector<Integer> out) throws Exception {
    int sum=0;
    for (    Tuple2<byte[],Integer> value : values) {
      sum+=value.f1;
    }
    out.collect(sum);
  }
}
