private static class Tuple2TimestampExtractor implements AssignerWithPunctuatedWatermarks<Tuple2<String,Integer>> {
  @Override public long extractTimestamp(  Tuple2<String,Integer> element,  long previousTimestamp){
    return element.f1;
  }
  @Override public Watermark checkAndGetNextWatermark(  Tuple2<String,Integer> lastElement,  long extractedTimestamp){
    return new Watermark(lastElement.f1 - 1);
  }
}
