/** 
 * Representation of point int 2d plane.
 */
public static class Point implements Serializable {
  public double x, y;
  public Point(){
  }
  public Point(  double x,  double y){
    this.x=x;
    this.y=y;
  }
  public Point add(  Point other){
    x+=other.x;
    y+=other.y;
    return this;
  }
  public Point div(  long val){
    x/=val;
    y/=val;
    return this;
  }
  public void clear(){
    x=y=0.0;
  }
  @Override public String toString(){
    return x + " " + y;
  }
}
