/** 
 * Our custom version of  {@link JavaSerializer} for checking whether restore with a registeredserializer works when no serializer was previously registered. <p>This  {@code Serializer} can only be used for writing, not for reading. With this weverify that state that was serialized without a registered  {@code Serializer} is in factnot restored with a  {@code Serializer} that was later registered.
 */
public static class CustomKryoTestSerializer extends JavaSerializer {
  @Override public void write(  Kryo kryo,  Output output,  Object object){
    super.write(kryo,output,object);
  }
  @Override public Object read(  Kryo kryo,  Input input,  Class type){
    throw new ExpectedKryoTestException();
  }
}
