private static class FullMetricInfo {
  private final String id;
  private final double min;
  private final double max;
  private final double avg;
  private final double sum;
  @JsonCreator private FullMetricInfo(  @JsonProperty("id") String id,  @JsonProperty("min") double min,  @JsonProperty("max") double max,  @JsonProperty("avg") double avg,  @JsonProperty("sum") double sum){
    this.id=id;
    this.min=min;
    this.max=max;
    this.avg=avg;
    this.sum=sum;
  }
}
