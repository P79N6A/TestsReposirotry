/** 
 * A factory to create a RestartPipelinedRegionStrategy that uses a direct (synchronous) executor for easier testing.
 */
private static class FailoverPipelinedRegionWithDirectExecutor implements Factory {
  @Override public FailoverStrategy create(  ExecutionGraph executionGraph){
    return new RestartPipelinedRegionStrategy(executionGraph,Executors.directExecutor());
  }
}
