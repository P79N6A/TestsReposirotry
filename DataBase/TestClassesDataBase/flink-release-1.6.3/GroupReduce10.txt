private static class GroupReduce10 implements GroupReduceFunction<Tuple2<Long,Long>,Boolean> {
  @Override public void reduce(  Iterable<Tuple2<Long,Long>> values,  Collector<Boolean> out) throws Exception {
    Iterator<Tuple2<Long,Long>> it=values.iterator();
    boolean f=it.hasNext();
    if (!f) {
      System.out.println();
    }
    if (f) {
      System.out.println();
    }
    out.collect(f);
  }
}
