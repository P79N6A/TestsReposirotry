private static class InputReturningTuple3GroupReduce implements GroupReduceFunction<Tuple3<Integer,Long,String>,Tuple3<Integer,Long,String>> {
  private static final long serialVersionUID=1L;
  @Override public void reduce(  Iterable<Tuple3<Integer,Long,String>> values,  Collector<Tuple3<Integer,Long,String>> out){
    for (    Tuple3<Integer,Long,String> t : values) {
      if (t.f0 < 4) {
        t.f2="Hi!";
        t.f0+=10;
        out.collect(t);
        t.f0+=10;
        t.f2="Hi again!";
        out.collect(t);
      }
    }
  }
}
