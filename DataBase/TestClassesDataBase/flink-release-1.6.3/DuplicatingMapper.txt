/** 
 * {@link Mapper} that duplicates records.
 */
public static class DuplicatingMapper implements Mapper<IntWritable,Text,IntWritable,Text> {
  @Override public void map(  final IntWritable k,  final Text v,  final OutputCollector<IntWritable,Text> out,  final Reporter r) throws IOException {
    out.collect(k,v);
    out.collect(k,new Text(v.toString().toUpperCase()));
  }
  @Override public void configure(  final JobConf arg0){
  }
  @Override public void close() throws IOException {
  }
}
