private static class ByteArrayIO implements IOReadableWritable {
  private final byte[] bytes;
  public ByteArrayIO(  byte[] bytes){
    this.bytes=bytes;
  }
  @Override public void write(  DataOutputView out) throws IOException {
    out.write(bytes);
  }
  @Override public void read(  DataInputView in) throws IOException {
    in.readFully(bytes);
  }
}
