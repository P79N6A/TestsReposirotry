@ForwardedFields("f0->*") private static class GroupReduce4 implements GroupReduceFunction<Tuple2<Long,Long>,Long> {
  @Override public void reduce(  Iterable<Tuple2<Long,Long>> values,  Collector<Long> out) throws Exception {
    Long id=0L;
    for (    Tuple2<Long,Long> value : values) {
      id=value.f0;
    }
    out.collect(id);
  }
}
