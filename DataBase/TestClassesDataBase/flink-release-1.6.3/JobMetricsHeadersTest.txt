/** 
 * Tests for  {@link JobMetricsHeaders}.
 */
public class JobMetricsHeadersTest extends TestLogger {
  private final JobMetricsHeaders jobMetricsHeaders=JobMetricsHeaders.getInstance();
  @Test public void testUrl(){
    assertThat(jobMetricsHeaders.getTargetRestEndpointURL(),equalTo("/jobs/:" + JobIDPathParameter.KEY + "/metrics"));
  }
  @Test public void testMessageParameters(){
    assertThat(jobMetricsHeaders.getUnresolvedMessageParameters(),instanceOf(JobMetricsMessageParameters.class));
  }
}
