@SuppressWarnings("serial") class SerializableMapCreator implements MapCreator, Serializable {
  private int add=0;
  public SerializableMapCreator(  int add){
    this.add=add;
  }
  @Override public MapFunction<Integer,Integer> getMap(){
    return new MapFunction<Integer,Integer>(){
      @Override public Integer map(      Integer value) throws Exception {
        return value + add;
      }
    }
;
  }
}
