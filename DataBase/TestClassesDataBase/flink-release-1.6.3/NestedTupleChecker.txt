@SuppressWarnings("serial") private static class NestedTupleChecker implements OrderChecker<Tuple2<Tuple2<Integer,Integer>,String>> {
  @Override public boolean inOrder(  Tuple2<Tuple2<Integer,Integer>,String> t1,  Tuple2<Tuple2<Integer,Integer>,String> t2){
    return t1.f0.f1 < t2.f0.f1 || t1.f0.f1.equals(t2.f0.f1) && t1.f1.compareTo(t2.f1) >= 0;
  }
}
