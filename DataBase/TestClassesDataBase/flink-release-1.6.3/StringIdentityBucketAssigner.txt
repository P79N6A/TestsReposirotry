/** 
 * A simple  {@link BucketAssigner} that accepts {@code String}'s and returns the element itself as the bucket id.
 */
static class StringIdentityBucketAssigner implements BucketAssigner<String,String> {
  private static final long serialVersionUID=1L;
  @Override public String getBucketId(  String element,  BucketAssigner.Context context){
    return element;
  }
  @Override public SimpleVersionedSerializer<String> getSerializer(){
    return SimpleVersionedStringSerializer.INSTANCE;
  }
}
