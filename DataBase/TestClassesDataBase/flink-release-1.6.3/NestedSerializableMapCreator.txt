@SuppressWarnings("serial") class NestedSerializableMapCreator implements MapCreator, Serializable {
  private int add=0;
  private InnerSerializableMapCreator inner;
  public NestedSerializableMapCreator(  int add){
    this.add=add;
    this.inner=new InnerSerializableMapCreator();
  }
  @Override public MapFunction<Integer,Integer> getMap(){
    return inner.getMap();
  }
class InnerSerializableMapCreator implements MapCreator, Serializable {
    @Override public MapFunction<Integer,Integer> getMap(){
      return new MapFunction<Integer,Integer>(){
        @Override public Integer map(        Integer value) throws Exception {
          return value + add;
        }
      }
;
    }
  }
}
