static class TupleToStringBucketer implements BucketAssigner<Tuple2<String,Integer>,String> {
  private static final long serialVersionUID=1L;
  @Override public String getBucketId(  Tuple2<String,Integer> element,  Context context){
    return element.f0;
  }
  @Override public SimpleVersionedSerializer<String> getSerializer(){
    return SimpleVersionedStringSerializer.INSTANCE;
  }
}
