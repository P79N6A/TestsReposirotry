@ForwardedFieldsFirst("f1") private static class AntiJoinVisits implements CoGroupFunction<Tuple3<Integer,String,Integer>,Tuple1<String>,Tuple3<Integer,String,Integer>> {
  @Override public void coGroup(  Iterable<Tuple3<Integer,String,Integer>> ranks,  Iterable<Tuple1<String>> visits,  Collector<Tuple3<Integer,String,Integer>> out){
    if (!visits.iterator().hasNext()) {
      for (      Tuple3<Integer,String,Integer> next : ranks) {
        out.collect(next);
      }
    }
  }
}
