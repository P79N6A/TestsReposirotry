static class FaultyJobIDPathParameter extends MessagePathParameter<JobID> {
  FaultyJobIDPathParameter(){
    super(JOB_ID_KEY);
  }
  @Override protected JobID convertFromString(  String value) throws ConversionException {
    return JobID.fromHexString(value);
  }
  @Override protected String convertToString(  JobID value){
    return "foobar";
  }
}
