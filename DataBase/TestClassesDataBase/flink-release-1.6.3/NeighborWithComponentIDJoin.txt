@ForwardedFieldsFirst("f1->f1") @ForwardedFieldsSecond("f1->f0") private static final class NeighborWithComponentIDJoin implements JoinFunction<Tuple2<Long,Long>,Tuple2<Long,Long>,Tuple2<Long,Long>> {
  @Override public Tuple2<Long,Long> join(  Tuple2<Long,Long> vertexWithComponent,  Tuple2<Long,Long> edge){
    return new Tuple2<Long,Long>(edge.f1,vertexWithComponent.f1);
  }
}
