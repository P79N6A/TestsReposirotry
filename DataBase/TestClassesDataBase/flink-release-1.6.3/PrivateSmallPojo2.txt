/** 
 * Another small POJO with private fields.
 */
@SuppressWarnings("unused") public static class PrivateSmallPojo2 {
  public PrivateSmallPojo2(){
  }
  public PrivateSmallPojo2(  String a,  int b,  double c,  String d){
    this.a=a;
    this.b=b;
    this.c=c;
    this.d=d;
  }
  private String a;
  private int b;
  private double c;
  private String d;
  public String getA(){
    return a;
  }
  public void setA(  String a){
    this.a=a;
  }
  public int getB(){
    return b;
  }
  public void setB(  int b){
    this.b=b;
  }
  public double getC(){
    return c;
  }
  public void setC(  double c){
    this.c=c;
  }
  public String getD(){
    return d;
  }
  public void setD(  String d){
    this.d=d;
  }
  @Override public String toString(){
    return a + "," + b+ ","+ c+ ","+ d;
  }
}
