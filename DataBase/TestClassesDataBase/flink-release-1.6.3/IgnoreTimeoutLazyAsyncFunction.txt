/** 
 * A special  {@link LazyAsyncFunction} for timeout handling.Complete the result future with 3 times the input when the timeout occurred.
 */
private static class IgnoreTimeoutLazyAsyncFunction extends LazyAsyncFunction {
  private static final long serialVersionUID=1428714561365346128L;
  @Override public void timeout(  Integer input,  ResultFuture<Integer> resultFuture) throws Exception {
    resultFuture.complete(Collections.singletonList(input * 3));
  }
}
