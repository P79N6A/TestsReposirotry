@Configuration @EnableCaching static class MultiCacheManagerConfigurer extends CachingConfigurerSupport {
  @Bean public CacheManager cm1(){
    return new NoOpCacheManager();
  }
  @Bean public CacheManager cm2(){
    return new NoOpCacheManager();
  }
  @Override public CacheManager cacheManager(){
    return cm1();
  }
  @Override public KeyGenerator keyGenerator(){
    return null;
  }
}
