@Controller @RequestMapping("/*.do") static class MyAdaptedController2 {
  @RequestMapping public void myHandle(  HttpServletRequest request,  HttpServletResponse response) throws IOException {
    response.getWriter().write("test");
  }
  @RequestMapping("/myPath2.do") public void myHandle(  @RequestParam("param1") String p1,  int param2,  HttpServletResponse response,  @RequestHeader("header1") String h1,  @CookieValue("cookie1") String c1) throws IOException {
    response.getWriter().write("test-" + p1 + "-"+ param2+ "-"+ h1+ "-"+ c1);
  }
  @RequestMapping("/myPath3") public void myHandle(  TestBean tb,  HttpServletResponse response) throws IOException {
    response.getWriter().write("test-" + tb.getName() + "-"+ tb.getAge());
  }
  @RequestMapping("/myPath4.*") public void myHandle(  TestBean tb,  Errors errors,  HttpServletResponse response) throws IOException {
    response.getWriter().write("test-" + tb.getName() + "-"+ errors.getFieldError("age").getCode());
  }
}
