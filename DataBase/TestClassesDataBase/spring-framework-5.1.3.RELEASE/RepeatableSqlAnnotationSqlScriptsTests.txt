/** 
 * This is a copy of  {@link TransactionalSqlScriptsTests} that verifies properhandling of  {@link Sql @Sql} as a {@link Repeatable} annotation.
 * @author Sam Brannen
 * @since 4.1
 */
@FixMethodOrder(MethodSorters.NAME_ASCENDING) @ContextConfiguration(classes=EmptyDatabaseConfig.class) @Sql("schema.sql") @Sql("data.sql") @DirtiesContext public class RepeatableSqlAnnotationSqlScriptsTests extends AbstractTransactionalJUnit4SpringContextTests {
  @Test public void test01_classLevelScripts(){
    assertNumUsers(1);
  }
  @Test @Sql("drop-schema.sql") @Sql("schema.sql") @Sql("data.sql") @Sql("data-add-dogbert.sql") public void test02_methodLevelScripts(){
    assertNumUsers(2);
  }
  protected void assertNumUsers(  int expected){
    assertEquals("Number of rows in the 'user' table.",expected,countRowsInTable("user"));
  }
}
