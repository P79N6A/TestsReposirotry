@Controller static class PrototypeController {
  private int count;
  @InitBinder public void initBinder(  WebDataBinder dataBinder){
    this.count++;
  }
  @ModelAttribute public void populate(  Model model){
    this.count++;
  }
  @RequestMapping("/") public void message(  int param,  Writer writer) throws IOException {
    this.count++;
    writer.write("count:" + this.count);
  }
}
