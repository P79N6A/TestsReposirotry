@CacheConfig(cacheNames="testCache") static class FooServiceImpl implements FooService {
  private final AtomicLong counter=new AtomicLong();
  @Override @Cacheable public Object getSimple(  Object key){
    return this.counter.getAndIncrement();
  }
  @Override @Cacheable(condition="@bar.enabled") public Object getWithCondition(  Object key){
    return this.counter.getAndIncrement();
  }
}
