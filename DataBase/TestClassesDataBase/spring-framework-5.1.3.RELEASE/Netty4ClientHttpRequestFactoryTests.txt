public class Netty4ClientHttpRequestFactoryTests extends AbstractHttpRequestFactoryTestCase {
  private static EventLoopGroup eventLoopGroup;
  @BeforeClass public static void createEventLoopGroup(){
    eventLoopGroup=new NioEventLoopGroup();
  }
  @AfterClass public static void shutdownEventLoopGroup() throws InterruptedException {
    eventLoopGroup.shutdownGracefully().sync();
  }
  @Override protected ClientHttpRequestFactory createRequestFactory(){
    return new Netty4ClientHttpRequestFactory(eventLoopGroup);
  }
  @Override @Test public void httpMethods() throws Exception {
    super.httpMethods();
    assertHttpMethod("patch",HttpMethod.PATCH);
  }
}
