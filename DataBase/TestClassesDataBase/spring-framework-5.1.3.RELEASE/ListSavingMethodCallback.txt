private static class ListSavingMethodCallback implements ReflectionUtils.MethodCallback {
  private List<String> methodNames=new LinkedList<>();
  private List<Method> methods=new LinkedList<>();
  @Override public void doWith(  Method m) throws IllegalArgumentException, IllegalAccessException {
    this.methodNames.add(m.getName());
    this.methods.add(m);
  }
  public List<String> getMethodNames(){
    return this.methodNames;
  }
  @SuppressWarnings("unused") public List<Method> getMethods(){
    return this.methods;
  }
}
