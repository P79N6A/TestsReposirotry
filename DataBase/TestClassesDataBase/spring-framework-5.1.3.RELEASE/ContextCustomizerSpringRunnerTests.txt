/** 
 * JUnit 4 based integration test which verifies support of {@link ContextCustomizerFactory} and {@link ContextCustomizer}.
 * @author Sam Brannen
 * @author Phillip Webb
 * @since 4.3
 */
@RunWith(SpringRunner.class) @BootstrapWith(CustomTestContextBootstrapper.class) public class ContextCustomizerSpringRunnerTests {
  @Autowired String foo;
  @Test public void injectedBean(){
    assertEquals("foo",foo);
  }
static class CustomTestContextBootstrapper extends DefaultTestContextBootstrapper {
    @Override protected List<ContextCustomizerFactory> getContextCustomizerFactories(){
      return singletonList((ContextCustomizerFactory)(testClass,configAttributes) -> (ContextCustomizer)(context,mergedConfig) -> context.getBeanFactory().registerSingleton("foo","foo"));
    }
  }
}
