 * @since 3.0
 */
@ContextConfiguration @RunWith(SpringJUnit4ClassRunner.class) public class AutowiredQualifierTests {
  @Autowired private String foo;
  @Autowired @Qualifier("customFoo") private String customFoo;
  @Test public void test(){
    assertThat(foo,equalTo("normal"));
    assertThat(customFoo,equalTo("custom"));
  }
}
