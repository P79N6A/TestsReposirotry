@SuppressWarnings("serial") protected static class TestStaticPointcutAdvice extends StaticMethodMatcherPointcutAdvisor {
  private String pattern;
  public TestStaticPointcutAdvice(  MethodInterceptor mi,  String pattern){
    super(mi);
    this.pattern=pattern;
  }
  @Override public boolean matches(  Method m,  @Nullable Class<?> targetClass){
    return m.getName().contains(pattern);
  }
}
