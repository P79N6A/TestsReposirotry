@Configuration @Order(2) class OverridingSingletonBeanConfig {
  public OverridingSingletonBeanConfig(  ObjectProvider<SingletonBeanConfig> other){
    other.getObject();
  }
  public @Bean ExtendedFoo foo(){
    return new ExtendedFoo();
  }
  public @Bean Bar bar(){
    return new Bar(foo());
  }
}
