public static class OptionalFieldResourceInjectionBean implements TestBeanHolder {
  @Autowired(required=false) @Lazy private TestBean testBean;
  @Autowired(required=false) @Lazy private List<TestBean> testBeans;
  public TestBean getTestBean(){
    return this.testBean;
  }
  public List<TestBean> getTestBeans(){
    return this.testBeans;
  }
}
