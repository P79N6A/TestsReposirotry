class SelectivePrototypeTargetSourceCreator extends AbstractBeanFactoryBasedTargetSourceCreator {
  @Override protected AbstractBeanFactoryBasedTargetSource createBeanFactoryBasedTargetSource(  Class<?> beanClass,  String beanName){
    if (!beanName.startsWith("prototype")) {
      return null;
    }
    return new PrototypeTargetSource();
  }
}
