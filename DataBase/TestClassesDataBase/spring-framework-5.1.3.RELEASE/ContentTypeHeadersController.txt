@Controller public static class ContentTypeHeadersController {
  @RequestMapping(value="/something",headers="content-type=application/pdf") public void handlePdf(  Writer writer) throws IOException {
    writer.write("pdf");
  }
  @RequestMapping(value="/something",headers="content-type=text/*") public void handleHtml(  Writer writer) throws IOException {
    writer.write("text");
  }
}
