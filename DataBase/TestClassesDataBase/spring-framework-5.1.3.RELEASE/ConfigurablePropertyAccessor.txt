private static class ConfigurablePropertyAccessor implements PropertyAccessor {
  private final Map<String,Object> values;
  public ConfigurablePropertyAccessor(  Map<String,Object> values){
    this.values=values;
  }
  @Override public Class<?>[] getSpecificTargetClasses(){
    return null;
  }
  @Override public boolean canRead(  EvaluationContext context,  Object target,  String name){
    return true;
  }
  @Override public TypedValue read(  EvaluationContext context,  Object target,  String name){
    return new TypedValue(this.values.get(name));
  }
  @Override public boolean canWrite(  EvaluationContext context,  Object target,  String name){
    return false;
  }
  @Override public void write(  EvaluationContext context,  Object target,  String name,  Object newValue){
  }
}
