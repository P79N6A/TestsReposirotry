@Configuration @EnableScheduling static class AspectConfig {
  @Bean public static AnnotationAwareAspectJAutoProxyCreator autoProxyCreator(){
    AnnotationAwareAspectJAutoProxyCreator apc=new AnnotationAwareAspectJAutoProxyCreator();
    apc.setProxyTargetClass(true);
    return apc;
  }
  @Bean public static MyAspect myAspect(){
    return new MyAspect();
  }
}
