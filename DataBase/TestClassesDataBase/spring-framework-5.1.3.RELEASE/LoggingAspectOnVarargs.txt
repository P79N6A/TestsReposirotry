@Aspect @SuppressWarnings("serial") public static class LoggingAspectOnVarargs implements Serializable {
  @Around("execution(* doWithVarargs(*))") public Object doLog(  ProceedingJoinPoint pjp) throws Throwable {
    LogFactory.getLog(LoggingAspectOnVarargs.class).debug(Arrays.asList(pjp.getArgs()));
    return pjp.proceed();
  }
}
