@Controller public static class DefaultValueParamController {
  @RequestMapping("/myPath.do") public void myHandle(  @RequestParam(value="id",defaultValue="foo") String id,  @RequestParam(value="otherId",defaultValue="") String id2,  @RequestHeader(defaultValue="bar") String header,  HttpServletResponse response) throws IOException {
    response.getWriter().write(String.valueOf(id) + "-" + String.valueOf(id2)+ "-"+ String.valueOf(header));
  }
}
