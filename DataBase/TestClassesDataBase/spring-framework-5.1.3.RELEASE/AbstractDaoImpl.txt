static abstract class AbstractDaoImpl<T,S> implements Dao<T,S> {
  protected T object;
  protected S otherObject;
  protected AbstractDaoImpl(  T object,  S otherObject){
    this.object=object;
    this.otherObject=otherObject;
  }
  @Override public S loadFromParent(){
    return otherObject;
  }
}
