@WebAppConfiguration @ContextHierarchy(@ContextConfiguration) public class RootWacEarTests extends EarTests {
@Configuration static class RootWacConfig {
    @Bean public String root(){
      return "root";
    }
  }
  @Autowired private WebApplicationContext wac;
  @Autowired private String ear;
  @Autowired private String root;
  @Ignore("Superseded by verifyRootWacConfig()") @Test @Override public void verifyEarConfig(){
  }
  @Test public void verifyRootWacConfig(){
    ApplicationContext parent=wac.getParent();
    assertNotNull(parent);
    assertFalse(parent instanceof WebApplicationContext);
    assertEquals("ear",ear);
    assertEquals("root",root);
  }
}
