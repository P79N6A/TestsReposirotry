@Configuration static class ValueConfigWithProviderMethodArguments {
  @Bean @Scope("prototype") public TestBean testBean(  @Value("#{systemProperties[myProp]}") Provider<String> name){
    return new TestBean(name.get());
  }
  @Bean @Scope("prototype") public TestBean testBean2(  @Value("#{systemProperties[myProp]}") Provider<String> name2){
    return new TestBean(name2.get());
  }
}
