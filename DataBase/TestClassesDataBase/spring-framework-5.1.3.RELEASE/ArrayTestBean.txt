static class ArrayTestBean {
  private final int[] ints=new int[10];
  private final Integer[] integers=new Integer[10];
  ArrayTestBean(){
    for (int i=0; i < 10; i++) {
      ints[i]=i;
      integers[i]=i;
    }
  }
  public int[] getInts(){
    return ints;
  }
  public Integer[] getIntegers(){
    return integers;
  }
}
