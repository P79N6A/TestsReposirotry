@ControllerAdvice private static class ApplicationExceptionHandler extends ResponseEntityExceptionHandler {
  @Override protected ResponseEntity<Object> handleServletRequestBindingException(  ServletRequestBindingException ex,  HttpHeaders headers,  HttpStatus status,  WebRequest request){
    headers.set("someHeader","someHeaderValue");
    return handleExceptionInternal(ex,"error content",headers,status,request);
  }
}
