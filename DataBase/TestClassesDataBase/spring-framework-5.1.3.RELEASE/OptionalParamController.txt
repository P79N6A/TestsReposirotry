@Controller public static class OptionalParamController {
  @RequestMapping("/myPath.do") public void myHandle(  @RequestParam(required=false) String id,  @RequestParam(required=false) boolean flag,  @RequestHeader(value="header",required=false) String header,  HttpServletResponse response) throws IOException {
    response.getWriter().write(String.valueOf(id) + "-" + flag+ "-"+ String.valueOf(header));
  }
}
