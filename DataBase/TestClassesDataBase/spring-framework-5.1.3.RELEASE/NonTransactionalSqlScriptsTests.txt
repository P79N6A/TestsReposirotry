/** 
 * Integration tests which verify that scripts executed via  {@link Sql @Sql}will persist between non-transactional test methods.
 * @author Sam Brannen
 * @since 4.1
 */
@RunWith(SpringJUnit4ClassRunner.class) @FixMethodOrder(MethodSorters.NAME_ASCENDING) @ContextConfiguration(classes=EmptyDatabaseConfig.class) @Sql({"schema.sql","data.sql"}) @DirtiesContext public class NonTransactionalSqlScriptsTests {
  protected JdbcTemplate jdbcTemplate;
  @Autowired public void setDataSource(  DataSource dataSource){
    this.jdbcTemplate=new JdbcTemplate(dataSource);
  }
  @Test public void test01_classLevelScripts(){
    assertNumUsers(1);
  }
  @Test @Sql("data-add-dogbert.sql") public void test02_methodLevelScripts(){
    assertNumUsers(2);
  }
  protected void assertNumUsers(  int expected){
    assertEquals("Number of rows in the 'user' table.",expected,JdbcTestUtils.countRowsInTable(jdbcTemplate,"user"));
  }
}
