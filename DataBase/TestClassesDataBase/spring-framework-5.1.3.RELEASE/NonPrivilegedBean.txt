@SuppressWarnings("unused") private static class NonPrivilegedBean {
  private String expectedName;
  public static boolean destroyed=false;
  public NonPrivilegedBean(  String expected){
    this.expectedName=expected;
    checkCurrentContext();
  }
  public void init(){
    checkCurrentContext();
  }
  public void destroy(){
    checkCurrentContext();
    destroyed=true;
  }
  public void setProperty(  Object value){
    checkCurrentContext();
  }
  public Object getProperty(){
    checkCurrentContext();
    return null;
  }
  public void setListProperty(  Object value){
    checkCurrentContext();
  }
  public Object getListProperty(){
    checkCurrentContext();
    return null;
  }
  private void checkCurrentContext(){
    assertEquals(expectedName,getCurrentSubjectName());
  }
}
