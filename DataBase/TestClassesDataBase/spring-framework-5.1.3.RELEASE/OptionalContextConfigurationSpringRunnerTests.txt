/** 
 * JUnit 4 based integration test which verifies that  {@link @ContextConfiguration}is optional.
 * @author Phillip Webb
 * @author Sam Brannen
 * @since 4.3
 */
@RunWith(SpringRunner.class) public class OptionalContextConfigurationSpringRunnerTests {
  @Autowired String foo;
  @Test public void contextConfigurationAnnotationIsOptional(){
    assertEquals("foo",foo);
  }
@Configuration static class Config {
    @Bean String foo(){
      return "foo";
    }
  }
}
