private static class TestMessageConverter extends AbstractMessageConverter {
  public TestMessageConverter(){
    super(MimeTypeUtils.TEXT_PLAIN);
  }
  public TestMessageConverter(  Collection<MimeType> supportedMimeTypes){
    super(supportedMimeTypes);
  }
  @Override protected boolean supports(  Class<?> clazz){
    return String.class.equals(clazz);
  }
  @Override protected Object convertFromInternal(  Message<?> message,  Class<?> targetClass,  @Nullable Object conversionHint){
    return "success-from";
  }
  @Override protected Object convertToInternal(  Object payload,  @Nullable MessageHeaders headers,  @Nullable Object conversionHint){
    return "success-to";
  }
}
