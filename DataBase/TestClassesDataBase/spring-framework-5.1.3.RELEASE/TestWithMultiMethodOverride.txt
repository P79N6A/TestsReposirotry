@Transactional public static class TestWithMultiMethodOverride {
  @Transactional(readOnly=true) public void doSomething(){
    assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
    assertTrue(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
  }
  @Transactional(readOnly=true) public void doSomethingElse(){
    assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
    assertTrue(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
  }
  public void doSomethingCompletelyElse(){
    assertTrue(TransactionSynchronizationManager.isActualTransactionActive());
    assertFalse(TransactionSynchronizationManager.isCurrentTransactionReadOnly());
  }
}
