  private final List<TypeDescriptor> sourceTypes=new ArrayList<>();

  }

    sourceTypes.add(sourceType);
    return false;
  }
  @Override @Nullable public Object convert(  @Nullable Object source,  TypeDescriptor sourceType,  TypeDescriptor targetType){
    return null;
  }
  public List<TypeDescriptor> getSourceTypes(){
    return sourceTypes;
  }
}
