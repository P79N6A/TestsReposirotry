@Controller public static class NegatedContentTypeHeadersController {
  @RequestMapping(value="/something",headers="content-type=application/pdf") public void handlePdf(  Writer writer) throws IOException {
    writer.write("pdf");
  }
  @RequestMapping(value="/something",headers="content-type!=application/pdf") public void handleNonPdf(  Writer writer) throws IOException {
    writer.write("non-pdf");
  }
}
