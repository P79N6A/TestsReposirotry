@SuppressWarnings("serial") private static class MyRestClientException extends RestClientException {
  private String foo;
  public MyRestClientException(  String msg){
    super(msg);
  }
  public MyRestClientException(  String msg,  Throwable ex){
    super(msg,ex);
  }
  public String getFoo(){
    return this.foo;
  }
  public void setFoo(  String foo){
    this.foo=foo;
  }
}
