@Controller @RequestMapping("/test") public static class VariableNamesController {
  @RequestMapping(value="/{foo}",method=RequestMethod.GET) public void foo(  @PathVariable String foo,  Writer writer) throws IOException {
    writer.write("foo-" + foo);
  }
  @RequestMapping(value="/{bar}",method=RequestMethod.DELETE) public void bar(  @PathVariable String bar,  Writer writer) throws IOException {
    writer.write("bar-" + bar);
  }
}
