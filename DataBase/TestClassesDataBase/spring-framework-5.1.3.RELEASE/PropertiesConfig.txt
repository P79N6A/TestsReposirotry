@Configuration static class PropertiesConfig {
  private String hostname;
  private Resource resource;
  @Value("#{myProps.hostname}") public void setHostname(  String hostname){
    this.hostname=hostname;
  }
  @Value("do_not_delete_me.txt") public void setResource(  Resource resource){
    this.resource=resource;
  }
  @Bean public TestBean testBean() throws IOException {
    return new TestBean(hostname,(int)resource.contentLength());
  }
}
