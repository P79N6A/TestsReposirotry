@Configuration static class BeanArgumentConfigWithPrototype {
  @Bean @Scope("prototype") public DependingFoo foo(  BarArgument bar){
    return new DependingFoo(bar);
  }
  @Bean public FooFactory fooFactory(){
    return new FooFactory(){
      @Override public DependingFoo createFoo(      BarArgument bar){
        return foo(bar);
      }
    }
;
  }
}
