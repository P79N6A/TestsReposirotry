@Configuration @EnableMBeanExport(server="server") static class PackagePrivateExtensionConfiguration {
  @Bean public MBeanServerFactoryBean server() throws Exception {
    return new MBeanServerFactoryBean();
  }
  @Bean public PackagePrivateTestBeanExtension testBean(){
    PackagePrivateTestBeanExtension bean=new PackagePrivateTestBeanExtension();
    bean.setName("TEST");
    return bean;
  }
}
