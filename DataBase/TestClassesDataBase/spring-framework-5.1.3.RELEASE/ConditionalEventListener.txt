@Component @Validated static class ConditionalEventListener extends TestEventListener implements ConditionalEventInterface {
  @EventListener(condition="'OK'.equals(#root.event.msg)") @Override public void handle(  TestEvent event){
    super.handle(event);
  }
  @Override @EventListener(condition="#payload.startsWith('OK')") public void handleString(  String payload){
    super.handleString(payload);
  }
  @ConditionalEvent("#root.event.timestamp > #p0") public void handleTimestamp(  Long timestamp){
    collectEvent(timestamp);
  }
  @ConditionalEvent("@conditionEvaluator.valid(#p0)") public void handleRatio(  Double ratio){
    collectEvent(ratio);
  }
}
