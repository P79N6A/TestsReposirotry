@Configuration public static class LazyMethodValidationConfigWithProxyTargetClass {
  @Bean public static MethodValidationPostProcessor methodValidationPostProcessor(  @Lazy Validator validator){
    MethodValidationPostProcessor postProcessor=new MethodValidationPostProcessor();
    postProcessor.setValidator(validator);
    postProcessor.setProxyTargetClass(true);
    return postProcessor;
  }
}
