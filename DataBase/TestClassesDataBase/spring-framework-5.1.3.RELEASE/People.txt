@XmlRootElement @SuppressWarnings({"WeakerAccess","unused"}) private static class People {
  private List<Person> persons=new ArrayList<>();
  public People(){
  }
  public People(  Person... persons){
    this.persons.addAll(Arrays.asList(persons));
  }
  @XmlElement public List<Person> getPerson(){
    return this.persons;
  }
}
