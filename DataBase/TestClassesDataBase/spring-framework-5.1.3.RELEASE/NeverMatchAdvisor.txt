@SuppressWarnings("serial") class NeverMatchAdvisor extends StaticMethodMatcherPointcutAdvisor {
  public NeverMatchAdvisor(){
    super(new NopInterceptor());
  }
  /** 
 * This method is solely to allow us to create a mixture of dependencies in the bean definitions. The dependencies don't have any meaning, and don't <b>do</b> anything.
 */
  public void setDependencies(  List<?> l){
  }
  /** 
 * @see org.springframework.aop.MethodMatcher#matches(java.lang.reflect.Method,java.lang.Class)
 */
  @Override public boolean matches(  Method m,  @Nullable Class<?> targetClass){
    return false;
  }
}
