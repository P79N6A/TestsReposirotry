@Configuration static class CustomConfigWithAttributeOverride {
  @InterestingBeanWithName(name="testBeanX") public TestBean testBean1(){
    return new TestBean("interesting");
  }
  @Bean @Qualifier("boring") public TestBean testBean2(  @Lazy TestBean testBean1){
    TestBean tb=new TestBean("boring");
    tb.setSpouse(testBean1);
    return tb;
  }
}
