@SuppressWarnings("serial") public static class ObjectFactoryMapMethodInjectionBean implements Serializable {
  private Provider<Map<String,TestBean>> testBeanFactory;
  @Inject public void setTestBeanFactory(  Provider<Map<String,TestBean>> testBeanFactory){
    this.testBeanFactory=testBeanFactory;
  }
  public TestBean getTestBean(){
    return this.testBeanFactory.get().values().iterator().next();
  }
}
