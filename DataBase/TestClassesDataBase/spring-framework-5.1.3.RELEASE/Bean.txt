private static class Bean implements MapEntry<String,String> {
  private String key;
  private String value;
  @Override public String getKey(){
    return key;
  }
  @Override public void setKey(  String aKey){
    key=aKey;
  }
  @Override public String getValue(){
    return value;
  }
  @Override public void setValue(  String aValue){
    value=aValue;
  }
}
