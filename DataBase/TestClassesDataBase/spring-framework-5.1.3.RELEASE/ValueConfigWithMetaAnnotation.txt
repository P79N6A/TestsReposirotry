  @MyProp private String name;
  private String name2;
  @MyProp public void setName2(  String name){
    this.name2=name;
  }
  @Bean @Scope("prototype") public TestBean testBean(){
    return new TestBean(name);
  }
  @Bean @Scope("prototype") public TestBean testBean2(){
    return new TestBean(name2);
  }
}
