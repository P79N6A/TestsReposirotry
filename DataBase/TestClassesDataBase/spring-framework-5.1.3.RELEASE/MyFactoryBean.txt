static class MyFactoryBean implements FactoryBean<String>, InitializingBean {
  private boolean initialized=false;
  @Override public void afterPropertiesSet() throws Exception {
    this.initialized=true;
  }
  @Override public String getObject() throws Exception {
    return "foo";
  }
  @Override public Class<String> getObjectType(){
    return String.class;
  }
  @Override public boolean isSingleton(){
    return true;
  }
  public String getString(){
    return Boolean.toString(this.initialized);
  }
}
