private static class MyAtomFeedView extends AbstractAtomFeedView {
  @Override protected void buildFeedMetadata(  Map<String,Object> model,  Feed feed,  HttpServletRequest request){
    feed.setTitle("Test Feed");
  }
  @Override protected List<Entry> buildFeedEntries(  Map<String,Object> model,  HttpServletRequest request,  HttpServletResponse response){
    List<Entry> entries=new ArrayList<>();
    for (    String name : model.keySet()) {
      Entry entry=new Entry();
      entry.setTitle(name);
      Content content=new Content();
      content.setValue((String)model.get(name));
      entry.setSummary(content);
      entries.add(entry);
    }
    return entries;
  }
}
