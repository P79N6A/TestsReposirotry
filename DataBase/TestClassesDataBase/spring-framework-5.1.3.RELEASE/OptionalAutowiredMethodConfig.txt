@Configuration static class OptionalAutowiredMethodConfig {
  @Bean public TestBean testBean(  Optional<Colour> colour,  Optional<List<Colour>> colours){
    if (!colour.isPresent() && !colours.isPresent()) {
      return new TestBean("");
    }
 else {
      return new TestBean(colour.get().toString() + "-" + colours.get().get(0).toString());
    }
  }
}
