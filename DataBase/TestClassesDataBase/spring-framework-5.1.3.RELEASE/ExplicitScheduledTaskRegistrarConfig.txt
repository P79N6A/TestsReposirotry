@Configuration @EnableScheduling static class ExplicitScheduledTaskRegistrarConfig implements SchedulingConfigurer {
  String threadName;
  @Bean public TaskScheduler taskScheduler1(){
    ThreadPoolTaskScheduler scheduler=new ThreadPoolTaskScheduler();
    scheduler.setThreadNamePrefix("explicitScheduler1");
    return scheduler;
  }
  @Bean public TaskScheduler taskScheduler2(){
    ThreadPoolTaskScheduler scheduler=new ThreadPoolTaskScheduler();
    scheduler.setThreadNamePrefix("explicitScheduler2");
    return scheduler;
  }
  @Bean public AtomicInteger counter(){
    return new AtomicInteger();
  }
  @Scheduled(fixedRate=10) public void task(){
    threadName=Thread.currentThread().getName();
    counter().incrementAndGet();
  }
  @Override public void configureTasks(  ScheduledTaskRegistrar taskRegistrar){
    taskRegistrar.setScheduler(taskScheduler1());
  }
}
