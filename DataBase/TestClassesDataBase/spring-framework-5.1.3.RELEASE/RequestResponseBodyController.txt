@Controller public static class RequestResponseBodyController {
  @RequestMapping(value="/something",method=RequestMethod.PUT) @ResponseBody public String handle(  @RequestBody String body) throws IOException {
    return body;
  }
  @RequestMapping(value="/something",method=RequestMethod.PATCH) @ResponseBody public String handlePartialUpdate(  @RequestBody String content) throws IOException {
    return content;
  }
}
