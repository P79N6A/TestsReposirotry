@Configuration static class ValueConfigWithProviderFields {
  @Value("#{systemProperties[myProp]}") private Provider<String> name;
  private Provider<String> name2;
  @Value("#{systemProperties[myProp]}") public void setName2(  Provider<String> name){
    this.name2=name;
  }
  @Bean @Scope("prototype") public TestBean testBean(){
    return new TestBean(name.get());
  }
  @Bean @Scope("prototype") public TestBean testBean2(){
    return new TestBean(name2.get());
  }
}
