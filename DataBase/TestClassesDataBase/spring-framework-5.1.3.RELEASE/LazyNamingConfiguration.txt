@Configuration @EnableMBeanExport(server="server") static class LazyNamingConfiguration {
  @Bean public MBeanServerFactoryBean server() throws Exception {
    return new MBeanServerFactoryBean();
  }
  @Bean @Lazy public AnnotationTestBean testBean(){
    AnnotationTestBean bean=new AnnotationTestBean();
    bean.setName("TEST");
    bean.setAge(100);
    return bean;
  }
}
