@Aspect("pertarget(execution(* *.getSpouse()))") class PerTargetAspect implements Ordered {
  public int count;
  private int order=Ordered.LOWEST_PRECEDENCE;
  @Around("execution(int *.getAge())") public int returnCountAsAge(){
    return count++;
  }
  @Before("execution(void *.set*(int))") public void countSetter(){
    ++count;
  }
  @Override public int getOrder(){
    return this.order;
  }
  public void setOrder(  int order){
    this.order=order;
  }
}
