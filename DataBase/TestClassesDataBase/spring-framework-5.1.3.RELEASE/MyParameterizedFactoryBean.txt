static class MyParameterizedFactoryBean<T> implements FactoryBean<T> {
  private final T obj;
  public MyParameterizedFactoryBean(  T obj){
    this.obj=obj;
  }
  @Override public T getObject() throws Exception {
    return obj;
  }
  @Override @SuppressWarnings("unchecked") public Class<T> getObjectType(){
    return (Class<T>)obj.getClass();
  }
  @Override public boolean isSingleton(){
    return true;
  }
}
