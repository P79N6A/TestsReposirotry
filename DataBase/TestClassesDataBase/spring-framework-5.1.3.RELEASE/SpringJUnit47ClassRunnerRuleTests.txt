/** 
 * Verifies support for JUnit 4.7  {@link Rule Rules} in conjunction with the{@link SpringRunner}. The body of this test class is taken from the JUnit 4.7 release notes.

 * @since 3.0

@RunWith(SpringRunner.class) @TestExecutionListeners({}) public class SpringJUnit47ClassRunnerRuleTests {
  @Rule public TestName name=new TestName();
  @Test public void testA(){
    assertEquals("testA",name.getMethodName());
  }
  @Test public void testB(){
    assertEquals("testB",name.getMethodName());
  }
}
