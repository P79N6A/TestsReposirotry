public static class Spr11124ServiceImpl implements Spr11124Service {
  private int multipleCount=0;
  @Override @Cacheable("smallCache") public List<String> single(  int id){
    if (this.multipleCount > 0) {
      fail("Called too many times");
    }
    this.multipleCount++;
    return Collections.emptyList();
  }
  @Override @Caching(cacheable={@Cacheable(cacheNames="bigCache",unless="#result.size() < 4"),@Cacheable(cacheNames="smallCache",unless="#result.size() > 3")}) public List<String> multiple(  int id){
    if (this.multipleCount > 0) {
      fail("Called too many times");
    }
    this.multipleCount++;
    return Collections.emptyList();
  }
}
