public class LogUserAdvice implements MethodBeforeAdvice, ThrowsAdvice {
  private int countBefore=0;
  private int countThrows=0;
  @Override public void before(  Method method,  Object[] objects,  @Nullable Object o) throws Throwable {
    countBefore++;
  }
  public void afterThrowing(  Exception ex) throws Throwable {
    countThrows++;
    throw ex;
  }
  public int getCountBefore(){
    return countBefore;
  }
  public int getCountThrows(){
    return countThrows;
  }
  public void reset(){
    countThrows=0;
    countBefore=0;
  }
}
