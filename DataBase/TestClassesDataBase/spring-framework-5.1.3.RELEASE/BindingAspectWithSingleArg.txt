@Aspect public static class BindingAspectWithSingleArg {
  @Pointcut(value="args(a)",argNames="a") public void setAge(  int a){
  }
  @Around(value="setAge(age)",argNames="age") public void changeReturnType(  ProceedingJoinPoint pjp,  int age) throws Throwable {
    pjp.proceed(new Object[]{age * 2});
  }
}
