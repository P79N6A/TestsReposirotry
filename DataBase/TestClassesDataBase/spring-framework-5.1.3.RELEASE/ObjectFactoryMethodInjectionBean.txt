@SuppressWarnings("serial") public static class ObjectFactoryMethodInjectionBean implements Serializable {
  private Provider<TestBean> testBeanFactory;
  @Inject public void setTestBeanFactory(  Provider<TestBean> testBeanFactory){
    this.testBeanFactory=testBeanFactory;
  }
  public TestBean getTestBean(){
    return this.testBeanFactory.get();
  }
}
