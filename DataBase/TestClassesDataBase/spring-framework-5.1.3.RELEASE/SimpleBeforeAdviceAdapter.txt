@SuppressWarnings("serial") class SimpleBeforeAdviceAdapter implements AdvisorAdapter, Serializable {
  @Override public boolean supportsAdvice(  Advice advice){
    return (advice instanceof SimpleBeforeAdvice);
  }
  @Override public MethodInterceptor getInterceptor(  Advisor advisor){
    SimpleBeforeAdvice advice=(SimpleBeforeAdvice)advisor.getAdvice();
    return new SimpleBeforeAdviceInterceptor(advice);
  }
}
