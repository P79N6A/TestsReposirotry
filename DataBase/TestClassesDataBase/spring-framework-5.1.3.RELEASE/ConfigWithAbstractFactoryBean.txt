@Configuration protected static class ConfigWithAbstractFactoryBean {
  @Bean public FactoryBean<Foo> foo(){
    return new AbstractFactoryBean<Foo>(){
      @Override public Foo createInstance(){
        return new Foo("x");
      }
      @Override public Class<?> getObjectType(){
        return Foo.class;
      }
    }
;
  }
  @Bean public Bar bar() throws Exception {
    assertTrue(foo().isSingleton());
    return new Bar(foo().getObject());
  }
}
