private static class OptionalHolder {
  private String id;
  private Optional<String> name;
  public String getId(){
    return id;
  }
  public void setId(  String id){
    this.id=id;
  }
  public Optional<String> getName(){
    return name;
  }
  public void setName(  Optional<String> name){
    this.name=name;
  }
}
