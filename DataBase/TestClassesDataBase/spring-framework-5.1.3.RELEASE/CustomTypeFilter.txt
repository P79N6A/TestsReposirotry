public static class CustomTypeFilter implements TypeFilter {
  /** 
 * Intentionally spelling "custom" with a "k" since there are numerous classes in this package named *Custom*.
 */
  @Override public boolean match(  MetadataReader metadataReader,  MetadataReaderFactory metadataReaderFactory){
    return metadataReader.getClassMetadata().getClassName().contains("Kustom");
  }
}
