@Configuration public static class ScopedProxyRepositoryConfiguration {
  @Bean @Scope(scopeName="prototype",proxyMode=ScopedProxyMode.TARGET_CLASS) public Repository<String> stringRepo(){
    return new Repository<String>(){
      @Override public String toString(){
        return "Repository<String>";
      }
    }
;
  }
  @Bean @PrototypeScoped public Repository<Integer> integerRepo(){
    return new Repository<Integer>(){
      @Override public String toString(){
        return "Repository<Integer>";
      }
    }
;
  }
}
