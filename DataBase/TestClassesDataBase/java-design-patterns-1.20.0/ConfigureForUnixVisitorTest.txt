/** 
 * ConfigureForUnixVisitor test class
 */
public class ConfigureForUnixVisitorTest {
  TestLogger logger=TestLoggerFactory.getTestLogger(ConfigureForUnixVisitor.class);
  @AfterEach public void clearLoggers(){
    TestLoggerFactory.clear();
  }
  @Test public void testVisitForZoom(){
    ConfigureForUnixVisitor conUnix=new ConfigureForUnixVisitor();
    Zoom zoom=new Zoom();
    conUnix.visit(zoom);
    assertThat(logger.getLoggingEvents()).extracting("level","message").contains(tuple(INFO,zoom + " used with Unix configurator."));
  }
}
