/** 
 * Application test
 */
public class DirtyFlagTest {
  @Test public void testIsDirty(){
    DataFetcher df=new DataFetcher();
    List<String> countries=df.fetch();
    assertTrue(!countries.isEmpty());
  }
  @Test public void testIsNotDirty(){
    DataFetcher df=new DataFetcher();
    df.fetch();
    List<String> countries=df.fetch();
    assertTrue(countries.isEmpty());
  }
}
