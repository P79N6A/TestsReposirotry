public static class GenericUserService implements GenericService<String,Long,User> {
  @HystrixCommand(fallbackMethod="fallback") @Override public User getByKey(  String sKey,  Long lKey){
    return new User(sKey,"name: " + lKey);
  }
  @HystrixCommand(fallbackMethod="fallback") @Override public User getByKeyForceFail(  String sKey,  Long lKey){
    throw new RuntimeException("force fail");
  }
  @Override public User fallback(  String sKey,  Long lKey){
    return new User(sKey,"name: " + lKey);
  }
}
