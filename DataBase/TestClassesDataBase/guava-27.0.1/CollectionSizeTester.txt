/** 
 * A generic JUnit test which tests  {@code size()} operations on a collection. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.CollectionTestSuiteBuilder}.
 * @author Kevin Bourrillion
 */
@GwtCompatible @Ignore public class CollectionSizeTester<E> extends AbstractCollectionTester<E> {
  public void testSize(){
    assertEquals("size():",getNumElements(),collection.size());
  }
}
