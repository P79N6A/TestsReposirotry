/** 
 * Testers for  {@link ListMultimap#putAll(Object,Iterable)}.
 * @author Louis Wasserman
 */
@GwtCompatible @Ignore public class ListMultimapPutAllTester<K,V> extends AbstractListMultimapTester<K,V> {
  @MapFeature.Require(SUPPORTS_PUT) public void testPutAllAddsAtEndInOrder(){
    @SuppressWarnings("unchecked") List<V> values=Arrays.asList(v3(),v1(),v4());
    for (    K k : sampleKeys()) {
      resetContainer();
      List<V> expectedValues=copyToList(multimap().get(k));
      assertTrue(multimap().putAll(k,values));
      expectedValues.addAll(values);
      assertGet(k,expectedValues);
    }
  }
}
