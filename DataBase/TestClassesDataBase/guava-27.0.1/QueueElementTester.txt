/** 
 * A generic JUnit test which tests  {@code element()} operations on a queue. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.CollectionTestSuiteBuilder}.
 * @author Jared Levy
 */
@GwtCompatible @Ignore public class QueueElementTester<E> extends AbstractQueueTester<E> {
  @CollectionSize.Require(ZERO) public void testElement_empty(){
    try {
      getQueue().element();
      fail("emptyQueue.element() should throw");
    }
 catch (    NoSuchElementException expected) {
    }
    expectUnchanged();
  }
  @CollectionSize.Require(ONE) public void testElement_size1(){
    assertEquals("size1Queue.element() should return first element",e0(),getQueue().element());
    expectUnchanged();
  }
  @CollectionFeature.Require(KNOWN_ORDER) @CollectionSize.Require(SEVERAL) public void testElement_sizeMany(){
    assertEquals("sizeManyQueue.element() should return first element",e0(),getQueue().element());
    expectUnchanged();
  }
}
