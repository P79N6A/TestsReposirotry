/** 
 * A generic JUnit test which tests  {@code stream} operations on a collection. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.CollectionTestSuiteBuilder}.
 * @author Louis Wasserman
 */
@GwtCompatible @Ignore public class CollectionStreamTester<E> extends AbstractCollectionTester<E> {
  @CollectionFeature.Require(absent=KNOWN_ORDER) public void testStreamToArrayUnknownOrder(){
synchronized (collection) {
      Helpers.assertEqualIgnoringOrder(getSampleElements(),Arrays.asList(collection.stream().toArray()));
    }
  }
  @CollectionFeature.Require(KNOWN_ORDER) public void testStreamToArrayKnownOrder(){
synchronized (collection) {
      assertEquals(getOrderedElements(),Arrays.asList(collection.stream().toArray()));
    }
  }
  public void testStreamCount(){
synchronized (collection) {
      assertEquals(getNumElements(),collection.stream().count());
    }
  }
}
