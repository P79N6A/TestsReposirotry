/** 
 * A generic JUnit test which tests  {@code lastIndexOf()} operations on a list. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.ListTestSuiteBuilder}.
 * @author Chris Povirk
 */
@GwtCompatible @Ignore public class ListLastIndexOfTester<E> extends AbstractListIndexOfTester<E> {
  @Override protected int find(  Object o){
    return getList().lastIndexOf(o);
  }
  @Override protected String getMethodName(){
    return "lastIndexOf";
  }
  @CollectionFeature.Require(absent=REJECTS_DUPLICATES_AT_CREATION) @CollectionSize.Require(absent={ZERO,ONE}) public void testLastIndexOf_duplicate(){
    E[] array=createSamplesArray();
    array[getNumElements() / 2]=e0();
    collection=getSubjectGenerator().create(array);
    assertEquals("lastIndexOf(duplicate) should return index of last occurrence",getNumElements() / 2,getList().lastIndexOf(e0()));
  }
}
