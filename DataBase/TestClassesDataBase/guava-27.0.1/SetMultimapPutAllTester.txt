/** 
 * Tests for  {@link SetMultimap#replaceValues}.
 * @author Louis Wasserman
 */
@GwtCompatible @Ignore public class SetMultimapPutAllTester<K,V> extends AbstractMultimapTester<K,V,SetMultimap<K,V>> {
  @MapFeature.Require(SUPPORTS_PUT) public void testPutAllHandlesDuplicates(){
    @SuppressWarnings("unchecked") List<V> valuesToPut=Arrays.asList(v0(),v1(),v0());
    for (    K k : sampleKeys()) {
      resetContainer();
      Set<V> expectedValues=copyToSet(multimap().get(k));
      multimap().putAll(k,valuesToPut);
      expectedValues.addAll(valuesToPut);
      assertGet(k,expectedValues);
    }
  }
}
