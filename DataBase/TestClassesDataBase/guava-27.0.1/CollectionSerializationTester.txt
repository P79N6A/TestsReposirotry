/** 
 * Basic reserialization test for collections.
 * @author Louis Wasserman
 */
@GwtCompatible @Ignore public class CollectionSerializationTester<E> extends AbstractCollectionTester<E> {
  @CollectionFeature.Require(SERIALIZABLE) public void testReserialize(){
    Helpers.assertEqualIgnoringOrder(actualContents(),SerializableTester.reserialize(actualContents()));
  }
}
