/** 
 * Tests for  {@link SetMultimap#replaceValues}.
 * @author Louis Wasserman
 */
@GwtCompatible @Ignore public class SetMultimapReplaceValuesTester<K,V> extends AbstractMultimapTester<K,V,SetMultimap<K,V>> {
  @MapFeature.Require({SUPPORTS_PUT,SUPPORTS_REMOVE}) public void testReplaceValuesHandlesDuplicates(){
    @SuppressWarnings("unchecked") List<V> values=Arrays.asList(v0(),v1(),v0());
    for (    K k : sampleKeys()) {
      resetContainer();
      multimap().replaceValues(k,values);
      assertGet(k,v0(),v1());
    }
  }
}
