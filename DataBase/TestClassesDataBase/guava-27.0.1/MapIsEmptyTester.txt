/** 
 * A generic JUnit test which tests  {@code isEmpty()} operations on a map. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.MapTestSuiteBuilder}.
 * @author Kevin Bourrillion
 */
@GwtCompatible @Ignore public class MapIsEmptyTester<K,V> extends AbstractMapTester<K,V> {
  @CollectionSize.Require(ZERO) public void testIsEmpty_yes(){
    assertTrue("isEmpty() should return true",getMap().isEmpty());
  }
  @CollectionSize.Require(absent=ZERO) public void testIsEmpty_no(){
    assertFalse("isEmpty() should return false",getMap().isEmpty());
  }
}
