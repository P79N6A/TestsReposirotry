/** 
 * A generic JUnit test which tests  {@code isEmpty()} operations on a collection. Can't be invokeddirectly; please see  {@link com.google.common.collect.testing.CollectionTestSuiteBuilder}.
 * @author Kevin Bourrillion
 */
@GwtCompatible @Ignore public class CollectionIsEmptyTester<E> extends AbstractCollectionTester<E> {
  @CollectionSize.Require(ZERO) public void testIsEmpty_yes(){
    assertTrue("isEmpty() should return true",collection.isEmpty());
  }
  @CollectionSize.Require(absent=ZERO) public void testIsEmpty_no(){
    assertFalse("isEmpty() should return false",collection.isEmpty());
  }
}
