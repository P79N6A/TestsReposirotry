/** 
 * A processor that removes custom timestamp information from messages and forwards modified messages to each child. A message contains custom timestamp information if the value is in ".*@[0-9]+" format.
 */
protected static class ValueTimestampProcessor extends AbstractProcessor<String,String> {
  @Override public void process(  final String key,  final String value){
    context().forward(key,value.split("@")[0]);
  }
}
