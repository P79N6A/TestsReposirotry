@ExtendWith(DefaultFileSystemExtension.class) class DefaultFileSystemExtensionTest {
  @Inject DefaultFileSystemAbstraction rootFileSystem;
  @Test void fileSystemInjectionCreateFileSystem(){
    assertNotNull(rootFileSystem);
  }
@Nested class NestedFileSystemTest {
    @Inject DefaultFileSystemAbstraction nestedFileSystem;
    @Test void nestedFileSystemInjection(){
      assertNotNull(nestedFileSystem);
    }
    @Test void rootFileSystemAvailable(){
      assertNotNull(rootFileSystem);
    }
    @Test void nestedAndRootFileSystemsAreTheSame(){
      assertSame(nestedFileSystem,rootFileSystem);
    }
  }
}
