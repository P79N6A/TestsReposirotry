public static class Await implements Task {
  private final CountDownLatch latch;
  private Await(  CountDownLatch latch){
    this.latch=latch;
  }
  public Signal signal(){
    return new Signal(latch);
  }
  public void release(){
    latch.countDown();
  }
  @Override public void perform() throws Exception {
    latch.await();
  }
}
