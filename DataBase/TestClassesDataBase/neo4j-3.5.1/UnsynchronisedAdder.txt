private static class UnsynchronisedAdder {
  private volatile long value;
  public void add(  long delta){
    long v=value;
    Thread.yield();
    usleep(ThreadLocalRandom.current().nextInt(50));
    value=v + delta;
  }
  public void increment(){
    add(1);
  }
  public long sum(){
    return value;
  }
}
