public class MandatoryTransactionsForSchemaTest extends AbstractMandatoryTransactionsTest<Schema> {
  @Test public void shouldRequireTransactionsWhenCallingMethodsOnSchema(){
    assertFacadeMethodsThrowNotInTransaction(obtainEntity(),SchemaFacadeMethods.values());
  }
  @Test public void shouldTerminateWhenCallingMethodsOnSchema(){
    assertFacadeMethodsThrowAfterTerminate(SchemaFacadeMethods.values());
  }
  @Override protected Schema obtainEntityInTransaction(  GraphDatabaseService graphDatabaseService){
    return graphDatabaseService.schema();
  }
}
