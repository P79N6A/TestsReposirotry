private class ConfiguredExecutionStatistics implements ExecutionStatistics {
  private long hits;
  private long faults;
  @Override public long pageHits(){
    return hits;
  }
  @Override public long pageFaults(){
    return faults;
  }
  void setHits(  long hits){
    this.hits=hits;
  }
  void setFaults(  long faults){
    this.faults=faults;
  }
}
