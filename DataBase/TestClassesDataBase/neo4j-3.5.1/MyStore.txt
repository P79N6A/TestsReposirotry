    super(new File("store"),new File("idFile"),config,IdType.NODE,new DefaultIdGeneratorFactory(fs.get()),pageCache,NullLogProvider.getInstance(),"T",format,format,"XYZ");
  }
  @Override public <FAILURE extends Exception>void accept(  Processor<FAILURE> processor,  IntRecord record){
    throw new UnsupportedOperationException();
  }

    Long override=nextPageId.poll();
    return override != null ? override : super.pageIdForRecord(id);
  }

    return override != null ? override : super.offsetForId(id);
  }
}
