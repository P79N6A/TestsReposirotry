private static class ExpensiveTask extends TestTask {
  private final int millis;
  ExpensiveTask(  int millis){
    this.millis=millis;
  }
  @Override public void run(  Void nothing){
    try {
      Thread.sleep(millis);
    }
 catch (    InterruptedException e) {
      throw new RuntimeException(e);
    }
    super.run(nothing);
  }
}
