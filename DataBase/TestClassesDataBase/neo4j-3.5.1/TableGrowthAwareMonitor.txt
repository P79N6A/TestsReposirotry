private static class TableGrowthAwareMonitor extends Monitor.Adapter {
  private boolean grew;
  @Override public boolean tableGrew(  int fromCapacity,  int toCapacity,  int currentSize){
    return grew=true;
  }
  boolean checkAndReset(){
    try {
      return grew;
    }
  finally {
      grew=false;
    }
  }
}
