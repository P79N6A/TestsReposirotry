static class Rhs {
  final int[] lhs;
  final int[] rhs;
  Rhs(  int[] lhs,  int[] rhs){
    this.lhs=lhs;
    this.rhs=rhs;
  }
  Object[] expect(  int... expected){
    return new Object[]{new Input(lhs,rhs,expected)};
  }
  Object[] expectLhs(){
    return new Object[]{new Input(lhs,rhs,lhs)};
  }
  Object[] expectRhs(){
    return new Object[]{new Input(lhs,rhs,rhs)};
  }
}
