private class WholeFileRacer implements Runnable {
  private LongArray array;
  WholeFileRacer(  LongArray array){
    this.array=array;
  }
  @Override public void run(){
    for (int o=0; o < LAPS; o++) {
      for (long i=0; i < COUNT; i++) {
        array.set(i,i);
        assertEquals(i,array.get(i));
      }
    }
  }
}
