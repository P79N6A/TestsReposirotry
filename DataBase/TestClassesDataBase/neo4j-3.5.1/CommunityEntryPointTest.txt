@ExtendWith(SuppressOutputExtension.class) class CommunityEntryPointTest {
  @Inject private SuppressOutput suppressOutput;
  @Test void mainPrintsVersion(){
    CommunityEntryPoint.main(new String[]{"--version"});
    assertTrue(suppressOutput.getOutputVoice().containsMessage("neo4j " + neo4jVersion()));
  }
}
