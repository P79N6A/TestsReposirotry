public class LogHeaderCacheTest {
  @Test public void shouldReturnNullWhenThereIsNoHeaderInTheCache(){
    final LogHeaderCache cache=new LogHeaderCache(2);
    final Long logHeader=cache.getLogHeader(5);
    assertNull(logHeader);
  }
  @Test public void shouldReturnTheHeaderIfInTheCache(){
    final LogHeaderCache cache=new LogHeaderCache(2);
    cache.putHeader(5,3);
    final long logHeader=cache.getLogHeader(5);
    assertEquals(3,logHeader);
  }
  @Test public void shouldClearTheCache(){
    final LogHeaderCache cache=new LogHeaderCache(2);
    cache.putHeader(5,3);
    cache.clear();
    final Long logHeader=cache.getLogHeader(5);
    assertNull(logHeader);
  }
}
