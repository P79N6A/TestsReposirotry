static class Pair {
  static Pair pair(  int key,  int value){
    return new Pair(key,value);
  }
  final int key;
  int value;
  Pair(  int key,  int value){
    this.key=key;
    this.value=value;
  }
  void visit(  WritableBuffer key,  WritableBuffer value){
    key.putInt(0,this.key);
    value.putInt(0,this.value);
  }
}
