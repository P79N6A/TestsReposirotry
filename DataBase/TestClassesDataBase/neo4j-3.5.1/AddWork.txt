private static class AddWork implements Work<Adder,AddWork> {
  private int delta;
  private AddWork(  int delta){
    this.delta=delta;
  }
  @Override public AddWork combine(  AddWork work){
    delta+=work.delta;
    return this;
  }
  @Override public void apply(  Adder adder){
    usleep(50);
    adder.add(delta);
  }
}
