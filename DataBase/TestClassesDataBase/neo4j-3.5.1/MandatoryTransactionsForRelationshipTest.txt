public class MandatoryTransactionsForRelationshipTest extends AbstractMandatoryTransactionsTest<Relationship> {
  @Test public void shouldRequireTransactionsWhenCallingMethodsOnRelationshipFacade(){
    assertFacadeMethodsThrowNotInTransaction(obtainEntity(),RelationshipFacadeMethods.values());
  }
  @Test public void shouldTerminateWhenCallingMethodsOnRelationshipFacade(){
    assertFacadeMethodsThrowAfterTerminate(RelationshipFacadeMethods.values());
  }
  @Override protected Relationship obtainEntityInTransaction(  GraphDatabaseService graphDatabaseService){
    return graphDatabaseService.createNode().createRelationshipTo(graphDatabaseService.createNode(),withName("foo"));
  }
}
