class BoltProtocolV3Test {
  @Test void shouldCreatePackForBoltV3() throws Throwable {
    BoltProtocolV3 protocolV3=new BoltProtocolV3(mock(BoltChannel.class),(ch,st) -> mock(BoltConnection.class),mock(BoltStateMachineFactory.class),NullLogService.getInstance());
    assertThat(protocolV3.createPack(),instanceOf(Neo4jPackV2.class));
  }
  @Test void shouldVersionReturnBoltV3() throws Throwable {
    BoltProtocolV3 protocolV3=new BoltProtocolV3(mock(BoltChannel.class),(ch,st) -> mock(BoltConnection.class),mock(BoltStateMachineFactory.class),NullLogService.getInstance());
    assertThat(protocolV3.version(),equalTo(3L));
  }
  @Test void shouldCreateMessageReaderForBoltV3() throws Throwable {
    BoltProtocolV3 protocolV3=new BoltProtocolV3(mock(BoltChannel.class),(ch,st) -> mock(BoltConnection.class),mock(BoltStateMachineFactory.class),NullLogService.getInstance());
    assertThat(protocolV3.createMessageReader(mock(BoltChannel.class),mock(Neo4jPack.class),mock(BoltConnection.class),NullLogService.getInstance()),instanceOf(BoltRequestMessageReaderV3.class));
  }
}
