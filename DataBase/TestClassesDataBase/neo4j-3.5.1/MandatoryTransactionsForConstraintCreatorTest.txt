public class MandatoryTransactionsForConstraintCreatorTest extends AbstractMandatoryTransactionsTest<ConstraintCreator> {
  @Test public void shouldRequireTransactionsWhenCallingMethodsConstraintCreators(){
    assertFacadeMethodsThrowNotInTransaction(obtainEntity(),ConstraintCreatorFacadeMethods.values());
  }
  @Test public void shouldTerminateWhenCallingMethodsConstraintCreators(){
    assertFacadeMethodsThrowAfterTerminate(ConstraintCreatorFacadeMethods.values());
  }
  @Override protected ConstraintCreator obtainEntityInTransaction(  GraphDatabaseService graphDatabaseService){
    return graphDatabaseService.schema().constraintFor(Label.label("Label"));
  }
}
