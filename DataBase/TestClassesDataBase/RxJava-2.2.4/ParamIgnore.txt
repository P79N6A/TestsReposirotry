static final class ParamIgnore {
  final Class<?> clazz;
  final String name;
  final Class<?>[] arguments;
  ParamIgnore(  Class<?> clazz,  String name,  Class<?>... arguments){
    this.clazz=clazz;
    this.name=name;
    this.arguments=arguments;
  }
  @Override public String toString(){
    return clazz.getName() + " " + name;
  }
}
