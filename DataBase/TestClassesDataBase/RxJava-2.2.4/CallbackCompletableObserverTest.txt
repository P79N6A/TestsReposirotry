public final class CallbackCompletableObserverTest {
  @Test public void emptyActionShouldReportNoCustomOnError(){
    CallbackCompletableObserver o=new CallbackCompletableObserver(Functions.EMPTY_ACTION);
    assertFalse(o.hasCustomOnError());
  }
  @Test public void customOnErrorShouldReportCustomOnError(){
    CallbackCompletableObserver o=new CallbackCompletableObserver(Functions.<Throwable>emptyConsumer(),Functions.EMPTY_ACTION);
    assertTrue(o.hasCustomOnError());
  }
}
