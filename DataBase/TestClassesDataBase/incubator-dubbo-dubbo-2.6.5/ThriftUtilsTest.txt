public class ThriftUtilsTest {
  @Test public void testGenerateMethodArgsClassName(){
    Assert.assertEquals($__DemoStub.echoString_args.class.getName(),ThriftUtils.generateMethodArgsClassName(com.alibaba.dubbo.rpc.gen.dubbo.Demo.class.getName(),"echoString"));
    Assert.assertEquals($__DemoStub.echoString_args.class.getName(),ExtensionLoader.getExtensionLoader(ClassNameGenerator.class).getExtension(DubboClassNameGenerator.NAME).generateArgsClassName(com.alibaba.dubbo.rpc.gen.dubbo.Demo.class.getName(),"echoString"));
  }
  @Test public void testGenerateMethodResultClassName(){

  }
  @Test public void testGenerateMethodArgsClassNameThrift(){
    Assert.assertEquals(com.alibaba.dubbo.rpc.gen.thrift.Demo.echoString_args.class.getName(),ThriftUtils.generateMethodArgsClassNameThrift(com.alibaba.dubbo.rpc.gen.thrift.Demo.Iface.class.getName(),"echoString"));

  }
  @Test public void testGenerateMethodResultClassNameThrift(){
