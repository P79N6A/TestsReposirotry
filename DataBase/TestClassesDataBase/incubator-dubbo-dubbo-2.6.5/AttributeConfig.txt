private static class AttributeConfig {
  private char letter;
  private boolean activate;
  private byte flag;
  public AttributeConfig(  char letter,  boolean activate,  byte flag){
    this.letter=letter;
    this.activate=activate;
    this.flag=flag;
  }
  @Parameter(attribute=true,key="let") public char getLetter(){
    return letter;
  }
  public void setLetter(  char letter){
    this.letter=letter;
  }
  @Parameter(attribute=true) public boolean isActivate(){
    return activate;
  }
  public void setActivate(  boolean activate){
    this.activate=activate;
  }
  public byte getFlag(){
    return flag;
  }
  public void setFlag(  byte flag){
    this.flag=flag;
  }
}
