/** 
 * Functional interface for creating committers, designed to allow different factories to be used to create different failure modes.
 */
@FunctionalInterface public interface CommitterFactory {
  /** 
 * Create a committer for a task.
 * @param context task context
 * @return new committer
 * @throws IOException failure
 */
  AbstractS3ACommitter createCommitter(  TaskAttemptContext context) throws IOException ;
}
