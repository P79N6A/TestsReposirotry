static class MRAppWithHistoryWithJobKilled extends MRAppWithHistory {
  public MRAppWithHistoryWithJobKilled(  int maps,  int reduces,  boolean autoComplete,  String testName,  boolean cleanOnStart){
    super(maps,reduces,autoComplete,testName,cleanOnStart);
  }
  @SuppressWarnings("unchecked") @Override protected void attemptLaunched(  TaskAttemptId attemptID){
    if (attemptID.getTaskId().getId() == 0) {
      getContext().getEventHandler().handle(new JobEvent(attemptID.getTaskId().getJobId(),JobEventType.JOB_KILL));
    }
 else {
      getContext().getEventHandler().handle(new TaskAttemptEvent(attemptID,TaskAttemptEventType.TA_DONE));
    }
  }
}
