public class MockClientContext extends ClientContext {
  private MockRemoteDirectoryManager remoteDirectoryMgr=new MockRemoteDirectoryManager();
  @Override public RemoteDirectoryManager getRemoteDirectoryManager(){
    return remoteDirectoryMgr;
  }
  @Override public synchronized YarnClient getOrCreateYarnClient(){
    YarnClient client=mock(YarnClient.class);
    try {
      when(client.getResourceTypeInfo()).thenReturn(ResourceUtils.getResourcesTypeInfo());
    }
 catch (    YarnException e) {
      fail(e.getMessage());
    }
catch (    IOException e) {
      fail(e.getMessage());
    }
    return client;
  }
}
