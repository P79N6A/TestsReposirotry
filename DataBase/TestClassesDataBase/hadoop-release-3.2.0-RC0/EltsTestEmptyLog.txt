/** 
 * Test the scenario where we have an empty edit log. This class is also useful in testing whether we can correctly handle various amounts of padding bytes at the end of the log.  We should be able to handle any amount of padding (including no padding) without throwing an exception.
 */
private static class EltsTestEmptyLog extends EditLogTestSetup {
  private final int paddingLength;
  public EltsTestEmptyLog(  int paddingLength){
    this.paddingLength=paddingLength;
  }
  @Override public void addTransactionsToLog(  EditLogOutputStream elos,  OpInstanceCache cache) throws IOException {
    padEditLog(elos,paddingLength);
  }
  @Override public long getLastValidTxId(){
    return -1;
  }
  @Override public Set<Long> getValidTxIds(){
    return new HashSet<Long>();
  }
}
