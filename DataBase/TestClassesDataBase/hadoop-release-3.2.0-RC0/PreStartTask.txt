public static class PreStartTask extends TaskRunner.Task {
  public static CountDownLatch latch=new CountDownLatch(1);
  public static boolean first;
  public PreStartTask(  long startTime){
    super.init(startTime);
  }
  @Override public void firstStep(){
    if (first) {
      Assert.fail();
    }
    first=true;
    latch.countDown();
  }
  @Override public void middleStep(){
  }
  @Override public void lastStep(){
  }
}
