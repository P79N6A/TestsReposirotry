static class ReducerClass extends Reducer<Text,Text,NullWritable,Text> {
  int count=0;
  @Override public void reduce(  Text key,  Iterable<Text> values,  Context ctx) throws IOException, InterruptedException {
    count=0;
    for (    Text value : values)     count++;
    ctx.write(NullWritable.get(),new Text(key.toString() + " " + count));
  }
}
