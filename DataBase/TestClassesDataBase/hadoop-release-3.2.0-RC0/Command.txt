private static class Command extends Shell {
  private int runCount=0;
  private Command(  long interval){
    super(interval);
  }
  @Override protected String[] getExecString(){
    return WINDOWS ? (new String[]{"cmd.exe","/c","echo","hello"}) : (new String[]{"echo","hello"});
  }
  @Override protected void parseExecResult(  BufferedReader lines) throws IOException {
    ++runCount;
  }
  public int getRunCount(){
    return runCount;
  }
}
