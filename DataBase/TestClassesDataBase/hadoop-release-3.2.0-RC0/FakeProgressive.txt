/** 
 * A dummy  {@link Progressive} implementation that allows users to set theprogress for testing. The  {@link Progressive#getProgress()} call will return the last progress value set using  {@link FakeProgressive#setProgress(float)}.
 */
static class FakeProgressive implements Progressive {
  private float progress=0F;
  @Override public float getProgress(){
    return progress;
  }
  void setProgress(  float progress){
    this.progress=progress;
  }
}
