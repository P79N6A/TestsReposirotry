public class StringLiteralEqualityCheckTest extends AbstractModuleTestSupport {
  @Override protected String getPackageLocation(){
    return "com/puppycrawl/tools/checkstyle/checks/coding/stringliteralequality";
  }
  @Test public void testIt() throws Exception {
    final DefaultConfiguration checkConfig=createModuleConfig(StringLiteralEqualityCheck.class);
    final String[] expected={"11:18: " + getCheckMessage(MSG_KEY,"=="),"16:20: " + getCheckMessage(MSG_KEY,"=="),"21:22: " + getCheckMessage(MSG_KEY,"==")};
    verify(checkConfig,getPath("InputStringLiteralEquality.java"),expected);
  }
  @Test public void testTokensNotNull(){
    final StringLiteralEqualityCheck check=new StringLiteralEqualityCheck();
    Assert.assertNotNull("Acceptable tokens should not be null",check.getAcceptableTokens());
    Assert.assertNotNull("Default tokens should not be null",check.getDefaultTokens());
    Assert.assertNotNull("Required tokens should not be null",check.getRequiredTokens());
  }
}
