private static final class MessageLevelPair {
  private final String msg;
  private final int level;
  MessageLevelPair(  String msg,  int level){
    this.msg=msg;
    this.level=level;
  }
  public String getMsg(){
    return msg;
  }
  public int getLevel(){
    return level;
  }
}
