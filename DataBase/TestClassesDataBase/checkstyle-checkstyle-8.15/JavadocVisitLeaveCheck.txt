private static class JavadocVisitLeaveCheck extends AbstractJavadocCheck {
  private static int visitCount;
  private static int leaveCount;
  public static void clearCounter(){
    visitCount=0;
    leaveCount=0;
  }
  @Override public int[] getRequiredJavadocTokens(){
    return new int[]{JavadocTokenTypes.TEXT};
  }
  @Override public int[] getDefaultJavadocTokens(){
    return getRequiredJavadocTokens();
  }
  @Override public int[] getAcceptableJavadocTokens(){
    return getRequiredJavadocTokens();
  }
  @Override public void visitJavadocToken(  DetailNode ast){
    visitCount++;
  }
  @Override public void leaveJavadocToken(  DetailNode ast){
    leaveCount++;
  }
}
