/** 
 * Checks that file can be parsed, when it ends on line-comment but without new-line.
 */
public class LineCommentAtTheEndOfFileTest extends AbstractModuleTestSupport {
  @Override protected String getPackageLocation(){
    return "com/puppycrawl/tools/checkstyle/grammar";
  }
  @Test public void testCanParse() throws Exception {
    final DefaultConfiguration checkConfig=createModuleConfig(MemberNameCheck.class);
    final String[] expected=CommonUtil.EMPTY_STRING_ARRAY;
    verify(checkConfig,getPath("InputLineCommentAtTheEndOfFile.java"),expected);
  }
}
