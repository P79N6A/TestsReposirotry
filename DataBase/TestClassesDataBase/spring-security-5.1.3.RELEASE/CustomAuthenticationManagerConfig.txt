@EnableWebFlux @EnableWebFluxSecurity static class CustomAuthenticationManagerConfig {
  @Bean SecurityWebFilterChain springSecurity(  ServerHttpSecurity http){
    http.oauth2ResourceServer().jwt().authenticationManager(authenticationManager());
    return http.build();
  }
  @Bean ReactiveAuthenticationManager authenticationManager(){
    return mock(ReactiveAuthenticationManager.class);
  }
}
