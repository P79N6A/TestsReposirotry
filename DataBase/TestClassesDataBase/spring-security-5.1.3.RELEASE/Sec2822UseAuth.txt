@Configuration static class Sec2822UseAuth {
  @Autowired public void useAuthenticationManager(  AuthenticationConfiguration auth) throws Exception {
    auth.getAuthenticationManager();
  }
  @Bean public static GlobalAuthenticationConfigurerAdapter bootGlobalAuthenticationConfigurerAdapter(){
    return new BootGlobalAuthenticationConfigurerAdapter();
  }
static class BootGlobalAuthenticationConfigurerAdapter extends GlobalAuthenticationConfigurerAdapter {
  }
}
