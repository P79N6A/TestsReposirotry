static class CopyUserPrincipal {
  public final String property;
  CopyUserPrincipal(  String property){
    this.property=property;
  }
  public CopyUserPrincipal(  CopyUserPrincipal toCopy){
    this.property=toCopy.property;
  }
  @Override public int hashCode(){
    final int prime=31;
    int result=1;
    result=prime * result + ((this.property == null) ? 0 : this.property.hashCode());
    return result;
  }
  @Override public boolean equals(  Object obj){
    if (this == obj) {
      return true;
    }
    if (obj == null) {
      return false;
    }
    if (getClass() != obj.getClass()) {
      return false;
    }
    CopyUserPrincipal other=(CopyUserPrincipal)obj;
    if (this.property == null) {
      if (other.property != null) {
        return false;
      }
    }
 else     if (!this.property.equals(other.property)) {
      return false;
    }
    return true;
  }
}
