@EnableWebFlux @EnableWebFluxSecurity static class CustomDecoderConfig {
  ReactiveJwtDecoder jwtDecoder=mock(ReactiveJwtDecoder.class);
  @Bean SecurityWebFilterChain springSecurity(  ServerHttpSecurity http){
    http.oauth2ResourceServer().jwt();
    return http.build();
  }
  @Bean ReactiveJwtDecoder jwtDecoder(){
    return this.jwtDecoder;
  }
}
