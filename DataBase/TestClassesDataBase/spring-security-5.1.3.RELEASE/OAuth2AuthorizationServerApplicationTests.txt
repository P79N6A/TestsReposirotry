/** 
 * Tests for  {@link OAuth2AuthorizationServerApplication}
 * @author Josh Cummings
 */
@RunWith(SpringRunner.class) @SpringBootTest @AutoConfigureMockMvc public class OAuth2AuthorizationServerApplicationTests {
  @Autowired MockMvc mvc;
  @Test public void requestTokenWhenUsingPasswordGrantTypeThenOk() throws Exception {
    this.mvc.perform(post("/oauth/token").param("grant_type","password").param("username","subject").param("password","password").header("Authorization","Basic cmVhZGVyOnNlY3JldA==")).andExpect(status().isOk());
  }
  @Test public void requestJwkSetWhenUsingDefaultsThenOk() throws Exception {
    this.mvc.perform(get("/.well-known/jwks.json")).andExpect(status().isOk());
  }
}
