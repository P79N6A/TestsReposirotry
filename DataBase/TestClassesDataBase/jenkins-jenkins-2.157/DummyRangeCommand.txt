@Extension public static class DummyRangeCommand extends RunRangeCommand {
  @Override public String getShortDescription(){
    return "DummyRangeCommand";
  }
  @Override protected int act(  List<Run<?,?>> builds) throws IOException {
    boolean comma=false;
    stdout.print("Builds: ");
    for (    Run<?,?> build : builds) {
      if (comma)       stdout.print(",");
 else       comma=true;
      stdout.print(build.getNumber());
    }
    stdout.println("");
    return 0;
  }
}
