class OverrideAbstractBuild extends InvisibleAction implements EnvironmentContributingAction {
  private boolean wasCalled=false;
  @Override @SuppressWarnings("deprecation") public void buildEnvVars(  AbstractBuild<?,?> abstractBuild,  EnvVars envVars){
    wasCalled=true;
  }
  boolean wasDeprecatedMethodCalled(){
    return wasCalled;
  }
}
