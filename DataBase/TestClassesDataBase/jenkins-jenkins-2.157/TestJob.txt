@SuppressWarnings("unchecked") private class TestJob extends Job implements TopLevelItem {
  int i;
  private final SortedMap<Integer,TestBuild> runs;
  public TestJob(  SortedMap<Integer,TestBuild> runs){
    super(rule.jenkins,"name");
    this.runs=runs;
    i=1;
  }
  @Override public int assignBuildNumber() throws IOException {
    return i++;
  }
  @Override public SortedMap<Integer,? extends Run> _getRuns(){
    return runs;
  }
  @Override public boolean isBuildable(){
    return true;
  }
  @Override protected void removeRun(  Run run){
  }
  public TopLevelItemDescriptor getDescriptor(){
    throw new AssertionError();
  }
}
