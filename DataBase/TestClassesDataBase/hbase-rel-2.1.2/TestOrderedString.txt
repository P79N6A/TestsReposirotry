@Category({MiscTests.class,SmallTests.class}) public class TestOrderedString {
  @ClassRule public static final HBaseClassTestRule CLASS_RULE=HBaseClassTestRule.forClass(TestOrderedString.class);
  static final String[] VALUES=new String[]{null,"","1","22","333","4444","55555","666666","7777777","88888888","999999999"};
  @Test public void testEncodedLength(){
    PositionedByteRange buff=new SimplePositionedMutableByteRange(20);
    for (    DataType<String> type : new OrderedString[]{OrderedString.ASCENDING,OrderedString.DESCENDING}) {
      for (      String val : VALUES) {
        buff.setPosition(0);
        type.encode(buff,val);
        assertEquals("encodedLength does not match actual, " + val,buff.getPosition(),type.encodedLength(val));
      }
    }
  }
}
