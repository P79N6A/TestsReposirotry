/** 
 * Test class for  {@link DropwizardMeter}.
 */
@Category(SmallTests.class) public class TestDropwizardMeter {
  @ClassRule public static final HBaseClassTestRule CLASS_RULE=HBaseClassTestRule.forClass(TestDropwizardMeter.class);
  private Meter meter;
  @Before public void setup(){
    this.meter=Mockito.mock(Meter.class);
  }
  @Test public void test(){
    DropwizardMeter dwMeter=new DropwizardMeter(this.meter);
    dwMeter.mark();
    dwMeter.mark(10L);
    dwMeter.mark();
    dwMeter.mark();
    Mockito.verify(meter,Mockito.times(3)).mark();
    Mockito.verify(meter).mark(10L);
  }
}
