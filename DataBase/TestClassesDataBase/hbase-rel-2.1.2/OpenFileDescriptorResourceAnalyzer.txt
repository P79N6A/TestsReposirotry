static class OpenFileDescriptorResourceAnalyzer extends ResourceChecker.ResourceAnalyzer {
  @Override public int getVal(  Phase phase){
    if (!JVM.isUnix()) {
      return 0;
    }
    JVM jvm=new JVM();
    return (int)jvm.getOpenFileDescriptorCount();
  }
  @Override public int getMax(){
    return 1024;
  }
}
