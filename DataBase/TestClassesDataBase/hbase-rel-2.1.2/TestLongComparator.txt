@Category(SmallTests.class) public class TestLongComparator {
  @ClassRule public static final HBaseClassTestRule CLASS_RULE=HBaseClassTestRule.forClass(TestLongComparator.class);
  private long values[]={Long.MIN_VALUE,-10000000000L,-1000000L,0L,1000000L,10000000000L,Long.MAX_VALUE};
  @Test public void testSimple(){
    for (int i=1; i < values.length; i++) {
      for (int j=0; j < i; j++) {
        LongComparator cp=new LongComparator(values[i]);
        assertEquals(1,cp.compareTo(Bytes.toBytes(values[j])));
        ByteBuffer data_bb=ByteBuffer.wrap(Bytes.toBytes(values[j]));
        assertEquals(1,cp.compareTo(data_bb,0,data_bb.capacity()));
      }
    }
  }
}
