/** 
 * A CacheableDeserializer implementation which just store reference to the  {@link ByteBuff} to bedeserialized. Use  {@link #getDeserializedByteBuff()} to get this reference.
 */
static class BufferGrabbingDeserializer implements CacheableDeserializer<Cacheable> {
  private ByteBuff buf;
  @Override public Cacheable deserialize(  ByteBuff b) throws IOException {
    return null;
  }
  @Override public Cacheable deserialize(  final ByteBuff b,  boolean reuse,  MemoryType memType) throws IOException {
    this.buf=b;
    return null;
  }
  @Override public int getDeserialiserIdentifier(){
    return 0;
  }
  public ByteBuff getDeserializedByteBuff(){
    return this.buf;
  }
}
