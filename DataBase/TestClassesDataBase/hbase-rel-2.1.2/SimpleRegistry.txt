/** 
 * Simple cluster registry inserted in place of our usual zookeeper based one.
 */
static class SimpleRegistry extends DoNothingAsyncRegistry {
  final ServerName META_HOST=META_SERVERNAME;
  public SimpleRegistry(  Configuration conf){
    super(conf);
  }
  @Override public CompletableFuture<RegionLocations> getMetaRegionLocation(){
    return CompletableFuture.completedFuture(new RegionLocations(new HRegionLocation(RegionInfoBuilder.FIRST_META_REGIONINFO,META_HOST)));
  }
  @Override public CompletableFuture<String> getClusterId(){
    return CompletableFuture.completedFuture(HConstants.CLUSTER_ID_DEFAULT);
  }
  @Override public CompletableFuture<Integer> getCurrentNrHRS(){
    return CompletableFuture.completedFuture(1);
  }
}
