/** 
 * Just counts when methods are called
 */
public static class DummyWALActionsListener implements WALActionsListener {
  public int preLogRollCounter=0;
  public int postLogRollCounter=0;
  public int closedCount=0;
  @Override public void preLogRoll(  Path oldFile,  Path newFile){
    preLogRollCounter++;
  }
  @Override public void postLogRoll(  Path oldFile,  Path newFile){
    postLogRollCounter++;
  }
  @Override public void logCloseRequested(){
    closedCount++;
  }
}
