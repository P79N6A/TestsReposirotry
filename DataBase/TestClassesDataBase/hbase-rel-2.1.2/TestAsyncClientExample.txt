@Category({ClientTests.class,MediumTests.class}) public class TestAsyncClientExample {
  @ClassRule public static final HBaseClassTestRule CLASS_RULE=HBaseClassTestRule.forClass(TestAsyncClientExample.class);
  private static final HBaseTestingUtility UTIL=new HBaseTestingUtility();
  private static final TableName TABLE_NAME=TableName.valueOf("test");
  @BeforeClass public static void setUp() throws Exception {
    UTIL.startMiniCluster(1);
    UTIL.createTable(TABLE_NAME,Bytes.toBytes("d"));
  }
  @AfterClass public static void tearDown() throws Exception {
    UTIL.shutdownMiniCluster();
  }
  @Test public void test() throws Exception {
    AsyncClientExample tool=new AsyncClientExample();
    tool.setConf(UTIL.getConfiguration());
    assertEquals(0,ToolRunner.run(tool,new String[]{TABLE_NAME.getNameAsString()}));
  }
}
