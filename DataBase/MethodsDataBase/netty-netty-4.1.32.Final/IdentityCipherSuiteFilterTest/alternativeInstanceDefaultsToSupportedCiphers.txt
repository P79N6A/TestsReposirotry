@Test public void alternativeInstanceDefaultsToSupportedCiphers(){
  List<String> defaultCiphers=Arrays.asList("FOO","BAR");
  Set<String> supportedCiphers=new HashSet<String>(Arrays.asList("BAZ","QIX"));
  String[] filtered=IdentityCipherSuiteFilter.INSTANCE_DEFAULTING_TO_SUPPORTED_CIPHERS.filterCipherSuites(null,defaultCiphers,supportedCiphers);
  assertArrayEquals(supportedCiphers.toArray(),filtered);
}
