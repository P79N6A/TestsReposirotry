@Test public void testEmptyHeaderValue(){
  EmbeddedChannel channel=new EmbeddedChannel(new HttpRequestDecoder());
  String crlf="\r\n";
  String request="GET /some/path HTTP/1.1" + crlf + "Host: localhost"+ crlf+ "EmptyHeader:"+ crlf+ crlf;
  channel.writeInbound(Unpooled.copiedBuffer(request,CharsetUtil.US_ASCII));
  HttpRequest req=channel.readInbound();
  assertEquals("",req.headers().get(of("EmptyHeader")));
}
