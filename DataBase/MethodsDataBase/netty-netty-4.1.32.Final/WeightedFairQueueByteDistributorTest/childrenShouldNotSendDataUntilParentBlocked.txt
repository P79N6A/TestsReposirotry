/** 
 * In this test, we block B which allows all bytes to be written by A. A should not share the data with its children since it's not blocked. <pre> 0 / \ A  [B] / \ C   D </pre>
 */
@Test public void childrenShouldNotSendDataUntilParentBlocked() throws Http2Exception {
  initState(STREAM_A,10,true);
  initState(STREAM_C,10,true);
  initState(STREAM_D,10,true);
  assertTrue(write(10));
  verifyWrite(STREAM_A,10);
  verifyNeverWrite(STREAM_B);
  verifyWrite(atMost(1),STREAM_C,0);
  verifyWrite(atMost(1),STREAM_D,0);
}
