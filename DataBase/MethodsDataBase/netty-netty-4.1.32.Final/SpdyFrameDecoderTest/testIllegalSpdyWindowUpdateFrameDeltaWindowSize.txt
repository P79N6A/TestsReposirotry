@Test public void testIllegalSpdyWindowUpdateFrameDeltaWindowSize() throws Exception {
  short type=9;
  byte flags=0;
  int length=8;
  int streamId=RANDOM.nextInt() & 0x7FFFFFFF;
  int deltaWindowSize=0;
  ByteBuf buf=Unpooled.buffer(SPDY_HEADER_SIZE + length);
  encodeControlFrameHeader(buf,type,flags,length);
  buf.writeInt(streamId);
  buf.writeInt(deltaWindowSize);
  decoder.decode(buf);
  verify(delegate).readFrameError(anyString());
  assertFalse(buf.isReadable());
  buf.release();
}
