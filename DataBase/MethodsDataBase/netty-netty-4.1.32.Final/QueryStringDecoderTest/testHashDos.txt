@Test public void testHashDos(){
  StringBuilder buf=new StringBuilder();
  buf.append('?');
  for (int i=0; i < 65536; i++) {
    buf.append('k');
    buf.append(i);
    buf.append("=v");
    buf.append(i);
    buf.append('&');
  }
  Assert.assertEquals(1024,new QueryStringDecoder(buf.toString()).parameters().size());
}
