@Test public void testStrictEquality(){
  DomainNameMapping<String> mapping=new DomainNameMappingBuilder<String>("NotFound").add("netty.io","Netty").add("downloads.netty.io","Netty-Downloads").build();
  assertEquals("Netty",mapping.map("netty.io"));
  assertEquals("Netty-Downloads",mapping.map("downloads.netty.io"));
  assertEquals("NotFound",mapping.map("x.y.z.netty.io"));
}
