@Test public void testFirstMatchWins(){
  assertEquals("Netty",new DomainNameMappingBuilder<String>("NotFound").add("*.netty.io","Netty").add("downloads.netty.io","Netty-Downloads").build().map("downloads.netty.io"));
  assertEquals("Netty-Downloads",new DomainNameMappingBuilder<String>("NotFound").add("downloads.netty.io","Netty-Downloads").add("*.netty.io","Netty").build().map("downloads.netty.io"));
}
