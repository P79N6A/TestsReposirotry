@Test(expected=Http2Exception.class) public void testLiteralWithIncrementalIndexingWithLargeValue() throws Http2Exception {
  StringBuilder sb=new StringBuilder();
  sb.append("4004");
  sb.append(hex("name"));
  sb.append("7F813F");
  for (int i=0; i < 8192; i++) {
    sb.append("61");
  }
  decode(sb.toString());
}
