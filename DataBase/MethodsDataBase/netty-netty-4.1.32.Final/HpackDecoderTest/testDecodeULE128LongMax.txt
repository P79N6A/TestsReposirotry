@Test public void testDecodeULE128LongMax() throws Http2Exception {
  byte[] input={(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0xFF,(byte)0x7F};
  ByteBuf in=Unpooled.wrappedBuffer(input);
  try {
    assertEquals(Long.MAX_VALUE,decodeULE128(in,0L));
  }
  finally {
    in.release();
  }
}
