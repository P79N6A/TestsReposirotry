@Test(expected=Http2Exception.class) public void failedWhenUnknownFrameInMiddleOfHeaderBlock() throws Http2Exception {
  final int streamId=1;
  ByteBuf input=Unpooled.buffer();
  try {
    Http2Headers headers=new DefaultHttp2Headers().authority("foo").method("get").path("/").scheme("https");
    Http2Flags flags=new Http2Flags().endOfHeaders(false).endOfStream(true);
    writeHeaderFrame(input,streamId,headers,flags);
    writeFrameHeader(input,0,(byte)0xff,new Http2Flags(),streamId);
    frameReader.readFrame(ctx,input,listener);
  }
  finally {
    input.release();
  }
}
