@Test public void readHeaderFrameAndContinuationFrame() throws Http2Exception {
  final int streamId=1;
  ByteBuf input=Unpooled.buffer();
  try {
    Http2Headers headers=new DefaultHttp2Headers().authority("foo").method("get").path("/").scheme("https");
    writeHeaderFrame(input,streamId,headers,new Http2Flags().endOfHeaders(false).endOfStream(true));
    writeContinuationFrame(input,streamId,new DefaultHttp2Headers().add("foo","bar"),new Http2Flags().endOfHeaders(true));
    frameReader.readFrame(ctx,input,listener);
    verify(listener).onHeadersRead(ctx,1,headers.add("foo","bar"),0,true);
  }
  finally {
    input.release();
  }
}
