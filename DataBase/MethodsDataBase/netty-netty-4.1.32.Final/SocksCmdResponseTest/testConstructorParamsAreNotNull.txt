@Test public void testConstructorParamsAreNotNull(){
  try {
    new SocksCmdResponse(null,SocksAddressType.UNKNOWN);
  }
 catch (  Exception e) {
    assertTrue(e instanceof NullPointerException);
  }
  try {
    new SocksCmdResponse(SocksCmdStatus.UNASSIGNED,null);
  }
 catch (  Exception e) {
    assertTrue(e instanceof NullPointerException);
  }
}
