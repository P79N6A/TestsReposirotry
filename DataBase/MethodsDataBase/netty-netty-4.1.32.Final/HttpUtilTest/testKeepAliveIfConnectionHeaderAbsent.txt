@Test public void testKeepAliveIfConnectionHeaderAbsent(){
  HttpMessage http11Message=new DefaultHttpRequest(HttpVersion.HTTP_1_1,HttpMethod.GET,"http:localhost/http_1_1");
  assertTrue(HttpUtil.isKeepAlive(http11Message));
  HttpMessage http10Message=new DefaultHttpRequest(HttpVersion.HTTP_1_0,HttpMethod.GET,"http:localhost/http_1_0");
  assertFalse(HttpUtil.isKeepAlive(http10Message));
}
