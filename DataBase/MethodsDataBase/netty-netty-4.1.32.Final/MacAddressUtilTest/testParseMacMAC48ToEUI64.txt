@Test public void testParseMacMAC48ToEUI64(){
  assertArrayEquals(new byte[]{0,(byte)0xaa,0x11,(byte)0xff,(byte)0xff,(byte)0xbb,0x22,(byte)0xcc},parseMAC("00-AA-11-FF-FF-BB-22-CC"));
  assertArrayEquals(new byte[]{0,(byte)0xaa,0x11,(byte)0xff,(byte)0xff,(byte)0xbb,0x22,(byte)0xcc},parseMAC("00:AA:11:FF:FF:BB:22:CC"));
}
