@Test public void testParseMacEUI48ToEUI64(){

  assertArrayEquals(new byte[]{0,(byte)0xaa,0x11,(byte)0xff,(byte)0xfe,(byte)0xbb,0x22,(byte)0xcc},parseMAC("00:AA:11:FF:FE:BB:22:CC"));
}
