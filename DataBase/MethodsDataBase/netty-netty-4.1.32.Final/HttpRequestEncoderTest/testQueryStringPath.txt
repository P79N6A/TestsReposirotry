@Test public void testQueryStringPath() throws Exception {
  for (  ByteBuf buffer : getBuffers()) {
    HttpRequestEncoder encoder=new HttpRequestEncoder();
    encoder.encodeInitialLine(buffer,new DefaultHttpRequest(HttpVersion.HTTP_1_1,HttpMethod.GET,"/?url=http://example.com"));
    String req=buffer.toString(Charset.forName("US-ASCII"));
    assertEquals("GET /?url=http://example.com HTTP/1.1\r\n",req);
    buffer.release();
  }
}
