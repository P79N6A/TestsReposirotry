@Test public void testGetCreationTime() throws Exception {
  clientSslCtx=SslContextBuilder.forClient().sslProvider(sslClientProvider()).sslContextProvider(clientSslContextProvider()).build();
  SSLEngine engine=null;
  try {
    engine=wrapEngine(clientSslCtx.newEngine(UnpooledByteBufAllocator.DEFAULT));
    assertTrue(engine.getSession().getCreationTime() <= System.currentTimeMillis());
  }
  finally {
    cleanupClientSslEngine(engine);
  }
}
