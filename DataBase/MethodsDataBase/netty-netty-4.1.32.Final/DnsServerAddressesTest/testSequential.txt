@Test public void testSequential(){
  DnsServerAddresses seq=DnsServerAddresses.sequential(ADDR1,ADDR2,ADDR3);
  assertThat(seq.stream(),is(not(sameInstance(seq.stream()))));
  for (int j=0; j < 2; j++) {
    DnsServerAddressStream i=seq.stream();
    assertNext(i,ADDR1);
    assertNext(i,ADDR2);
    assertNext(i,ADDR3);
    assertNext(i,ADDR1);
    assertNext(i,ADDR2);
    assertNext(i,ADDR3);
  }
}
