@Test public void testRotational(){
  DnsServerAddresses seq=DnsServerAddresses.rotational(ADDR1,ADDR2,ADDR3);
  DnsServerAddressStream i=seq.stream();
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  i=seq.stream();
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  i=seq.stream();
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  i=seq.stream();
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
  assertNext(i,ADDR1);
  assertNext(i,ADDR2);
  assertNext(i,ADDR3);
}
