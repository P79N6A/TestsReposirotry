@Test(expected=EncoderException.class) public void encodeNonFullHttpResponse100ContinueIsRejected() throws Exception {
  EmbeddedChannel ch=new EmbeddedChannel(new Http2StreamFrameToHttpObjectCodec(true));
  try {
    ch.writeOutbound(new DefaultHttpResponse(HttpVersion.HTTP_1_1,HttpResponseStatus.CONTINUE));
  }
  finally {
    ch.finishAndReleaseAll();
  }
}
