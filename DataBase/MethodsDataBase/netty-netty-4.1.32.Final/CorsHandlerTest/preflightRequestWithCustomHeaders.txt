@Test public void preflightRequestWithCustomHeaders(){
  final String headerName="CustomHeader";
  final String value1="value1";
  final String value2="value2";
  final CorsConfig config=forOrigin("http://localhost:8888").preflightResponseHeader(headerName,value1,value2).build();
  final HttpResponse response=preflightRequest(config,"http://localhost:8888","content-type, xheader1");
  assertValues(response,headerName,value1,value2);
  assertThat(response.headers().get(VARY),equalTo(ORIGIN.toString()));
}
