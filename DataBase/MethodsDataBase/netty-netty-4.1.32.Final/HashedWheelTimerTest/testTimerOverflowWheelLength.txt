@Test(timeout=5000) public void testTimerOverflowWheelLength() throws InterruptedException {
  final HashedWheelTimer timer=new HashedWheelTimer(Executors.defaultThreadFactory(),100,TimeUnit.MILLISECONDS,32);
  final CountDownLatch latch=new CountDownLatch(3);
  timer.newTimeout(new TimerTask(){
    @Override public void run(    final Timeout timeout) throws Exception {
      timer.newTimeout(this,100,TimeUnit.MILLISECONDS);
      latch.countDown();
    }
  }
,100,TimeUnit.MILLISECONDS);
  latch.await();
  assertFalse(timer.stop().isEmpty());
}
