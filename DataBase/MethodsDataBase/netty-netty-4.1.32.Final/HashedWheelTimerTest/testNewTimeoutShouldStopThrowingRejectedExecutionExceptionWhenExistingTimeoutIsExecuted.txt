@Test(timeout=3000) public void testNewTimeoutShouldStopThrowingRejectedExecutionExceptionWhenExistingTimeoutIsExecuted() throws InterruptedException {
  final CountDownLatch latch=new CountDownLatch(1);
  final HashedWheelTimer timer=new HashedWheelTimer(Executors.defaultThreadFactory(),25,TimeUnit.MILLISECONDS,4,true,2);
  timer.newTimeout(createNoOpTimerTask(),5,TimeUnit.SECONDS);
  timer.newTimeout(createCountDownLatchTimerTask(latch),90,TimeUnit.MILLISECONDS);
  latch.await();
  final CountDownLatch secondLatch=new CountDownLatch(1);
  timer.newTimeout(createCountDownLatchTimerTask(secondLatch),90,TimeUnit.MILLISECONDS);
  secondLatch.await();
  timer.stop();
}
