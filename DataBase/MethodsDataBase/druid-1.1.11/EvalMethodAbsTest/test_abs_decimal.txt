public void test_abs_decimal() throws Exception {
  Assert.assertEquals(new BigDecimal("12"),SQLEvalVisitorUtils.evalExpr(JdbcConstants.MYSQL,"abs(?)",new BigDecimal("12")));
  Assert.assertEquals(new BigDecimal("12"),SQLEvalVisitorUtils.evalExpr(JdbcConstants.MYSQL,"abs(?)",new BigDecimal("-12")));
}
