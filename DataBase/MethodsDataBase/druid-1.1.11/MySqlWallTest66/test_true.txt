public void test_true() throws Exception {
  WallProvider provider=new MySqlWallProvider();
  provider.getConfig().setSchemaCheck(true);
  Assert.assertTrue(provider.checkValid("SELECT LOGFILE_GROUP_NAME, FILE_NAME, TOTAL_EXTENTS, INITIAL_SIZE, ENGINE, EXTRA " + "FROM INFORMATION_SCHEMA.FILES WHERE FILE_TYPE = 'UNDO LOG' AND FILE_NAME IS NOT NULL " + "AND LOGFILE_GROUP_NAME IN ("+ "   SELECT DISTINCT LOGFILE_GROUP_NAME FROM INFORMATION_SCHEMA.FILES "+ "       WHERE FILE_TYPE = 'DATAFILE' "+ "           AND TABLESPACE_NAME IN ("+ "               SELECT DISTINCT TABLESPACE_NAME FROM INFORMATION_SCHEMA.PARTITIONS "+ "               WHERE TABLE_SCHEMA IN ('stat'))"+ ") "+ "GROUP BY LOGFILE_GROUP_NAME, FILE_NAME, ENGINE "+ "ORDER BY LOGFILE_GROUP_NAME"));
  Assert.assertEquals(2,provider.getTableStats().size());
}
