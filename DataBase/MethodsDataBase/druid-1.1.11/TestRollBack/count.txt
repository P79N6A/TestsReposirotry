public int count() throws Exception {
  Connection conn=null;
  Statement stmt=null;
  ResultSet rs=null;
  try {
    conn=dataSource.getConnection();
    stmt=conn.createStatement();
    rs=stmt.executeQuery("SELECT COUNT(*) FROM T_0");
    rs.next();
    return rs.getInt(1);
  }
  finally {
    JdbcUtils.close(rs);
    JdbcUtils.close(stmt);
    JdbcUtils.close(conn);
  }
}
