public void test_alter_first() throws Exception {
  String sql="alter table tabelname alter column operateVersion set default 0";
  MySqlStatementParser parser=new MySqlStatementParser(sql);
  SQLStatement stmt=parser.parseStatementList().get(0);
  parser.match(Token.EOF);
  Assert.assertEquals("ALTER TABLE tabelname" + "\n\tALTER COLUMN operateVersion SET DEFAULT 0",SQLUtils.toMySqlString(stmt));
  Assert.assertEquals("alter table tabelname" + "\n\talter column operateVersion set default 0",SQLUtils.toMySqlString(stmt,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
}
