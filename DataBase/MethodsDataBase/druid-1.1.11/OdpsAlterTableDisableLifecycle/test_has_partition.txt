public void test_has_partition() throws Exception {
  String sql="ALTER TABLE trans PARTITION(dt='20141111') DISABLE LIFECYCLE;";
  Assert.assertEquals("ALTER TABLE trans" + "\n\tPARTITION (dt = '20141111') DISABLE LIFECYCLE;",SQLUtils.formatOdps(sql));
  Assert.assertEquals("alter table trans" + "\n\tpartition (dt = '20141111') disable lifecycle;",SQLUtils.formatOdps(sql,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
}
