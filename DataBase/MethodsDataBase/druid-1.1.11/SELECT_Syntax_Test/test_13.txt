public void test_13() throws Exception {
  String sql="SELECT s1 FROM t1 WHERE s1 IN (SELECT s1 FROM t2);";
  SQLStatementParser parser=new MySqlStatementParser(sql);
  List<SQLStatement> stmtList=parser.parseStatementList();
  SQLStatement stmt=stmtList.get(0);
  assertEquals("SELECT s1\n" + "FROM t1\n" + "WHERE s1 IN (\n"+ "\tSELECT s1\n"+ "\tFROM t2\n"+ ");",SQLUtils.toMySqlString(stmt));
  assertEquals("select s1\n" + "from t1\n" + "where s1 in (\n"+ "\tselect s1\n"+ "\tfrom t2\n"+ ");",SQLUtils.toMySqlString(stmt,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
}
