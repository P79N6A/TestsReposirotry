public void test_permitTable_join() throws Exception {
  Assert.assertTrue(WallUtils.isValidateOracle("select * from t1, TAB"));
  Assert.assertTrue(WallUtils.isValidateOracle("select * from t1, tab"));
  Assert.assertTrue(WallUtils.isValidateOracle("select * from t1, SYS.TAB"));
  Assert.assertTrue(WallUtils.isValidateOracle("select * from t1, SYS.\"TAB\""));
}
