public void test_page() throws Exception {
  String sql="SELECT XX.*, ROWNUM AS RN" + "\nFROM (SELECT *" + "\n\tFROM t"+ "\n\tORDER BY id"+ "\n\t) XX"+ "\nWHERE ROWNUM <= 10";
  String mysqlSql=SQLUtils.translateOracleToMySql(sql);
  Assert.assertEquals("SELECT *" + "\nFROM t" + "\nORDER BY id"+ "\nLIMIT 10",mysqlSql);
  System.out.println(mysqlSql);
}
