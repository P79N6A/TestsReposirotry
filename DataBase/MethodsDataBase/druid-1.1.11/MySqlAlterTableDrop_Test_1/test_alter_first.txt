public void test_alter_first() throws Exception {
  String sql="ALTER TABLE test.student DROP Column9, DROP Column10;";
  MySqlStatementParser parser=new MySqlStatementParser(sql);
  SQLStatement stmt=parser.parseStatementList().get(0);
  parser.match(Token.EOF);
  assertEquals("ALTER TABLE test.student\n" + "\tDROP COLUMN Column9,\n" + "\tDROP COLUMN Column10;",SQLUtils.toMySqlString(stmt));
  assertEquals("alter table test.student\n" + "\tdrop column Column9,\n" + "\tdrop column Column10;",SQLUtils.toMySqlString(stmt,SQLUtils.DEFAULT_LCASE_FORMAT_OPTION));
}
