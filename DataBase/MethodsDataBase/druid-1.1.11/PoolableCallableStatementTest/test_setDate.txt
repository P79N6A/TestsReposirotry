public void test_setDate() throws Exception {
  stmt.setDate(1,(Date)null);
  stmt.setDate("1",(Date)null);
{
    SQLException error=null;
    try {
      stmt.setDate("0",(Date)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setDate(0,(Date)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
