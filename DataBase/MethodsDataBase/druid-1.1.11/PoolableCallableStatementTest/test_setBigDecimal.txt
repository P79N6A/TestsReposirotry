public void test_setBigDecimal() throws Exception {
  stmt.setBigDecimal(1,null);
  stmt.setBigDecimal("1",null);
{
    SQLException error=null;
    try {
      stmt.setBigDecimal("0",null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setBigDecimal(0,null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
