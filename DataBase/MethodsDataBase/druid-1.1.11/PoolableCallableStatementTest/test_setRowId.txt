public void test_setRowId() throws Exception {
  stmt.setRowId(1,(RowId)null);
  stmt.setRowId("1",(RowId)null);
{
    SQLException error=null;
    try {
      stmt.setRowId("0",(RowId)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setRowId(0,(RowId)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
