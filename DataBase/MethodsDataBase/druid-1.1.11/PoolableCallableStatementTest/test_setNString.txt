public void test_setNString() throws Exception {
  stmt.setNString(1,(String)null);
  stmt.setNString("1",(String)null);
{

    try {
      stmt.setNString("0",(String)null);
    }
 catch (    SQLException ex) {

    }
    Assert.assertNotNull(error);
  }
{

    try {
      stmt.setNString(0,(String)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
