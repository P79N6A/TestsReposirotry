public void test_setURL() throws Exception {
  stmt.setURL(1,null);
  stmt.setURL("1",null);
{
    SQLException error=null;
    try {
      stmt.setURL("0",null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setURL(0,null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
