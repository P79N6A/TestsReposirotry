public void test_setNClob() throws Exception {
  stmt.setNClob(1,(Reader)null);
  stmt.setNClob("1",(Reader)null);
{
    SQLException error=null;
    try {
      stmt.setNClob("0",(Reader)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setNClob(0,(Reader)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
