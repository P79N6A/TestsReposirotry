public void test_setBoolean() throws Exception {
  stmt.setBoolean(1,true);
  stmt.setBoolean("1",true);
{
    SQLException error=null;
    try {
      stmt.setBoolean("0",true);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setBoolean(0,true);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
