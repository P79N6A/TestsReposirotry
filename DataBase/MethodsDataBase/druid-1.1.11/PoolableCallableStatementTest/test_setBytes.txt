public void test_setBytes() throws Exception {
  stmt.setBytes(1,null);
  stmt.setBytes("1",null);
{
    SQLException error=null;
    try {
      stmt.setBytes("0",null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setBytes(0,null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
