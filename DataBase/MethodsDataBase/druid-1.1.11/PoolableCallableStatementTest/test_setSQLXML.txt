public void test_setSQLXML() throws Exception {
  stmt.setSQLXML(1,(SQLXML)null);
  stmt.setSQLXML("1",(SQLXML)null);
{
    SQLException error=null;
    try {
      stmt.setSQLXML("0",(SQLXML)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setSQLXML(0,(SQLXML)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
