public void test_getSQLXML() throws Exception {
  stmt.getSQLXML(1);
  stmt.getSQLXML("1");
{
    SQLException error=null;
    try {
      stmt.getSQLXML(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getSQLXML("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
