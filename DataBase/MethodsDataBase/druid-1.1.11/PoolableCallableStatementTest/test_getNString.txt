public void test_getNString() throws Exception {
  stmt.getNString(1);
  stmt.getNString("1");
{
    SQLException error=null;
    try {
      stmt.getNString(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getNString("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
