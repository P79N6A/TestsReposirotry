public void test_getDate() throws Exception {
  stmt.getDate(1);
  stmt.getDate("1");
{
    SQLException error=null;
    try {
      stmt.getDate(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getDate("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
