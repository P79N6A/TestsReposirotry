public void test_getURL() throws Exception {
  stmt.getURL(1);
  stmt.getURL("1");
{
    SQLException error=null;
    try {
      stmt.getURL(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getURL("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
