public void test_setInt() throws Exception {
  stmt.setInt(1,2);
  stmt.setInt("1",2);
{
    SQLException error=null;
    try {
      stmt.setInt("0",2);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setInt(0,2);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
