public void test_getCharacterStream() throws Exception {
  stmt.getCharacterStream(1);
  stmt.getCharacterStream("1");
{
    SQLException error=null;
    try {
      stmt.getCharacterStream(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getCharacterStream("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
