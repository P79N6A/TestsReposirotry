public void test_getObject() throws Exception {
  stmt.getObject(1,(java.util.Map)null);
  stmt.getObject("1",(java.util.Map)null);
{
    SQLException error=null;
    try {
      stmt.getObject(0,(java.util.Map)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getObject("0",(java.util.Map)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
