public void test_setLong() throws Exception {
  stmt.setLong(1,2);
  stmt.setLong("1",2);
{
    SQLException error=null;
    try {
      stmt.setLong("0",2);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setLong(0,2);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
