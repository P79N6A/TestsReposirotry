public void test_setTime() throws Exception {
  stmt.setTime(1,(Time)null);
  stmt.setTime("1",(Time)null);
{
    SQLException error=null;
    try {
      stmt.setTime("0",(Time)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setTime(0,(Time)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
