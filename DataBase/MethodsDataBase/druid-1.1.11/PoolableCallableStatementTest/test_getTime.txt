public void test_getTime() throws Exception {
  stmt.getTime(1);
  stmt.getTime("1");
{
    SQLException error=null;
    try {
      stmt.getTime(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getTime("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
