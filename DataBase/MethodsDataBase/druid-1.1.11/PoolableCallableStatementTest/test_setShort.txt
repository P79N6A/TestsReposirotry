public void test_setShort() throws Exception {
  stmt.setShort(1,Short.MAX_VALUE);
  stmt.setShort("1",Short.MAX_VALUE);
{
    SQLException error=null;
    try {
      stmt.setShort("0",Short.MAX_VALUE);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setShort(0,Short.MAX_VALUE);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
