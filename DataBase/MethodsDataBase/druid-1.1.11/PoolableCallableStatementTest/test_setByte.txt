public void test_setByte() throws Exception {
  stmt.setByte(1,Byte.MAX_VALUE);
  stmt.setByte("1",Byte.MAX_VALUE);
{
    SQLException error=null;
    try {
      stmt.setByte("0",Byte.MAX_VALUE);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setByte(0,Byte.MAX_VALUE);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
