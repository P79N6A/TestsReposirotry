public void test_setNull() throws Exception {
  stmt.setNull(1,Types.INTEGER,"Int");
  stmt.setNull("1",Types.INTEGER,"Int");
{
    SQLException error=null;
    try {
      stmt.setNull("0",Types.INTEGER,"Int");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setNull(0,Types.INTEGER,"Int");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
