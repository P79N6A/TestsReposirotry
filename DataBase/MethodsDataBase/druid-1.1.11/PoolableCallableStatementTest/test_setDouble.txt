public void test_setDouble() throws Exception {
  stmt.setDouble(1,1.0D);
  stmt.setDouble("1",1.0D);
{
    SQLException error=null;
    try {
      stmt.setDouble("0",1.0D);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setDouble(0,1.0D);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
