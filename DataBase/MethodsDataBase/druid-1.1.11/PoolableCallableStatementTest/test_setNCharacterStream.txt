public void test_setNCharacterStream() throws Exception {
  stmt.setNCharacterStream(1,(Reader)null);
  stmt.setNCharacterStream("1",(Reader)null);
{
    SQLException error=null;
    try {
      stmt.setNCharacterStream("0",(Reader)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setNCharacterStream(0,(Reader)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
