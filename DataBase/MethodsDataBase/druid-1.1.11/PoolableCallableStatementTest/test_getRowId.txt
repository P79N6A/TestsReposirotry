public void test_getRowId() throws Exception {
  stmt.getRowId(1);
  stmt.getRowId("1");
{
    SQLException error=null;
    try {
      stmt.getRowId(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getRowId("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
