public void test_getArray() throws Exception {
  stmt.getArray(1);
  stmt.getArray("1");
{
    SQLException error=null;
    try {
      stmt.getArray(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getArray("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
