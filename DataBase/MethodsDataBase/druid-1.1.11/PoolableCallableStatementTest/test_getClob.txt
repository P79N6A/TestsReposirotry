public void test_getClob() throws Exception {
  stmt.getClob(1);
  stmt.getClob("1");
{
    SQLException error=null;
    try {
      stmt.getClob(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getClob("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
