public void test_getTimestamp() throws Exception {
  stmt.getTimestamp(1);
  stmt.getTimestamp("1");
{
    SQLException error=null;
    try {
      stmt.getTimestamp(0);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.getTimestamp("0");
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
