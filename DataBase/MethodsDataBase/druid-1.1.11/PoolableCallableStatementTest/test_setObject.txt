public void test_setObject() throws Exception {
  stmt.setObject(1,null);
  stmt.setObject("1",null);
{
    SQLException error=null;
    try {
      stmt.setObject("0",null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setObject(0,null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
