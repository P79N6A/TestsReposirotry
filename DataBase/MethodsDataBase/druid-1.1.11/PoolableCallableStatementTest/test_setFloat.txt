public void test_setFloat() throws Exception {
  stmt.setFloat(1,1.0F);
  stmt.setFloat("1",1.0F);
{
    SQLException error=null;
    try {
      stmt.setFloat("0",1.0F);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setFloat(0,1.0F);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
