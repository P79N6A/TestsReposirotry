public void test_setClob_1() throws Exception {
  stmt.setClob(1,(Clob)null);
  stmt.setClob("1",(Clob)null);
{
    SQLException error=null;
    try {
      stmt.setClob("0",(Clob)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
{
    SQLException error=null;
    try {
      stmt.setClob(0,(Clob)null);
    }
 catch (    SQLException ex) {
      error=ex;
    }
    Assert.assertNotNull(error);
  }
}
