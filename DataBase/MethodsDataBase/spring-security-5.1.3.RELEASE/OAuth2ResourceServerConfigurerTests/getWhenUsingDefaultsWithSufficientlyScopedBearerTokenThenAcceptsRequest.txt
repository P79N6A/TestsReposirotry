@Test public void getWhenUsingDefaultsWithSufficientlyScopedBearerTokenThenAcceptsRequest() throws Exception {
  this.spring.register(WebServerConfig.class,DefaultConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidMessageReadScope");
  this.mvc.perform(get("/requires-read-scope").with(bearerToken(token))).andExpect(status().isOk()).andExpect(content().string("SCOPE_message:read"));
}
