@Test public void getWhenUsingDefaultsWithBearerTokenInTwoPlacesThenInvalidRequest() throws Exception {
  this.spring.register(DefaultConfig.class).autowire();
  this.mvc.perform(get("/").with(bearerToken("token")).with(bearerToken("token").asParam())).andExpect(status().isBadRequest()).andExpect(invalidRequestHeader("Found multiple bearer tokens in the request"));
}
