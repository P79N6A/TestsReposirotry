@Test public void postWhenUsingDefaultsWithExpiredBearerTokenAndNoCsrfThenInvalidToken() throws Exception {
  this.spring.register(WebServerConfig.class,DefaultConfig.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("Expired");
  this.mvc.perform(post("/authenticated").with(bearerToken(token))).andExpect(status().isUnauthorized()).andExpect(invalidTokenHeader("An error occurred while attempting to decode the Jwt"));
}
