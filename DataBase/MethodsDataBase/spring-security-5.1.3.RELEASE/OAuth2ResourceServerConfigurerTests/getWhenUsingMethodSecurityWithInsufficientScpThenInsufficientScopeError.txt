@Test public void getWhenUsingMethodSecurityWithInsufficientScpThenInsufficientScopeError() throws Exception {
  this.spring.register(WebServerConfig.class,MethodSecurityConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidMessageWriteScp");
  this.mvc.perform(get("/ms-requires-read-scope").with(bearerToken(token))).andExpect(status().isForbidden()).andExpect(insufficientScopeHeader("message:write"));
}
