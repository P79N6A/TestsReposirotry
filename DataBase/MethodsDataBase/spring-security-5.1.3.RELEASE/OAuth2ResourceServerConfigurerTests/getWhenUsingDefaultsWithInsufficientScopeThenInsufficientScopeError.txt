@Test public void getWhenUsingDefaultsWithInsufficientScopeThenInsufficientScopeError() throws Exception {
  this.spring.register(WebServerConfig.class,DefaultConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidNoScopes");
  this.mvc.perform(get("/requires-read-scope").with(bearerToken(token))).andExpect(status().isForbidden()).andExpect(insufficientScopeHeader(""));
}
