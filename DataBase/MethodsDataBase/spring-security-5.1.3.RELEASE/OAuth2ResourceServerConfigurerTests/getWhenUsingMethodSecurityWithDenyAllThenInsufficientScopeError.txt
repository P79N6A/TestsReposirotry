@Test public void getWhenUsingMethodSecurityWithDenyAllThenInsufficientScopeError() throws Exception {
  this.spring.register(WebServerConfig.class,MethodSecurityConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidMessageReadScope");
  this.mvc.perform(get("/ms-deny").with(bearerToken(token))).andExpect(status().isForbidden()).andExpect(insufficientScopeHeader("message:read"));
}
