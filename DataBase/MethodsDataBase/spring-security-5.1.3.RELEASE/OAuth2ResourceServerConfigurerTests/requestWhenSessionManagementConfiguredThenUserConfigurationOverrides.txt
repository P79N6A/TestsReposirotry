@Test public void requestWhenSessionManagementConfiguredThenUserConfigurationOverrides() throws Exception {
  this.spring.register(WebServerConfig.class,AlwaysSessionCreationConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidNoScopes");
  MvcResult result=this.mvc.perform(get("/").with(bearerToken(token))).andExpect(status().isOk()).andReturn();
  assertThat(result.getRequest().getSession(false)).isNotNull();
}
