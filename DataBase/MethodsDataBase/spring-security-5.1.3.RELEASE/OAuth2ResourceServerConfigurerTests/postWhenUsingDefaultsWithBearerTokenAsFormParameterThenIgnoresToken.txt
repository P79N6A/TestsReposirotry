@Test public void postWhenUsingDefaultsWithBearerTokenAsFormParameterThenIgnoresToken() throws Exception {
  this.spring.register(DefaultConfig.class).autowire();
  this.mvc.perform(post("/").with(bearerToken("token").asParam())).andExpect(status().isForbidden()).andExpect(header().doesNotExist(HttpHeaders.WWW_AUTHENTICATE));
}
