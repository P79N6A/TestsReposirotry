@Test public void getWhenUsingDefaultsWithValidBearerTokenThenAcceptsRequest() throws Exception {
  this.spring.register(WebServerConfig.class,DefaultConfig.class,BasicController.class).autowire();
  this.authz.enqueue(this.jwks("Default"));
  String token=this.token("ValidNoScopes");
  this.mvc.perform(get("/").with(bearerToken(token))).andExpect(status().isOk()).andExpect(content().string("ok"));
}
