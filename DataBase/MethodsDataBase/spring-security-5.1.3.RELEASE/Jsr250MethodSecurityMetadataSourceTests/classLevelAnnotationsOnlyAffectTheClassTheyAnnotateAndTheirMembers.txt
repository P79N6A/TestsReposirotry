/** 
 * Class-level annotations only affect the class they annotate and their members, that is, its methods and fields. They never affect a member declared by a superclass, even if it is not hidden or overridden by the class in question.
 * @throws Exception
 */
@Test public void classLevelAnnotationsOnlyAffectTheClassTheyAnnotateAndTheirMembers() throws Exception {
  Child target=new Child();
  MockMethodInvocation mi=new MockMethodInvocation(target,target.getClass(),"notOverriden");
  Collection<ConfigAttribute> accessAttributes=this.mds.getAttributes(mi);
  assertThat(accessAttributes).isNull();
}
