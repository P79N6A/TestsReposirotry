@Test public void poolingFlagIsNotSetWhenAuthenticationDnIsNotManagerUserDn() throws Exception {
  EnvExposingDefaultSpringSecurityContextSource ctxSrc=new EnvExposingDefaultSpringSecurityContextSource("ldap://blah:789/dc=springframework,dc=org");
  ctxSrc.setUserDn("manager");
  ctxSrc.setPassword("password");
  ctxSrc.afterPropertiesSet();
  assertThat(ctxSrc.getAuthenticatedEnvForTest("user","password")).doesNotContainKey(AbstractContextSource.SUN_LDAP_POOLING_FLAG);
}
