@Test public void requestWhenHstsDisabledThenExcludesHeader() throws Exception {
  Collection<String> hsts=Arrays.asList("Strict-Transport-Security");
  Map<String,String> allButHsts=remove(defaultHeaders,hsts);
  this.spring.configLocations(this.xml("HstsDisabled")).autowire();
  this.mvc.perform(get("/").secure(true)).andExpect(status().isOk()).andExpect(includes(allButHsts)).andExpect(excludes(hsts));
}
