@Test public void requestWhenUsingFrameOptionsSameOriginThenRespondsWithSameOrigin() throws Exception {
  Set<String> excludedHeaders=new HashSet<>(defaultHeaders.keySet());
  excludedHeaders.remove("X-Frame-Options");
  this.spring.configLocations(this.xml("DefaultsDisabledWithFrameOptionsSameOrigin")).autowire();
  this.mvc.perform(get("/")).andExpect(status().isOk()).andExpect(header().string("X-Frame-Options","SAMEORIGIN")).andExpect(excludes(excludedHeaders));
}
