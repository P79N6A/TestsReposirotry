@Test public void requestWhenUsingCustomHeaderThenRespondsWithThatHeader() throws Exception {
  this.spring.configLocations(this.xml("DefaultsDisabledWithCustomHeader")).autowire();
  this.mvc.perform(get("/")).andExpect(status().isOk()).andExpect(header().string("a","b")).andExpect(header().string("c","d")).andExpect(excludesDefaults());
}
