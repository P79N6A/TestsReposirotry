@Test public void requestWhenXssProtectionDisabledThenExcludesHeader() throws Exception {
  Collection<String> xssProtection=Arrays.asList("X-XSS-Protection");
  Map<String,String> allButXssProtection=remove(defaultHeaders,xssProtection);
  this.spring.configLocations(this.xml("XssProtectionDisabled")).autowire();
  this.mvc.perform(get("/").secure(true)).andExpect(status().isOk()).andExpect(includes(allButXssProtection)).andExpect(excludes(xssProtection));
}
