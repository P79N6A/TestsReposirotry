@Test public void requestWhenReferrerPolicyConfiguredThenResponseDefaultsToNoReferrer() throws Exception {
  this.spring.configLocations(this.xml("DefaultsDisabledWithReferrerPolicy")).autowire();
  this.mvc.perform(get("/")).andExpect(status().isOk()).andExpect(excludesDefaults()).andExpect(header().string("Referrer-Policy","no-referrer"));
}
