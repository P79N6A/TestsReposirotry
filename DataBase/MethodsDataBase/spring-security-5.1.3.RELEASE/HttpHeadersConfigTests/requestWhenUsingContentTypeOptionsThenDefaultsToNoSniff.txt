@Test public void requestWhenUsingContentTypeOptionsThenDefaultsToNoSniff() throws Exception {
  Set<String> excludedHeaders=new HashSet<>(defaultHeaders.keySet());
  excludedHeaders.remove("X-Content-Type-Options");
  this.spring.configLocations(this.xml("DefaultsDisabledWithContentTypeOptions")).autowire();
  this.mvc.perform(get("/")).andExpect(status().isOk()).andExpect(header().string("X-Content-Type-Options","nosniff")).andExpect(excludes(excludedHeaders));
}
