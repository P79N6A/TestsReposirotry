@Test public void requestWhenDisablingXssProtectionThenDefaultsToZero() throws Exception {
  Set<String> excludedHeaders=new HashSet<>(defaultHeaders.keySet());
  excludedHeaders.remove("X-XSS-Protection");
  this.spring.configLocations(this.xml("DefaultsDisabledWithXssProtectionDisabled")).autowire();
  this.mvc.perform(get("/")).andExpect(status().isOk()).andExpect(header().string("X-XSS-Protection","0")).andExpect(excludes(excludedHeaders));
}
