@Override @Test public void testBasePopulation(){
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from DIRECTORY",Integer.class)).isEqualTo(9);
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from FILE",Integer.class)).isEqualTo(90);
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from ACL_SID",Integer.class)).isEqualTo(4);
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from ACL_CLASS",Integer.class)).isEqualTo(2);
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from ACL_OBJECT_IDENTITY",Integer.class)).isEqualTo(100);
  assertThat(this.jdbcTemplate.queryForObject("select count(id) from ACL_ENTRY",Integer.class)).isEqualTo(115);
}
