@Test public void buildSuccessResponseWhenAllAttributesProvidedThenAllAttributesAreSet(){
  OAuth2AuthorizationResponse authorizationResponse=OAuth2AuthorizationResponse.success(AUTH_CODE).redirectUri(REDIRECT_URI).state(STATE).build();
  assertThat(authorizationResponse.getCode()).isEqualTo(AUTH_CODE);
  assertThat(authorizationResponse.getRedirectUri()).isEqualTo(REDIRECT_URI);
  assertThat(authorizationResponse.getState()).isEqualTo(STATE);
}
