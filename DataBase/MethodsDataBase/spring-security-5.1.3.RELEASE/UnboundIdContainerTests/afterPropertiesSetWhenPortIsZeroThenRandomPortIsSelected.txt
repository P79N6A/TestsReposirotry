@Test public void afterPropertiesSetWhenPortIsZeroThenRandomPortIsSelected() throws Exception {
  UnboundIdContainer server=new UnboundIdContainer("dc=springframework,dc=org",null);
  server.setPort(0);
  try {
    server.afterPropertiesSet();
    assertThat(server.getPort()).isNotEqualTo(0);
  }
  finally {
    server.destroy();
  }
}
