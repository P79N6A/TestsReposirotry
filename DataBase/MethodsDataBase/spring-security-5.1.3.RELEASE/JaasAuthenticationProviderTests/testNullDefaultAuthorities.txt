@Test public void testNullDefaultAuthorities(){
  UsernamePasswordAuthenticationToken token=new UsernamePasswordAuthenticationToken("user","password");
  assertThat(jaasProvider.supports(UsernamePasswordAuthenticationToken.class)).isTrue();
  Authentication auth=jaasProvider.authenticate(token);
  assertThat(auth.getAuthorities()).withFailMessage("Only ROLE_TEST1 and ROLE_TEST2 should have been returned").hasSize(2);
}
