@Test public void defaultPrefixIsCorrectlyApplied(){
  SimpleAuthorityMapper mapper=new SimpleAuthorityMapper();
  Set<String> mapped=AuthorityUtils.authorityListToSet(mapper.mapAuthorities(AuthorityUtils.createAuthorityList("AaA","ROLE_bbb")));
  assertThat(mapped.contains("ROLE_AaA")).isTrue();
  assertThat(mapped.contains("ROLE_bbb")).isTrue();
}
