/** 
 * Test if beeline prompts for a password when optional password option is used in conjunction with additional BeeLineOpts options after -p. Also, verifies the beelineOpt value is set as expected
 */
@Test public void testPromptPasswordVerifyBeelineOpts() throws Exception {
  List<String> argList=getBaseArgs(miniHS2.getBaseJdbcURL());
  argList.add("-n");
  argList.add("hive");
  argList.add("-p");
  argList.add("--maxColumnWidth=57");
  argList.add("-e");
  argList.add("show tables;");
  String output=connectWithPromptAndVerify(argList,"hivepassword",true,57,null,null);
  Assert.assertTrue("Table name " + tableName + " not found in the output",output.contains(tableName.toLowerCase()));
}
