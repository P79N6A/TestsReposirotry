@Test public void testPosMod(){
  HiveDecimal hd1=HiveDecimal.create("-100.91");
  assertTrue(hd1 != null);
  hd1.validate();
  HiveDecimal hd2=HiveDecimal.create("9.8");
  assertTrue(hd2 != null);
  hd2.validate();
  HiveDecimal dec=hd1.remainder(hd2).add(hd2).remainder(hd2);
  assertTrue(dec != null);
  dec.validate();
  Assert.assertEquals("6.89",dec.toString());
}
