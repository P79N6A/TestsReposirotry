/** 
 * Execute non-select statements using execute() and executeUpdated() APIs of PreparedStatement interface
 * @throws Exception
 */
@Test public void testExecutePreparedStatement() throws Exception {
  String key="testKey";
  String val1="val1";
  String val2="val2";
  PreparedStatement ps=con.prepareStatement("set " + key + " = ?");
  ps.setString(1,val1);
  ps.execute();
  verifyConfValue(con,key,val1);
  ps.clearParameters();
  ps.setString(1,val2);
  ps.executeUpdate();
  verifyConfValue(con,key,val2);
}
