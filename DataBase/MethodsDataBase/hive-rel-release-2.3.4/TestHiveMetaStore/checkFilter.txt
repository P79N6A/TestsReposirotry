private void checkFilter(HiveMetaStoreClient client,String dbName,String tblName,String filter,int expectedCount) throws MetaException, NoSuchObjectException, TException {
  LOG.debug("Testing filter: " + filter);
  List<Partition> partitions=client.listPartitionsByFilter(dbName,tblName,filter,(short)-1);
  assertEquals("Partition count expected for filter " + filter,expectedCount,partitions.size());
}
