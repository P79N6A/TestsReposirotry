private void cleanUp(String dbName,String tableName,String typeName) throws Exception {
  if (dbName != null && tableName != null) {
    client.dropTable(dbName,tableName);
  }
  if (dbName != null) {
    silentDropDatabase(dbName);
  }
  if (typeName != null) {
    client.dropType(typeName);
  }
}
