/** 
 * Test that Beeline queries don't treat semicolons inside quotations as query-ending characters.
 */
@Test public void testQueryNonEscapedSemiColon() throws Throwable {
  String SCRIPT_TEXT="drop table if exists nonEscapedSemiColon;create table nonEscapedSemiColon " + "(key int) ROW FORMAT DELIMITED FIELDS TERMINATED BY ';';show tables;";
  final String EXPECTED_PATTERN=" nonEscapedSemiColon ";
  List<String> argList=getBaseArgs(miniHS2.getBaseJdbcURL());
  testScriptFile(SCRIPT_TEXT,argList,EXPECTED_PATTERN,true);
}
