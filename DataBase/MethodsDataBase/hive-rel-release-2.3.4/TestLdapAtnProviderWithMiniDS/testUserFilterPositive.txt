@Test public void testUserFilterPositive(){
  testCase=defaultBuilder().userDNPatterns("uid=%s,ou=People,dc=example,dc=com").userFilters(USER1.getId()).build();
  testCase.assertAuthenticatePasses(USER1.credentialsWithId());
  testCase.assertAuthenticatePasses(USER1.credentialsWithDn());
  testCase=defaultBuilder().userDNPatterns("uid=%s,ou=People,dc=example,dc=com").userFilters(USER2.getId()).build();
  testCase.assertAuthenticatePasses(USER2.credentialsWithId());
  testCase.assertAuthenticatePasses(USER2.credentialsWithDn());
  testCase=defaultBuilder().userDNPatterns("uid=%s,ou=People,dc=example,dc=com").userFilters(USER1.getId(),USER2.getId()).build();
  testCase.assertAuthenticatePasses(USER1.credentialsWithId());
  testCase.assertAuthenticatePasses(USER1.credentialsWithDn());
  testCase.assertAuthenticatePasses(USER2.credentialsWithId());
  testCase.assertAuthenticatePasses(USER2.credentialsWithDn());
}
