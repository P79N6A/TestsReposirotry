@Test public void testParcelable(){
  String[] children=new String[]{"child0","child1"};
  Id3Frame[] subFrames=new Id3Frame[]{new TextInformationFrame("TIT2",null,"title"),new UrlLinkFrame("WXXX","description","url")};
  ChapterTocFrame chapterTocFrameToParcel=new ChapterTocFrame("id",false,true,children,subFrames);
  Parcel parcel=Parcel.obtain();
  chapterTocFrameToParcel.writeToParcel(parcel,0);
  parcel.setDataPosition(0);
  ChapterTocFrame chapterTocFrameFromParcel=ChapterTocFrame.CREATOR.createFromParcel(parcel);
  assertThat(chapterTocFrameFromParcel).isEqualTo(chapterTocFrameToParcel);
  parcel.recycle();
}
