@Test public void defaultInitialBitrateEstimate_forWifi_isGreaterThanEstimateFor2G(){
  setActiveNetworkInfo(networkInfoWifi);
  DefaultBandwidthMeter bandwidthMeterWifi=new DefaultBandwidthMeter.Builder(RuntimeEnvironment.application).build();
  long initialEstimateWifi=bandwidthMeterWifi.getBitrateEstimate();
  setActiveNetworkInfo(networkInfo2g);
  DefaultBandwidthMeter bandwidthMeter2g=new DefaultBandwidthMeter.Builder(RuntimeEnvironment.application).build();
  long initialEstimate2g=bandwidthMeter2g.getBitrateEstimate();
  assertThat(initialEstimateWifi).isGreaterThan(initialEstimate2g);
}
