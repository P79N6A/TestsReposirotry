@Test public void defaultInitialBitrateEstimate_for4G_isGreaterThanEstimateFor2G(){
  setActiveNetworkInfo(networkInfo4g);
  DefaultBandwidthMeter bandwidthMeter4g=new DefaultBandwidthMeter.Builder(RuntimeEnvironment.application).build();
  long initialEstimate4g=bandwidthMeter4g.getBitrateEstimate();
  setActiveNetworkInfo(networkInfo2g);
  DefaultBandwidthMeter bandwidthMeter2g=new DefaultBandwidthMeter.Builder(RuntimeEnvironment.application).build();
  long initialEstimate2g=bandwidthMeter2g.getBitrateEstimate();
  assertThat(initialEstimate4g).isGreaterThan(initialEstimate2g);
}
