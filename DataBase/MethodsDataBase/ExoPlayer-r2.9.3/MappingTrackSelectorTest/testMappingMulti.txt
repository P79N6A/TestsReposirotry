/** 
 * Tests video and audio track groups are mapped onto the correct renderers when there are multiple track groups of the same type.
 */
@Test public void testMappingMulti() throws ExoPlaybackException {
  FakeMappingTrackSelector trackSelector=new FakeMappingTrackSelector();
  TrackGroupArray multiTrackGroups=new TrackGroupArray(VIDEO_TRACK_GROUP,AUDIO_TRACK_GROUP,VIDEO_TRACK_GROUP);
  trackSelector.selectTracks(RENDERER_CAPABILITIES,multiTrackGroups);
  trackSelector.assertMappedTrackGroups(0,VIDEO_TRACK_GROUP,VIDEO_TRACK_GROUP);
  trackSelector.assertMappedTrackGroups(1,AUDIO_TRACK_GROUP);
}
