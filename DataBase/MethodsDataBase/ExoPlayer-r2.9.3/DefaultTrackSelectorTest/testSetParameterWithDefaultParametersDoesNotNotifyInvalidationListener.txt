/** 
 * Tests that track selector will not call {@link InvalidationListener#onTrackSelectionsInvalidated()} when it's set with defaultvalues of  {@link Parameters}.
 */
@Test public void testSetParameterWithDefaultParametersDoesNotNotifyInvalidationListener() throws Exception {
  trackSelector.init(invalidationListener,null);
  verify(invalidationListener,never()).onTrackSelectionsInvalidated();
}
