/** 
 * Tests that track selector will call  {@link InvalidationListener#onTrackSelectionsInvalidated()}when it's set with non-default values of  {@link Parameters}.
 */
@Test public void testSetParameterWithNonDefaultParameterNotifyInvalidationListener() throws Exception {
  Parameters parameters=new ParametersBuilder().setPreferredAudioLanguage("eng").build();
  trackSelector.init(invalidationListener,null);
  trackSelector.setParameters(parameters);
  verify(invalidationListener).onTrackSelectionsInvalidated();
}
