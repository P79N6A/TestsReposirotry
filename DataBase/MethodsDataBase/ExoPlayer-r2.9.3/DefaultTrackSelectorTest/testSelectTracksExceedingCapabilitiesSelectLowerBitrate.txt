/** 
 * Tests that track selector will select audio tracks with lower bit-rate when other factors are the same, and tracks exceed renderer's capabilities.
 */
@Test public void testSelectTracksExceedingCapabilitiesSelectLowerBitrate() throws Exception {
  Format lowerBitrateFormat=Format.createAudioSampleFormat("audioFormat",MimeTypes.AUDIO_AAC,null,15000,Format.NO_VALUE,2,44100,null,null,0,null);
  Format higherBitrateFormat=Format.createAudioSampleFormat("audioFormat",MimeTypes.AUDIO_AAC,null,30000,Format.NO_VALUE,2,44100,null,null,0,null);
  TrackSelectorResult result=trackSelector.selectTracks(new RendererCapabilities[]{ALL_AUDIO_FORMAT_EXCEEDED_RENDERER_CAPABILITIES},singleTrackGroup(lowerBitrateFormat,higherBitrateFormat));
  assertThat(result.selections.get(0).getSelectedFormat()).isEqualTo(lowerBitrateFormat);
}
