@Test public void testLoadActions() throws Exception {
  byte[] data1=Util.getUtf8Bytes("123");
  byte[] data2=Util.getUtf8Bytes("321");
  DownloadAction[] actions=loadActions(new Object[]{ActionFile.VERSION,2,"type1",FakeDownloadAction.VERSION,data1,"type2",FakeDownloadAction.VERSION,data2},new FakeDeserializer("type1"),new FakeDeserializer("type2"));
  assertThat(actions).isNotNull();
  assertThat(actions).hasLength(2);
  assertAction(actions[0],"type1",FakeDownloadAction.VERSION,data1);
  assertAction(actions[1],"type2",FakeDownloadAction.VERSION,data2);
}
