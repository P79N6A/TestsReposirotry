@Test public void testReadBit(){
  VorbisBitArray bitArray=new VorbisBitArray(TestUtil.createByteArray(0x5c,0x50));
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isFalse();
  assertThat(bitArray.readBit()).isTrue();
  assertThat(bitArray.readBit()).isFalse();
}
