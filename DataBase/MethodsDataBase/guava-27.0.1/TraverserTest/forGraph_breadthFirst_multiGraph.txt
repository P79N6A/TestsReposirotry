@Test public void forGraph_breadthFirst_multiGraph(){

  assertEqualCharNodes(traverser.breadthFirst('a'),"abcd");
  assertEqualCharNodes(traverser.breadthFirst('b'),"bd");

  assertEqualCharNodes(traverser.breadthFirst('d'),"d");
}
