@Test public void failsInAndroidPackage(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package android.test;\n" + "import android.view.View;\n"+ "import butterknife.BindView;\n"+ "public class Test {\n"+ "  @BindView(1) View thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindView-annotated class incorrectly in Android framework package. (android.test.Test)").in(source).onLine(5);
}
