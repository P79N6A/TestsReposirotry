@Test public void duplicateBindingFails() throws Exception {
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import android.view.View;\n"+ "import butterknife.BindView;\n"+ "public class Test {\n"+ "    @BindView(1) View thing1;\n"+ "    @BindView(1) View thing2;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("Attempt to use @BindView for an already bound ID 1 on 'thing1'. (test.Test.thing2)").in(source).onLine(6);
}
