@Test public void failsInJavaPackage(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package java.test;\n" + "import android.view.View;\n"+ "import butterknife.BindView;\n"+ "public class Test {\n"+ "  @BindView(1) View thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindView-annotated class incorrectly in Java framework package. (java.test.Test)").in(source).onLine(5);
}
