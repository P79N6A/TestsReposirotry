@Test public void failsIfNotView(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import butterknife.BindView;\n"+ "public class Test {\n"+ "  @BindView(1) String thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindView fields must extend from View or be an interface. (test.Test.thing)").in(source).onLine(4);
}
