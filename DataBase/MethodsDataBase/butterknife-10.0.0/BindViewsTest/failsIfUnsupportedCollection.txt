@Test public void failsIfUnsupportedCollection(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import android.view.View;\n"+ "import butterknife.BindViews;\n"+ "import java.util.Deque;\n"+ "public class Test {\n"+ "  @BindViews(1) Deque<View> thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindViews must be a List or array. (test.Test.thing)").in(source).onLine(6);
}
