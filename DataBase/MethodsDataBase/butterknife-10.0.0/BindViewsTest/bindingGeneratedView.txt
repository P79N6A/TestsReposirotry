@Test public void bindingGeneratedView(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import butterknife.BindViews;\n"+ "import java.util.List;\n"+ "@PerformGeneration\n"+ "public class Test {\n"+ "    @BindViews({1, 2}) List<GeneratedView> things;\n"+ "}");
  assertAbout(javaSources()).that(ImmutableList.of(source,TestGeneratingProcessor.ANNOTATION)).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("cannot find symbol");
  assertAbout(javaSources()).that(ImmutableList.of(source,TestGeneratingProcessor.ANNOTATION)).processedWith(new ButterKnifeProcessor(),new TestGeneratingProcessor("GeneratedView","package test;","import android.content.Context;","import android.view.View;","public class GeneratedView extends View {","  public GeneratedView(Context context) {","    super(context);","  }","}")).compilesWithoutError().withNoteContaining("@BindViews List or array with unresolved type (GeneratedView)").and().withNoteContaining("must elsewhere be generated as a View or interface").and().and().generatesFileNamed(StandardLocation.CLASS_OUTPUT,"test","Test_ViewBinding.class");
}
