@Test public void failsIfArrayNotView(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import butterknife.BindViews;\n"+ "public class Test {\n"+ "  @BindViews(1) String[] thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindViews List or array type must extend from View or be an interface. (test.Test.thing)").in(source).onLine(4);
}
