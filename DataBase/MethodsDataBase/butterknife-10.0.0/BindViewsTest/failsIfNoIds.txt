@Test public void failsIfNoIds(){
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import android.view.View;\n"+ "import butterknife.BindViews;\n"+ "import java.util.List;\n"+ "public class Test {\n"+ "  @BindViews({}) List<View> thing;\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("@BindViews must specify at least one ID. (test.Test.thing)").in(source).onLine(6);
}
