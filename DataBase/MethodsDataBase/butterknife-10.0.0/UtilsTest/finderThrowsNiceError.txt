@Test public void finderThrowsNiceError(){
  Context context=InstrumentationRegistry.getContext();
  View view=new View(context);
  try {
    Utils.findRequiredView(view,android.R.id.button1,"yo mama");
    fail();
  }
 catch (  IllegalStateException e) {
    assertThat(e).hasMessage("Required view 'button1' with ID " + android.R.id.button1 + " for yo mama was not found. If this view is optional add '@Nullable' (fields) or '@Optional' (methods) annotation.");
  }
}
