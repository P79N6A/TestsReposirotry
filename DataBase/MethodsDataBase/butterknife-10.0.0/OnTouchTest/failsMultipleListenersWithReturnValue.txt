@Test public void failsMultipleListenersWithReturnValue() throws Exception {
  JavaFileObject source=JavaFileObjects.forSourceString("test.Test","" + "package test;\n" + "import butterknife.OnTouch;\n"+ "public class Test {\n"+ "  @OnTouch(1) boolean doStuff1() {}\n"+ "  @OnTouch(1) boolean doStuff2() {}\n"+ "}");
  assertAbout(javaSource()).that(source).processedWith(new ButterKnifeProcessor()).failsToCompile().withErrorContaining("Multiple listener methods with return value specified for ID 1. (test.Test.doStuff2)").in(source).onLine(5);
}
