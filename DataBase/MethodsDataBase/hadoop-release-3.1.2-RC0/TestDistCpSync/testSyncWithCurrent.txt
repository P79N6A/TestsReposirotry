/** 
 * Similar test with testSync, but the "to" snapshot is specified as "."
 * @throws Exception
 */
@Test public void testSyncWithCurrent() throws Exception {
  final DistCpOptions options=new DistCpOptions.Builder(Collections.singletonList(source),target).withSyncFolder(true).withUseDiff("s1",".").build();
  context=new DistCpContext(options);
  initData(source);
  initData(target);
  enableAndCreateFirstSnapshot();
  changeData(source);
  sync();
  Assert.assertEquals(source,context.getSourcePaths().get(0));
}
