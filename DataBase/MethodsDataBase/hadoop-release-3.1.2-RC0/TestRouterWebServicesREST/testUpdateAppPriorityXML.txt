/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#updateApplicationPriority()} inside Router.
 */
@Test(timeout=2000) public void testUpdateAppPriorityXML() throws Exception {
  String appId=submitApplication();
  ClientResponse badResponse=performCall(RM_WEB_SERVICE_PATH + format(APPS_APPID_PRIORITY,appId),null,null,null,POST);
  assertEquals(SC_INTERNAL_SERVER_ERROR,badResponse.getStatus());
  AppPriority appPriority=new AppPriority(1);
  ClientResponse response=performCall(RM_WEB_SERVICE_PATH + format(APPS_APPID_PRIORITY,appId),null,null,appPriority,PUT);
  assertEquals(SC_OK,response.getStatus());
  AppPriority ci=response.getEntity(AppPriority.class);
  assertNotNull(ci);
}
