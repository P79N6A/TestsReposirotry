/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#dumpSchedulerLogs()} inside Router.
 */
@Test(timeout=2000) public void testDumpSchedulerLogsXML() throws Exception {
  ClientResponse badResponse=performCall(RM_WEB_SERVICE_PATH + SCHEDULER_LOGS,null,null,null,PUT);
  assertEquals(SC_INTERNAL_SERVER_ERROR,badResponse.getStatus());
  ClientResponse response=performCall(RM_WEB_SERVICE_PATH + SCHEDULER_LOGS,TIME,"1",null,POST);
  assertEquals(SC_BAD_REQUEST,response.getStatus());
  String ci=response.getEntity(String.class);
  assertNotNull(ci);
}
