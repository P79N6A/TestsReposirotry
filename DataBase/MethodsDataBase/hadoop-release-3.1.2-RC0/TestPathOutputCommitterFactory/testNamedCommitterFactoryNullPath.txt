/** 
 * Verify that if you explicitly name a committer and there's no path, the committer is picked up.
 */
@Test public void testNamedCommitterFactoryNullPath() throws Throwable {
  Configuration conf=new Configuration();
  conf.set(COMMITTER_FACTORY_CLASS,NAMED_COMMITTER_FACTORY);
  conf.set(NAMED_COMMITTER_CLASS,SimpleCommitter.class.getName());
  SimpleCommitter sc=createCommitter(NamedCommitterFactory.class,SimpleCommitter.class,null,conf);
  assertNull(sc.getOutputPath());
}
