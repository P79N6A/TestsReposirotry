/** 
 * Verify that if the committer output path is null, you get back a FileOutputCommitter with null output & work paths.
 */
@Test public void testCommitterNullOutputPath() throws Throwable {
  Configuration conf=newBondedConfiguration();
  FileOutputCommitter committer=createCommitter(FileOutputCommitterFactory.class,FileOutputCommitter.class,null,conf);
  assertNull(committer.getOutputPath());
  assertNull(committer.getWorkPath());
}
