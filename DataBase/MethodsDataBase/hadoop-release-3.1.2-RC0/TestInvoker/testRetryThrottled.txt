@Test public void testRetryThrottled() throws Throwable {
  S3ARetryPolicy policy=RETRY_POLICY;
  IOException ex=translateException("GET","/",newThrottledException());
  assertRetryAction("Expected retry on first throttle",policy,RetryPolicy.RetryAction.RETRY,ex,0,true);
  int retries=SAFE_RETRY_COUNT;
  assertRetryAction("Expected retry on repeated throttle",policy,RetryPolicy.RetryAction.RETRY,ex,retries,true);
  assertRetryAction("Expected retry on non-idempotent throttle",policy,RetryPolicy.RetryAction.RETRY,ex,retries,false);
}
