@Test public void testBadGetService(){
  final String serviceName="nonexistent-jenkins";
  final Response actual=apiServer.getService(request,serviceName);
  assertEquals("Get service is ",Response.status(Status.NOT_FOUND).build().getStatus(),actual.getStatus());
  ServiceStatus serviceStatus=(ServiceStatus)actual.getEntity();
  assertEquals("Response code don't match",RestApiConstants.ERROR_CODE_APP_NAME_INVALID,serviceStatus.getCode());
  assertEquals("Response diagnostics don't match","Service " + serviceName + " not found",serviceStatus.getDiagnostics());
}
