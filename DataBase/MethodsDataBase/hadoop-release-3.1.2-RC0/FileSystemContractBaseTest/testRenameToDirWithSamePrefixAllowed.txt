/** 
 * This a sanity check to make sure that any filesystem's handling of renames non-empty dirs doesn't cause any regressions.
 */
@Test public void testRenameToDirWithSamePrefixAllowed() throws Throwable {
  assumeTrue(renameSupported());
  final Path parentdir=path("testRenameToDirWithSamePrefixAllowed");
  fs.mkdirs(parentdir);
  createFile(new Path(parentdir,"mychild"));
  final Path dest=path("testRenameToDirWithSamePrefixAllowedDest");
  rename(parentdir,dest,true,false,true);
}
