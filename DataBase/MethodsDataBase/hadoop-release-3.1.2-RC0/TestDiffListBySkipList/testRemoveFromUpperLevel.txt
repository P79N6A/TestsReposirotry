@Test public void testRemoveFromUpperLevel() throws Exception {
  testRemove("FromUpperLevel",NUM_SNAPSHOTS,new ToIntBiFunction<DiffListBySkipList,Integer>(){
    private int level=MAX_LEVEL;
    @Override public int applyAsInt(    DiffListBySkipList skipList,    Integer integer){
      for (; level >= 0; level--) {
        final int index=findIndex(skipList,level);
        if (index != -1) {
          return index;
        }
        DiffListBySkipList.LOG.info("change from level " + level);
      }
      return -1;
    }
  }
);
}
