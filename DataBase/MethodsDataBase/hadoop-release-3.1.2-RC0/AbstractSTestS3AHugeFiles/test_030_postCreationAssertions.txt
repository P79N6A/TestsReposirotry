/** 
 * This is the set of actions to perform when verifying the file actually was created. With the s3guard committer, the file doesn't come into existence; a different set of assertions must be checked.
 */
@Test public void test_030_postCreationAssertions() throws Throwable {
  S3AFileSystem fs=getFileSystem();
  ContractTestUtils.assertPathExists(fs,"Huge file",hugefile);
  FileStatus status=fs.getFileStatus(hugefile);
  ContractTestUtils.assertIsFile(hugefile,status);
  assertEquals("File size in " + status,filesize,status.getLen());
}
