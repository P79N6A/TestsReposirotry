@Test public void testGetNonEmptyDir() throws Exception {
  final String dirPath="/a1/b1/c1";
  setupListStatus();
  PathMetadata meta=ms.get(strToPath(dirPath),true);
  if (!allowMissing() || meta != null) {

  }
}
