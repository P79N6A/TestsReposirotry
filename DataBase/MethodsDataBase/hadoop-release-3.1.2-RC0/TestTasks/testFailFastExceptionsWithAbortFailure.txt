@Test public void testFailFastExceptionsWithAbortFailure() throws Throwable {
  CounterTask failFirst=new CounterTask("task",1,Item::commit);
  CounterTask a=new CounterTask("aborter",1,Item::abort);
  intercept(IOException.class,() -> builder().stopOnFailure().abortWith(a).run(failFirst));
  if (!isParallel()) {
    a.assertInvokedAtLeast("abort",ITEM_COUNT - 1);
  }
}
