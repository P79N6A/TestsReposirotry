@Test public void testDisabledSandboxWithWhitelist() throws ContainerExecutionException {
  String[] inputCommand={"java jar MyJob.jar"};
  List<String> commands=Arrays.asList(inputCommand);
  conf.set(YarnConfiguration.YARN_CONTAINER_SANDBOX_WHITELIST_GROUP,WHITELIST_GROUP);
  runtimeContextBuilder.setExecutionAttribute(USER,WHITELIST_USER);
  runtimeContextBuilder.setExecutionAttribute(CONTAINER_RUN_CMDS,commands);
  runtime.prepareContainer(runtimeContextBuilder.build());
  Assert.assertTrue("Command should not be modified when user is " + "member of whitelisted group",inputCommand[0].equals(commands.get(0)));
}
