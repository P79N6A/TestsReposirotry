/** 
 * This is here to check up on the S3ATestUtils probes themselves.
 * @see S3ATestUtils#authenticationContains(Configuration, String).
 */
@Test public void testAuthenticationContainsProbes(){
  Configuration conf=new Configuration(false);
  assertFalse("found AssumedRoleCredentialProvider",authenticationContains(conf,AssumedRoleCredentialProvider.NAME));
  conf.set(AWS_CREDENTIALS_PROVIDER,AssumedRoleCredentialProvider.NAME);
  assertTrue("didn't find AssumedRoleCredentialProvider",authenticationContains(conf,AssumedRoleCredentialProvider.NAME));
}
