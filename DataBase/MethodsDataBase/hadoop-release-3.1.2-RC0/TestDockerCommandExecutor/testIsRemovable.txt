@Test public void testIsRemovable(){
  assertTrue(DockerCommandExecutor.isRemovable(DockerContainerStatus.STOPPED));
  assertTrue(DockerCommandExecutor.isRemovable(DockerContainerStatus.RESTARTING));
  assertTrue(DockerCommandExecutor.isRemovable(DockerContainerStatus.EXITED));
  assertTrue(DockerCommandExecutor.isRemovable(DockerContainerStatus.CREATED));
  assertTrue(DockerCommandExecutor.isRemovable(DockerContainerStatus.DEAD));
  assertFalse(DockerCommandExecutor.isRemovable(DockerContainerStatus.NONEXISTENT));
  assertFalse(DockerCommandExecutor.isRemovable(DockerContainerStatus.REMOVING));
  assertFalse(DockerCommandExecutor.isRemovable(DockerContainerStatus.UNKNOWN));
  assertFalse(DockerCommandExecutor.isRemovable(DockerContainerStatus.RUNNING));
}
