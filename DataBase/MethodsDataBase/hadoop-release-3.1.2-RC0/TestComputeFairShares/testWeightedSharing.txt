/** 
 * Basic test for weighted shares with no minimum shares and no low demands. Each pool should get slots in proportion to its weight.
 */
@Test public void testWeightedSharing(){
  scheds.add(new FakeSchedulable(0,2.0f));
  scheds.add(new FakeSchedulable(0,1.0f));
  scheds.add(new FakeSchedulable(0,1.0f));
  scheds.add(new FakeSchedulable(0,0.5f));
  ComputeFairShares.computeShares(scheds,Resources.createResource(45),ResourceInformation.MEMORY_MB.getName());
  verifyMemoryShares(20,10,10,5);
}
