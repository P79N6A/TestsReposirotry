/** 
 * Tests the converters indirectly through the public methods of the corresponding rowkey.
 */
@Test public void testAppToFlowRowKey(){
  byte[] byteRowKey=new AppToFlowRowKey(APPLICATION_ID).getRowKey();
  AppToFlowRowKey rowKey=AppToFlowRowKey.parseRowKey(byteRowKey);
  assertEquals(APPLICATION_ID,rowKey.getAppId());
}
