@Test public void testSubClusterHeartbeatUnknownSubCluster() throws Exception {
  SubClusterId subClusterId=SubClusterId.newInstance("SC");
  SubClusterHeartbeatRequest heartbeatRequest=SubClusterHeartbeatRequest.newInstance(subClusterId,SubClusterState.SC_RUNNING,"capability");
  try {
    stateStore.subClusterHeartbeat(heartbeatRequest);
    Assert.fail();
  }
 catch (  FederationStateStoreException e) {
    Assert.assertTrue(e.getMessage().startsWith("SubCluster SC does not exist; cannot heartbeat"));
  }
}
