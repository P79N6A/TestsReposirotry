@Test public void testGetApplicationHomeSubCluster() throws Exception {
  ApplicationId appId=ApplicationId.newInstance(1,1);
  SubClusterId subClusterId=SubClusterId.newInstance("SC");
  addApplicationHomeSC(appId,subClusterId);
  GetApplicationHomeSubClusterRequest getRequest=GetApplicationHomeSubClusterRequest.newInstance(appId);

}
