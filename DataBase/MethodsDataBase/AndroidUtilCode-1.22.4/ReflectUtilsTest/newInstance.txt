@Test public void newInstance(){
  assertEquals("",ReflectUtils.reflect(String.class).newInstance().get());
  assertEquals("abc",ReflectUtils.reflect(String.class).newInstance("abc").get());
  assertEquals("abc",ReflectUtils.reflect(String.class).newInstance("abc".getBytes()).get());
  assertEquals("abc",ReflectUtils.reflect(String.class).newInstance("abc".toCharArray()).get());
  assertEquals("b",ReflectUtils.reflect(String.class).newInstance("abc".toCharArray(),1,1).get());
}
