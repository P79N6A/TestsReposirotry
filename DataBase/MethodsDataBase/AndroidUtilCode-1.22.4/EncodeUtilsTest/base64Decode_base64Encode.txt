@Test public void base64Decode_base64Encode(){
  assertTrue(Arrays.equals("blankj".getBytes(),EncodeUtils.base64Decode(EncodeUtils.base64Encode("blankj"))));
  assertTrue(Arrays.equals("blankj".getBytes(),EncodeUtils.base64Decode(EncodeUtils.base64Encode2String("blankj".getBytes()))));
  assertEquals("Ymxhbmtq",EncodeUtils.base64Encode2String("blankj".getBytes()));
  assertTrue(Arrays.equals("Ymxhbmtq".getBytes(),EncodeUtils.base64Encode("blankj".getBytes())));
}
