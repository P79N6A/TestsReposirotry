@Test public void fusionLive(){
  AsyncProcessor<Integer> ap=new AsyncProcessor<Integer>();
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  ap.subscribe(ts);
  ts.assertOf(SubscriberFusion.<Integer>assertFuseable()).assertOf(SubscriberFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC));
  ts.assertNoValues().assertNoErrors().assertNotComplete();
  ap.onNext(1);
  ts.assertNoValues().assertNoErrors().assertNotComplete();
  ap.onComplete();
  ts.assertResult(1);
}
