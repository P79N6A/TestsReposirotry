@Test public void onNextCrossCancel(){
  AsyncProcessor<Object> p=AsyncProcessor.create();
  final TestSubscriber<Object> ts2=new TestSubscriber<Object>();
  TestSubscriber<Object> ts1=new TestSubscriber<Object>(){
    @Override public void onNext(    Object t){
      ts2.cancel();
      super.onNext(t);
    }
  }
;
  p.subscribe(ts1);
  p.subscribe(ts2);
  p.onNext(1);
  p.onComplete();
  ts1.assertResult(1);
  ts2.assertEmpty();
}
