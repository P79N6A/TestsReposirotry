@Test public void emptyInnerThenSingleBackpressured(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(1);
  Flowable.range(1,2).concatMapIterable(new Function<Integer,Iterable<Integer>>(){
    @Override public Iterable<Integer> apply(    Integer v){
      return v == 2 ? Collections.singleton(1) : Collections.<Integer>emptySet();
    }
  }
).subscribe(ts);
  ts.assertValue(1);
  ts.assertNoErrors();
  ts.assertComplete();
}
