@Test public void asyncFusedInput(){
  MulticastProcessor<Integer> mp=new MulticastProcessor<Integer>(128,true);
  UnicastProcessor<Integer> up=UnicastProcessor.create();
  up.subscribe(mp);
  TestSubscriber<Integer> ts1=mp.test();
  TestSubscriber<Integer> ts2=mp.take(1).test();
  up.onNext(1);
  up.onNext(2);
  up.onComplete();
  ts1.assertResult(1,2);
  ts2.assertResult(1);
}
