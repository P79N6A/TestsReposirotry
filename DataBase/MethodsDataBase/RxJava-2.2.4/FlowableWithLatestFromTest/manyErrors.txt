@Test public void manyErrors(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    new Flowable<Integer>(){
      @Override protected void subscribeActual(      Subscriber<? super Integer> subscriber){
        subscriber.onSubscribe(new BooleanSubscription());
        subscriber.onError(new TestException("First"));
        subscriber.onNext(1);
        subscriber.onError(new TestException("Second"));
        subscriber.onComplete();
      }
    }
.withLatestFrom(Flowable.just(2),Flowable.just(3),new Function3<Integer,Integer,Integer,Object>(){
      @Override public Object apply(      Integer a,      Integer b,      Integer c) throws Exception {
        return a;
      }
    }
).test().assertFailureAndMessage(TestException.class,"First");
    TestHelper.assertUndeliverable(errors,0,TestException.class,"Second");
  }
  finally {
    RxJavaPlugins.reset();
  }
}
