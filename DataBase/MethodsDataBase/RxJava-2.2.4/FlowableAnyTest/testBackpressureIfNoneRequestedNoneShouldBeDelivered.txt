@Test @Ignore("Single doesn't do backpressure") public void testBackpressureIfNoneRequestedNoneShouldBeDelivered(){
  TestObserver<Boolean> to=new TestObserver<Boolean>();
  Flowable.just(1).any(new Predicate<Integer>(){
    @Override public boolean test(    Integer t){
      return true;
    }
  }
).subscribe(to);
  to.assertNoValues();
  to.assertNoErrors();
  to.assertNotComplete();
}
