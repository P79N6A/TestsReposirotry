@Test public void normalBackpressure(){
  TestSubscriber<Integer> ts=TestSubscriber.create(0);
  PublishProcessor<Integer> pp=PublishProcessor.create();
  pp.onErrorResumeNext(new Function<Throwable,Flowable<Integer>>(){
    @Override public Flowable<Integer> apply(    Throwable v){
      return Flowable.range(3,2);
    }
  }
).subscribe(ts);
  ts.request(2);
  pp.onNext(1);
  pp.onNext(2);
  pp.onError(new TestException("Forced failure"));
  ts.assertValues(1,2);
  ts.assertNoErrors();
  ts.assertNotComplete();
  ts.request(2);
  ts.assertValues(1,2,3,4);
  ts.assertNoErrors();
  ts.assertComplete();
}
