@Test public void otherPublisherNextSlipsThrough(){
  Maybe.just(1).delay(new Flowable<Integer>(){
    @Override protected void subscribeActual(    Subscriber<? super Integer> s){
      s.onSubscribe(new BooleanSubscription());
      s.onNext(1);
      s.onNext(2);
    }
  }
).test().assertResult(1);
}
