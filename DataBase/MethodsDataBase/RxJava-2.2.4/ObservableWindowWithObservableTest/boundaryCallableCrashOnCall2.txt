@Test public void boundaryCallableCrashOnCall2(){
  BehaviorSubject.createDefault(1).window(new Callable<Observable<Integer>>(){
    int calls;
    @Override public Observable<Integer> call() throws Exception {
      if (++calls == 2) {
        throw new TestException();
      }
      return Observable.just(1);
    }
  }
).test().assertError(TestException.class).assertNotComplete();
}
