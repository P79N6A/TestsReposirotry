@Test public void testFirstOrDefaultWithPredicateAndOneElementFlowable(){
  Flowable<Integer> flowable=Flowable.just(1,2).filter(new Predicate<Integer>(){
    @Override public boolean test(    Integer t1){
      return t1 % 2 == 0;
    }
  }
).first(4).toFlowable();
  Subscriber<Integer> subscriber=TestHelper.mockSubscriber();
  flowable.subscribe(subscriber);
  InOrder inOrder=inOrder(subscriber);
  inOrder.verify(subscriber,times(1)).onNext(2);
  inOrder.verify(subscriber,times(1)).onComplete();
  inOrder.verifyNoMoreInteractions();
}
