@SuppressWarnings("unchecked") @Test public void noSubsequentSubscription(){
  final int[] calls={0};
  Maybe<Integer> source=Maybe.create(new MaybeOnSubscribe<Integer>(){
    @Override public void subscribe(    MaybeEmitter<Integer> s) throws Exception {
      calls[0]++;
      s.onSuccess(1);
    }
  }
);
  Maybe.concat(Arrays.asList(source,source)).firstElement().test().assertResult(1);
  assertEquals(1,calls[0]);
}
