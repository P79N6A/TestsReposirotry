@Test public void testInnerBackpressureWithAlignedBoundaries(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
  Flowable.range(0,Flowable.bufferSize() * 2).concatWith(Flowable.range(0,Flowable.bufferSize() * 2)).observeOn(Schedulers.computation()).subscribe(ts);
  ts.awaitTerminalEvent();
  ts.assertNoErrors();
  assertEquals(Flowable.bufferSize() * 4,ts.valueCount());
}
