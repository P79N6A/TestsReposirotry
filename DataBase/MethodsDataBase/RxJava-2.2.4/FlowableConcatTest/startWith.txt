@SuppressWarnings("unchecked") @Test @Ignore("startWith(a, b, ...) replaced by startWithArray(T...)") public void startWith() throws Exception {
  for (int i=2; i < 10; i++) {
    Class<?>[] clazz=new Class[i];
    Arrays.fill(clazz,Object.class);
    Object[] obs=new Object[i];
    Arrays.fill(obs,1);
    Integer[] expected=new Integer[i];
    Arrays.fill(expected,1);
    Method m=Flowable.class.getMethod("startWith",clazz);
    TestSubscriber<Integer> ts=TestSubscriber.create();
    ((Flowable<Integer>)m.invoke(Flowable.empty(),obs)).subscribe(ts);
    ts.assertValues(expected);
    ts.assertNoErrors();
    ts.assertComplete();
  }
}
