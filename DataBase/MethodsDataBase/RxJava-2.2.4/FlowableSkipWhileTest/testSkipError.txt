@Test public void testSkipError(){
  Flowable<Integer> src=Flowable.just(1,2,42,5,3,1);
  src.skipWhile(LESS_THAN_FIVE).subscribe(w);
  InOrder inOrder=inOrder(w);
  inOrder.verify(w,never()).onNext(anyInt());
