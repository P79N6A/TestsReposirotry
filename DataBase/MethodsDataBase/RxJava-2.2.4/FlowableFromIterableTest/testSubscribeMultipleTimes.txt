@Test public void testSubscribeMultipleTimes(){
  Flowable<Integer> f=Flowable.fromIterable(Arrays.asList(1,2,3));
  for (int i=0; i < 10; i++) {
    TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
    f.subscribe(ts);
    ts.assertValues(1,2,3);
    ts.assertNoErrors();
    ts.assertComplete();
  }
}
