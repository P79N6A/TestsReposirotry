@Test public void biconsumer(){
  final Object[] value={null,null};
  Single.just(1).subscribe(new BiConsumer<Integer,Throwable>(){
    @Override public void accept(    Integer v,    Throwable e) throws Exception {
      value[0]=v;
      value[1]=e;
    }
  }
);
  assertEquals(1,value[0]);
  assertNull(value[1]);
}
