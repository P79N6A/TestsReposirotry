@Test public void biConsumerIsDisposedOnError(){
  final Object[] result={null,null};
  Disposable d=Single.<Integer>error(new IOException()).subscribe(new BiConsumer<Integer,Throwable>(){
    @Override public void accept(    Integer t1,    Throwable t2) throws Exception {
      result[0]=t1;
      result[1]=t2;
    }
  }
);
  assertTrue("Not disposed?!",d.isDisposed());
  assertNull(result[0]);
  assertTrue("" + result[1],result[1] instanceof IOException);
}
