@Test public void biconsumerError(){
  final Object[] value={null,null};
  TestException ex=new TestException();
  Single.error(ex).subscribe(new BiConsumer<Object,Throwable>(){
    @Override public void accept(    Object v,    Throwable e) throws Exception {
      value[0]=v;
      value[1]=e;
    }
  }
);
  assertNull(value[0]);
  assertEquals(ex,value[1]);
}
