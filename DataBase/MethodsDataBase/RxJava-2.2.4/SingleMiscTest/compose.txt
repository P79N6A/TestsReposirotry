@Test public void compose(){
  Single.just(1).compose(new SingleTransformer<Integer,Object>(){
    @Override public SingleSource<Object> apply(    Single<Integer> f){
      return f.map(new Function<Integer,Object>(){
        @Override public Object apply(        Integer v) throws Exception {
          return v + 1;
        }
      }
);
    }
  }
).test().assertResult(2);
}
