@Test public void retryBiPredicate(){
  Single.fromCallable(new Callable<Object>(){
    int c;
    @Override public Object call() throws Exception {
      if (++c != 5) {
        throw new TestException();
      }
      return 1;
    }
  }
).retry(new BiPredicate<Integer,Throwable>(){
    @Override public boolean test(    Integer i,    Throwable e) throws Exception {
      return true;
    }
  }
).test().assertResult(1);
}
