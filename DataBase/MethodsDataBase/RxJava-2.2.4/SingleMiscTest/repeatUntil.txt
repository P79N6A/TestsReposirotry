@Test public void repeatUntil(){
  final AtomicBoolean flag=new AtomicBoolean();
  Single.just(1).doOnSuccess(new Consumer<Integer>(){
    int c;
    @Override public void accept(    Integer v) throws Exception {
      if (++c == 5) {
        flag.set(true);
      }
    }
  }
).repeatUntil(new BooleanSupplier(){
    @Override public boolean getAsBoolean() throws Exception {
      return flag.get();
    }
  }
).test().assertResult(1,1,1,1,1);
}
