@Test public void testBackpressureNonEmpty(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(0L);
  Flowable.just(1,2,3).defaultIfEmpty(1).subscribe(ts);
  ts.assertNoValues();
  ts.assertNotTerminated();
  ts.request(2);
  ts.assertValues(1,2);
  ts.request(1);
  ts.assertValues(1,2,3);
  ts.assertNoErrors();
  ts.assertComplete();
}
