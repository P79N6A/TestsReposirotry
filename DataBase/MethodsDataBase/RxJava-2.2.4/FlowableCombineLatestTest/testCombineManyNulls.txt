@Ignore("Nulls are not allowed") @Test public void testCombineManyNulls(){
  int n=Flowable.bufferSize() * 3;
  Flowable<Integer> source=Flowable.just((Integer)null);
  List<Flowable<Integer>> sources=new ArrayList<Flowable<Integer>>();
  for (int i=0; i < n; i++) {
    sources.add(source);
  }
  TestSubscriber<Integer> ts=TestSubscriber.create();
  Flowable.combineLatest(sources,new Function<Object[],Integer>(){
    @Override public Integer apply(    Object[] args){
      int sum=0;
      for (      Object o : args) {
        if (o == null) {
          sum++;
        }
      }
      return sum;
    }
  }
).subscribe(ts);
  ts.assertValue(n);
  ts.assertNoErrors();
  ts.assertComplete();
}
