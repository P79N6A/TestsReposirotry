@SuppressWarnings("unchecked") @Test public void combineLatestArrayOfSources(){
  Flowable.combineLatest(new Flowable[]{Flowable.just(1),Flowable.just(2)},new Function<Object[],Object>(){
    @Override public Object apply(    Object[] a) throws Exception {
      return Arrays.toString(a);
    }
  }
).test().assertResult("[1, 2]");
}
