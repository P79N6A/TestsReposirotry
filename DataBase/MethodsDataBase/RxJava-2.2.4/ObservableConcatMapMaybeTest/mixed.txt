@Test public void mixed(){
  Observable.range(1,10).concatMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      if (v % 2 == 0) {
        return Maybe.just(v);
      }
      return Maybe.empty();
    }
  }
).test().assertResult(2,4,6,8,10);
}
