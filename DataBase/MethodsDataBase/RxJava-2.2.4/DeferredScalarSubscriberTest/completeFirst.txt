@Test public void completeFirst(){
  TestSubscriber<Integer> ts=TestSubscriber.create(0L);
  TestingDeferredScalarSubscriber ds=new TestingDeferredScalarSubscriber(ts);
  ds.setupDownstream();
  ds.onNext(1);
  ts.assertNoValues();
  ds.onComplete();
  ts.assertNoValues();
  ts.request(1);
  ts.assertValues(1);
  ts.assertNoErrors();
  ts.assertComplete();
}
