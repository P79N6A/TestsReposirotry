@Test public void rangeSource(){
  TestSubscriber<Integer> ts=TestSubscriber.create(0);
  TestingDeferredScalarSubscriber ds=new TestingDeferredScalarSubscriber(ts);
  ds.subscribeTo(Flowable.range(1,10));
  ts.assertNoValues();
  ts.request(1);
  ts.assertValue(10);
  ts.assertNoErrors();
  ts.assertComplete();
}
