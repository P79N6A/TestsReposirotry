@Test public void justSource(){
  TestSubscriber<Integer> ts=TestSubscriber.create(0L);
  TestingDeferredScalarSubscriber ds=new TestingDeferredScalarSubscriber(ts);
  ds.subscribeTo(Flowable.just(1));
  ts.assertNoValues();
  ts.request(1);
  ts.assertValue(1);
  ts.assertNoErrors();
  ts.assertComplete();
}
