@Test(expected=TestException.class) public void testToIteratorWithException(){
  Flowable<String> obs=Flowable.unsafeCreate(new Publisher<String>(){
    @Override public void subscribe(    Subscriber<? super String> subscriber){
      subscriber.onSubscribe(new BooleanSubscription());
      subscriber.onNext("one");
      subscriber.onError(new TestException());
    }
  }
);
  Iterator<String> it=obs.blockingIterable().iterator();
  assertEquals(true,it.hasNext());
  assertEquals("one",it.next());
  assertEquals(true,it.hasNext());
  it.next();
}
