@Test(expected=TestException.class) public void testToIteratorWithException(){
  Observable<String> obs=Observable.unsafeCreate(new ObservableSource<String>(){
    @Override public void subscribe(    Observer<? super String> observer){
      observer.onSubscribe(Disposables.empty());
      observer.onNext("one");
      observer.onError(new TestException());
    }
  }
);
  Iterator<String> it=obs.blockingIterable().iterator();
  assertEquals(true,it.hasNext());
  assertEquals("one",it.next());
  assertEquals(true,it.hasNext());
  it.next();
}
