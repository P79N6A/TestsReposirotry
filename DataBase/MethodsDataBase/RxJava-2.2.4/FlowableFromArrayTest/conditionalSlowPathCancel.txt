@Test public void conditionalSlowPathCancel(){
  Flowable.fromArray(new Integer[]{1,2,3,4,5}).filter(Functions.alwaysTrue()).subscribeWith(new TestSubscriber<Integer>(5L){
    @Override public void onNext(    Integer t){
      super.onNext(t);
      if (t == 1) {
        cancel();
        onComplete();
      }
    }
  }
).assertResult(1);
}
