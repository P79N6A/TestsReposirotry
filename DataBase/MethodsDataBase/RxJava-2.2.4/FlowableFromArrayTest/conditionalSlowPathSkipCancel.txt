@Test public void conditionalSlowPathSkipCancel(){
  Flowable.fromArray(new Integer[]{1,2,3,4,5}).filter(new Predicate<Integer>(){
    @Override public boolean test(    Integer v) throws Exception {
      return v < 2;
    }
  }
).subscribeWith(new TestSubscriber<Integer>(5L){
    @Override public void onNext(    Integer t){
      super.onNext(t);
      if (t == 1) {
        cancel();
        onComplete();
      }
    }
  }
).assertResult(1);
}
