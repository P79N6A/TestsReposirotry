@Test public void inputOutputAsyncFusedErrorDelayed(){
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  UnicastProcessor<Integer> us=UnicastProcessor.create();
  us.observeOn(Schedulers.single(),true).subscribe(ts);
  us.onError(new TestException());
  ts.awaitDone(5,TimeUnit.SECONDS).assertFailure(TestException.class);
  SubscriberFusion.assertFusion(ts,QueueFuseable.ASYNC).awaitDone(5,TimeUnit.SECONDS).assertFailure(TestException.class);
}
