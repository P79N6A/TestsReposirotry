@Test(timeout=2000) public void testUnsubscribeFromSynchronousInfiniteFlowable(){
  final AtomicLong count=new AtomicLong();
  INFINITE_OBSERVABLE.take(10).subscribe(new Consumer<Long>(){
    @Override public void accept(    Long l){
      count.set(l);
    }
  }
);
  assertEquals(10,count.get());
}
