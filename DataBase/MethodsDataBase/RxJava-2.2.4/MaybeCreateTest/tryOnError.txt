@Test public void tryOnError(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    final Boolean[] response={null};
    Maybe.create(new MaybeOnSubscribe<Object>(){
      @Override public void subscribe(      MaybeEmitter<Object> e) throws Exception {
        e.onSuccess(1);
        response[0]=e.tryOnError(new TestException());
      }
    }
).test().assertResult(1);
    assertFalse(response[0]);
    assertTrue(errors.toString(),errors.isEmpty());
  }
  finally {
    RxJavaPlugins.reset();
  }
}
