@Test public void boundedBlockingSubscribeConsumerConsumerBufferExceed(){
  final List<Object> list=new ArrayList<Object>();
  Flowable.range(1,5).subscribeOn(Schedulers.computation()).blockingSubscribe(new Consumer<Integer>(){
    @Override public void accept(    Integer v) throws Exception {
      list.add(v);
    }
  }
,Functions.emptyConsumer(),3);
  assertEquals(Arrays.asList(1,2,3,4,5),list);
}
