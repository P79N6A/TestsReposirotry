@SuppressWarnings("unchecked") @Test public void bufferExactBoundarySecondBufferCrash(){
  PublishProcessor<Integer> pp=PublishProcessor.create();
  PublishProcessor<Integer> b=PublishProcessor.create();
  TestSubscriber<List<Integer>> ts=pp.buffer(b,new Callable<List<Integer>>(){
    int calls;
    @Override public List<Integer> call() throws Exception {
      if (++calls == 2) {
        throw new TestException();
      }
      return new ArrayList<Integer>();
    }
  }
).test();
  b.onNext(1);
  ts.assertFailure(TestException.class);
}
