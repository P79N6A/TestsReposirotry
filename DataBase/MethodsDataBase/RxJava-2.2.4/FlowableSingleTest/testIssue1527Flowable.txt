@Test(timeout=30000) public void testIssue1527Flowable() throws InterruptedException {
  Flowable<Integer> source=Flowable.just(1,2,3,4,5,6);
  Flowable<Integer> reduced=source.reduce(new BiFunction<Integer,Integer,Integer>(){
    @Override public Integer apply(    Integer i1,    Integer i2){
      return i1 + i2;
    }
  }
).toFlowable();
  Integer r=reduced.blockingFirst();
  assertEquals(21,r.intValue());
}
