@Test public void testSingleDoesNotRequestMoreThanItNeedsToEmitItem(){
  final AtomicLong request=new AtomicLong();
  Flowable.just(1).doOnRequest(new LongConsumer(){
    @Override public void accept(    long n){
      request.addAndGet(n);
    }
  }
).blockingSingle();
  assertEquals(Long.MAX_VALUE,request.get());
}
