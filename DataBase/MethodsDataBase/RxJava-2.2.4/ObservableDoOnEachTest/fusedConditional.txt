@Test @Ignore("Fusion not supported yet") public void fusedConditional(){
  TestObserver<Integer> to=ObserverFusion.newTest(QueueFuseable.ANY);
  final int[] call={0,0};
  Observable.range(1,5).doOnNext(new Consumer<Integer>(){
    @Override public void accept(    Integer v) throws Exception {
      call[0]++;
    }
  }
).doOnComplete(new Action(){
    @Override public void run() throws Exception {
      call[1]++;
    }
  }
).filter(Functions.alwaysTrue()).subscribe(to);
  to.assertOf(ObserverFusion.<Integer>assertFuseable()).assertOf(ObserverFusion.<Integer>assertFusionMode(QueueFuseable.SYNC)).assertResult(1,2,3,4,5);
  assertEquals(5,call[0]);
  assertEquals(1,call[1]);
}
