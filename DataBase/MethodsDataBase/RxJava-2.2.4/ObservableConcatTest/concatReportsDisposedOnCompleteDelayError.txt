@Test @SuppressWarnings("unchecked") public void concatReportsDisposedOnCompleteDelayError(){
  final Disposable[] disposable={null};
  Observable.concatArrayDelayError(Observable.just(1),Observable.just(2)).subscribe(new Observer<Integer>(){
    @Override public void onSubscribe(    Disposable d){
      disposable[0]=d;
    }
    @Override public void onNext(    Integer t){
    }
    @Override public void onError(    Throwable e){
    }
    @Override public void onComplete(){
    }
  }
);
  assertTrue(disposable[0].isDisposed());
}
