@Test public void backpressured(){
  TestSubscriber<Integer> ts=Flowable.range(1,1024).switchMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      if (v % 2 == 0) {
        return Maybe.just(v);
      }
      return Maybe.empty();
    }
  }
).test(0L);
  ts.requestMore(1).assertResult(1024);
}
