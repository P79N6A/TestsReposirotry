@Test public void disposeBeforeSwitchInOnNext(){
  final TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
  Flowable.just(1).switchMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      ts.cancel();
      return Maybe.just(1);
    }
  }
).subscribe(ts);
  ts.assertEmpty();
}
