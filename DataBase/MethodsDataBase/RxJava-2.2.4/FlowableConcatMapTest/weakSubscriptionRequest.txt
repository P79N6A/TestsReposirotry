@Test public void weakSubscriptionRequest(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(0);
  WeakScalarSubscription<Integer> ws=new WeakScalarSubscription<Integer>(1,ts);
  ts.onSubscribe(ws);
  ws.request(0);
  ts.assertEmpty();
  ws.request(1);
  ts.assertResult(1);
  ws.request(1);
  ts.assertResult(1);
}
