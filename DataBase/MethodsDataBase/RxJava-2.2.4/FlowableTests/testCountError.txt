@Test public void testCountError(){
  Flowable<String> f=Flowable.error(new Callable<Throwable>(){
    @Override public Throwable call(){
      return new RuntimeException();
    }
  }
);
  f.count().subscribe(wo);
  verify(wo,never()).onSuccess(anyInt());

}
