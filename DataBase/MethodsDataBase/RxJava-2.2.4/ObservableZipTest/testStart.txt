@Test public void testStart(){
  Observable<String> os=OBSERVABLE_OF_5_INTEGERS.zipWith(OBSERVABLE_OF_5_INTEGERS,new BiFunction<Integer,Integer,String>(){
    @Override public String apply(    Integer a,    Integer b){
      return a + "-" + b;
    }
  }
);
  final ArrayList<String> list=new ArrayList<String>();
  os.subscribe(new Consumer<String>(){
    @Override public void accept(    String s){
      System.out.println(s);
      list.add(s);
    }
  }
);
  assertEquals(5,list.size());
  assertEquals("1-1",list.get(0));
  assertEquals("2-2",list.get(1));
  assertEquals("5-5",list.get(4));
}
