@Test public void testStartAsync() throws InterruptedException {
  Observable<String> os=ASYNC_OBSERVABLE_OF_INFINITE_INTEGERS(new CountDownLatch(1)).zipWith(ASYNC_OBSERVABLE_OF_INFINITE_INTEGERS(new CountDownLatch(1)),new BiFunction<Integer,Integer,String>(){
    @Override public String apply(    Integer a,    Integer b){
      return a + "-" + b;
    }
  }
).take(5);
  TestObserver<String> to=new TestObserver<String>();
  os.subscribe(to);
  to.awaitTerminalEvent();
  to.assertNoErrors();
  assertEquals(5,to.valueCount());
  assertEquals("1-1",to.values().get(0));
  assertEquals("2-2",to.values().get(1));
  assertEquals("5-5",to.values().get(4));
}
