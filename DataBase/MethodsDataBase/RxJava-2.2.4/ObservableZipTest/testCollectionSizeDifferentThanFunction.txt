@SuppressWarnings("unchecked") @Test public void testCollectionSizeDifferentThanFunction(){
  Function<Object[],String> zipr=Functions.toFunction(getConcatStringIntegerIntArrayZipr());
  Observer<String> observer=TestHelper.mockObserver();
  @SuppressWarnings("rawtypes") Collection ws=java.util.Collections.singleton(Observable.just("one","two"));
  Observable<String> w=Observable.zip(ws,zipr);
  w.subscribe(observer);
  verify(observer,times(1)).onError(any(Throwable.class));
  verify(observer,never()).onComplete();
  verify(observer,never()).onNext(any(String.class));
}
