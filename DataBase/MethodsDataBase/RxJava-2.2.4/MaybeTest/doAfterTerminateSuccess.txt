@Test public void doAfterTerminateSuccess(){
  final int[] call={0};
  Maybe.just(1).doOnSuccess(new Consumer<Integer>(){
    @Override public void accept(    Integer v) throws Exception {
      call[0]++;
    }
  }
).doAfterTerminate(new Action(){
    @Override public void run() throws Exception {
      if (call[0] == 1) {
        call[0]=-1;
      }
    }
  }
).test().assertResult(1);
  assertEquals(-1,call[0]);
}
