@Test public void doAfterTerminateComplete(){
  final int[] call={0};
  Maybe.empty().doOnComplete(new Action(){
    @Override public void run() throws Exception {
      call[0]++;
    }
  }
).doAfterTerminate(new Action(){
    @Override public void run() throws Exception {
      if (call[0] == 1) {
        call[0]=-1;
      }
    }
  }
).test().assertResult();
  assertEquals(-1,call[0]);
}
