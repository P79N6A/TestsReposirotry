@Test public void compose(){
  Maybe.just(1).compose(new MaybeTransformer<Integer,Integer>(){
    @Override public MaybeSource<Integer> apply(    Maybe<Integer> m){
      return m.map(new Function<Integer,Integer>(){
        @Override public Integer apply(        Integer w) throws Exception {
          return w + 1;
        }
      }
);
    }
  }
).test().assertResult(2);
}
