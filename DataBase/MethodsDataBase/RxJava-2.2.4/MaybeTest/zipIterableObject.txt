@Test public void zipIterableObject(){
  @SuppressWarnings("unchecked") final List<Maybe<Integer>> maybes=Arrays.asList(Maybe.just(1),Maybe.just(4));
  Maybe.zip(maybes,new Function<Object[],Object>(){
    @Override public Object apply(    final Object[] o) throws Exception {
      int sum=0;
      for (      Object i : o) {
        sum+=(Integer)i;
      }
      return sum;
    }
  }
).test().assertResult(5);
}
