@Test public void doOnEventComplete(){
  final List<Object> list=new ArrayList<Object>();
  assertTrue(Maybe.<Integer>empty().doOnEvent(new BiConsumer<Integer,Throwable>(){
    @Override public void accept(    Integer v,    Throwable e) throws Exception {
      list.add(v);
      list.add(e);
    }
  }
).subscribe().isDisposed());
  assertEquals(Arrays.asList(null,null),list);
}
