@SuppressWarnings("unchecked") @Test public void mergeArrayFused(){
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  Maybe.mergeArray(Maybe.just(1),Maybe.just(2),Maybe.just(3)).subscribe(ts);
  ts.assertSubscribed().assertOf(SubscriberFusion.<Integer>assertFuseable()).assertOf(SubscriberFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC)).assertResult(1,2,3);
}
