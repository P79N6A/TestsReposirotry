@Test public void testNoUnsubscribeDownstream(){
  Observable<Integer> source=Observable.range(1,1000).takeWhile(new Predicate<Integer>(){
    @Override public boolean test(    Integer t1){
      return t1 < 2;
    }
  }
);
  TestObserver<Integer> to=new TestObserver<Integer>();
  source.subscribe(to);
  to.assertNoErrors();
  to.assertValue(1);
}
