@Test public void cancel(){
  PublishProcessor<Integer> pp=PublishProcessor.create();
  TestSubscriber<List<Integer>> ts=pp.parallel().collect(new Callable<List<Integer>>(){
    @Override public List<Integer> call() throws Exception {
      return new ArrayList<Integer>();
    }
  }
,new BiConsumer<List<Integer>,Integer>(){
    @Override public void accept(    List<Integer> a,    Integer b) throws Exception {
      a.add(b);
    }
  }
).sequential().test();
  assertTrue(pp.hasSubscribers());
  ts.cancel();
  assertFalse(pp.hasSubscribers());
}
