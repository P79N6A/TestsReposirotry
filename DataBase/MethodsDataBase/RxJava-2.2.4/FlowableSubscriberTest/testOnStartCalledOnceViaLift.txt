@Test public void testOnStartCalledOnceViaLift(){
  final AtomicInteger c=new AtomicInteger();
  Flowable.just(1,2,3,4).lift(new FlowableOperator<Integer,Integer>(){
    @Override public Subscriber<? super Integer> apply(    final Subscriber<? super Integer> child){
      return new DefaultSubscriber<Integer>(){
        @Override public void onStart(){
          c.incrementAndGet();
          request(1);
        }
        @Override public void onComplete(){
          child.onComplete();
        }
        @Override public void onError(        Throwable e){
          child.onError(e);
        }
        @Override public void onNext(        Integer t){
          child.onNext(t);
          request(1);
        }
      }
;
    }
  }
).subscribe();
  assertEquals(1,c.get());
}
