@Test public void testErrorInParentObservable(){
  TestObserver<Integer> to=new TestObserver<Integer>();
  Observable.mergeDelayError(Observable.just(Observable.just(1),Observable.just(2)).startWith(Observable.<Integer>error(new RuntimeException()))).subscribe(to);
  to.awaitTerminalEvent();
  to.assertTerminated();
  to.assertValues(1,2);
  assertEquals(1,to.errorCount());
}
