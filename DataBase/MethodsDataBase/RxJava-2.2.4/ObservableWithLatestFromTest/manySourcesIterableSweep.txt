@Test public void manySourcesIterableSweep(){
  for (  String val : new String[]{"1"}) {
    int n=35;
    for (int i=0; i < n; i++) {
      List<Observable<?>> sources=new ArrayList<Observable<?>>();
      List<String> expected=new ArrayList<String>();
      expected.add(val);
      for (int j=0; j < i; j++) {
        sources.add(Observable.just(val));
        expected.add(String.valueOf(val));
      }
      TestObserver<String> to=new TestObserver<String>();
      PublishSubject<String> main=PublishSubject.create();
      main.withLatestFrom(sources,toArray).subscribe(to);
      to.assertNoValues();
      main.onNext(val);
      main.onComplete();
      to.assertValue(expected.toString());
      to.assertNoErrors();
      to.assertComplete();
    }
  }
}
