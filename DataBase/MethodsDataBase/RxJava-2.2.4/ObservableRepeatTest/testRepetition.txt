@Test(timeout=2000) public void testRepetition(){
  int num=10;
  final AtomicInteger count=new AtomicInteger();
  int value=Observable.unsafeCreate(new ObservableSource<Integer>(){
    @Override public void subscribe(    final Observer<? super Integer> o){
      o.onNext(count.incrementAndGet());
      o.onComplete();
    }
  }
).repeat().subscribeOn(Schedulers.computation()).take(num).blockingLast();
  assertEquals(num,value);
}
