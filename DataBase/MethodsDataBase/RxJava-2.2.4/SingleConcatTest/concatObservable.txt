@SuppressWarnings("unchecked") @Test public void concatObservable(){
  for (int i=1; i < 100; i++) {
    Single<Integer>[] array=new Single[i];
    Arrays.fill(array,Single.just(1));
    Single.concat(Observable.fromArray(array)).test().assertSubscribed().assertValueCount(i).assertNoErrors().assertComplete();
  }
}
