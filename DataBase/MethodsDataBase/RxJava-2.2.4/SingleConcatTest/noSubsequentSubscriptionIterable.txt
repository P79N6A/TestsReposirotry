@SuppressWarnings("unchecked") @Test public void noSubsequentSubscriptionIterable(){
  final int[] calls={0};
  Single<Integer> source=Single.create(new SingleOnSubscribe<Integer>(){
    @Override public void subscribe(    SingleEmitter<Integer> s) throws Exception {
      calls[0]++;
      s.onSuccess(1);
    }
  }
);
  Single.concat(Arrays.asList(source,source)).firstElement().test().assertResult(1);
  assertEquals(1,calls[0]);
}
