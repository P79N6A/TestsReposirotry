@Test public void onSubscribeAfterDone(){
  AsyncSubject<Object> p=AsyncSubject.create();
  Disposable bs=Disposables.empty();
  p.onSubscribe(bs);
  assertFalse(bs.isDisposed());
  p.onComplete();
  bs=Disposables.empty();
  p.onSubscribe(bs);
  assertTrue(bs.isDisposed());
  p.test().assertResult();
}
