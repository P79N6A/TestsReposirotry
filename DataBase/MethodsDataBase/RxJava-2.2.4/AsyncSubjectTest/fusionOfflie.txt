@Test public void fusionOfflie(){
  AsyncSubject<Integer> ap=new AsyncSubject<Integer>();
  ap.onNext(1);
  ap.onComplete();
  TestObserver<Integer> to=ObserverFusion.newTest(QueueFuseable.ANY);
  ap.subscribe(to);
  to.assertOf(ObserverFusion.<Integer>assertFuseable()).assertOf(ObserverFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC)).assertResult(1);
}
