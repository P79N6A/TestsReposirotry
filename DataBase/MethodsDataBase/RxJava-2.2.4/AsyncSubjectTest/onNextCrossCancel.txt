@Test public void onNextCrossCancel(){
  AsyncSubject<Object> p=AsyncSubject.create();
  final TestObserver<Object> to2=new TestObserver<Object>();
  TestObserver<Object> to1=new TestObserver<Object>(){
    @Override public void onNext(    Object t){
      to2.cancel();
      super.onNext(t);
    }
  }
;
  p.subscribe(to1);
  p.subscribe(to2);
  p.onNext(1);
  p.onComplete();
  to1.assertResult(1);
  to2.assertEmpty();
}
