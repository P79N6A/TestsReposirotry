@Test public void nullKeyTakeInner(){
  Flowable.just(1).groupBy(new Function<Integer,Object>(){
    @Override public Object apply(    Integer v) throws Exception {
      return null;
    }
  }
).flatMap(new Function<GroupedFlowable<Object,Integer>,Publisher<Integer>>(){
    @Override public Publisher<Integer> apply(    GroupedFlowable<Object,Integer> g) throws Exception {
      return g.take(1);
    }
  }
).test().assertResult(1);
}
