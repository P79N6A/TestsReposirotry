@Test public void frontOverflow(){
  new Flowable<Integer>(){
    @Override protected void subscribeActual(    Subscriber<? super Integer> s){
      s.onSubscribe(new BooleanSubscription());
      for (int i=0; i < 9; i++) {
        s.onNext(i);
      }
    }
  }
.publish(Functions.<Flowable<Integer>>identity(),8).test(0).assertFailure(MissingBackpressureException.class);
}
