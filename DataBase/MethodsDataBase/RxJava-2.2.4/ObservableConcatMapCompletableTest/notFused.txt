@Test public void notFused() throws Exception {
  UnicastSubject<Integer> us=UnicastSubject.create();
  TestObserver<Void> to=us.hide().concatMapCompletable(completableComplete(),2).test();
  us.onNext(1);
  us.onNext(2);
  us.onComplete();
  to.assertComplete();
  to.assertValueCount(0);
  to.assertNoErrors();
}
