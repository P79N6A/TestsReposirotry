@Test public void cancel(){
  PublishSubject<Integer> ps=PublishSubject.create();
  final MaybeSubject<Integer> ms=MaybeSubject.create();
  TestObserver<Integer> to=ps.switchMapMaybeDelayError(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      return ms;
    }
  }
).test();
  to.assertEmpty();
  ps.onNext(1);
  to.assertEmpty();
  assertTrue(ps.hasObservers());
  assertTrue(ms.hasObservers());
  to.cancel();
  assertFalse(ps.hasObservers());
  assertFalse(ms.hasObservers());
}
