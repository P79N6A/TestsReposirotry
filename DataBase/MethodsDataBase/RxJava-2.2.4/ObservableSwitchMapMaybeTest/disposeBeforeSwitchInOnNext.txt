@Test public void disposeBeforeSwitchInOnNext(){
  final TestObserver<Integer> to=new TestObserver<Integer>();
  Observable.just(1).switchMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      to.cancel();
      return Maybe.just(1);
    }
  }
).subscribe(to);
  to.assertEmpty();
}
