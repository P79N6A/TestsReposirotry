@Test public void testErrorInPredicate(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
  ts.onError(new RuntimeException());
  try {
    ts.assertError(new Predicate<Throwable>(){
      @Override public boolean test(      Throwable throwable) throws Exception {
        throw new TestException();
      }
    }
);
  }
 catch (  TestException ex) {
    return;
  }
  fail("Error in predicate but not thrown!");
}
