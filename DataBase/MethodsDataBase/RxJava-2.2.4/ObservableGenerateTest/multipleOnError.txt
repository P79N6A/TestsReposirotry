@Test public void multipleOnError(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    Observable.generate(new Consumer<Emitter<Object>>(){
      @Override public void accept(      Emitter<Object> e) throws Exception {
        e.onError(new TestException("First"));
        e.onError(new TestException("Second"));
      }
    }
).test().assertFailure(TestException.class);
    TestHelper.assertUndeliverable(errors,0,TestException.class,"Second");
  }
  finally {
    RxJavaPlugins.reset();
  }
}
