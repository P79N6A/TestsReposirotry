@Test public void nullSourceSuccessRace(){
  for (int i=0; i < TestHelper.RACE_DEFAULT_LOOPS; i++) {
    List<Throwable> errors=TestHelper.trackPluginErrors();
    try {
      final Subject<Integer> ps=ReplaySubject.create();
      ps.onNext(1);
      @SuppressWarnings("unchecked") final Single<Integer> source=Single.ambArray(ps.singleOrError(),Single.<Integer>never(),Single.<Integer>never(),null);
      Runnable r1=new Runnable(){
        @Override public void run(){
          source.test();
        }
      }
;
      Runnable r2=new Runnable(){
        @Override public void run(){
          ps.onComplete();
        }
      }
;
      TestHelper.race(r1,r2);
      if (!errors.isEmpty()) {
        TestHelper.assertError(errors,0,NullPointerException.class);
      }
    }
  finally {
      RxJavaPlugins.reset();
    }
  }
}
