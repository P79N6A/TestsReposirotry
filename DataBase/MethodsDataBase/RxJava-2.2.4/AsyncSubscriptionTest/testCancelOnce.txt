@Test public void testCancelOnce(){
  Disposable r=mock(Disposable.class);
  AsyncSubscription as=new AsyncSubscription(r);
  Subscription s=mock(Subscription.class);
  as.setSubscription(s);
  as.cancel();
  as.cancel();
  as.cancel();
  verify(s,never()).request(anyLong());
  verify(s).cancel();
  verify(r).dispose();
}
