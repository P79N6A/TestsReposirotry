@Test public void deleteRace(){
  for (int i=0; i < TestHelper.RACE_DEFAULT_LOOPS; i++) {

    cd.add(d1);
    Runnable run=new Runnable(){
      @Override public void run(){
        cd.delete(d1);
      }
    }
;
    TestHelper.race(run,run);
  }
}
