@Test public void testReduceObservable(){
  Observable<Integer> o=Observable.just(1,2,3,4);
  o.reduce(new BiFunction<Integer,Integer,Integer>(){
    @Override public Integer apply(    Integer t1,    Integer t2){
      return t1 + t2;
    }
  }
).toObservable().subscribe(w);
  verify(w,times(1)).onNext(anyInt());
  verify(w).onNext(10);
  verify(w,never()).onError(any(Throwable.class));
  verify(w).onComplete();
}
