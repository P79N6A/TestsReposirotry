@Test public void combineLatestObject(){
  @SuppressWarnings("unchecked") final List<Observable<Integer>> observables=Arrays.asList(Observable.just(1,2,3),Observable.just(1,2,3));
  Observable.combineLatest(observables,new Function<Object[],Object>(){
    @Override public Object apply(    final Object[] o) throws Exception {
      int sum=1;
      for (      Object i : o) {
        sum*=(Integer)i;
      }
      return sum;
    }
  }
).test().assertResult(3,6,9);
}
