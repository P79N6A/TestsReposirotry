@Ignore("Throwing is not allowed from the unsafeCreate?!") @Test public void testOnSubscribeFails(){
  Observer<String> observer=TestHelper.mockObserver();
  final RuntimeException re=new RuntimeException("bad impl");
  Observable<String> o=Observable.unsafeCreate(new ObservableSource<String>(){
    @Override public void subscribe(    Observer<? super String> observer){
      throw re;
    }
  }
);
  o.subscribe(observer);
  verify(observer,times(0)).onNext(anyString());
  verify(observer,times(0)).onComplete();
  verify(observer,times(1)).onError(re);
}
