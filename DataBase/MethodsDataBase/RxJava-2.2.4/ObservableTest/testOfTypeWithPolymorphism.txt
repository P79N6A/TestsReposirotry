@Test public void testOfTypeWithPolymorphism(){
  ArrayList<Integer> l1=new ArrayList<Integer>();
  l1.add(1);
  LinkedList<Integer> l2=new LinkedList<Integer>();
  l2.add(2);
  @SuppressWarnings("rawtypes") Observable<List> o=Observable.<Object>just(l1,l2,"123").ofType(List.class);
  Observer<Object> observer=TestHelper.mockObserver();
  o.subscribe(observer);
  verify(observer,times(1)).onNext(l1);
  verify(observer,times(1)).onNext(l2);
  verify(observer,never()).onNext("123");
  verify(observer,never()).onError(any(Throwable.class));
  verify(observer,times(1)).onComplete();
}
