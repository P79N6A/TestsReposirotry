@Test public void combinerMapperFunctionReturnsNull(){
  Observable.just(1).flatMap(new Function<Integer,Observable<Object>>(){
    @Override public Observable<Object> apply(    Integer v) throws Exception {
      return null;
    }
  }
,new BiFunction<Integer,Object,Object>(){
    @Override public Object apply(    Integer v,    Object w) throws Exception {
      return v;
    }
  }
).test().assertFailureAndMessage(NullPointerException.class,"The mapper returned a null ObservableSource");
}
