@Test public void testNoUnsubscribeDownstream(){
  Flowable<Integer> source=Flowable.range(1,1000).takeWhile(new Predicate<Integer>(){
    @Override public boolean test(    Integer t1){
      return t1 < 2;
    }
  }
);
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
  source.subscribe(ts);
  ts.assertNoErrors();
  ts.assertValue(1);
  Assert.assertFalse("Unsubscribed!",ts.isCancelled());
}
