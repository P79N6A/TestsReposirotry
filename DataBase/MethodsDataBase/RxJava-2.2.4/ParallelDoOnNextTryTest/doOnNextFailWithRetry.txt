@Test public void doOnNextFailWithRetry(){
  Flowable.range(0,2).parallel(1).doOnNext(new Consumer<Integer>(){
    int count;
    @Override public void accept(    Integer v) throws Exception {
      if (count++ == 1) {
        return;
      }
      if (1 / v < 0) {
        System.out.println("Should not happen!");
      }
    }
  }
,ParallelFailureHandling.RETRY).sequential().test().assertResult(0,1);
}
