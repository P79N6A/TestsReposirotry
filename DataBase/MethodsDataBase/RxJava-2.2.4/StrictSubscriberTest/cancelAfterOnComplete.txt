@Test public void cancelAfterOnComplete(){
  final List<Object> list=new ArrayList<Object>();
  Subscriber<Object> sub=new Subscriber<Object>(){
    Subscription upstream;
    @Override public void onSubscribe(    Subscription s){
      this.upstream=s;
    }
    @Override public void onNext(    Object t){
      list.add(t);
    }
    @Override public void onError(    Throwable t){
      upstream.cancel();
      list.add(t);
    }
    @Override public void onComplete(){
      upstream.cancel();
      list.add("Done");
    }
  }
;
  new Flowable<Object>(){
    @Override protected void subscribeActual(    Subscriber<? super Object> s){
      BooleanSubscription b=new BooleanSubscription();
      s.onSubscribe(b);
      s.onComplete();
      list.add(b.isCancelled());
    }
  }
.subscribe(sub);
  assertEquals(Arrays.<Object>asList("Done",false),list);
}
