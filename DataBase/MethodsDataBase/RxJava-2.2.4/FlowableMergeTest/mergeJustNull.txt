@Test @Ignore("Nulls are not allowed with RS") public void mergeJustNull(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(0);
  Flowable.range(1,2).flatMap(new Function<Integer,Flowable<Integer>>(){
    @Override public Flowable<Integer> apply(    Integer t){
      return Flowable.just(null);
    }
  }
).subscribe(ts);
  ts.request(2);
  ts.assertValues(null,null);
  ts.assertNoErrors();
  ts.assertComplete();
}
