@Test @Ignore("Subscribe should not throw") public void testThrownErrorHandling(){
  TestSubscriber<String> ts=new TestSubscriber<String>();
  Flowable<String> f1=Flowable.unsafeCreate(new Publisher<String>(){
    @Override public void subscribe(    Subscriber<? super String> s){
      throw new RuntimeException("fail");
    }
  }
);
  Flowable.merge(f1,f1).subscribe(ts);
  ts.awaitTerminalEvent(1000,TimeUnit.MILLISECONDS);
  ts.assertTerminated();
  System.out.println("Error: " + ts.errors());
}
