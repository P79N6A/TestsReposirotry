@Test public void testCurrentStateMethodsEmpty(){
  ReplaySubject<Object> as=ReplaySubject.create();
  assertFalse(as.hasThrowable());
  assertFalse(as.hasComplete());
  assertNull(as.getThrowable());
  as.onComplete();
  assertFalse(as.hasThrowable());
  assertTrue(as.hasComplete());
  assertNull(as.getThrowable());
}
