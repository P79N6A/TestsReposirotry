@Test public void fused(){
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  Flowable.just(1,2,2,3,3,4,5).distinctUntilChanged(new BiPredicate<Integer,Integer>(){
    @Override public boolean test(    Integer a,    Integer b) throws Exception {
      return a.equals(b);
    }
  }
).subscribe(ts);
  ts.assertOf(SubscriberFusion.<Integer>assertFuseable()).assertOf(SubscriberFusion.<Integer>assertFusionMode(QueueFuseable.SYNC)).assertResult(1,2,3,4,5);
}
