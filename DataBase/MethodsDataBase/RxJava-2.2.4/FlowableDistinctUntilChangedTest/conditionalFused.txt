@Test public void conditionalFused(){
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  Flowable.just(1,2,1,3,3,4,3,5,5).distinctUntilChanged().filter(new Predicate<Integer>(){
    @Override public boolean test(    Integer v) throws Exception {
      return v % 2 == 0;
    }
  }
).subscribe(ts);
  SubscriberFusion.assertFusion(ts,QueueFuseable.SYNC).assertResult(2,4);
}
