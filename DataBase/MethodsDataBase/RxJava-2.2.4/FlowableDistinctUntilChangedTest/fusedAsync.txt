@Test public void fusedAsync(){
  TestSubscriber<Integer> ts=SubscriberFusion.newTest(QueueFuseable.ANY);
  UnicastProcessor<Integer> up=UnicastProcessor.create();
  up.distinctUntilChanged(new BiPredicate<Integer,Integer>(){
    @Override public boolean test(    Integer a,    Integer b) throws Exception {
      return a.equals(b);
    }
  }
).subscribe(ts);
  TestHelper.emit(up,1,2,2,3,3,4,5);
  ts.assertOf(SubscriberFusion.<Integer>assertFuseable()).assertOf(SubscriberFusion.<Integer>assertFusionMode(QueueFuseable.ASYNC)).assertResult(1,2,3,4,5);
}
