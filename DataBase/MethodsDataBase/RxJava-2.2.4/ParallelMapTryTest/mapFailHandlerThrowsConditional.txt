@SuppressWarnings("unchecked") @Test public void mapFailHandlerThrowsConditional(){
  TestSubscriber<Integer> ts=Flowable.range(0,2).parallel(1).map(new Function<Integer,Integer>(){
    @Override public Integer apply(    Integer v) throws Exception {
      return 1 / v;
    }
  }
,new BiFunction<Long,Throwable,ParallelFailureHandling>(){
    @Override public ParallelFailureHandling apply(    Long n,    Throwable e) throws Exception {
      throw new TestException();
    }
  }
).filter(Functions.alwaysTrue()).sequential().test().assertFailure(CompositeException.class);
  TestHelper.assertCompositeExceptions(ts,ArithmeticException.class,TestException.class);
}
