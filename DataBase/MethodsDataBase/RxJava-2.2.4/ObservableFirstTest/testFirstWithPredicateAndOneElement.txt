@Test public void testFirstWithPredicateAndOneElement(){
  Maybe<Integer> o=Observable.just(1,2).filter(new Predicate<Integer>(){
    @Override public boolean test(    Integer t1){
      return t1 % 2 == 0;
    }
  }
).firstElement();
  o.subscribe(wm);
  InOrder inOrder=inOrder(wm);
  inOrder.verify(wm,times(1)).onSuccess(2);
  inOrder.verifyNoMoreInteractions();
}
