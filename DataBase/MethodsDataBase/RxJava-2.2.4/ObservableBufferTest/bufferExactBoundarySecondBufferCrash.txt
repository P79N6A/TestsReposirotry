@SuppressWarnings("unchecked") @Test public void bufferExactBoundarySecondBufferCrash(){
  PublishSubject<Integer> ps=PublishSubject.create();
  PublishSubject<Integer> b=PublishSubject.create();
  TestObserver<List<Integer>> to=ps.buffer(b,new Callable<List<Integer>>(){
    int calls;
    @Override public List<Integer> call() throws Exception {
      if (++calls == 2) {
        throw new TestException();
      }
      return new ArrayList<Integer>();
    }
  }
).test();
  b.onNext(1);
  to.assertFailure(TestException.class);
}
