@Test public void mapperCancels(){
  final TestObserver<Void> to=new TestObserver<Void>();
  Observable.range(1,5).switchMapCompletable(new Function<Integer,CompletableSource>(){
    @Override public CompletableSource apply(    Integer f) throws Exception {
      to.cancel();
      return Completable.complete();
    }
  }
).subscribe(to);
  to.assertEmpty();
}
