@Test public void testFactoryFailureResultsInErrorEmission(){
  final RuntimeException e=new RuntimeException();
  Flowable.just(1).collect(new Callable<List<Integer>>(){
    @Override public List<Integer> call() throws Exception {
      throw e;
    }
  }
,new BiConsumer<List<Integer>,Integer>(){
    @Override public void accept(    List<Integer> list,    Integer t){
      list.add(t);
    }
  }
).test().assertNoValues().assertError(e).assertNotComplete();
}
