@Test public void onErrorNull(){
  TestObserver<Integer> to=Maybe.<Integer>error(new TestException()).flatMap(Functions.justFunction(Maybe.just(1)),Functions.justFunction((Maybe<Integer>)null),Functions.justCallable(Maybe.just(1))).test().assertFailure(CompositeException.class);
  List<Throwable> ce=TestHelper.compositeList(to.errors().get(0));
  TestHelper.assertError(ce,0,TestException.class);
  TestHelper.assertError(ce,1,NullPointerException.class);
}
