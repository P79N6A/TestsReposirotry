@SuppressWarnings({"rawtypes","unchecked"}) @Test public void repeatWhenDefaultScheduler(){
  TestSubscriber<Integer> ts=TestSubscriber.create();
  Flowable.just(1).repeatWhen((Function)new Function<Flowable,Flowable>(){
    @Override public Flowable apply(    Flowable f){
      return f.take(2);
    }
  }
).subscribe(ts);
  ts.assertValues(1,1);
  ts.assertNoErrors();
  ts.assertComplete();
}
