@Test public void iterableMapperFunctionReturnsNull(){
  Flowable.just(1).flatMapIterable(new Function<Integer,Iterable<Object>>(){
    @Override public Iterable<Object> apply(    Integer v) throws Exception {
      return null;
    }
  }
,new BiFunction<Integer,Object,Object>(){
    @Override public Object apply(    Integer v,    Object w) throws Exception {
      return v;
    }
  }
).test().assertFailureAndMessage(NullPointerException.class,"The mapper returned a null Iterable");
}
