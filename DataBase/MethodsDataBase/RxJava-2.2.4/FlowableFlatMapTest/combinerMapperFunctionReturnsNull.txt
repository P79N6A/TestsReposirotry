@Test public void combinerMapperFunctionReturnsNull(){
  Flowable.just(1).flatMap(new Function<Integer,Publisher<Object>>(){
    @Override public Publisher<Object> apply(    Integer v) throws Exception {
      return null;
    }
  }
,new BiFunction<Integer,Object,Object>(){
    @Override public Object apply(    Integer v,    Object w) throws Exception {
      return v;
    }
  }
).test().assertFailureAndMessage(NullPointerException.class,"The mapper returned a null Publisher");
}
