@Test public void flatMapBiMapper(){
  Flowable.just(1).flatMap(new Function<Integer,Publisher<Integer>>(){
    @Override public Publisher<Integer> apply(    Integer v) throws Exception {
      return Flowable.just(v * 10);
    }
  }
,new BiFunction<Integer,Integer,Integer>(){
    @Override public Integer apply(    Integer a,    Integer b) throws Exception {
      return a + b;
    }
  }
,true).test().assertResult(11);
}
