@Test public void backpressure(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(0);
  FutureTask<Integer> f=new FutureTask<Integer>(new Runnable(){
    @Override public void run(){
    }
  }
,1);
  f.run();
  Flowable.fromFuture(f).subscribe(ts);
  ts.assertNoValues();
  ts.request(1);
  ts.assertValue(1);
  ts.assertNoErrors();
  ts.assertComplete();
}
