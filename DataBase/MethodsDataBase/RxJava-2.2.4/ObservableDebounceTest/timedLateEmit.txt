@Test public void timedLateEmit(){
  TestObserver<Integer> to=new TestObserver<Integer>();
  DebounceTimedObserver<Integer> sub=new DebounceTimedObserver<Integer>(to,1,TimeUnit.SECONDS,new TestScheduler().createWorker());
  sub.onSubscribe(Disposables.empty());
  DebounceEmitter<Integer> de=new DebounceEmitter<Integer>(1,50,sub);
  de.run();
  de.run();
  to.assertEmpty();
}
