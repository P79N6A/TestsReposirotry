@Test public void disposeAfterRun(){
  final ScheduledRunnable run=new ScheduledRunnable(Functions.EMPTY_RUNNABLE,null);
  run.run();
  assertEquals(ScheduledRunnable.DONE,run.get(ScheduledRunnable.FUTURE_INDEX));
  run.dispose();
  assertEquals(ScheduledRunnable.DONE,run.get(ScheduledRunnable.FUTURE_INDEX));
}
