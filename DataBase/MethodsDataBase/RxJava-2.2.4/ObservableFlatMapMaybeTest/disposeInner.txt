@Test public void disposeInner(){
  final TestObserver<Object> to=new TestObserver<Object>();
  Observable.just(1).flatMapMaybe(new Function<Integer,MaybeSource<Object>>(){
    @Override public MaybeSource<Object> apply(    Integer v) throws Exception {
      return new Maybe<Object>(){
        @Override protected void subscribeActual(        MaybeObserver<? super Object> observer){
          observer.onSubscribe(Disposables.empty());
          assertFalse(((Disposable)observer).isDisposed());
          to.dispose();
          assertTrue(((Disposable)observer).isDisposed());
        }
      }
;
    }
  }
).subscribe(to);
  to.assertEmpty();
}
