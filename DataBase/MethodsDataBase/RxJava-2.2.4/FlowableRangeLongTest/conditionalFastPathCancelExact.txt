@Test public void conditionalFastPathCancelExact(){
  TestSubscriber<Long> ts=new TestSubscriber<Long>(){
    @Override public void onNext(    Long t){
      super.onNext(t);
      if (t == 5L) {
        cancel();
        onComplete();
      }
    }
  }
;
  Flowable.rangeLong(1L,5L).filter(new Predicate<Long>(){
    @Override public boolean test(    Long v) throws Exception {
      return v % 2 == 0;
    }
  }
).subscribe(ts);
  ts.assertResult(2L,4L);
}
