@Test public void conditionalFastPathCancel(){
  TestSubscriber<Long> ts=new TestSubscriber<Long>(){
    @Override public void onNext(    Long t){
      super.onNext(t);
      cancel();
      onComplete();
    }
  }
;
  Flowable.rangeLong(1L,5L).filter(Functions.alwaysTrue()).subscribe(ts);
  ts.assertResult(1L);
}
