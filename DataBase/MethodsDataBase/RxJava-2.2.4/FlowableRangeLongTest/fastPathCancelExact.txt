@Test public void fastPathCancelExact(){
  TestSubscriber<Long> ts=new TestSubscriber<Long>(){
    @Override public void onNext(    Long t){
      super.onNext(t);
      if (t == 5L) {
        cancel();
        onComplete();
      }
    }
  }
;
  Flowable.rangeLong(1L,5L).subscribe(ts);
  ts.assertResult(1L,2L,3L,4L,5L);
}
