@Test public void testUnsubscribeJustBeforeCompletionNotificationShouldPreventThatNotificationArriving(){
  TestSubscriber<Notification<Integer>> ts=new TestSubscriber<Notification<Integer>>(0L);
  Flowable.<Integer>empty().materialize().subscribe(ts);
  ts.assertNoValues();
  ts.dispose();
  ts.request(1);
  ts.assertNoValues();
}
