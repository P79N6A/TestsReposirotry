@Test public void filterFailWithRetry(){
  Flowable.range(0,2).parallel(1).filter(new Predicate<Integer>(){
    int count;
    @Override public boolean test(    Integer v) throws Exception {
      if (count++ == 1) {
        return true;
      }
      return 1 / v > 0;
    }
  }
,ParallelFailureHandling.RETRY).sequential().test().assertResult(0,1);
}
