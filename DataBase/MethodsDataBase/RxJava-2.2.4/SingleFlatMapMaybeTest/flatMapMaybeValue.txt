@Test public void flatMapMaybeValue(){
  Single.just(1).flatMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    final Integer integer) throws Exception {
      if (integer == 1) {
        return Maybe.just(2);
      }
      return Maybe.just(1);
    }
  }
).test().assertResult(2);
}
