@Test public void flatMapMaybeValueDifferentType(){
  Single.just(1).flatMapMaybe(new Function<Integer,MaybeSource<String>>(){
    @Override public MaybeSource<String> apply(    final Integer integer) throws Exception {
      if (integer == 1) {
        return Maybe.just("2");
      }
      return Maybe.just("1");
    }
  }
).test().assertResult("2");
}
