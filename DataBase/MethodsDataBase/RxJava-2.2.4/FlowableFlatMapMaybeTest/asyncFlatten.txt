@Test public void asyncFlatten(){
  Flowable.range(1,1000).flatMapMaybe(new Function<Integer,MaybeSource<Integer>>(){
    @Override public MaybeSource<Integer> apply(    Integer v) throws Exception {
      return Maybe.just(1).subscribeOn(Schedulers.computation());
    }
  }
).take(500).test().awaitDone(5,TimeUnit.SECONDS).assertSubscribed().assertValueCount(500).assertNoErrors().assertComplete();
}
