@Test public void mergeErrors(){
  List<Throwable> errors=TestHelper.trackPluginErrors();
  try {
    Observable<Integer> source1=Observable.error(new TestException("First"));
    Observable<Integer> source2=Observable.error(new TestException("Second"));
    Observable.merge(source1,source2).test().assertFailureAndMessage(TestException.class,"First");
    assertTrue(errors.toString(),errors.isEmpty());
  }
  finally {
    RxJavaPlugins.reset();
  }
}
