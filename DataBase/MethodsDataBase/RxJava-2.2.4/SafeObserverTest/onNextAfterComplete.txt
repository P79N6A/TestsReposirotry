@Test public void onNextAfterComplete(){
  TestObserver<Integer> to=new TestObserver<Integer>();
  SafeObserver<Integer> so=new SafeObserver<Integer>(to);
  Disposable d=Disposables.empty();
  so.onSubscribe(d);
  so.onComplete();
  so.onNext(1);
  so.onError(new TestException());
  so.onComplete();
  to.assertResult();
}
