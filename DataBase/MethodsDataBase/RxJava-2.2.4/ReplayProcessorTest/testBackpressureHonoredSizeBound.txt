@Test public void testBackpressureHonoredSizeBound(){
  ReplayProcessor<Integer> rs=ReplayProcessor.createWithSize(100);
  rs.onNext(1);
  rs.onNext(2);
  rs.onNext(3);
  rs.onComplete();
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(0L);
  rs.subscribe(ts);
  ts.request(1);
  ts.assertValue(1);
  ts.assertNotComplete();
  ts.assertNoErrors();
  ts.request(1);
  ts.assertValues(1,2);
  ts.assertNotComplete();
  ts.assertNoErrors();
  ts.request(1);
  ts.assertValues(1,2,3);
  ts.assertComplete();
  ts.assertNoErrors();
}
