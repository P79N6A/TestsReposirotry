@Test public void testSuccessOperatesOnSuppliedScheduler() throws Exception {
  @SuppressWarnings("unchecked") Future<Object> future=mock(Future.class);
  Object value=new Object();
  when(future.get()).thenReturn(value);
  Observer<Object> o=TestHelper.mockObserver();
  TestScheduler scheduler=new TestScheduler();
  TestObserver<Object> to=new TestObserver<Object>(o);
  Observable.fromFuture(future,scheduler).subscribe(to);
  verify(o,never()).onNext(value);
  scheduler.triggerActions();
  verify(o,times(1)).onNext(value);
}
