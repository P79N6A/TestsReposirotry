@Test public void conditionalCancel1(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(2L){
    @Override public void onNext(    Integer t){
      super.onNext(t);
      if (t == 1) {
        cancel();
        onComplete();
      }
    }
  }
;
  Flowable.range(1,2).filter(Functions.alwaysTrue()).subscribe(ts);
  ts.assertResult(1);
}
