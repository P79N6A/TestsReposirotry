@Test public void errorLatest(){
  Flowable.create(source,BackpressureStrategy.LATEST).subscribe(ts);
  source.onNext(1);
  source.onNext(2);
  source.onError(new TestException());
  ts.assertNoValues();
  ts.request(1);
  ts.assertValues(2);
  ts.assertError(TestException.class);
  ts.assertNotComplete();
}
