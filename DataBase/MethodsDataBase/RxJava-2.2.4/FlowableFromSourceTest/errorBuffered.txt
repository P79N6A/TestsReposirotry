@Test public void errorBuffered(){
  Flowable.create(source,BackpressureStrategy.BUFFER).subscribe(ts);
  source.onNext(1);
  source.onNext(2);
  source.onError(new TestException());
  ts.request(1);
  ts.assertValue(1);
  ts.request(1);
  ts.assertValues(1,2);
  ts.assertError(TestException.class);
  ts.assertNotComplete();
}
