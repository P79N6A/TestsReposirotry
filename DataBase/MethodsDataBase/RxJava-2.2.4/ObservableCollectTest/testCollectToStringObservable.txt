@Test public void testCollectToStringObservable(){
  String value=Observable.just(1,2,3).collect(new Callable<StringBuilder>(){
    @Override public StringBuilder call(){
      return new StringBuilder();
    }
  }
,new BiConsumer<StringBuilder,Integer>(){
    @Override public void accept(    StringBuilder sb,    Integer v){
      if (sb.length() > 0) {
        sb.append("-");
      }
      sb.append(v);
    }
  }
).toObservable().blockingLast().toString();
  assertEquals("1-2-3",value);
}
