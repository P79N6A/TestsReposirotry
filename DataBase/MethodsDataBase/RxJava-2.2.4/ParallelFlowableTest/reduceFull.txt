@Test public void reduceFull(){
  for (int i=1; i <= Runtime.getRuntime().availableProcessors() * 2; i++) {
    TestSubscriber<Integer> ts=new TestSubscriber<Integer>();
    Flowable.range(1,10).parallel(i).reduce(new BiFunction<Integer,Integer,Integer>(){
      @Override public Integer apply(      Integer a,      Integer b) throws Exception {
        return a + b;
      }
    }
).subscribe(ts);
    ts.assertResult(55);
  }
}
