@Test public void parallelismAndPrefetch(){
  for (int parallelism=1; parallelism <= 8; parallelism++) {
    for (int prefetch=1; prefetch <= 1024; prefetch*=2) {
      Flowable.range(1,1024 * 1024).parallel(parallelism,prefetch).map(Functions.<Integer>identity()).sequential().test().assertSubscribed().assertValueCount(1024 * 1024).assertNoErrors().assertComplete();
    }
  }
}
