@Test public void unsubscribeSubscribeRace(){
  for (int i=0; i < 1000; i++) {
    final Flowable<Integer> source=Flowable.range(1,5).replay().refCount(1);
    final TestSubscriber<Integer> ts1=source.test(0);
    final TestSubscriber<Integer> ts2=new TestSubscriber<Integer>(0);
    Runnable r1=new Runnable(){
      @Override public void run(){
        ts1.cancel();
      }
    }
;
    Runnable r2=new Runnable(){
      @Override public void run(){
        source.subscribe(ts2);
      }
    }
;
    TestHelper.race(r1,r2,Schedulers.single());
    ts2.requestMore(6).withTag("Round: " + i).assertResult(1,2,3,4,5);
  }
}
