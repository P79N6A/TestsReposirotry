@Test public void backpressured() throws Exception {
  o=new Object();
  WeakReference<Object> wr=new WeakReference<Object>(o);
  TestSubscriber<Object> ts=new TestSubscriber<Object>(0L);
  Flowable.just(o).count().toFlowable().onTerminateDetach().subscribe(ts);
  ts.assertNoValues();
  ts.request(1);
  ts.assertValue(1L);
  ts.assertComplete();
  ts.assertNoErrors();
  o=null;
  System.gc();
  Thread.sleep(200);
  Assert.assertNull("Object retained!",wr.get());
}
