@Test public void workerNotDisposedPrematurelyAsyncInNormalOut(){
  DisposeTrackingScheduler s=new DisposeTrackingScheduler();
  UnicastSubject<Integer> up=UnicastSubject.create();
  up.onNext(1);
  up.onComplete();
  Observable.concat(up.observeOn(s),Observable.just(2)).test().assertResult(1,2);
  assertEquals(1,s.disposedCount.get());
}
