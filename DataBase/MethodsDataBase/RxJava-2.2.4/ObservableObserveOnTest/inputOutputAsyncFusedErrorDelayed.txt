@Test public void inputOutputAsyncFusedErrorDelayed(){
  TestObserver<Integer> to=ObserverFusion.newTest(QueueFuseable.ANY);
  UnicastSubject<Integer> us=UnicastSubject.create();
  us.observeOn(Schedulers.single(),true).subscribe(to);
  us.onError(new TestException());
  to.awaitDone(5,TimeUnit.SECONDS).assertFailure(TestException.class);
  ObserverFusion.assertFusion(to,QueueFuseable.ASYNC).awaitDone(5,TimeUnit.SECONDS).assertFailure(TestException.class);
}
