@Test public void cancel(){
  Observable.range(1,5).concatWith(Observable.<Integer>never()).concatMapSingle(new Function<Integer,SingleSource<Integer>>(){
    @Override public SingleSource<Integer> apply(    Integer v) throws Exception {
      return Single.just(v);
    }
  }
).test().assertValues(1,2,3,4,5).assertNoErrors().assertNotComplete().cancel();
}
