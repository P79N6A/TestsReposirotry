@Test public void checkUnboundedInnerQueue(){

  @SuppressWarnings("unchecked") TestObserver<Integer> to=Observable.fromArray(ss,Single.just(2),Single.just(3),Single.just(4)).concatMapSingle(Functions.<Single<Integer>>identity(),2).test();

  to.assertResult(1,2,3,4);
}
