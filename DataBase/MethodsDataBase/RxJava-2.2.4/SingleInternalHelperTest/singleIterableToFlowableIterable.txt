@Test public void singleIterableToFlowableIterable(){
  Iterable<? extends Flowable<Integer>> it=SingleInternalHelper.iterableToFlowable(Collections.singletonList(Single.just(1)));
  Iterator<? extends Flowable<Integer>> iter=it.iterator();
  if (iter.hasNext()) {
    iter.next().test().assertResult(1);
    if (iter.hasNext()) {
      fail("Iterator reports an additional element");
    }
  }
 else {
    fail("Iterator was empty");
  }
}
