@Test public void fusionWithConcatMap(){
  TestObserver<Integer> to=new TestObserver<Integer>();
  Observable.fromIterable(Arrays.asList(1,2,3,4)).concatMap(new Function<Integer,ObservableSource<Integer>>(){
    @Override public ObservableSource<Integer> apply(    Integer v){
      return Observable.range(v,2);
    }
  }
).subscribe(to);
  to.assertValues(1,2,2,3,3,4,4,5);
  to.assertNoErrors();
  to.assertComplete();
}
