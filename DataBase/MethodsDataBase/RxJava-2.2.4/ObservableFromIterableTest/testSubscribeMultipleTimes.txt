@Test public void testSubscribeMultipleTimes(){
  Observable<Integer> o=Observable.fromIterable(Arrays.asList(1,2,3));
  for (int i=0; i < 10; i++) {
    TestObserver<Integer> to=new TestObserver<Integer>();
    o.subscribe(to);
    to.assertValues(1,2,3);
    to.assertNoErrors();
    to.assertComplete();
  }
}
