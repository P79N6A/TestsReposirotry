@Test public void unbounded(){
  SubscriptionArbiter sa=new SubscriptionArbiter(true);
  sa.request(Long.MAX_VALUE);
  assertEquals(Long.MAX_VALUE,sa.requested);
  assertTrue(sa.isUnbounded());
  sa.unbounded=false;
  sa.request(Long.MAX_VALUE);
  assertEquals(Long.MAX_VALUE,sa.requested);
  sa.produced(1);
  assertEquals(Long.MAX_VALUE,sa.requested);
  sa.unbounded=false;
  sa.produced(Long.MAX_VALUE);
  assertEquals(Long.MAX_VALUE,sa.requested);
}
