@Test public void drainExactRequestCancel(){
  final PublishProcessor<Integer> pp=PublishProcessor.create();
  final SingleSubject<Integer> cs=SingleSubject.create();
  TestSubscriber<Integer> ts=pp.mergeWith(cs).limit(2).subscribeWith(new TestSubscriber<Integer>(2){
    @Override public void onNext(    Integer t){
      super.onNext(t);
      if (t == 1) {
        cs.onSuccess(2);
      }
    }
  }
);
  pp.onNext(1);
  pp.onComplete();
  ts.request(2);
  ts.assertResult(1,2);
}
