@Test public void cancelOnBackpressureBoundary(){
  TestSubscriber<Integer> ts=new TestSubscriber<Integer>(1L){
    @Override public void onNext(    Integer t){
      super.onNext(t);
      cancel();
      onComplete();
    }
  }
;
  Flowable.zip(Flowable.range(1,2),Flowable.range(3,2),new BiFunction<Integer,Integer,Integer>(){
    @Override public Integer apply(    Integer a,    Integer b) throws Exception {
      return a + b;
    }
  }
).subscribe(ts);
  ts.assertResult(4);
}
