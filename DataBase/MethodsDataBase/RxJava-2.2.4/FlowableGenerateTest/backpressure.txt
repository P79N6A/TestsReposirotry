@Test public void backpressure(){
  Flowable.generate(new Callable<Object>(){
    @Override public Object call() throws Exception {
      return 1;
    }
  }
,new BiConsumer<Object,Emitter<Object>>(){
    @Override public void accept(    Object s,    Emitter<Object> e) throws Exception {
      e.onNext(1);
    }
  }
,Functions.emptyConsumer()).rebatchRequests(1).test(5).assertSubscribed().assertValues(1,1,1,1,1).assertNoErrors().assertNotComplete();
}
