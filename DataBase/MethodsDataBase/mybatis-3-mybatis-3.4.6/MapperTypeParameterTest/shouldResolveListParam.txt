@Test public void shouldResolveListParam(){
  SqlSession sqlSession=sqlSessionFactory.openSession();

    PersonMapper mapper=sqlSession.getMapper(PersonMapper.class);
    Person person1=new Person("James");

    sqlSession.close();
  }
