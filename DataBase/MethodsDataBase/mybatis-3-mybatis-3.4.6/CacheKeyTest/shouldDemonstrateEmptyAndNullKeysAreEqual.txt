@Test public void shouldDemonstrateEmptyAndNullKeysAreEqual(){
  CacheKey key1=new CacheKey();
  CacheKey key2=new CacheKey();
  assertEquals(key1,key2);
  assertEquals(key2,key1);
  key1.update(null);
  key2.update(null);
  assertEquals(key1,key2);
  assertEquals(key2,key1);
  key1.update(null);
  key2.update(null);
  assertEquals(key1,key2);
  assertEquals(key2,key1);
}
