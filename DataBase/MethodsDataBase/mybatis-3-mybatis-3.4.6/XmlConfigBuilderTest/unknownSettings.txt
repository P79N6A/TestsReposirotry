@Test public void unknownSettings(){
  final String MAPPER_CONFIG="<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n" + "<!DOCTYPE configuration PUBLIC \"-//mybatis.org//DTD Config 3.0//EN\" \"http://mybatis.org/dtd/mybatis-3-config.dtd\">\n" + "<configuration>\n"+ "  <settings>\n"+ "    <setting name=\"foo\" value=\"bar\"/>\n"+ "  </settings>\n"+ "</configuration>\n";
  XMLConfigBuilder builder=new XMLConfigBuilder(new StringReader(MAPPER_CONFIG));
  when(builder).parse();
  then(caughtException()).isInstanceOf(BuilderException.class).hasMessageContaining("The setting foo is not known.  Make sure you spelled it correctly (case sensitive).");
}
