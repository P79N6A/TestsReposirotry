@Test public void testTrimToNothing() throws Exception {
  byte[] b1=mPool.get(1);
  mPool.release(b1);
  mPool.get(3);
  mStats.refresh();
  Assert.assertEquals(2,mStats.mFreeBytes);
  Assert.assertEquals(4,mStats.mUsedBytes);
  mPool.trimToNothing();
  mStats.refresh();
  Assert.assertEquals(0,mStats.mFreeBytes);
  Assert.assertEquals(4,mStats.mUsedBytes);
}
