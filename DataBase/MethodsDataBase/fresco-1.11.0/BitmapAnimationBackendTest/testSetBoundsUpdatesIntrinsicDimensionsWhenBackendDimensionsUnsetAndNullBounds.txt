@Test public void testSetBoundsUpdatesIntrinsicDimensionsWhenBackendDimensionsUnsetAndNullBounds(){
  int boundsWidth=160;
  int boundsHeight=90;
  int backendIntrinsicWidth=AnimationBackend.INTRINSIC_DIMENSION_UNSET;
  int backendIntrinsicHeight=AnimationBackend.INTRINSIC_DIMENSION_UNSET;
  setupBoundsAndRendererDimensions(boundsWidth,boundsHeight,backendIntrinsicWidth,backendIntrinsicHeight);
  mBitmapAnimationBackend.setBounds(null);
  assertThat(mBitmapAnimationBackend.getIntrinsicWidth()).isEqualTo(AnimationBackend.INTRINSIC_DIMENSION_UNSET);
  assertThat(mBitmapAnimationBackend.getIntrinsicHeight()).isEqualTo(AnimationBackend.INTRINSIC_DIMENSION_UNSET);
}
