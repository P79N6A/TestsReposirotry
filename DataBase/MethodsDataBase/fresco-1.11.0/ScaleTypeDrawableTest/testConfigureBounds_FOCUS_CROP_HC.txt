/** 
 * Underlying drawable's aspect ratio is bigger than view's, so it has to be slided horizontally after scaling. Focus point is at 40% and it can be completely centered.
 */
@Test public void testConfigureBounds_FOCUS_CROP_HC(){
  Rect bounds=new Rect(10,10,410,310);
  int width=400;
  int height=200;
  PointF focusPoint=new PointF(0.40f,0.5f);
  Matrix expectedMatrix=new Matrix();
  expectedMatrix.setScale(1.5f,1.5f);
  expectedMatrix.postTranslate(-29,10);
  testConfigureBounds(bounds,width,height,ScaleType.FOCUS_CROP,focusPoint,expectedMatrix);
}
