@Test public void testMultipleMarks() throws IOException {
  mLimitedStream.mark(BYTES_LENGTH);
  assertEquals(LIMITED_LENGTH / 2,mLimitedStream.read(mReadBuffer,0,LIMITED_LENGTH / 2));
  mLimitedStream.mark(BYTES_LENGTH);
  assertEquals(LIMITED_LENGTH / 2,mLimitedStream.read(mReadBuffer,LIMITED_LENGTH / 2,LIMITED_LENGTH / 2));
  mLimitedStream.reset();
  assertEquals(LIMITED_LENGTH / 2,mLimitedStream.read(mReadBuffer));
  assertArrayEquals(Arrays.copyOfRange(mReadBuffer,0,LIMITED_LENGTH / 2),Arrays.copyOfRange(mReadBuffer,LIMITED_LENGTH / 2,LIMITED_LENGTH));
}
