@Test public void testDoesNotTransformIfImageRotationAngleUnkown(){
  whenResizingEnabled();
  whenRequestSpecificRotation(RotationOptions.NO_ROTATION);
  provideIntermediateResult(DefaultImageFormats.JPEG,800,800,EncodedImage.UNKNOWN_ROTATION_ANGLE,ExifInterface.ORIENTATION_UNDEFINED);
  verifyIntermediateResultPassedThroughUnchanged();
  provideFinalResult(DefaultImageFormats.JPEG,800,800,EncodedImage.UNKNOWN_ROTATION_ANGLE,ExifInterface.ORIENTATION_UNDEFINED);
  verifyFinalResultPassedThroughUnchanged();
  verifyZeroJpegTranscoderInteractions();
}
