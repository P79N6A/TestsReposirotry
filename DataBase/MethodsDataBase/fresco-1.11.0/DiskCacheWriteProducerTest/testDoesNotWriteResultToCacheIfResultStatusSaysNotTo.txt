@Test public void testDoesNotWriteResultToCacheIfResultStatusSaysNotTo(){
  setupInputProducerSuccessWithStatusFlags(Consumer.DO_NOT_CACHE_ENCODED,mFinalEncodedImageFormatUnknown);
  mDiskCacheWriteProducer.produceResults(mConsumer,mProducerContext);
  verify(mConsumer).onNewResult(mIntermediateEncodedImage,Consumer.DO_NOT_CACHE_ENCODED);
  verify(mConsumer).onNewResult(mFinalEncodedImageFormatUnknown,Consumer.IS_LAST | Consumer.DO_NOT_CACHE_ENCODED);
  verifyNoMoreInteractions(mProducerListener,mCacheKeyFactory,mDefaultBufferedDiskCache,mSmallImageBufferedDiskCache);
}
