@Test public void testToggleExclusive(){
  CloseableReference<Integer> cachedRef=mCache.cache(KEY,newReference(100),mEntryStateObserver);
  cachedRef.close();
  verify(mEntryStateObserver).onExclusivityChanged(KEY,true);
  mCache.get(KEY);
  verify(mEntryStateObserver).onExclusivityChanged(KEY,false);
}
