@Test public void testHexStringToByteArray(){
  assertArrayEquals(new byte[]{0x3F},JfifTestUtils.hexStringToByteArray("    3F        "));
  assertArrayEquals(new byte[]{0x1A,0x2B,0x3C,0x4D},JfifTestUtils.hexStringToByteArray("1A 2B 3C 4D"));
  assertArrayEquals(new byte[]{0x1A,0x2B,0x3C,0x4D,0x5E,0x6F},JfifTestUtils.hexStringToByteArray("1A2B 3C4D5E6F"));
}
