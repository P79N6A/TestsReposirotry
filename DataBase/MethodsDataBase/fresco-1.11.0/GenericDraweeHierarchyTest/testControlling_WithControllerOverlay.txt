@Test public void testControlling_WithControllerOverlay(){
  GenericDraweeHierarchy dh=mBuilder.setPlaceholderImage(mPlaceholderImage,null).setActualImageScaleType(null).setFadeDuration(250).build();

  assertSame(controllerOverlay,rootDrawable.mControllerOverlay);

  assertNull(rootDrawable.mControllerOverlay);
}
