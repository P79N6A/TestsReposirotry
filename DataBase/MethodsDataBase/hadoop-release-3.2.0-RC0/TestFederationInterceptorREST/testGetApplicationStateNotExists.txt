/** 
 * This test validates the correctness of GetApplicationState in case the application does not exist in StateStore.
 */
@Test public void testGetApplicationStateNotExists() throws YarnException, IOException, InterruptedException {
  ApplicationId appId=ApplicationId.newInstance(System.currentTimeMillis(),1);
  AppState response=interceptor.getAppState(null,appId.toString());
  Assert.assertNull(response);
}
