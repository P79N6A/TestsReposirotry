/** 
 * Test replica placement policy in case last node is invalid. We create 6 nodes but the last node is in fault topology (with rack info), so cannot be added to cluster. We should test proper exception is thrown in  adding node but shouldn't affect the cluster.
 */

  String scope="~" + NodeBase.getPath(dataNodes[0]);
  Map<Node,Integer> frequency=pickNodesAtRandom(100,scope);
  for (  Node key : dataNodes) {
    assertTrue(frequency.get(key) > 0 || key == dataNodes[0]);
  }
}
