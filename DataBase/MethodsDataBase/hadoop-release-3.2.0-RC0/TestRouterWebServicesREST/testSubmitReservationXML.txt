/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#submitReservation()} inside Router.
 */
@Test(timeout=2000) public void testSubmitReservationXML() throws Exception {
  ClientResponse badResponse=performCall(RM_WEB_SERVICE_PATH + RESERVATION_SUBMIT,null,null,null,PUT);
  assertEquals(SC_INTERNAL_SERVER_ERROR,badResponse.getStatus());
  ReservationSubmissionRequestInfo context=new ReservationSubmissionRequestInfo();
  context.setReservationId(getNewReservationId().getReservationId());
  ClientResponse response=performCall(RM_WEB_SERVICE_PATH + RESERVATION_SUBMIT,null,null,context,POST);
  assertEquals(SC_BAD_REQUEST,response.getStatus());
  String ci=response.getEntity(String.class);
  assertNotNull(ci);
}
