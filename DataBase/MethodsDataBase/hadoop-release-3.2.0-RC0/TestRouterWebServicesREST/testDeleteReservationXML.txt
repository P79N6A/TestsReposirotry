/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#deleteReservation()} inside Router.
 */
@Test(timeout=2000) public void testDeleteReservationXML() throws Exception {
  ClientResponse badResponse=performCall(RM_WEB_SERVICE_PATH + RESERVATION_DELETE,null,null,null,PUT);
  assertEquals(SC_INTERNAL_SERVER_ERROR,badResponse.getStatus());
  String reservationId=getNewReservationId().getReservationId();
  ReservationDeleteRequestInfo context=new ReservationDeleteRequestInfo();
  context.setReservationId(reservationId);
  ClientResponse response=performCall(RM_WEB_SERVICE_PATH + RESERVATION_DELETE,null,null,context,POST);
  assertEquals(SC_BAD_REQUEST,response.getStatus());
  String ci=response.getEntity(String.class);
  assertNotNull(ci);
}
