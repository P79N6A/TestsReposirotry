/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#getAppAttempts()} inside Router.
 */
@Test(timeout=2000) public void testAppAttemptXML() throws Exception {
  String appId=submitApplication();
  List<AppAttemptsInfo> responses=performGetCalls(RM_WEB_SERVICE_PATH + format(APPS_APPID_APPATTEMPTS,appId),AppAttemptsInfo.class,null,null);
  AppAttemptsInfo routerResponse=responses.get(0);
  AppAttemptsInfo rmResponse=responses.get(1);
  assertNotNull(routerResponse);
  assertNotNull(rmResponse);
  assertEquals(rmResponse.getAttempts().size(),routerResponse.getAttempts().size());
}
