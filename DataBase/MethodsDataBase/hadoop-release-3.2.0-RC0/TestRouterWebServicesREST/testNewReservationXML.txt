/** 
 * This test validates the correctness of {@link RMWebServiceProtocol#createNewReservation()} inside Router.
 */
@Test(timeout=2000) public void testNewReservationXML() throws Exception {
  ClientResponse badResponse=performCall(RM_WEB_SERVICE_PATH + RESERVATION_NEW,null,null,null,PUT);
  assertEquals(SC_INTERNAL_SERVER_ERROR,badResponse.getStatus());
  ClientResponse response=performCall(RM_WEB_SERVICE_PATH + RESERVATION_NEW,null,null,null,POST);
  assertEquals(SC_OK,response.getStatus());
  NewReservation ci=response.getEntity(NewReservation.class);
  assertNotNull(ci);
}
