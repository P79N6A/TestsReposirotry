/** 
 * Test valid value range for the config namenode.maintenance.replication.min.
 */
@Test(timeout=60000) public void testMaintenanceMinReplConfigRange(){
  LOG.info("Setting testMaintenanceMinReplConfigRange");
  setMinMaintenanceR(-1);
  try {
    startCluster(1,1);
    fail("Cluster start should fail when 'dfs.namenode.maintenance" + ".replication.min=-1'");
  }
 catch (  IOException e) {
    LOG.info("Expected exception: " + e);
  }
  int defaultRepl=getConf().getInt(DFSConfigKeys.DFS_REPLICATION_KEY,DFSConfigKeys.DFS_REPLICATION_DEFAULT);
  setMinMaintenanceR(defaultRepl + 1);
  try {
    startCluster(1,1);
    fail("Cluster start should fail when 'dfs.namenode.maintenance" + ".replication.min > " + defaultRepl + "'");
  }
 catch (  IOException e) {
    LOG.info("Expected exception: " + e);
  }
}
