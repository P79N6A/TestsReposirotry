/** 
 * Ensure that setStoragePolicy throws IOException when dfs.storage.policy.enabled is set to false.
 * @throws IOException
 */
@Test(timeout=300000,expected=IOException.class) public void testConfigKeyDisabled() throws IOException {
  Configuration conf=new HdfsConfiguration();
  conf.setBoolean(DFSConfigKeys.DFS_STORAGE_POLICY_ENABLED_KEY,false);
  MiniDFSCluster cluster=new MiniDFSCluster.Builder(conf).numDataNodes(1).build();
  try {
    cluster.waitActive();
    cluster.getFileSystem().setStoragePolicy(new Path("/"),HdfsConstants.COLD_STORAGE_POLICY_NAME);
  }
  finally {
    cluster.shutdown();
  }
}
