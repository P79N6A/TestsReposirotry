@Test public void testReadCredentialsFromConfigFile() throws Exception {
  Credentials credentials=DockerClientConfigHandler.readCredentialsFromConfigFile(new Path(file.toURI()),conf,APPLICATION_ID);
  Token token1=credentials.getToken(new Text("https://index.docker.io/v1/-" + APPLICATION_ID));
  assertEquals(DockerCredentialTokenIdentifier.KIND,token1.getKind());
  assertEquals("foobarbaz",new String(token1.getPassword()));
  DockerCredentialTokenIdentifier ti1=(DockerCredentialTokenIdentifier)token1.decodeIdentifier();
  assertEquals("https://index.docker.io/v1/",ti1.getRegistryUrl());
  assertEquals(APPLICATION_ID,ti1.getApplicationId());
  Token token2=credentials.getToken(new Text("registry.example.com-" + APPLICATION_ID));
  assertEquals(DockerCredentialTokenIdentifier.KIND,token2.getKind());
  assertEquals("bazbarfoo",new String(token2.getPassword()));
  DockerCredentialTokenIdentifier ti2=(DockerCredentialTokenIdentifier)token2.decodeIdentifier();
  assertEquals("registry.example.com",ti2.getRegistryUrl());
  assertEquals(APPLICATION_ID,ti2.getApplicationId());
}
