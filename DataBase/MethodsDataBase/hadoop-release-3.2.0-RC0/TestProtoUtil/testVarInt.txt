/** 
 * Test that readRawVarint32 is compatible with the varints encoded by ProtoBuf's CodedOutputStream.
 */
@Test public void testVarInt() throws IOException {
  for (  int value : TEST_VINT_VALUES) {
    doVarIntTest(value);
  }
  for (int i=1; i != 0; i<<=1) {
    doVarIntTest(i);
    doVarIntTest(-i);
    doVarIntTest(i - 1);
    doVarIntTest(~i);
  }
}
