@Test public void testContainsJavaRegex(){
  String[] javaCmds={"$JAVA_HOME/bin/java -cp App.jar AppClass","$JAVA_HOME/bin/java -jar App.jar AppClass &> logfile"};
  String[] nonJavaCmds={"$JAVA_HOME/bin/jajavava -cp App.jar AppClass","/nm/app/container/usercache/badjava -cp Bad.jar ChaosClass"};
  for (  String javaCmd : javaCmds) {
    Assert.assertTrue(javaCmd.matches(CONTAINS_JAVA_CMD));
  }
  for (  String nonJavaCmd : nonJavaCmds) {
    Assert.assertFalse(nonJavaCmd.matches(CONTAINS_JAVA_CMD));
  }
}
