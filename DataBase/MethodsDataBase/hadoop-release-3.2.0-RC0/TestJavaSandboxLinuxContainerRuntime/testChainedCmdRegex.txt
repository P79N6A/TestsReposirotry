@Test public void testChainedCmdRegex(){
  String[] multiCmds={"cmd1 && cmd2","cmd1 || cmd2","cmd1 `cmd2`","cmd1 $(cmd2)","cmd1; \\\n cmd2","cmd1; cmd2","cmd1|&cmd2","cmd1|cmd2","cmd1&cmd2"};
  Arrays.stream(multiCmds).forEach(cmd -> Assert.assertTrue(cmd.matches(MULTI_COMMAND_REGEX)));
  Assert.assertFalse("cmd1 &> logfile".matches(MULTI_COMMAND_REGEX));
}
