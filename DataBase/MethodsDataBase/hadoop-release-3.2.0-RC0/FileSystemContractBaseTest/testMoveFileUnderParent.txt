/** 
 * trying to move a file into it's parent dir should succeed again: no-op
 */
@Test public void testMoveFileUnderParent() throws Throwable {
  assumeTrue(renameSupported());
  Path filepath=path("testMoveFileUnderParent");
  createFile(filepath);
  rename(filepath,filepath,true,true,true);
  assertIsFile(filepath);
}
