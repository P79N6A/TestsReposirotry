/** 
 * Asserts that a zero byte file has a status of file and not directory or symlink
 * @throws Exception on failures
 */
@Test public void testMultiByteFilesAreFiles() throws Exception {
  Path src=path("testMultiByteFilesAreFiles");
  FSDataOutputStream out=fs.create(src);
  out.writeUTF("testMultiByteFilesAreFiles");
  out.close();
  assertIsFile(src);
}
