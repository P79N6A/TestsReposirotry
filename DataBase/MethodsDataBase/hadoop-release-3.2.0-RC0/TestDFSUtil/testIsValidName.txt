@Test(timeout=15000) public void testIsValidName(){
  assertFalse(DFSUtil.isValidName("/foo/../bar"));
  assertFalse(DFSUtil.isValidName("/foo/./bar"));
  assertFalse(DFSUtil.isValidName("/foo//bar"));
  assertTrue(DFSUtil.isValidName("/"));
  assertTrue(DFSUtil.isValidName("/bar/"));
  assertFalse(DFSUtil.isValidName("/foo/:/bar"));
  assertFalse(DFSUtil.isValidName("/foo:bar"));
}
