@Test public void testDockerImageNamePattern() throws Exception {
  String[] validNames={"ubuntu","fedora/httpd:version1.0","fedora/httpd:version1.0.test","fedora/httpd:version1.0.TEST","myregistryhost:5000/ubuntu","myregistryhost:5000/fedora/httpd:version1.0","myregistryhost:5000/fedora/httpd:version1.0.test","myregistryhost:5000/fedora/httpd:version1.0.TEST"};
  String[] invalidNames={"Ubuntu","ubuntu || fedora","ubuntu#","myregistryhost:50AB0/ubuntu","myregistry#host:50AB0/ubuntu",":8080/ubuntu"};
  for (  String name : validNames) {
    DockerLinuxContainerRuntime.validateImageName(name);
  }
  for (  String name : invalidNames) {
    try {
      DockerLinuxContainerRuntime.validateImageName(name);
      Assert.fail(name + " is an invalid name and should fail the regex");
    }
 catch (    ContainerExecutionException ce) {
      continue;
    }
  }
}
