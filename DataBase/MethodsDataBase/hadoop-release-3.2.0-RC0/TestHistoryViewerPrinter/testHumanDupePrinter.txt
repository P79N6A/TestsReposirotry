@Test public void testHumanDupePrinter() throws Exception {
  JobHistoryParser.JobInfo job=createJobInfo2();
  HumanReadableHistoryViewerPrinter printer=new HumanReadableHistoryViewerPrinter(job,false,"http://",TimeZone.getTimeZone("GMT"));
  String outStr=run(printer);
  int count1=outStr.indexOf("|Map-Reduce Framework          |Map input records             |");
  Assert.assertNotEquals("First counter occurrence not found",-1,count1);
  int count2=outStr.indexOf("|Map-Reduce Framework          |Map input records             |",count1 + 1);
  Assert.assertEquals("Duplicate counter found at: " + count1 + " and "+ count2,-1,count2);
}
