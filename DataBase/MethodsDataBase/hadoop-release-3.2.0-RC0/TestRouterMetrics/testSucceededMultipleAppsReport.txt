/** 
 * This test validates the correctness of the metric: Retrieved Multiple Apps successfully.
 */
@Test public void testSucceededMultipleAppsReport(){
  long totalGoodBefore=metrics.getNumSucceededMultipleAppsRetrieved();
  goodSubCluster.getApplicationsReport(100);
  Assert.assertEquals(totalGoodBefore + 1,metrics.getNumSucceededMultipleAppsRetrieved());
  Assert.assertEquals(100,metrics.getLatencySucceededMultipleGetAppReport(),0);
  goodSubCluster.getApplicationsReport(200);
  Assert.assertEquals(totalGoodBefore + 2,metrics.getNumSucceededMultipleAppsRetrieved());
  Assert.assertEquals(150,metrics.getLatencySucceededMultipleGetAppReport(),0);
}
