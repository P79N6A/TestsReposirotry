/** 
 * This test validates the correctness of the metric: Created Apps successfully.
 */
@Test public void testSucceededAppsCreated(){
  long totalGoodBefore=metrics.getNumSucceededAppsCreated();
  goodSubCluster.getNewApplication(100);
  Assert.assertEquals(totalGoodBefore + 1,metrics.getNumSucceededAppsCreated());
  Assert.assertEquals(100,metrics.getLatencySucceededAppsCreated(),0);
  goodSubCluster.getNewApplication(200);
  Assert.assertEquals(totalGoodBefore + 2,metrics.getNumSucceededAppsCreated());
  Assert.assertEquals(150,metrics.getLatencySucceededAppsCreated(),0);
}
