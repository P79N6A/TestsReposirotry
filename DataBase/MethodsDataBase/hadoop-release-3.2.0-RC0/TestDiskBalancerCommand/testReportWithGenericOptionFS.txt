/** 
 * This test simulates DiskBalancerCLI Report command run from a shell with a generic option 'fs'.
 * @throws Exception
 */
@Test(timeout=60000) public void testReportWithGenericOptionFS() throws Exception {
  final String topReportArg="5";
  final String reportArgs=String.format("-%s file:%s -%s -%s %s","fs",clusterJson.getPath(),REPORT,"top",topReportArg);
  final String cmdLine=String.format("%s",reportArgs);
  final List<String> outputs=runCommand(cmdLine);
  assertThat(outputs.get(0),containsString("Processing report command"));
  assertThat(outputs.get(1),is(allOf(containsString("Reporting top"),containsString(topReportArg),containsString("DataNode(s) benefiting from running DiskBalancer"))));
}
