@Test public void testDeregisterSubCluster() throws Exception {
  SubClusterId subClusterId=SubClusterId.newInstance("SC");
  registerSubCluster(createSubClusterInfo(subClusterId));
  SubClusterDeregisterRequest deregisterRequest=SubClusterDeregisterRequest.newInstance(subClusterId,SubClusterState.SC_UNREGISTERED);
  stateStore.deregisterSubCluster(deregisterRequest);
  Assert.assertEquals(SubClusterState.SC_UNREGISTERED,querySubClusterInfo(subClusterId).getState());
}
