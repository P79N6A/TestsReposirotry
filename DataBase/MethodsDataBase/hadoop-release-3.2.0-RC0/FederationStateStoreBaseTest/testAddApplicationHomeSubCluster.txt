@Test public void testAddApplicationHomeSubCluster() throws Exception {
  ApplicationId appId=ApplicationId.newInstance(1,1);
  SubClusterId subClusterId=SubClusterId.newInstance("SC");
  ApplicationHomeSubCluster ahsc=ApplicationHomeSubCluster.newInstance(appId,subClusterId);
  AddApplicationHomeSubClusterRequest request=AddApplicationHomeSubClusterRequest.newInstance(ahsc);
  AddApplicationHomeSubClusterResponse response=stateStore.addApplicationHomeSubCluster(request);
  Assert.assertEquals(subClusterId,response.getHomeSubCluster());
  Assert.assertEquals(subClusterId,queryApplicationHomeSC(appId));
}
