@Test public void testAddApplicationHomeSubClusterAppAlreadyExistsInTheSameSC() throws Exception {
  ApplicationId appId=ApplicationId.newInstance(1,1);
  SubClusterId subClusterId1=SubClusterId.newInstance("SC1");
  addApplicationHomeSC(appId,subClusterId1);
  ApplicationHomeSubCluster ahsc2=ApplicationHomeSubCluster.newInstance(appId,subClusterId1);
  AddApplicationHomeSubClusterResponse response=stateStore.addApplicationHomeSubCluster(AddApplicationHomeSubClusterRequest.newInstance(ahsc2));
  Assert.assertEquals(subClusterId1,response.getHomeSubCluster());
  Assert.assertEquals(subClusterId1,queryApplicationHomeSC(appId));
}
