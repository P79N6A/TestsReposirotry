@Test public void testUsernameInconsistentWithPassword2() throws Exception {
  conf=new Configuration();
  conf.setInt(Constants.MAX_ERROR_RETRIES,2);
  conf.set(Constants.PROXY_HOST,"127.0.0.1");
  conf.setInt(Constants.PROXY_PORT,1);
  conf.set(Constants.PROXY_PASSWORD,"password");
  IllegalArgumentException e=expectFSCreateFailure(IllegalArgumentException.class,conf,"Expected a connection error for proxy server");
  assertIsProxyUsernameError(e);
}
