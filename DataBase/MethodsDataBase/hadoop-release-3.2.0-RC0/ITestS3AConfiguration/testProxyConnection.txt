@Test public void testProxyConnection() throws Exception {
  conf=new Configuration();
  conf.setInt(Constants.MAX_ERROR_RETRIES,2);
  conf.set(Constants.PROXY_HOST,"127.0.0.1");
  conf.setInt(Constants.PROXY_PORT,1);
  String proxy=conf.get(Constants.PROXY_HOST) + ":" + conf.get(Constants.PROXY_PORT);
  expectFSCreateFailure(AWSClientIOException.class,conf,"when using proxy " + proxy);
}
