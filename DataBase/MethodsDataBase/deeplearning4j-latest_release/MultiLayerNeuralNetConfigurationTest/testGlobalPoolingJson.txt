@Test public void testGlobalPoolingJson(){
  MultiLayerConfiguration conf=new NeuralNetConfiguration.Builder().updater(new NoOp()).weightInit(WeightInit.DISTRIBUTION).dist(new NormalDistribution(0,1.0)).seed(12345L).list().layer(0,new ConvolutionLayer.Builder().kernelSize(2,2).stride(1,1).nOut(5).build()).layer(1,new GlobalPoolingLayer.Builder().poolingType(PoolingType.PNORM).pnorm(3).build()).layer(2,new OutputLayer.Builder(LossFunctions.LossFunction.MCXENT).activation(Activation.SOFTMAX).nOut(3).build()).setInputType(InputType.convolutional(32,32,1)).build();
  String str=conf.toJson();
  MultiLayerConfiguration fromJson=conf.fromJson(str);
  assertEquals(conf,fromJson);
}
