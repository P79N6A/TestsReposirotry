@Test @Ignore public void testDistance() throws Exception {
  INDArray matrix=Nd4j.rand(new int[]{400,10});
  INDArray rowVector=matrix.getRow(70);
  INDArray resultArr=Nd4j.zeros(400,1);
  Executor executor=Executors.newSingleThreadExecutor();
  executor.execute(new Runnable(){
    @Override public void run(){
      Nd4j.getExecutioner().exec(new EuclideanDistance(matrix,rowVector,resultArr,matrix.lengthLong()),-1);
      System.out.println("Ran!");
    }
  }
);
  Thread.sleep(600000);
}
