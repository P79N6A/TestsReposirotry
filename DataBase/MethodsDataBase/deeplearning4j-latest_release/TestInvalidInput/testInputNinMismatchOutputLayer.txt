@Test public void testInputNinMismatchOutputLayer(){
  MultiLayerConfiguration conf=new NeuralNetConfiguration.Builder().list().layer(0,new DenseLayer.Builder().nIn(10).nOut(20).build()).layer(1,new OutputLayer.Builder().nIn(10).nOut(10).activation(Activation.SOFTMAX).build()).build();
  MultiLayerNetwork net=new MultiLayerNetwork(conf);
  net.init();
  try {
    net.feedForward(Nd4j.create(1,10));
    fail("Expected DL4JException");
  }
 catch (  DL4JException e) {
    System.out.println("testInputNinMismatchOutputLayer(): " + e.getMessage());
  }
catch (  Exception e) {
    e.printStackTrace();
    fail("Expected DL4JException");
  }
}
