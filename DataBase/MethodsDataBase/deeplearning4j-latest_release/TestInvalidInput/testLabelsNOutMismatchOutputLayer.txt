@Test public void testLabelsNOutMismatchOutputLayer(){
  MultiLayerConfiguration conf=new NeuralNetConfiguration.Builder().list().layer(0,new DenseLayer.Builder().nIn(10).nOut(10).build()).layer(1,new OutputLayer.Builder().nIn(10).nOut(10).activation(Activation.SOFTMAX).build()).build();
  MultiLayerNetwork net=new MultiLayerNetwork(conf);
  net.init();
  try {
    net.fit(Nd4j.create(1,10),Nd4j.create(1,20));
    fail("Expected IllegalArgumentException");
  }
 catch (  IllegalArgumentException e) {
    System.out.println("testLabelsNOutMismatchOutputLayer(): " + e.getMessage());
  }
catch (  Exception e) {
    e.printStackTrace();
    fail("Expected DL4JException");
  }
}
