@Test public void testUnescape(){
  assertEquals("A",DictionaryEntryLineParser.unescape("\"A\""));
  assertEquals("\"A\"",DictionaryEntryLineParser.unescape("\"\"\"A\"\"\""));
  assertEquals("\"",DictionaryEntryLineParser.unescape("\"\"\"\""));
  assertEquals("\"\"",DictionaryEntryLineParser.unescape("\"\"\"\"\"\""));
  assertEquals("\"\"\"",DictionaryEntryLineParser.unescape("\"\"\"\"\"\"\"\""));
  assertEquals("\"\"\"\"\"",DictionaryEntryLineParser.unescape("\"\"\"\"\"\"\"\"\"\"\"\""));
}
