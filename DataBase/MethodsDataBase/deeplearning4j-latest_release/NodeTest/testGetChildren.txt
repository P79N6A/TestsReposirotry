@Test public void testGetChildren(){
  Trie trie=new Trie();
  Trie.Node node=trie.new Node('a');
  node.add("bcd");
  node.add("bde");
  node.add("xyz");
  assertEquals(2,node.getChildren().size());
  assertEquals('b',node.getChildren().get(0).getKey());
  assertEquals('x',node.getChildren().get(1).getKey());
}
