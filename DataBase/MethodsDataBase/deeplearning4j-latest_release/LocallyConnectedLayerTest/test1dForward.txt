@Test public void test1dForward(){
  MultiLayerConfiguration.Builder builder=new NeuralNetConfiguration.Builder().seed(123).optimizationAlgo(OptimizationAlgorithm.STOCHASTIC_GRADIENT_DESCENT).l2(2e-4).updater(new Nesterovs(0.9)).dropOut(0.5).list().layer(new LocallyConnected1D.Builder().kernelSize(8).nIn(3).stride(1).nOut(16).dropOut(0.5).convolutionMode(ConvolutionMode.Strict).setInputSize(28).activation(Activation.RELU).weightInit(WeightInit.XAVIER).build()).layer(new OutputLayer.Builder(LossFunctions.LossFunction.SQUARED_LOSS).nOut(10).weightInit(WeightInit.XAVIER).activation(Activation.SOFTMAX).build()).setInputType(InputType.recurrent(3,28));
  MultiLayerConfiguration conf=builder.build();
  MultiLayerNetwork network=new MultiLayerNetwork(conf);
  network.init();
  INDArray input=Nd4j.ones(10,3,28);
  INDArray output=network.output(input,false);
  ;
  for (int i=0; i < 100; i++) {
    output=network.output(input,false);
  }
  assertArrayEquals(new long[]{(28 - 8 + 1) * 10,10},output.shape());
  network.fit(input,output);
}
