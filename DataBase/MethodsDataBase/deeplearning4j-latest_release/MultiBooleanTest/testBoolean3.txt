@Test public void testBoolean3() throws Exception {
  MultiBoolean bool=new MultiBoolean(5);
  bool.set(true,0);
  bool.set(true,1);
  bool.set(true,2);
  bool.set(true,3);
  assertFalse(bool.allTrue());
  bool.set(true,4);
  assertFalse(bool.allFalse());
  assertTrue(bool.allTrue());
  bool.set(false,2);
  assertFalse(bool.allTrue());
  bool.set(true,2);
  assertTrue(bool.allTrue());
}
