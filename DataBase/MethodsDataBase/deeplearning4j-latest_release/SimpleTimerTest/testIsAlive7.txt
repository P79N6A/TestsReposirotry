@Test public void testIsAlive7() throws Exception {
  SimpleTimer timer=new SimpleTimer(5,TimeUnit.SECONDS);
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(1000);
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(1000);
  timer.triggerEvent();
  timer.triggerEvent();
  assertEquals(6,timer.getNumberOfEvents());
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(6000);
  assertEquals(0,timer.getNumberOfEvents());
}
