@Test public void testIsAlive6() throws Exception {
  SimpleTimer timer=new SimpleTimer(20,TimeUnit.SECONDS);
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(1000);
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(1000);
  timer.triggerEvent();
  timer.triggerEvent();
  assertEquals(6,timer.getNumberOfEvents());
  timer.triggerEvent();
  timer.triggerEvent();
  Thread.sleep(8000);
  assertEquals(8,timer.getNumberOfEvents());
}
