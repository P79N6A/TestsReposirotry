@Test public void testAcropolisInSentence() throws IOException {
  String userDictionary="クロ,クロ,クロ,カスタム名詞";
  Tokenizer tokenizer=makeTokenizer(userDictionary);
  String input="この丘はアクロポリスと呼ばれている。";
  assertTokenSurfacesEquals(Arrays.asList("この","丘","は","ア","クロ","ポリス","と","呼ば","れ","て","いる","。"),tokenizer.tokenize(input));
}
