@Test public void testIntegerToCategoricalTransform(){
  Schema schema=getSchema(ColumnType.Integer);
  Transform transform=new IntegerToCategoricalTransform("column",Arrays.asList("zero","one","two"));
  transform.setInputSchema(schema);
  Schema out=transform.transform(schema);
  assertEquals(1,out.getColumnMetaData().size());
  assertEquals(ColumnType.Categorical,out.getMetaData(0).getColumnType());
  CategoricalMetaData meta=(CategoricalMetaData)out.getMetaData(0);
  assertEquals(Arrays.asList("zero","one","two"),meta.getStateNames());
  assertEquals(Collections.singletonList((Writable)new Text("zero")),transform.map(Collections.singletonList((Writable)new IntWritable(0))));
  assertEquals(Collections.singletonList((Writable)new Text("one")),transform.map(Collections.singletonList((Writable)new IntWritable(1))));
  assertEquals(Collections.singletonList((Writable)new Text("two")),transform.map(Collections.singletonList((Writable)new IntWritable(2))));
}
