@Test public void shouldRecognizeInvalidCharactersInTopicNames(){
  char[] invalidChars={'/','\\',',','\u0000',':','"','\'',';','*','?',' ','\t','\r','\n','='};
  for (  char c : invalidChars) {
    String topicName="Is " + c + "illegal";
    assertFalse(Topic.containsValidPattern(topicName));
  }
}
