@Test public void testPutGet(){
  Cache<String,String> cache=new LRUCache<>(4);
  cache.put("a","b");
  cache.put("c","d");
  cache.put("e","f");
  cache.put("g","h");
  assertEquals(4,cache.size());
  assertEquals("b",cache.get("a"));
  assertEquals("d",cache.get("c"));
  assertEquals("f",cache.get("e"));
  assertEquals("h",cache.get("g"));
}
