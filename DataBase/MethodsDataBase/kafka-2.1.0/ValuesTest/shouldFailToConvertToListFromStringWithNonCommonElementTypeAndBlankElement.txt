/** 
 * Schema of type ARRAY requires a schema for the values, but Connect has no union or "any" schema type. Therefore, we can't represent this.
 */
@Test(expected=DataException.class) public void shouldFailToConvertToListFromStringWithNonCommonElementTypeAndBlankElement(){
  Values.convertToList(Schema.STRING_SCHEMA,"[1, 2, 3, \"four\",,,]");
}
