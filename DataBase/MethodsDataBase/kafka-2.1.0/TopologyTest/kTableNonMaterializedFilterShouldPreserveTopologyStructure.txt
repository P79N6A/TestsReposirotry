@Test public void kTableNonMaterializedFilterShouldPreserveTopologyStructure(){
  final StreamsBuilder builder=new StreamsBuilder();
  final KTable<Object,Object> table=builder.table("input-topic");
  table.filter((key,value) -> false);
  final TopologyDescription describe=builder.build().describe();
  Assert.assertEquals("Topologies:\n" + "   Sub-topology: 0\n" + "    Source: KSTREAM-SOURCE-0000000001 (topics: [input-topic])\n"+ "      --> KTABLE-SOURCE-0000000002\n"+ "    Processor: KTABLE-SOURCE-0000000002 (stores: [input-topic-STATE-STORE-0000000000])\n"+ "      --> KTABLE-FILTER-0000000003\n"+ "      <-- KSTREAM-SOURCE-0000000001\n"+ "    Processor: KTABLE-FILTER-0000000003 (stores: [])\n"+ "      --> none\n"+ "      <-- KTABLE-SOURCE-0000000002\n\n",describe.toString());
}
