@Test public void sinkShouldReturnTopicNameExtractorWithDynamicRouting(){
  final TopicNameExtractor topicNameExtractor=(key,value,record) -> record.topic() + "-" + key;
  final TopologyDescription.Sink expectedSinkNode=new InternalTopologyBuilder.Sink("sink",topicNameExtractor);
  assertThat(expectedSinkNode.topicNameExtractor(),equalTo(topicNameExtractor));
}
