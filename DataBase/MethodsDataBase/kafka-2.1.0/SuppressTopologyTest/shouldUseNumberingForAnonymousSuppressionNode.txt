@Test public void shouldUseNumberingForAnonymousSuppressionNode(){
  final StreamsBuilder anonymousNodeBuilder=new StreamsBuilder();
  anonymousNodeBuilder.stream("input",Consumed.with(STRING_SERDE,STRING_SERDE)).groupByKey().count().suppress(untilTimeLimit(Duration.ofSeconds(1),unbounded())).toStream().to("output",Produced.with(STRING_SERDE,Serdes.Long()));
