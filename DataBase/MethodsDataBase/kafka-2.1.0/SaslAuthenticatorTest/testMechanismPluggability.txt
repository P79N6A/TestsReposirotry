/** 
 * Tests that mechanisms that are not supported in Kafka can be plugged in without modifying Kafka code if Sasl client and server providers are available.
 */
@Test public void testMechanismPluggability() throws Exception {
  String node="0";
  SecurityProtocol securityProtocol=SecurityProtocol.SASL_SSL;
  configureMechanisms("DIGEST-MD5",Arrays.asList("DIGEST-MD5"));
  configureDigestMd5ServerCallback(securityProtocol);
  server=createEchoServer(securityProtocol);
  createAndCheckClientConnection(securityProtocol,node);
}
