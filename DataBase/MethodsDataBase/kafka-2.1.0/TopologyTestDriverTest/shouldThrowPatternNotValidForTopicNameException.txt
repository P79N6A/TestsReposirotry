@Test public void shouldThrowPatternNotValidForTopicNameException(){
  final String sourceName="source";
  final String pattern2Source1="source-topic-\\d";
  final Topology topology=new Topology();
  topology.addSource(sourceName,pattern2Source1);
  topology.addSink("sink",SINK_TOPIC_1,sourceName);
  testDriver=new TopologyTestDriver(topology,config);
  try {
    testDriver.pipeInput(consumerRecord1);
  }
 catch (  final TopologyException exception) {
    final String str=String.format("Invalid topology: Topology add source of type String for topic: %s cannot contain regex pattern for " + "input record topic: %s and hence cannot process the message.",pattern2Source1,SOURCE_TOPIC_1);
    assertEquals(str,exception.getMessage());
  }
}
