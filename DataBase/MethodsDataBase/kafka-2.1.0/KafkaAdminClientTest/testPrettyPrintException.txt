@Test public void testPrettyPrintException(){
  assertEquals("Null exception.",KafkaAdminClient.prettyPrintException(null));
  assertEquals("TimeoutException",KafkaAdminClient.prettyPrintException(new TimeoutException()));
  assertEquals("TimeoutException: The foobar timed out.",KafkaAdminClient.prettyPrintException(new TimeoutException("The foobar timed out.")));
}
