@Test public void testGenerateClientId(){
  Set<String> ids=new HashSet<>();
  for (int i=0; i < 10; i++) {
    String id=KafkaAdminClient.generateClientId(newConfMap(AdminClientConfig.CLIENT_ID_CONFIG,""));
    assertTrue("Got duplicate id " + id,!ids.contains(id));
    ids.add(id);
  }
  assertEquals("myCustomId",KafkaAdminClient.generateClientId(newConfMap(AdminClientConfig.CLIENT_ID_CONFIG,"myCustomId")));
}
