/** 
 * Tests that server accepts connections from clients with a trusted certificate when client authentication is required.
 */
@Test public void testClientAuthenticationRequiredValidProvided() throws Exception {
  String node="0";
  sslServerConfigs.put(BrokerSecurityConfigs.SSL_CLIENT_AUTH_CONFIG,"required");
  server=createEchoServer(SecurityProtocol.SSL);
  createSelector(sslClientConfigs);
  InetSocketAddress addr=new InetSocketAddress("localhost",server.port());
  selector.connect(node,addr,BUFFER_SIZE,BUFFER_SIZE);
  NetworkTestUtils.checkClientConnection(selector,node,100,10);
}
