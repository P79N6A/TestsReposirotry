/** 
 * Tests that server certificate with CN containing valid hostname is accepted by a client that connects using hostname and validates server endpoint.
 */
@Test public void testValidEndpointIdentificationCN() throws Exception {
  String node="0";
  serverCertStores=new CertStores(true,"localhost");
  clientCertStores=new CertStores(false,"localhost");
  sslServerConfigs=serverCertStores.getTrustingConfig(clientCertStores);
  sslClientConfigs=clientCertStores.getTrustingConfig(serverCertStores);
  server=createEchoServer(SecurityProtocol.SSL);
  sslClientConfigs.put(SslConfigs.SSL_ENDPOINT_IDENTIFICATION_ALGORITHM_CONFIG,"HTTPS");
  createSelector(sslClientConfigs);
  InetSocketAddress addr=new InetSocketAddress("localhost",server.port());
  selector.connect(node,addr,BUFFER_SIZE,BUFFER_SIZE);
  NetworkTestUtils.checkClientConnection(selector,node,100,10);
}
