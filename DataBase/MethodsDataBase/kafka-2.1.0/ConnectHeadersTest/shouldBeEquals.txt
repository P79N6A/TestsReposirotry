@Test public void shouldBeEquals(){
  Headers other=new ConnectHeaders();

  assertEquals(headers.hashCode(),other.hashCode());
  populate(headers);
  assertNotEquals(headers,other);
  assertNotEquals(headers.hashCode(),other.hashCode());

  assertNotEquals(headers,other);
