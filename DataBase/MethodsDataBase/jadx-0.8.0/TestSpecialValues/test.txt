@Test public void test(){
  ClassNode cls=getClassNode(TestCls.class);
  String code=cls.getCode().toString();
  assertThat(code,containsOne("Float.NaN, Float.NEGATIVE_INFINITY, Float.POSITIVE_INFINITY, " + "Float.MIN_VALUE, Float.MAX_VALUE, Float.MIN_NORMAL"));
  assertThat(code,containsOne("Double.NaN, Double.NEGATIVE_INFINITY, Double.POSITIVE_INFINITY, " + "Double.MIN_VALUE, Double.MAX_VALUE, Double.MIN_NORMAL"));
  assertThat(code,containsOne("Short.MIN_VALUE, Short.MAX_VALUE"));
  assertThat(code,containsOne("Byte.MIN_VALUE, Byte.MAX_VALUE"));
  assertThat(code,containsOne("Integer.MIN_VALUE, Integer.MAX_VALUE"));
  assertThat(code,containsOne("Long.MIN_VALUE, Long.MAX_VALUE"));
}
