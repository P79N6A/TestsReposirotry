@Test public void multipleTopLevelValuesInOneDocument() throws IOException {
  assumeTrue(factory == Factory.STRING_READER);
  JsonReader reader=factory.create("[][]");
  reader.setLenient(true);
  reader.beginArray();
  reader.endArray();
  assertEquals("$",reader.getPath());
  reader.beginArray();
  reader.endArray();
  assertEquals("$",reader.getPath());
}
