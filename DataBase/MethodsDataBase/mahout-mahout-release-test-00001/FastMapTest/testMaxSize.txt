@Test public void testMaxSize(){
  Map<String,String> map=new FastMap<>(1,1);
  map.put("foo","bar");
  assertEquals(1,map.size());
  map.put("baz","bang");
  assertEquals(1,map.size());
  assertNull(map.get("foo"));
  map.put("baz","buzz");
  assertEquals(1,map.size());
  assertEquals("buzz",map.get("baz"));
}
