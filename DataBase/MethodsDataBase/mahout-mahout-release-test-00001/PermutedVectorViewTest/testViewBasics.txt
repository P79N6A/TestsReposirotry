@Test public void testViewBasics(){
  Vector v=randomVector();
  int[] pivot=pivot();
  Vector pvv=new PermutedVectorView(v,pivot);
  for (int i=0; i < 20; i++) {
    assertEquals("Element " + i,v.get(pivot[i]),pvv.get(i),0);
  }
  pvv.set(6,321);
  v.set(9,512);
  for (int i=0; i < 20; i++) {
    assertEquals("Element " + i,v.get(pivot[i]),pvv.get(i),0);
  }
}
