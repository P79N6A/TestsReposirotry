@Test public void testInsert(){
  Random rand=RandomUtils.getRandom();
  Multinomial<Integer> table=new Multinomial<>();
  double[] p=new double[10];
  for (int i=0; i < 10; i++) {
    p[i]=rand.nextDouble();
    table.add(i,p[i]);
  }
  checkSelfConsistent(table);
  for (int i=0; i < 10; i++) {
    assertEquals(p[i],table.getWeight(i),0);
  }
}
