@Test public void testSizeEmpty(){
  FastByIDMap<Long> map=new FastByIDMap<>();
  assertEquals(0,map.size());
  assertTrue(map.isEmpty());
  map.put(500000L,2L);
  assertEquals(1,map.size());
  assertFalse(map.isEmpty());
  map.remove(500000L);
  assertEquals(0,map.size());
  assertTrue(map.isEmpty());
}
