@Test public void testIteratorSet(){
  Vector clone=test.clone();
  for (  Element e : clone.nonZeroes()) {
    e.set(e.get() * 2.0);
  }
  for (  Element e : clone.nonZeroes()) {
    assertEquals(test.get(e.index()) * 2.0,e.get(),EPSILON);
  }
  clone=test.clone();
  for (  Element e : clone.all()) {
    e.set(e.get() * 2.0);
  }
  for (  Element e : clone.all()) {
    assertEquals(test.get(e.index()) * 2.0,e.get(),EPSILON);
  }
}
