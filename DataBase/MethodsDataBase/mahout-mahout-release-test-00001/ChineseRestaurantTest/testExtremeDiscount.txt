@Test public void testExtremeDiscount(){
  ChineseRestaurant x=new ChineseRestaurant(100,1);
  Multiset<Integer> counts=HashMultiset.create();
  for (int i=0; i < 10000; i++) {
    counts.add(x.sample());
  }
  assertEquals(10000,x.size());
  for (int i=0; i < 10000; i++) {
    assertEquals(1,x.count(i));
  }
}
