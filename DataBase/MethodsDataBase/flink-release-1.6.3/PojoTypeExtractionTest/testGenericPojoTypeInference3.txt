@SuppressWarnings({"unchecked","rawtypes"}) @Test public void testGenericPojoTypeInference3(){
  MyMapper3<Boolean,Character> function=new MyMapper3<>();
  TypeInformation<?> ti=TypeExtractor.getMapReturnTypes(function,TypeInformation.of(new TypeHint<PojoTuple<Character,Boolean,Boolean>>(){
  }
));
  Assert.assertTrue(ti instanceof TupleTypeInfo<?>);
  TupleTypeInfo<?> tti=(TupleTypeInfo<?>)ti;
  Assert.assertEquals(BasicTypeInfo.CHAR_TYPE_INFO,tti.getTypeAt(0));
  Assert.assertEquals(BasicTypeInfo.BOOLEAN_TYPE_INFO,tti.getTypeAt(1));
}
