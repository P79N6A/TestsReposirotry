@Test public void testCreateEntropyAwarePlainFs() throws Exception {
  File folder=TMP_FOLDER.newFolder();
  Path path=new Path(Path.fromLocalFile(folder),"_entropy_/file");
  OutputStreamAndPath out=EntropyInjector.createEntropyAware(LocalFileSystem.getSharedInstance(),path,WriteMode.NO_OVERWRITE);
  out.stream().close();

}
