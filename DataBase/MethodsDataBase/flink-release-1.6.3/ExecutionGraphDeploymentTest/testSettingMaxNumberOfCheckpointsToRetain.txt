@Test public void testSettingMaxNumberOfCheckpointsToRetain() throws Exception {
  final int maxNumberOfCheckpointsToRetain=10;
  final Configuration jobManagerConfig=new Configuration();
  jobManagerConfig.setInteger(CheckpointingOptions.MAX_RETAINED_CHECKPOINTS,maxNumberOfCheckpointsToRetain);
  final ExecutionGraph eg=createExecutionGraph(jobManagerConfig);
  assertEquals(maxNumberOfCheckpointsToRetain,eg.getCheckpointCoordinator().getCheckpointStore().getMaxNumberOfRetainedCheckpoints());
}
