@Test public void testRowSerialization() throws IOException {
  final TypeInformation<Row> rowSchema=Types.ROW_NAMED(new String[]{"f1","f2","f3"},Types.INT,Types.BOOLEAN,Types.STRING);
  final Row row=new Row(3);
  row.setField(0,1);
  row.setField(1,true);
  row.setField(2,"str");
  final Row resultRow=serializeAndDeserialize(rowSchema,row);
  assertEquals(row,resultRow);
}
