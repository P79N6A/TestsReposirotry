 * Test test verifies that the execution environment can be used to execute multiple bounded streaming jobs after one another.
 */

  LocalStreamEnvironment env=new LocalStreamEnvironment();

  env.execute();
  addSmallBoundedJob(env,5);
  env.execute();
}
