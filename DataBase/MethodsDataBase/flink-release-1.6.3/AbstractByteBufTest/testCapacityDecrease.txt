@Test public void testCapacityDecrease(){
  ByteBuf buffer=newBuffer(3,13);
  assertEquals(13,buffer.maxCapacity());
  assertEquals(3,buffer.capacity());
  try {
    buffer.capacity(2);
    assertEquals(2,buffer.capacity());
    assertEquals(13,buffer.maxCapacity());
  }
  finally {
    buffer.release();
  }
}
