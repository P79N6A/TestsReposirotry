@Test public void testInstantiate(){
  try {
    TypeSerializer<T> serializer=getSerializer();
    if (serializer.getClass().getName().endsWith("KryoSerializer")) {
      return;
    }
    T instance=serializer.createInstance();
    assertNotNull("The created instance must not be null.",instance);
    Class<T> type=getTypeClass();
    assertNotNull("The test is corrupt: type class is null.",type);
    if (!type.isAssignableFrom(instance.getClass())) {
      fail("Type of the instantiated object is wrong. " + "Expected Type: " + type + " present type "+ instance.getClass());
    }
  }
 catch (  Exception e) {
    System.err.println(e.getMessage());
    e.printStackTrace();
    fail("Exception in test: " + e.getMessage());
  }
}
