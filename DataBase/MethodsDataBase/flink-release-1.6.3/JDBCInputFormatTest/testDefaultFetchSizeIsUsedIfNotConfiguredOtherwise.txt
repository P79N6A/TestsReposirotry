@Test public void testDefaultFetchSizeIsUsedIfNotConfiguredOtherwise() throws SQLException, ClassNotFoundException {
  jdbcInputFormat=JDBCInputFormat.buildJDBCInputFormat().setDrivername(DRIVER_CLASS).setDBUrl(DB_URL).setQuery(SELECT_ALL_BOOKS).setRowTypeInfo(ROW_TYPE_INFO).finish();
  jdbcInputFormat.openInputFormat();
  Class.forName(DRIVER_CLASS);
  final int defaultFetchSize=DriverManager.getConnection(DB_URL).createStatement().getFetchSize();
  Assert.assertEquals(defaultFetchSize,jdbcInputFormat.getStatement().getFetchSize());
}
