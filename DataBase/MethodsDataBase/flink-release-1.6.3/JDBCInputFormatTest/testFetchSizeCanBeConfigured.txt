@Test public void testFetchSizeCanBeConfigured() throws SQLException {
  final int desiredFetchSize=10_000;
  jdbcInputFormat=JDBCInputFormat.buildJDBCInputFormat().setDrivername(DRIVER_CLASS).setDBUrl(DB_URL).setQuery(SELECT_ALL_BOOKS).setRowTypeInfo(ROW_TYPE_INFO).setFetchSize(desiredFetchSize).finish();
  jdbcInputFormat.openInputFormat();
  Assert.assertEquals(desiredFetchSize,jdbcInputFormat.getStatement().getFetchSize());
}
