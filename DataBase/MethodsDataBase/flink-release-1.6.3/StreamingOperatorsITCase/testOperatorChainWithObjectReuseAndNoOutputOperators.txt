@Test public void testOperatorChainWithObjectReuseAndNoOutputOperators() throws Exception {
  StreamExecutionEnvironment env=StreamExecutionEnvironment.getExecutionEnvironment();
  env.getConfig().enableObjectReuse();
  DataStream<Integer> input=env.fromElements(1,2,3);
  input.flatMap(new FlatMapFunction<Integer,Integer>(){
    @Override public void flatMap(    Integer value,    Collector<Integer> out) throws Exception {
      out.collect(value << 1);
    }
  }
);
  env.execute();
}
