@Test public void testTableSchema() throws Exception {
  final Executor executor=createDefaultExecutor(clusterClient);
  final SessionContext session=new SessionContext("test-session",new Environment());
  final TableSchema actualTableSchema=executor.getTableSchema(session,"TableNumber2");
  final TableSchema expectedTableSchema=new TableSchema(new String[]{"IntegerField2","StringField2"},new TypeInformation[]{Types.INT,Types.STRING});
  assertEquals(expectedTableSchema,actualTableSchema);
}
