@Test public void testIsValid() throws Exception {
  ResourceSpec rs=ResourceSpec.newBuilder().setCpuCores(1.0).setHeapMemoryInMB(100).build();
  assertTrue(rs.isValid());
  rs=ResourceSpec.newBuilder().setCpuCores(1.0).setHeapMemoryInMB(100).setGPUResource(1).build();
  assertTrue(rs.isValid());
  rs=ResourceSpec.newBuilder().setCpuCores(1.0).setHeapMemoryInMB(100).setGPUResource(-1).build();
  assertFalse(rs.isValid());
}
