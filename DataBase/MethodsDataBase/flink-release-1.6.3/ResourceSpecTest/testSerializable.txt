@Test public void testSerializable() throws Exception {
  ResourceSpec rs1=ResourceSpec.newBuilder().setCpuCores(1.0).setHeapMemoryInMB(100).setGPUResource(1.1).build();
  byte[] buffer=InstantiationUtil.serializeObject(rs1);
  ResourceSpec rs2=InstantiationUtil.deserializeObject(buffer,ClassLoader.getSystemClassLoader());
  assertEquals(rs1,rs2);
}
