/** 
 * Tests forwarding of both  {@link ReadOnlySlicedNetworkBuffer#recycleBuffer()} and{@link ReadOnlySlicedNetworkBuffer#isRecycled()}.
 */
@Test public void testForwardsRecycleBuffer1(){
  ReadOnlySlicedNetworkBuffer slice=buffer.readOnlySlice();
  assertFalse(slice.isRecycled());
  slice.recycleBuffer();
  assertTrue(slice.isRecycled());
  assertTrue(buffer.isRecycled());
}
