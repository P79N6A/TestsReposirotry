/** 
 * Tests forwarding of both  {@link ReadOnlySlicedNetworkBuffer#retainBuffer()} and{@link ReadOnlySlicedNetworkBuffer#isRecycled()}.
 */
@Test public void testForwardsRetainBuffer2(){
  ReadOnlySlicedNetworkBuffer slice=buffer.readOnlySlice(1,2);
  assertEquals(buffer.refCnt(),slice.refCnt());
  slice.retainBuffer();
  assertEquals(buffer.refCnt(),slice.refCnt());
}
