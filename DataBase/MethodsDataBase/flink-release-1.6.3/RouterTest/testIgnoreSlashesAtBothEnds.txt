@Test public void testIgnoreSlashesAtBothEnds(){
  assertEquals("index",router.route(GET,"articles").target());
  assertEquals("index",router.route(GET,"/articles").target());
  assertEquals("index",router.route(GET,"//articles").target());
  assertEquals("index",router.route(GET,"articles/").target());
  assertEquals("index",router.route(GET,"articles//").target());
  assertEquals("index",router.route(GET,"/articles/").target());
  assertEquals("index",router.route(GET,"//articles//").target());
}
