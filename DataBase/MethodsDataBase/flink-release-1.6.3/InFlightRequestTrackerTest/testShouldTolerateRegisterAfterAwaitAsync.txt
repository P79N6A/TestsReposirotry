@Test public void testShouldTolerateRegisterAfterAwaitAsync(){
  final CompletableFuture<Void> awaitFuture=inFlightRequestTracker.awaitAsync();
  assertTrue(awaitFuture.isDone());
  inFlightRequestTracker.registerRequest();

}
