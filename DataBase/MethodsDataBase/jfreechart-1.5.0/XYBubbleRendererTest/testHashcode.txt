/** 
 * Two objects that are equal are required to return the same hashCode.
 */
@Test public void testHashcode(){
  XYBubbleRenderer r1=new XYBubbleRenderer();
  XYBubbleRenderer r2=new XYBubbleRenderer();
  assertTrue(r1.equals(r2));
  int h1=r1.hashCode();
  int h2=r2.hashCode();
  assertEquals(h1,h2);
}
