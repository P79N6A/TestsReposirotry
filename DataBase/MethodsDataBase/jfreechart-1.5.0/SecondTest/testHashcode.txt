/** 
 * Two objects that are equal are required to return the same hashCode.
 */
@Test public void testHashcode(){
  Second s1=new Second(13,45,5,1,2,2003);
  Second s2=new Second(13,45,5,1,2,2003);
  assertTrue(s1.equals(s2));
  int h1=s1.hashCode();
  int h2=s2.hashCode();
  assertEquals(h1,h2);
}
