/** 
 * Two objects that are equal are required to return the same hashCode.
 */
@Test public void testHashcode(){
  XYSplineRenderer r1=new XYSplineRenderer();
  XYSplineRenderer r2=new XYSplineRenderer();
  assertTrue(r1.equals(r2));
  int h1=r1.hashCode();
  int h2=r2.hashCode();
  assertEquals(h1,h2);
}
