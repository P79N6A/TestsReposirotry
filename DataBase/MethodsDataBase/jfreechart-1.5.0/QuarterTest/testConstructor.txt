/** 
 * Some tests for the constructor with (int, int) arguments.  Covers bug report 1377239.
 */
@Test public void testConstructor(){
  boolean pass=false;
  try {
    new Quarter(0,2005);
  }
 catch (  IllegalArgumentException e) {
    pass=true;
  }
  assertTrue(pass);
  pass=false;
  try {
    new Quarter(5,2005);
  }
 catch (  IllegalArgumentException e) {
    pass=true;
  }
  assertTrue(pass);
}
