/** 
 * Two objects that are equal are required to return the same hashCode.
 */
@Test public void testHashcode(){
  XYAreaRenderer2 r1=new XYAreaRenderer2();
  XYAreaRenderer2 r2=new XYAreaRenderer2();
  assertTrue(r1.equals(r2));
  int h1=r1.hashCode();
  int h2=r2.hashCode();
  assertEquals(h1,h2);
}
