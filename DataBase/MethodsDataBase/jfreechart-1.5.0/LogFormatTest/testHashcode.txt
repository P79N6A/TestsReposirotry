/** 
 * Two objects that are equal are required to return the same hashCode.
 */
@Test public void testHashcode(){
  LogFormat f1=new LogFormat(10.0,"10",true);
  LogFormat f2=new LogFormat(10.0,"10",true);
  assertTrue(f1.equals(f2));
  int h1=f1.hashCode();
  int h2=f2.hashCode();
  assertEquals(h1,h2);
}
