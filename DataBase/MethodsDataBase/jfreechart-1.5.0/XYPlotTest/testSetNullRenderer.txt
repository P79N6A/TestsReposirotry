/** 
 * Setting a null renderer should be allowed, but is generating a null pointer exception in 0.9.7.
 */
@Test public void testSetNullRenderer(){
  boolean failed=false;
  try {
    XYPlot plot=new XYPlot(null,new NumberAxis("X"),new NumberAxis("Y"),null);
    plot.setRenderer(null);
  }
 catch (  Exception e) {
    failed=true;
  }
  assertTrue(!failed);
}
