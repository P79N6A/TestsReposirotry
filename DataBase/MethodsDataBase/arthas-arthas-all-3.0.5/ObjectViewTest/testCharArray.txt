@Test public void testCharArray(){
  char[] data={'a','b','c','d'};
  ObjectView objectView=new ObjectView(data,3);
  String expected="@char[][\n" + "    @Character[a],\n" + "    @Character[b],\n"+ "    @Character[c],\n"+ "    @Character[d],\n"+ "]";
  Assert.assertEquals(expected,objectView.draw());
}
