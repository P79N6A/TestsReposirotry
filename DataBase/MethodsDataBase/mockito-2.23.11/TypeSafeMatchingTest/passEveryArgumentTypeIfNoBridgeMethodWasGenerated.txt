@Test public void passEveryArgumentTypeIfNoBridgeMethodWasGenerated(){
  final AtomicBoolean wasCalled=new AtomicBoolean();
class GenericMatcher<T> implements ArgumentMatcher<T> {
    @Override public boolean matches(    T argument){
      wasCalled.set(true);
      return true;
    }
  }
  wasCalled.set(false);
  matchesTypeSafe().apply(new GenericMatcher<Integer>(),123);
  assertThat(wasCalled.get()).isTrue();
  wasCalled.set(false);
  matchesTypeSafe().apply(new GenericMatcher<Integer>(),"");
  assertThat(wasCalled.get()).isTrue();
}
