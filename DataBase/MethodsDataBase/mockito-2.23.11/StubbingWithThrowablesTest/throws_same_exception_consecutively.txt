@Test public void throws_same_exception_consecutively(){
  when(mock.add("")).thenThrow(new ExceptionOne());
  Assertions.assertThatThrownBy(new ThrowableAssert.ThrowingCallable(){
    public void call(){
      mock.add("");
    }
  }
).isInstanceOf(ExceptionOne.class);
  mock.add("1");
  Assertions.assertThatThrownBy(new ThrowableAssert.ThrowingCallable(){
    public void call(){
      mock.add("");
    }
  }
).isInstanceOf(ExceptionOne.class);
}
