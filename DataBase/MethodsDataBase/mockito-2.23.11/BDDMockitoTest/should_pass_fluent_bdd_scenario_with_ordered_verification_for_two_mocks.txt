@Test public void should_pass_fluent_bdd_scenario_with_ordered_verification_for_two_mocks(){
  Car car=new Car();
  Person person=mock(Person.class);
  Police police=mock(Police.class);
  person.drive(car);
  person.drive(car);
  police.chase(car);
  InOrder inOrder=inOrder(person,police);
  then(person).should(inOrder,times(2)).drive(car);
  then(police).should(inOrder).chase(car);
}
