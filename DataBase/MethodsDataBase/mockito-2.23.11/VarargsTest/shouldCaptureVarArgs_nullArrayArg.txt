/** 
 * Relates to Github issue #583 "ArgumentCaptor: NPE when an null array is passed to a varargs method"
 */
@Test public void shouldCaptureVarArgs_nullArrayArg(){
  String[] argArray=null;
  mock.varargs(argArray);
  verify(mock).varargs(captor.capture());
  assertThat(captor).areExactly(1,NULL);
}
