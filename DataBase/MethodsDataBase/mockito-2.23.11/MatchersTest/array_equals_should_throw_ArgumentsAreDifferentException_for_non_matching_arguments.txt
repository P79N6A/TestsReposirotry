@Test(expected=ArgumentsAreDifferent.class) public void array_equals_should_throw_ArgumentsAreDifferentException_for_non_matching_arguments(){
  List<Object> list=Mockito.mock(List.class);
  list.add("test");
  list.contains(new Object[]{"1"});
  Mockito.verify(list).contains(new Object[]{"1","2","3"});
}
