@Test public void runWithSystemPropertiesShouldSetAndRemoveProperties(){
  String key="test." + UUID.randomUUID();
  assertThat(System.getProperties().containsKey(key)).isFalse();
  get().withSystemProperties(key + "=value").run((context) -> assertThat(System.getProperties()).containsEntry(key,"value"));
  assertThat(System.getProperties().containsKey(key)).isFalse();
}
