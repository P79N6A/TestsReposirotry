@Test public void runWithSystemPropertiesWhenContextFailsShouldRemoveProperties(){
  String key="test." + UUID.randomUUID();
  assertThat(System.getProperties().containsKey(key)).isFalse();
  get().withSystemProperties(key + "=value").withUserConfiguration(FailingConfig.class).run((context) -> assertThat(context).hasFailed());
  assertThat(System.getProperties().containsKey(key)).isFalse();
}
