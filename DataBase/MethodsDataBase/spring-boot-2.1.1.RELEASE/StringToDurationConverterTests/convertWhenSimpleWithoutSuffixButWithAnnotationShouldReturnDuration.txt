@Test public void convertWhenSimpleWithoutSuffixButWithAnnotationShouldReturnDuration(){
  assertThat(convert("10",ChronoUnit.SECONDS,null)).isEqualTo(Duration.ofSeconds(10));
  assertThat(convert("+10",ChronoUnit.SECONDS,null)).isEqualTo(Duration.ofSeconds(10));
  assertThat(convert("-10",ChronoUnit.SECONDS,null)).isEqualTo(Duration.ofSeconds(-10));
}
