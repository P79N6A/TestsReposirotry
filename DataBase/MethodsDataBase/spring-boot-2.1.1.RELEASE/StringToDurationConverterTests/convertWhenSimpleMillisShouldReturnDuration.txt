@Test public void convertWhenSimpleMillisShouldReturnDuration(){
  assertThat(convert("10ms")).isEqualTo(Duration.ofMillis(10));
  assertThat(convert("10MS")).isEqualTo(Duration.ofMillis(10));
  assertThat(convert("+10ms")).isEqualTo(Duration.ofMillis(10));
  assertThat(convert("-10ms")).isEqualTo(Duration.ofMillis(-10));
}
