@Test public void convertWhenSimpleNanosShouldReturnDuration(){
  assertThat(convert("10ns")).isEqualTo(Duration.ofNanos(10));
  assertThat(convert("10NS")).isEqualTo(Duration.ofNanos(10));
  assertThat(convert("+10ns")).isEqualTo(Duration.ofNanos(10));
  assertThat(convert("-10ns")).isEqualTo(Duration.ofNanos(-10));
}
