@Test public void convertWhenSimpleSecondsShouldReturnDuration(){
  assertThat(convert("10s")).isEqualTo(Duration.ofSeconds(10));
  assertThat(convert("10S")).isEqualTo(Duration.ofSeconds(10));
  assertThat(convert("+10s")).isEqualTo(Duration.ofSeconds(10));
  assertThat(convert("-10s")).isEqualTo(Duration.ofSeconds(-10));
}
