@Test public void convertWhenSimpleHoursShouldReturnDuration(){
  assertThat(convert("10h")).isEqualTo(Duration.ofHours(10));
  assertThat(convert("10H")).isEqualTo(Duration.ofHours(10));
  assertThat(convert("+10h")).isEqualTo(Duration.ofHours(10));
  assertThat(convert("-10h")).isEqualTo(Duration.ofHours(-10));
}
