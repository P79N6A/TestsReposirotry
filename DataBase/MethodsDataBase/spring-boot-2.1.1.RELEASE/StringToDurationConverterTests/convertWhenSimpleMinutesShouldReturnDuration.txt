@Test public void convertWhenSimpleMinutesShouldReturnDuration(){
  assertThat(convert("10m")).isEqualTo(Duration.ofMinutes(10));
  assertThat(convert("10M")).isEqualTo(Duration.ofMinutes(10));
  assertThat(convert("+10m")).isEqualTo(Duration.ofMinutes(10));
  assertThat(convert("-10m")).isEqualTo(Duration.ofMinutes(-10));
}
