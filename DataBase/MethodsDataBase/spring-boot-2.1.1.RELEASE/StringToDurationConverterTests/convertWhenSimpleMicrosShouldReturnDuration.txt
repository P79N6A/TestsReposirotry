@Test public void convertWhenSimpleMicrosShouldReturnDuration(){
  assertThat(convert("10us")).isEqualTo(Duration.ofNanos(10000));
  assertThat(convert("10US")).isEqualTo(Duration.ofNanos(10000));
  assertThat(convert("+10us")).isEqualTo(Duration.ofNanos(10000));
  assertThat(convert("-10us")).isEqualTo(Duration.ofNanos(-10000));
}
