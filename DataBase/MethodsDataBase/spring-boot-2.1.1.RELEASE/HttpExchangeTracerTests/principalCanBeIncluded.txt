@Test public void principalCanBeIncluded(){
  HttpTrace trace=new HttpTrace(createRequest());
  new HttpExchangeTracer(EnumSet.of(Include.PRINCIPAL)).sendingResponse(trace,createResponse(),() -> createPrincipal(),null);
  assertThat(trace.getPrincipal()).isNotNull();
  assertThat(trace.getPrincipal().getName()).isEqualTo("alice");
}
