@Test public void customProjectInfo(){
  this.request.setGroupId("org.acme");
  this.request.setArtifactId("sample");
  this.request.setVersion("1.0.1-SNAPSHOT");
  this.request.setDescription("Spring Boot Test");
  assertThat(this.request.generateUrl(createDefaultMetadata())).isEqualTo(createDefaultUrl("?groupId=org.acme&artifactId=sample&version=1.0.1-SNAPSHOT" + "&description=Spring+Boot+Test&type=test-type"));
}
