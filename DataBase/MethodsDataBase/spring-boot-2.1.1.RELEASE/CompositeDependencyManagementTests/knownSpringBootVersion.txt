@Test public void knownSpringBootVersion(){
  given(this.dependencyManagement1.getSpringBootVersion()).willReturn("1.2.3");
  given(this.dependencyManagement2.getSpringBootVersion()).willReturn("1.2.4");
  assertThat(new CompositeDependencyManagement(this.dependencyManagement1,this.dependencyManagement2).getSpringBootVersion()).isEqualTo("1.2.3");
}
