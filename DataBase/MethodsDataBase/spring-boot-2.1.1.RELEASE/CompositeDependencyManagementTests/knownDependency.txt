@Test public void knownDependency(){
  given(this.dependencyManagement1.find("artifact")).willReturn(new Dependency("test","artifact","1.2.3"));
  given(this.dependencyManagement2.find("artifact")).willReturn(new Dependency("test","artifact","1.2.4"));
  assertThat(new CompositeDependencyManagement(this.dependencyManagement1,this.dependencyManagement2).find("artifact")).isEqualTo(new Dependency("test","artifact","1.2.3"));
}
