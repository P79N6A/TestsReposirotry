@Test public void bootJarCanBePublished() throws FileNotFoundException, IOException {
  BuildResult result=this.gradleBuild.build("publish");
  assertThat(result.task(":publish").getOutcome()).isEqualTo(TaskOutcome.SUCCESS);
  assertThat(artifactWithSuffix("jar")).isFile();
  assertThat(artifactWithSuffix("pom")).is(pomWith().groupId("com.example").artifactId(this.gradleBuild.getProjectDir().getName()).version("1.0").noPackaging().noDependencies());
}
