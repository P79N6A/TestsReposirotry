@Test public void statusException(){
  this.contextRunner.withPropertyValues("server.error.include-exception=true").run((context) -> {
    WebTestClient client=WebTestClient.bindToApplicationContext(context).build();
    client.get().uri("/badRequest").exchange().expectStatus().isBadRequest().expectBody().jsonPath("status").isEqualTo("400").jsonPath("error").isEqualTo(HttpStatus.BAD_REQUEST.getReasonPhrase()).jsonPath("exception").isEqualTo(ResponseStatusException.class.getName());
  }
);
}
