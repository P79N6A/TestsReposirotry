@Test public void whenWhenCombinedWithAsUsesSourceValue(){
  Count<String> source=new Count<>(() -> "123");
  Long result=this.map.from(source).when("123"::equals).as(Integer::valueOf).when((v) -> v == 123).as(Integer::longValue).toInstance(Long::new);
  assertThat(result).isEqualTo(123);
  assertThat(source.getCount()).isOne();
}
