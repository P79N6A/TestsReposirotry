@Test public void nullsArePassedToTheOperationWhenPostRequestHasNoBody(){
  load(TestEndpointConfiguration.class,(context,client) -> {
    client.post().uri("/test").contentType(MediaType.APPLICATION_JSON).exchange().expectStatus().isNoContent().expectBody().isEmpty();
    verify(context.getBean(EndpointDelegate.class)).write(null,null);
  }
);
}
