@Test public void operationWithAQueryNamedPrincipalCanBeAccessedWhenAuthenticated(){
  load((context) -> {
    this.authenticatedContextCustomizer.accept(context);
    context.register(PrincipalQueryEndpointConfiguration.class);
  }
,(client) -> client.get().uri("/principalquery?principal=Zoe").accept(MediaType.APPLICATION_JSON).exchange().expectStatus().isOk().expectBody(String.class).isEqualTo("Zoe"));
}
