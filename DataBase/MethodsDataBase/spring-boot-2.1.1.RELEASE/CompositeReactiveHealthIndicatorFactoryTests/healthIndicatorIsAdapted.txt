@Test public void healthIndicatorIsAdapted(){
  ReactiveHealthIndicator healthIndicator=createHealthIndicator(Collections.singletonMap("test",() -> Mono.just(UP)),Collections.singletonMap("regular",() -> DOWN));
  StepVerifier.create(healthIndicator.health()).consumeNextWith((h) -> {
    assertThat(h.getStatus()).isEqualTo(Status.DOWN);
    assertThat(h.getDetails()).containsOnlyKeys("test","regular");
  }
).verifyComplete();
}
