@Test public void noHealthIndicator(){
  ReactiveHealthIndicator healthIndicator=createHealthIndicator(Collections.singletonMap("test",() -> Mono.just(UP)),null);
  StepVerifier.create(healthIndicator.health()).consumeNextWith((h) -> {
    assertThat(h.getStatus()).isEqualTo(Status.UP);
    assertThat(h.getDetails()).containsOnlyKeys("test");
  }
).verifyComplete();
}
