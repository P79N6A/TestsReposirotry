@Test public void convertWhenHasDelimiterNoneShouldConvert(){
  Data data=new Data();
  data.none=new String[]{"1","2","3"};
  String converted=(String)this.conversionService.convert(data.none,TypeDescriptor.nested(ReflectionUtils.findField(Data.class,"none"),0),TypeDescriptor.valueOf(String.class));
  assertThat(converted).isEqualTo("123");
}
