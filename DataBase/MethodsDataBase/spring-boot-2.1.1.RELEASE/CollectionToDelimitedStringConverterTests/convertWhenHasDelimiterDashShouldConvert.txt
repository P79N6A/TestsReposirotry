@Test public void convertWhenHasDelimiterDashShouldConvert(){
  Data data=new Data();
  data.dash=Arrays.asList("1","2","3");
  String converted=(String)this.conversionService.convert(data.dash,TypeDescriptor.nested(ReflectionUtils.findField(Data.class,"dash"),0),TypeDescriptor.valueOf(String.class));
  assertThat(converted).isEqualTo("1-2-3");
}
