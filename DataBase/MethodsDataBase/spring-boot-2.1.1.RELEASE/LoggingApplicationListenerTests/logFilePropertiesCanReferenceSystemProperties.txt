@Test public void logFilePropertiesCanReferenceSystemProperties(){
  TestPropertySourceUtils.addInlinedPropertiesToEnvironment(this.context,"logging.file=target/${PID}.log");
  this.initializer.initialize(this.context.getEnvironment(),this.context.getClassLoader());
  assertThat(System.getProperty(LoggingSystemProperties.LOG_FILE)).isEqualTo("target/" + new ApplicationPid().toString() + ".log");
}
