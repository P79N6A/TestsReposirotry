@Test public void overrideConfigDoesNotExist(){
  TestPropertySourceUtils.addInlinedPropertiesToEnvironment(this.context,"logging.config=doesnotexist.xml");
  assertThatIllegalStateException().isThrownBy(() -> {
    this.outputCapture.expect(containsString("Logging system failed to initialize using configuration from 'doesnotexist.xml'"));
    this.initializer.initialize(this.context.getEnvironment(),this.context.getClassLoader());
  }
);
}
