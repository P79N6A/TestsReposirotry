@Test public void getMissingEntryUrl() throws Exception {
  URL url=new URL(this.jarFile.getUrl(),"missing.dat");
  assertThat(url.toString()).isEqualTo("jar:" + this.rootJarFile.toURI() + "!/missing.dat");
  assertThatExceptionOfType(FileNotFoundException.class).isThrownBy(((JarURLConnection)url.openConnection())::getJarEntry);
}
