@Test public void testUserSpecifiedClassPathOrder() throws Exception {
  System.setProperty("loader.path","more-jars/app.jar,jars/app.jar");
  System.setProperty("loader.classLoader",URLClassLoader.class.getName());
  PropertiesLauncher launcher=new PropertiesLauncher();
  assertThat(ReflectionTestUtils.getField(launcher,"paths").toString()).isEqualTo("[more-jars/app.jar, jars/app.jar]");
  launcher.launch(new String[0]);
  waitFor("Hello Other World");
}
