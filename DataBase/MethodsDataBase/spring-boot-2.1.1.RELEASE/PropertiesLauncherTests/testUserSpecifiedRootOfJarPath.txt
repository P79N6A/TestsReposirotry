@Test public void testUserSpecifiedRootOfJarPath() throws Exception {
  System.setProperty("loader.path","jar:file:./src/test/resources/nested-jars/app.jar!/");
  PropertiesLauncher launcher=new PropertiesLauncher();
  assertThat(ReflectionTestUtils.getField(launcher,"paths").toString()).isEqualTo("[jar:file:./src/test/resources/nested-jars/app.jar!/]");
  List<Archive> archives=launcher.getClassPathArchives();
  assertThat(archives).areExactly(1,endingWith("foo.jar!/"));
  assertThat(archives).areExactly(1,endingWith("app.jar!/"));
}
