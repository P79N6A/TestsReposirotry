@Test public void jtaTransactionManagerTakesPrecedence(){
  this.contextRunner.withConfiguration(AutoConfigurations.of(DataSourceTransactionManagerAutoConfiguration.class)).run((context) -> {
    assertThat(context).hasSingleBean(DataSource.class);
    assertThat(context).hasSingleBean(JpaTransactionManager.class);
    assertThat(context).getBean("transactionManager").isInstanceOf(JpaTransactionManager.class);
  }
);
}
