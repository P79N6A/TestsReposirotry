@Test @SuppressWarnings("unchecked") public void convertWhenHasConvertibleElementTypeShouldReturnConvertedType(){
  if (this.conversionService instanceof ApplicationConversionService) {
    TypeDescriptor sourceType=TypeDescriptor.valueOf(String.class);
    TypeDescriptor targetType=TypeDescriptor.nested(ReflectionUtils.findField(Values.class,"convertibleElementType"),0);
    List<Integer> converted=(List<Integer>)this.conversionService.convert(" 1 |  2| 3  ",sourceType,targetType);
    assertThat(converted).containsExactly(1,2,3);
  }
}
