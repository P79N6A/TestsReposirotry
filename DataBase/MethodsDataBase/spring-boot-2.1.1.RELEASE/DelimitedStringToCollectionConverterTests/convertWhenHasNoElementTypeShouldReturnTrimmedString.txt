@Test @SuppressWarnings("unchecked") public void convertWhenHasNoElementTypeShouldReturnTrimmedString(){
  TypeDescriptor sourceType=TypeDescriptor.valueOf(String.class);
  TypeDescriptor targetType=TypeDescriptor.nested(ReflectionUtils.findField(Values.class,"noElementType"),0);
  Collection<String> converted=(Collection<String>)this.conversionService.convert(" a |  b| c  ",sourceType,targetType);
  assertThat(converted).containsExactly("a","b","c");
}
