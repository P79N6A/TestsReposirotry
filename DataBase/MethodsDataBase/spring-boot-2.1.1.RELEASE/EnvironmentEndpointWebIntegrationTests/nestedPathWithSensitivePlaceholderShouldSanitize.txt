@Test public void nestedPathWithSensitivePlaceholderShouldSanitize(){
  Map<String,Object> map=new HashMap<>();
  map.put("my.foo","${my.password}");
  map.put("my.password","hello");
  context.getEnvironment().getPropertySources().addFirst(new MapPropertySource("placeholder",map));
  client.get().uri("/actuator/env/my.foo").exchange().expectStatus().isOk().expectBody().jsonPath("property.value").isEqualTo("******").jsonPath(forPropertyEntry("placeholder")).isEqualTo("******");
}
