@Test public void nestedPathMatchedByRegexWithSensitivePlaceholderShouldSanitize(){
  Map<String,Object> map=new HashMap<>();
  map.put("my.foo","${my.password}");
  map.put("my.password","hello");
  context.getEnvironment().getPropertySources().addFirst(new MapPropertySource("placeholder",map));
  client.get().uri("/actuator/env?pattern=my.*").exchange().expectStatus().isOk().expectBody().jsonPath(forProperty("placeholder","my.foo")).isEqualTo("******");
}
