@Test public void nestedPathMatchedByRegexWhenPlaceholderCannotBeResolvedShouldReturnUnresolvedProperty(){
  Map<String,Object> map=new HashMap<>();
  map.put("my.foo","${my.bar}");
  context.getEnvironment().getPropertySources().addFirst(new MapPropertySource("unresolved-placeholder",map));
  client.get().uri("/actuator/env?pattern=my.*").exchange().expectStatus().isOk().expectBody().jsonPath("propertySources[?(@.name=='unresolved-placeholder')].properties.['my.foo'].value").isEqualTo("${my.bar}");
}
