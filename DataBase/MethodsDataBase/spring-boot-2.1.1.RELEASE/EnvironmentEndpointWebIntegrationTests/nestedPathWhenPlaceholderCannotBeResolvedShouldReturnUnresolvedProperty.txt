@Test public void nestedPathWhenPlaceholderCannotBeResolvedShouldReturnUnresolvedProperty(){
  Map<String,Object> map=new HashMap<>();
  map.put("my.foo","${my.bar}");
  context.getEnvironment().getPropertySources().addFirst(new MapPropertySource("unresolved-placeholder",map));
  client.get().uri("/actuator/env/my.foo").exchange().expectStatus().isOk().expectBody().jsonPath("property.value").isEqualTo("${my.bar}").jsonPath(forPropertyEntry("unresolved-placeholder")).isEqualTo("${my.bar}");
}
