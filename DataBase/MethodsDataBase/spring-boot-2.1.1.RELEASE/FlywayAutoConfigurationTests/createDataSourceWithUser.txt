@Test public void createDataSourceWithUser(){
  this.contextRunner.withUserConfiguration(EmbeddedDataSourceConfiguration.class).withPropertyValues("spring.datasource.url:jdbc:hsqldb:mem:normal","spring.flyway.user:sa").run((context) -> {
    assertThat(context).hasSingleBean(Flyway.class);
    assertThat(context.getBean(Flyway.class).getDataSource()).isNotNull();
  }
);
}
