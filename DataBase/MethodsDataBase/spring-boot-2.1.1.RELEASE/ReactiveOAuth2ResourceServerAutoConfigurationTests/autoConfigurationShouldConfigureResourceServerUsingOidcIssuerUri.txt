@Test public void autoConfigurationShouldConfigureResourceServerUsingOidcIssuerUri() throws IOException {
  this.server=new MockWebServer();
  this.server.start();
  String issuer=this.server.url("").toString();
  String cleanIssuerPath=cleanIssuerPath(issuer);
  setupMockResponse(cleanIssuerPath);
  this.contextRunner.withPropertyValues("spring.security.oauth2.resourceserver.jwt.issuer-uri=http://" + this.server.getHostName() + ":"+ this.server.getPort()).run((context) -> {
    assertThat(context.getBean(ReactiveJwtDecoder.class)).isInstanceOf(NimbusReactiveJwtDecoder.class);
    assertFilterConfiguredWithJwtAuthenticationManager(context);
  }
);
}
