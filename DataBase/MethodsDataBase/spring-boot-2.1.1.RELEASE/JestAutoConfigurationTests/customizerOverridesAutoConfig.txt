@Test public void customizerOverridesAutoConfig(){
  this.contextRunner.withUserConfiguration(BuilderCustomizer.class).withPropertyValues("spring.elasticsearch.jest.uris=http://localhost:9200").run((context) -> {
    JestHttpClient client=(JestHttpClient)context.getBean(JestClient.class);
    assertThat(client.getGson()).isSameAs(context.getBean(BuilderCustomizer.class).getGson());
  }
);
}
