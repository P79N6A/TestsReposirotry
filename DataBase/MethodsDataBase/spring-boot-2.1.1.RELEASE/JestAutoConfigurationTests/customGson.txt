@Test public void customGson(){
  this.contextRunner.withUserConfiguration(CustomGson.class).withPropertyValues("spring.elasticsearch.jest.uris=http://localhost:9200").run((context) -> {
    JestHttpClient client=(JestHttpClient)context.getBean(JestClient.class);
    assertThat(client.getGson()).isSameAs(context.getBean("customGson"));
  }
);
}
