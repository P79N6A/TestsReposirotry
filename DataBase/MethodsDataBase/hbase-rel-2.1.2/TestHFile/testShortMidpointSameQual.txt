@Test public void testShortMidpointSameQual(){
  Cell left=CellUtil.createCell(Bytes.toBytes("a"),Bytes.toBytes("a"),Bytes.toBytes("a"),11,KeyValue.Type.Maximum.getCode(),HConstants.EMPTY_BYTE_ARRAY);
  Cell right=CellUtil.createCell(Bytes.toBytes("a"),Bytes.toBytes("a"),Bytes.toBytes("a"),9,KeyValue.Type.Maximum.getCode(),HConstants.EMPTY_BYTE_ARRAY);
  Cell mid=HFileWriterImpl.getMidpoint(CellComparatorImpl.COMPARATOR,left,right);
  assertTrue(PrivateCellUtil.compareKeyIgnoresMvcc(CellComparatorImpl.COMPARATOR,left,mid) <= 0);
  assertTrue(PrivateCellUtil.compareKeyIgnoresMvcc(CellComparatorImpl.COMPARATOR,mid,right) == 0);
}
