@Test public void testUpdate(){
  HistogramImpl histogram=new HistogramImpl();

  histogram.update(0);
  assertEquals(1,histogram.getCount());
  histogram.update(10);
  assertEquals(2,histogram.getCount());

  histogram.update(30);
  assertEquals(4,histogram.getCount());
}
