/** 
 * Tests the case where if there is some data in the primary region, reopening the region replicas (enable/disable table, etc) makes the region replicas readable.
 * @throws IOException
 */
@Test public void testSecondaryRegionWithNonEmptyRegion() throws IOException {
  try (Connection connection=ConnectionFactory.createConnection(HTU.getConfiguration());Table table=connection.getTable(htd.getTableName())){
    HTU.loadNumericRows(table,fam,0,1000);
    HTU.getAdmin().disableTable(htd.getTableName());
    HTU.getAdmin().enableTable(htd.getTableName());
    HTU.verifyNumericRows(table,fam,0,1000,1);
  }
 }
