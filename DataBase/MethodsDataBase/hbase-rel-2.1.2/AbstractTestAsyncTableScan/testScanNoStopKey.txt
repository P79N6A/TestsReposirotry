@Test public void testScanNoStopKey() throws Exception {
  int start=345;
  List<Result> results=doScan(createScan().withStartRow(Bytes.toBytes(String.format("%03d",start))));
  assertEquals(COUNT - start,results.size());
  IntStream.range(0,COUNT - start).forEach(i -> assertResultEquals(results.get(i),start + i));
}
