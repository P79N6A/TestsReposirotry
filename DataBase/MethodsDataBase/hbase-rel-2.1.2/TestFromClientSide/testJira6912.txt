@Test public void testJira6912() throws Exception {
  final TableName tableName=TableName.valueOf(name.getMethodName());
  Table foo=TEST_UTIL.createTable(tableName,new byte[][]{FAMILY},10);
  List<Put> puts=new ArrayList<Put>();
  for (int i=0; i != 100; i++) {
    Put put=new Put(Bytes.toBytes(i));
    put.addColumn(FAMILY,FAMILY,Bytes.toBytes(i));
    puts.add(put);
  }
  foo.put(puts);
  TEST_UTIL.flush();
  Scan scan=new Scan();
  scan.setStartRow(Bytes.toBytes(1));
  scan.setStopRow(Bytes.toBytes(3));
  scan.addColumn(FAMILY,FAMILY);
  scan.setFilter(new RowFilter(CompareOperator.NOT_EQUAL,new BinaryComparator(Bytes.toBytes(1))));
  ResultScanner scanner=foo.getScanner(scan);
  Result[] bar=scanner.next(100);
  assertEquals(1,bar.length);
}
