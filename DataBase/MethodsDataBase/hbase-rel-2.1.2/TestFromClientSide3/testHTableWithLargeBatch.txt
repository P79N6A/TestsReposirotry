@Test public void testHTableWithLargeBatch() throws Exception {
  Table table=TEST_UTIL.createTable(TableName.valueOf(name.getMethodName()),new byte[][]{FAMILY});
  int sixtyFourK=64 * 1024;
  try {
    List actions=new ArrayList();
    Object[] results=new Object[(sixtyFourK + 1) * 2];
    for (int i=0; i < sixtyFourK + 1; i++) {
      Put put1=new Put(ROW);
      put1.addColumn(FAMILY,QUALIFIER,VALUE);
      actions.add(put1);
      Put put2=new Put(ANOTHERROW);
      put2.addColumn(FAMILY,QUALIFIER,VALUE);
      actions.add(put2);
    }
    table.batch(actions,results);
  }
  finally {
    table.close();
  }
}
