@Test public void testCheckAndDelete() throws IOException {
  Get get=new Get(ROW_1);
  Result result=remoteTable.get(get);
  byte[] value1=result.getValue(COLUMN_1,QUALIFIER_1);
  byte[] value2=result.getValue(COLUMN_2,QUALIFIER_2);
  assertNotNull(value1);
  assertTrue(Bytes.equals(VALUE_1,value1));
  assertNull(value2);
  assertTrue(remoteTable.exists(get));
  assertEquals(1,remoteTable.existsAll(Collections.singletonList(get)).length);
  Delete delete=new Delete(ROW_1);
  remoteTable.checkAndMutate(ROW_1,COLUMN_1).qualifier(QUALIFIER_1).ifEquals(VALUE_1).thenDelete(delete);
  assertFalse(remoteTable.exists(get));
  Put put=new Put(ROW_1);
  put.addColumn(COLUMN_1,QUALIFIER_1,VALUE_1);
  remoteTable.put(put);
  assertTrue(remoteTable.checkAndMutate(ROW_1,COLUMN_1).qualifier(QUALIFIER_1).ifEquals(VALUE_1).thenPut(put));
  assertFalse(remoteTable.checkAndMutate(ROW_1,COLUMN_1).qualifier(QUALIFIER_1).ifEquals(VALUE_2).thenPut(put));
}
