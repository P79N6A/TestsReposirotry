  c.set(HConstants.BUCKET_CACHE_IOENGINE_KEY,"offheap");

  try {
    CacheConfig.getBucketCache(c);
    fail("Should throw IllegalArgumentException when passing illegal value for bucket size");
  }
 catch (  IllegalArgumentException e) {
  }
}
