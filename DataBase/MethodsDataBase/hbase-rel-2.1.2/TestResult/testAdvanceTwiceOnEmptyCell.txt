public void testAdvanceTwiceOnEmptyCell() throws IOException {

  assertFalse(r.advance());
  try {
    r.advance();
    fail("NoSuchElementException should have been thrown!");
  }
 catch (  NoSuchElementException ex) {
    LOG.debug("As expected: " + ex.getMessage());
  }
}
