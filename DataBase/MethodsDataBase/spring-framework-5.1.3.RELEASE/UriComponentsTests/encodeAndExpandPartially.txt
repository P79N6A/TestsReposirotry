@Test public void encodeAndExpandPartially(){
  UriComponents uri=UriComponentsBuilder.fromPath("/hotel list/{city} specials").queryParam("q","{value}").encode().uriVariables(Collections.singletonMap("city","Z\u00fcrich")).build();
  assertEquals("/hotel%20list/Z%C3%BCrich%20specials?q=a%2Bb",uri.expand("a+b").toString());
}
