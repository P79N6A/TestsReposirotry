@Test public void fromMethodWithPrefix(){
  initWebApplicationContext(PathPrefixWebConfig.class);
  this.request.setServerName("example.org");
  this.request.setServerPort(9999);
  this.request.setContextPath("/base");
  assertEquals("http://example.org:9999/base/api/people/123/addresses/DE",fromMethodCall(on(PersonsAddressesController.class).getAddressesForCountry("DE")).buildAndExpand("123").toString());
}
