@Test public void shouldNotDoubleEncode() throws Exception {
  HttpHeaders headers=new HttpHeaders();
  headers.add("Forwarded","host=84.198.58.199;proto=https");
  ServerHttpRequest request=MockServerHttpRequest.method(HttpMethod.GET,new URI("http://example.com/a%20b?q=a%2Bb")).headers(headers).build();
  request=this.requestMutator.apply(request);
  assertEquals(new URI("https://84.198.58.199/a%20b?q=a%2Bb"),request.getURI());
  assertForwardedHeadersRemoved(request);
}
