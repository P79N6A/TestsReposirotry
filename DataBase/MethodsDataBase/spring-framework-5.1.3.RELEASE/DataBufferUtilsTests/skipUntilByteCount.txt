@Test public void skipUntilByteCount(){
  Flux<DataBuffer> source=Flux.concat(deferStringBuffer("foo"),deferStringBuffer("bar"),deferStringBuffer("baz"));
  Flux<DataBuffer> result=DataBufferUtils.skipUntilByteCount(source,5L);
  StepVerifier.create(result).consumeNextWith(stringConsumer("r")).consumeNextWith(stringConsumer("baz")).expectComplete().verify(Duration.ofSeconds(5));
}
