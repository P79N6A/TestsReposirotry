@Test public void takeUntilByteCountExact(){
  Flux<DataBuffer> source=Flux.concat(deferStringBuffer("foo"),deferStringBuffer("bar"),deferStringBuffer("baz"));
  Flux<DataBuffer> result=DataBufferUtils.takeUntilByteCount(source,6L);

}
