@Test public void parseInvalidDates(){
  ContentDisposition disposition=ContentDisposition.parse("attachment; creation-date=\"-1\"; modification-date=\"-1\"; " + "read-date=\"Wed, 14 Feb 2007 10:15:30 -0500\"");
  DateTimeFormatter formatter=DateTimeFormatter.RFC_1123_DATE_TIME;
  assertEquals(ContentDisposition.builder("attachment").readDate(ZonedDateTime.parse("Wed, 14 Feb 2007 10:15:30 -0500",formatter)).build(),disposition);
}
