@Test public void customHeader() throws Exception {
  HttpHeaders headers=new HttpHeaders();
  headers.add("my-header","my-value");
  MonoProcessor<Object> output=MonoProcessor.create();
  this.client.execute(getUrl("/custom-header"),headers,session -> session.receive().map(WebSocketMessage::getPayloadAsText).subscribeWith(output).then()).block(TIMEOUT);
  assertEquals("my-header:my-value",output.block(TIMEOUT));
}
