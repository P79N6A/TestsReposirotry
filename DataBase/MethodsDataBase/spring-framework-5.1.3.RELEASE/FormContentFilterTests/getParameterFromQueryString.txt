@Test public void getParameterFromQueryString() throws Exception {
  this.request.addParameter("name","value1");
  this.request.setContent("name=value2".getBytes("ISO-8859-1"));
  this.filter.doFilter(this.request,this.response,this.filterChain);
  assertNotSame("Request not wrapped",this.request,this.filterChain.getRequest());
  assertEquals("Query string parameters should be listed ahead of form parameters","value1",this.filterChain.getRequest().getParameter("name"));
}
