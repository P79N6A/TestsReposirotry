/** 
 * We'll typically align by indenting with tabs or spaces. These should be ignored if at the beginning of a line. We must ensure that comment lines beginning with whitespace are still ignored: The standard syntax doesn't allow this on JDK 1.3.
 */
@Test public void ignoresLeadingSpacesAndTabs(){
  String s="    #Ignore this comment\n" + "\t\tfoo=bar\n" + "\t#Another comment more junk \n"+ " me=mi\n"+ "x=x\n"+ "\n";
  PropertiesEditor pe=new PropertiesEditor();
  pe.setAsText(s);
  Properties p=(Properties)pe.getValue();
  assertTrue("contains 3 entries, not " + p.size(),p.size() == 3);
  assertTrue("foo is bar",p.get("foo").equals("bar"));
  assertTrue("me=mi",p.get("me").equals("mi"));
}
