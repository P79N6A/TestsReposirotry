@Test(expected=TypeMismatchDataAccessException.class) public void doesNotUseConversionService() throws SQLException {
  SingleColumnRowMapper<LocalDateTime> rowMapper=SingleColumnRowMapper.newInstance(LocalDateTime.class,null);
  ResultSet resultSet=mock(ResultSet.class);
  ResultSetMetaData metaData=mock(ResultSetMetaData.class);
  given(metaData.getColumnCount()).willReturn(1);
  given(resultSet.getMetaData()).willReturn(metaData);
  given(resultSet.getObject(1,LocalDateTime.class)).willThrow(new SQLFeatureNotSupportedException());
  given(resultSet.getTimestamp(1)).willReturn(new Timestamp(0));
  rowMapper.mapRow(resultSet,1);
}
