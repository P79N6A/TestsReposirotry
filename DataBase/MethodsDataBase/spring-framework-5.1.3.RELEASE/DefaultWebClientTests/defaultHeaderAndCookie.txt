@Test public void defaultHeaderAndCookie(){
  WebClient client=this.builder.defaultHeader("Accept","application/json").defaultCookie("id","123").build();
  client.get().uri("/path").exchange();
  ClientRequest request=verifyAndGetRequest();
  assertEquals("application/json",request.headers().getFirst("Accept"));
  assertEquals("123",request.cookies().getFirst("id"));
  verifyNoMoreInteractions(this.exchangeFunction);
}
