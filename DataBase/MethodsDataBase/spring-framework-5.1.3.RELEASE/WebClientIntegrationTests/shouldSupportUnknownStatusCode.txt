@Test public void shouldSupportUnknownStatusCode(){
  int errorStatus=555;
  assertNull(HttpStatus.resolve(errorStatus));
  String errorMessage="Something went wrong";
  prepareResponse(response -> response.setResponseCode(errorStatus).setHeader("Content-Type","text/plain").setBody(errorMessage));
  Mono<ClientResponse> result=this.webClient.get().uri("/unknownPage").exchange();
  StepVerifier.create(result).consumeNextWith(response -> assertEquals(555,response.rawStatusCode())).expectComplete().verify(Duration.ofSeconds(3));
  expectRequestCount(1);
  expectRequest(request -> {
    assertEquals("*/*",request.getHeader(HttpHeaders.ACCEPT));
    assertEquals("/unknownPage",request.getPath());
  }
);
}
