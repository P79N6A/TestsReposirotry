/** 
 * When inheritance is not involved, it is still possible to override a bean method from the container's point of view. This is not strictly 'overloading' of a method per se, so it's referred to here as 'shadowing' to distinguish the difference.
 */
@Test public void beanMethodShadowing(){
  AnnotationConfigApplicationContext ctx=new AnnotationConfigApplicationContext(ShadowConfig.class);
  assertThat(ctx.getBean(String.class),equalTo("shadow"));
}
