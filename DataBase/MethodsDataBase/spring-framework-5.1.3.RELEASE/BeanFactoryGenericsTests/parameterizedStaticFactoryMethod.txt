/** 
 * Tests support for parameterized static  {@code factory-method} declarations such asMockito's  {@code mock()} method which has the following signature.<pre> {@code public static <T> T mock(Class<T> classToMock)}</pre> <p>See SPR-9493
 */
@Test public void parameterizedStaticFactoryMethod(){
  RootBeanDefinition rbd=new RootBeanDefinition(Mockito.class);
  rbd.setFactoryMethodName("mock");
  rbd.getConstructorArgumentValues().addGenericArgumentValue(Runnable.class);
  DefaultListableBeanFactory bf=new DefaultListableBeanFactory();
  bf.registerBeanDefinition("mock",rbd);
  assertEquals(Runnable.class,bf.getType("mock"));
  assertEquals(Runnable.class,bf.getType("mock"));
  Map<String,Runnable> beans=bf.getBeansOfType(Runnable.class);
  assertEquals(1,beans.size());
}
