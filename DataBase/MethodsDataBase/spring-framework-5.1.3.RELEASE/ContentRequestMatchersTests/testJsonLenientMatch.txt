@Test public void testJsonLenientMatch() throws Exception {
  String content="{\n \"foo array\":[\"first\",\"second\"] , \"someExtraProperty\": \"which is allowed\" \n}";
  this.request.getBody().write(content.getBytes());
  MockRestRequestMatchers.content().json("{\n \"foo array\":[\"second\",\"first\"] \n}").match(this.request);
  MockRestRequestMatchers.content().json("{\n \"foo array\":[\"second\",\"first\"] \n}",false).match(this.request);
}
