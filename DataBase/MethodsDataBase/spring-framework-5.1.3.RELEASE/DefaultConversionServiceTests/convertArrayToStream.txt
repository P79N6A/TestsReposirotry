@Test public void convertArrayToStream() throws Exception {
  String[] source={"1","3","4"};
  @SuppressWarnings("unchecked") Stream<Integer> result=(Stream<Integer>)this.conversionService.convert(source,TypeDescriptor.valueOf(String[].class),new TypeDescriptor(getClass().getDeclaredField("genericStream")));
  assertEquals(8,result.mapToInt((x) -> x).sum());
}
