@Test public void checkNotModifiedETagStrongWeak(){
  String etag="\"Foo\"";
  servletRequest.addHeader("If-None-Match",String.format("W/%s",etag));
  assertTrue(request.checkNotModified(etag));
  assertEquals(304,servletResponse.getStatus());

}
