@Test public void writeAndAutoFlushOnComplete(){
  Mono<String> result=this.webClient.get().uri("/write-and-complete").retrieve().bodyToMono(String.class);
  try {
    StepVerifier.create(result).consumeNextWith(value -> assertEquals(64 * 1024,value.length())).expectComplete().verify(Duration.ofSeconds(10L));
  }
 catch (  AssertionError err) {
    String os=System.getProperty("os.name").toLowerCase();
    if (os.contains("windows") && err.getMessage() != null && err.getMessage().startsWith("VerifySubscriber timed out")) {
      err.printStackTrace();
      return;
    }
    throw err;
  }
}
