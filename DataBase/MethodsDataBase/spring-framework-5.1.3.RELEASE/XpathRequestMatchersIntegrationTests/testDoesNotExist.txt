@Test public void testDoesNotExist() throws Exception {
  String composer="/ns:people/composers/composer[%s]";
  String performer="/ns:people/performers/performer[%s]";
  this.mockServer.expect(requestTo("/composers")).andExpect(content().contentType("application/xml")).andExpect(xpath(composer,NS,0).doesNotExist()).andExpect(xpath(composer,NS,5).doesNotExist()).andExpect(xpath(performer,NS,0).doesNotExist()).andExpect(xpath(performer,NS,3).doesNotExist()).andRespond(withSuccess());
  executeAndVerify();
}
