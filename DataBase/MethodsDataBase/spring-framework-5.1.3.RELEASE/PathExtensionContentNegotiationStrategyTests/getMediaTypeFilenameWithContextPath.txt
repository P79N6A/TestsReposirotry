@Test public void getMediaTypeFilenameWithContextPath() throws Exception {
  PathExtensionContentNegotiationStrategy strategy=new PathExtensionContentNegotiationStrategy();
  this.servletRequest.setContextPath("/project-1.0.0.M3");
  this.servletRequest.setRequestURI("/project-1.0.0.M3/");
  assertEquals("Context path should be excluded",ContentNegotiationStrategy.MEDIA_TYPE_ALL_LIST,strategy.resolveMediaTypes(webRequest));
  this.servletRequest.setRequestURI("/project-1.0.0.M3");
  assertEquals("Context path should be excluded",ContentNegotiationStrategy.MEDIA_TYPE_ALL_LIST,strategy.resolveMediaTypes(webRequest));
}
