@Test public void preferredLocalesFromAcceptLanguageHeader(){
  String headerValue="fr-ch, fr;q=0.9, en-*;q=0.8, de;q=0.7, *;q=0.5";
  request.addHeader("Accept-Language",headerValue);
  List<Locale> actual=Collections.list(request.getLocales());
  assertEquals(Arrays.asList(Locale.forLanguageTag("fr-ch"),Locale.forLanguageTag("fr"),Locale.forLanguageTag("en"),Locale.forLanguageTag("de")),actual);
}
