@Test public void propertyAccessWithInstanceMethodResolverAndTypedRootObject(){
  Person target=new Person("p1");
  EvaluationContext context=SimpleEvaluationContext.forReadOnlyDataBinding().withInstanceMethods().withTypedRootObject(target,TypeDescriptor.valueOf(Object.class)).build();
  assertEquals("1",parser.parseExpression("name.substring(1)").getValue(context,target));
  assertSame(target,context.getRootObject().getValue());
  assertSame(Object.class,context.getRootObject().getTypeDescriptor().getType());
}
