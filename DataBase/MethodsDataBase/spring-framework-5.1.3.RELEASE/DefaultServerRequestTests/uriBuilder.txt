@Test public void uriBuilder() throws URISyntaxException {
  URI uri=new URI("http","localhost","/path","a=1",null);
  DefaultServerRequest request=new DefaultServerRequest(MockServerWebExchange.from(MockServerHttpRequest.method(HttpMethod.GET,uri)),this.messageReaders);
  URI result=request.uriBuilder().build();
  assertEquals("http",result.getScheme());
  assertEquals("localhost",result.getHost());
  assertEquals(-1,result.getPort());
  assertEquals("/path",result.getPath());
  assertEquals("a=1",result.getQuery());
}
