@Test public void pathContainer(){
  assertEquals("[/][abc][/][def]",elementsToString(toPathContainer("/abc/def").elements()));
  assertEquals("[abc][/][def]",elementsToString(toPathContainer("abc/def").elements()));
  assertEquals("[abc][/][def][/]",elementsToString(toPathContainer("abc/def/").elements()));
  assertEquals("[abc][/][/][def][/][/]",elementsToString(toPathContainer("abc//def//").elements()));
  assertEquals("[/]",elementsToString(toPathContainer("/").elements()));
  assertEquals("[/][/][/]",elementsToString(toPathContainer("///").elements()));
}
