@Test public void caseInsensitive(){
  PathPatternParser pp=new PathPatternParser();
  pp.setCaseSensitive(false);
  PathPattern p=pp.parse("/group/{groupName}/members");
  assertMatches(p,"/group/sales/members");
  assertMatches(p,"/Group/Sales/Members");
  assertMatches(p,"/group/Sales/members");
}
