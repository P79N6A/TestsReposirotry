@Test public void multipleSeparatorsInPatternAndPath(){
  checkNoMatch("///one///two///three","//one/////two///////three");
  checkMatches("//one/////two///////three","//one/////two///////three");
  checkNoMatch("//one//two//three","/one/////two/three");
  checkMatches("/one/////two/three","/one/////two/three");
  checkCapture("///{foo}///bar","///one///bar","foo","one");
}
