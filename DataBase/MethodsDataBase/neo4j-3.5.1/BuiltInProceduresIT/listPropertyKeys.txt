@Test public void listPropertyKeys() throws Throwable {
  TokenWrite ops=tokenWriteInNewTransaction();
  ops.propertyKeyGetOrCreateForName("MyProp");
  commit();
  RawIterator<Object[],ProcedureException> stream=procs().procedureCallRead(procs().procedureGet(procedureName("db","propertyKeys")).id(),new Object[0]);
  assertThat(asList(stream),contains(equalTo(new Object[]{"MyProp"})));
}
