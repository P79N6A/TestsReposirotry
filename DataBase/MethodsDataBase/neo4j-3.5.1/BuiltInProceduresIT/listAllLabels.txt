@Test public void listAllLabels() throws Throwable {
  Transaction transaction=newTransaction(AnonymousContext.writeToken());
  long nodeId=transaction.dataWrite().nodeCreate();
  int labelId=transaction.tokenWrite().labelGetOrCreateForName("MyLabel");
  transaction.dataWrite().nodeAddLabel(nodeId,labelId);
  commit();
  RawIterator<Object[],ProcedureException> stream=procs().procedureCallRead(procs().procedureGet(procedureName("db","labels")).id(),new Object[0]);
  assertThat(asList(stream),contains(equalTo(new Object[]{"MyLabel"})));
}
