@Test public void shouldCloseMultiPopulatorOnSuccessfulPopulation(){
  NullLogProvider logProvider=NullLogProvider.getInstance();
  TrackingMultipleIndexPopulator populator=new TrackingMultipleIndexPopulator(IndexStoreView.EMPTY,logProvider,EntityType.NODE,new DatabaseSchemaState(logProvider));
  IndexPopulationJob populationJob=new IndexPopulationJob(populator,NO_MONITOR,false);
  populationJob.run();
  assertTrue(populator.closed);
}
