@Documented("Get all nodes with a label.") @Test @GraphDescription.Graph(nodes={@NODE(name="Clint Eastwood",setNameProperty=true,labels={@LABEL("Actor"),@LABEL("Director")}),@NODE(name="Donald Sutherland",setNameProperty=true,labels={@LABEL("Actor")}),@NODE(name="Steven Spielberg",setNameProperty=true,labels={@LABEL("Director")})}) public void get_all_nodes_with_label() throws JsonParseException {
  data.get();
  String uri=getNodesWithLabelUri("Actor");
  String body=gen.get().expectedStatus(200).get(uri).entity();
  List<?> parsed=(List<?>)readJson(body);
  assertEquals(asSet("Clint Eastwood","Donald Sutherland"),Iterables.asSet(map(getProperty("name",String.class),parsed)));
}
