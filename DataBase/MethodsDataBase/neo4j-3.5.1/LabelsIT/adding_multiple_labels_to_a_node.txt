@Documented("Adding multiple labels to a node.") @Test @GraphDescription.Graph(nodes={@NODE(name="Clint Eastwood",setNameProperty=true)}) public void adding_multiple_labels_to_a_node(){
  Map<String,Node> nodes=data.get();
  String nodeUri=getNodeUri(nodes.get("Clint Eastwood"));
  gen.get().expectedStatus(204).payload(createJsonFrom(new String[]{"Person","Actor"})).post(nodeUri + "/labels");
  assertThat(nodes.get("Clint Eastwood"),inTx(graphdb(),hasLabels("Person","Actor")));
}
