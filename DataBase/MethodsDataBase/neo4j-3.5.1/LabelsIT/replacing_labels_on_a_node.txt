@Documented("Replacing labels on a node.\n" + "\n" + "This removes any labels currently on a node, and replaces them with the labels passed in as the\n"+ "request body.") @Test @GraphDescription.Graph(nodes={@NODE(name="Clint Eastwood",setNameProperty=true,labels={@LABEL("Person")})}) public void replacing_labels_on_a_node(){
  Map<String,Node> nodes=data.get();
  String nodeUri=getNodeUri(nodes.get("Clint Eastwood"));
  gen.get().expectedStatus(204).payload(createJsonFrom(new String[]{"Actor","Director"})).put(nodeUri + "/labels");
  assertThat(nodes.get("Clint Eastwood"),inTx(graphdb(),hasLabels("Actor","Director")));
}
