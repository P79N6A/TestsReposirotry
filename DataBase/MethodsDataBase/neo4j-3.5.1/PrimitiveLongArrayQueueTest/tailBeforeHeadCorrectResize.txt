@Test void tailBeforeHeadCorrectResize(){
  PrimitiveLongArrayQueue queue=createQueue();
  for (int i=0; i < 14; i++) {
    queue.enqueue(i);
  }
  for (int i=0; i < 10; i++) {
    assertEquals(i,queue.dequeue());
  }
  for (int i=14; i < 34; i++) {
    queue.enqueue(i);
  }
  assertEquals(24,queue.size());
  for (int j=10; j < 34; j++) {
    assertEquals(j,queue.dequeue());
  }
}
