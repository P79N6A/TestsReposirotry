@Test void freeze(){
  set.addAll(1,2,3,100,200,300);
  final LongSet frozen=set.freeze();
  assertEquals(set,frozen);
  assertEquals(6,set.size());
  assertEquals(6,frozen.size());
  set.removeAll(1,100);
  assertNotEquals(set,frozen);
  assertEquals(4,set.size());
  assertEquals(6,frozen.size());
}
