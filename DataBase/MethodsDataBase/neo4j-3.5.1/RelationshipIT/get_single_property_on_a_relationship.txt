@Test @Graph(nodes={@NODE(name="Romeo",setNameProperty=true),@NODE(name="Juliet",setNameProperty=true)},relationships={@REL(start="Romeo",end="Juliet",type="LOVES",properties={@PROP(key="cost",value="high",type=GraphDescription.PropType.STRING)})}) public void get_single_property_on_a_relationship(){
  Relationship loves=getFirstRelationshipFromRomeoNode();
  String response=gen().expectedStatus(ClientResponse.Status.OK).get(getRelPropURI(loves,"cost")).entity();
  assertTrue(response.contains("high"));
}
