@Test public void shouldFailWhenZonedDateTimeZoneIdIsNotKnown() throws Exception {

    packer.pack(Values.of("Europe/Marmaris"));
  }
,"Unable to construct ZonedDateTime value: `Unknown time-zone ID: Europe/Marmaris`");
