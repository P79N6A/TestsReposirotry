@Test public void shouldAllowGraphCheckDisabled() throws ConsistencyCheckIncompleteException {
  GraphDatabaseService gds=getGraphDatabaseService();
  try (Transaction tx=gds.beginTx()){
    gds.createNode();
    tx.success();
  }
   gds.shutdown();
  ConsistencyCheckService service=new ConsistencyCheckService();
  Config configuration=Config.defaults(settings(ConsistencyCheckSettings.consistency_check_graph.name(),Settings.FALSE));
  Result result=runFullConsistencyCheck(service,configuration);
  assertTrue(result.isSuccessful());
}
