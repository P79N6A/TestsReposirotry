@Test void shouldWriteDuration(){
  for (  DurationValue duration : new DurationValue[]{duration(0,0,0,0),duration(1,0,0,0),duration(0,1,0,0),duration(0,0,1,0),duration(0,0,0,1)}) {
    List<DurationValue> values=new ArrayList<>(1);
    ValueWriter<RuntimeException> writer=new ThrowingValueWriter.AssertOnly(){
      @Override public void writeDuration(      long months,      long days,      long seconds,      int nanos){
        values.add(duration(months,days,seconds,nanos));
      }
    }
;
    duration.writeTo(writer);
    assertEquals(singletonList(duration),values);
  }
}
