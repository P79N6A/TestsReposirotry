@Test void shouldConcatSeveralArrays(){
  Integer[] a={0,1,2};
  Integer[] b={3,4};
  Integer[] c={5,6,7,8};
  Integer[] result=ArrayUtil.concatArrays(a,b,c);
  assertEquals(a.length + b.length + c.length,result.length);
  for (int i=0; i < result.length; i++) {
    assertEquals((Integer)i,result[i]);
  }
}
