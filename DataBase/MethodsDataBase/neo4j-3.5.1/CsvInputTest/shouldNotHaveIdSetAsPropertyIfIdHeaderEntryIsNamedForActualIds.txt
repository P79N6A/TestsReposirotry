@Test public void shouldNotHaveIdSetAsPropertyIfIdHeaderEntryIsNamedForActualIds() throws Exception {
  DataFactory data=data("myId:ID,name:string,level:int\n" + "0,Mattias,1\n" + "1,Johan,2\n");
  Iterable<DataFactory> dataIterable=dataIterable(data);
  Input input=new CsvInput(dataIterable,defaultFormatNodeFileHeader(),datas(),defaultFormatRelationshipFileHeader(),IdType.ACTUAL,config(COMMAS),silentBadCollector(0));
  try (InputIterator nodes=input.nodes().iterator()){
    assertNextNode(nodes,0L,new Object[]{"name","Mattias","level",1},labels());
    assertNextNode(nodes,1L,new Object[]{"name","Johan","level",2},labels());
    assertFalse(readNext(nodes));
  }
 }
