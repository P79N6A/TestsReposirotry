@Test public void shouldCopeWithLinesThatHasTooFewValuesButStillValidates() throws Exception {
  Iterable<DataFactory> data=dataIterable(data("1,ultralisk,ZERG,10\n" + "2,corruptor,ZERG\n" + "3,mutalisk,ZERG,3"));
  Input input=new CsvInput(data,header(entry(null,Type.ID,extractors.long_()),entry("unit",Type.PROPERTY,extractors.string()),entry("type",Type.LABEL,extractors.string()),entry("kills",Type.PROPERTY,extractors.int_())),datas(),defaultFormatRelationshipFileHeader(),IdType.ACTUAL,config(COMMAS),silentBadCollector(0));
  try (InputIterator nodes=input.nodes().iterator()){
    assertNextNode(nodes,1L,new Object[]{"unit","ultralisk","kills",10},labels("ZERG"));
    assertNextNode(nodes,2L,new Object[]{"unit","corruptor"},labels("ZERG"));
    assertNextNode(nodes,3L,new Object[]{"unit","mutalisk","kills",3},labels("ZERG"));
    assertFalse(readNext(nodes));
  }
 }
