@Test public void shouldNotIncludeEmptyArraysInEntities() throws Exception {
  Iterable<DataFactory> data=DataFactories.datas(CsvInputTest.data(":ID,sprop:String[],lprop:long[]\n" + "1,,\n" + "2,a;b,10;20"));
  Input input=new CsvInput(data,defaultFormatNodeFileHeader(),datas(),defaultFormatNodeFileHeader(),IdType.INTEGER,config(COMMAS),silentBadCollector(0));
  try (InputIterator nodes=input.nodes().iterator()){
    assertNextNode(nodes,1L,NO_PROPERTIES,labels());
    assertNextNode(nodes,2L,properties("sprop",new String[]{"a","b"},"lprop",new long[]{10,20}),labels());
    assertFalse(readNext(nodes));
  }
 }
