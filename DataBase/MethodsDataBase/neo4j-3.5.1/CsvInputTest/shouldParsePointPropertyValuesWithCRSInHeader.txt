@Test public void shouldParsePointPropertyValuesWithCRSInHeader() throws Exception {
  DataFactory data=data(":ID,name,point:Point{crs:WGS-84-3D}\n" + "0,Johan,\" { height :0.01 ,longitude:5, latitude : -4.2 } \"\n");
  Iterable<DataFactory> dataIterable=dataIterable(data);
  Input input=new CsvInput(dataIterable,defaultFormatNodeFileHeader(),datas(),defaultFormatRelationshipFileHeader(),IdType.ACTUAL,config(COMMAS),silentBadCollector(0));
  try (InputIterator nodes=input.nodes().iterator()){
    assertNextNode(nodes,0L,new Object[]{"name","Johan","point",Values.pointValue(CoordinateReferenceSystem.WGS84_3D,5,-4.2,0.01)},labels());
    assertFalse(readNext(nodes));
  }
 }
