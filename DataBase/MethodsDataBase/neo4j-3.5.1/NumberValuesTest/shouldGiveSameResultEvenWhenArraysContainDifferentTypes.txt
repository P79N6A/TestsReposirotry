@Test void shouldGiveSameResultEvenWhenArraysContainDifferentTypes(){
  int[] ints=new int[32];
  long[] longs=new long[32];
  Random r=ThreadLocalRandom.current();
  for (int i=0; i < 32; i++) {
    int nextInt=r.nextInt();
    ints[i]=nextInt;
    longs[i]=nextInt;
  }
  assertThat(hash(ints),equalTo(hash(longs)));
}
