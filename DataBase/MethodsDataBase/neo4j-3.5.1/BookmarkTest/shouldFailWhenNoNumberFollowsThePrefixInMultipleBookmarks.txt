@Test void shouldFailWhenNoNumberFollowsThePrefixInMultipleBookmarks(){
  String bookmarkString="neo4j:bookmark:v1:tx10";
  String wrongBookmarkString="neo4j:bookmark:v1:tx";
  BookmarkFormatException e=assertThrows(BookmarkFormatException.class,() -> Bookmark.fromParamsOrNull(singletonMap("bookmarks",asList(bookmarkString,wrongBookmarkString))));
  assertTrue(e.causesFailureMessage());
}
