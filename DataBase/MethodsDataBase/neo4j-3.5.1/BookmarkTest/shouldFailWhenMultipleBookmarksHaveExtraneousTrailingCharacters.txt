@Test void shouldFailWhenMultipleBookmarksHaveExtraneousTrailingCharacters(){
  String bookmarkString="neo4j:bookmark:v1:tx1234";
  String wrongBookmarkString="neo4j:bookmark:v1:tx1234supercalifragilisticexpialidocious";
  BookmarkFormatException e=assertThrows(BookmarkFormatException.class,() -> Bookmark.fromParamsOrNull(singletonMap("bookmarks",asList(bookmarkString,wrongBookmarkString))));
  assertTrue(e.causesFailureMessage());
}
