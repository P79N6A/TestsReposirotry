@Test public void mustRescheduleRecurringTasks() throws Exception {
  scheduler.submit(Group.STORAGE_MAINTENANCE,semaphore::release,100,100);
  clock.forward(100,TimeUnit.NANOSECONDS);
  scheduler.tick();
  assertSemaphoreAcquire();
  clock.forward(100,TimeUnit.NANOSECONDS);
  scheduler.tick();
  assertSemaphoreAcquire();
}
