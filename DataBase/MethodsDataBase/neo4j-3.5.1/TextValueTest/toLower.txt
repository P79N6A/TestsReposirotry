@Test public void toLower(){
  assertThat(value.apply("HELLO").toLower(),equalTo(value.apply("hello")));
  assertThat(value.apply("Hello").toLower(),equalTo(value.apply("hello")));
  assertThat(value.apply("hello").toLower(),equalTo(value.apply("hello")));
  assertThat(value.apply("").toLower(),equalTo(value.apply("")));
}
