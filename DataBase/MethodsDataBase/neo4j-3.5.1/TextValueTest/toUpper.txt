@Test public void toUpper(){
  assertThat(value.apply("HELLO").toUpper(),equalTo(value.apply("HELLO")));
  assertThat(value.apply("Hello").toUpper(),equalTo(value.apply("HELLO")));
  assertThat(value.apply("hello").toUpper(),equalTo(value.apply("HELLO")));
  assertThat(value.apply("").toUpper(),equalTo(value.apply("")));
}
