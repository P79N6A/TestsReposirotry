@Test public void trim(){
  assertThat(value.apply("  hello  ").trim(),equalTo(value.apply("hello")));
  assertThat(value.apply("  hello ").trim(),equalTo(value.apply("hello")));
  assertThat(value.apply("hello ").trim(),equalTo(value.apply("hello")));
  assertThat(value.apply("  hello").trim(),equalTo(value.apply("hello")));
  assertThat(value.apply("\u2009㺂࿝鋦毠\u2009").trim(),equalTo(value.apply("㺂࿝鋦毠")));
}
