@Test public void ltrim(){
  assertThat(value.apply("  HELLO").ltrim(),equalTo(value.apply("HELLO")));
  assertThat(value.apply(" Hello").ltrim(),equalTo(value.apply("Hello")));
  assertThat(value.apply("  hello  ").ltrim(),equalTo(value.apply("hello  ")));
  assertThat(value.apply("\u2009㺂࿝鋦毠\u2009").ltrim(),equalTo(value.apply("㺂࿝鋦毠\u2009")));
}
