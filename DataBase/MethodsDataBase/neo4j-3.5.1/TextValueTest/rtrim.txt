@Test public void rtrim(){
  assertThat(value.apply("HELLO  ").rtrim(),equalTo(value.apply("HELLO")));
  assertThat(value.apply("Hello  ").rtrim(),equalTo(value.apply("Hello")));
  assertThat(value.apply("  hello  ").rtrim(),equalTo(value.apply("  hello")));
  assertThat(value.apply("\u2009㺂࿝鋦毠\u2009").rtrim(),equalTo(value.apply("\u2009㺂࿝鋦毠")));
}
