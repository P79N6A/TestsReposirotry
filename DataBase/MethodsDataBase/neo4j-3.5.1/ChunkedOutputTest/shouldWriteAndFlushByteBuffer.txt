@Test public void shouldWriteAndFlushByteBuffer() throws Exception {
  out.beginMessage();
  out.writeBytes(ByteBuffer.wrap(new byte[]{9,8,7,6,5,4,3,2,1}));
  out.messageSucceeded();
  out.flush();
  ByteBuf outboundMessage=peekSingleOutboundMessage();
  assertByteBufEqual(outboundMessage,chunkContaining((byte)9,(byte)8,(byte)7,(byte)6,(byte)5,(byte)4,(byte)3,(byte)2,(byte)1) + messageBoundary());
}
