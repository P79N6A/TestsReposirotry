@Test public void shouldNotSeeDeletedNode() throws Exception {
  long node;
  try (Transaction tx=beginTransaction()){
    node=tx.dataWrite().nodeCreate();
    tx.success();
  }
   try (Transaction tx=beginTransaction()){
    tx.dataWrite().nodeDelete(node);
    assertFalse(tx.dataRead().nodeExists(node));
  }
 }
