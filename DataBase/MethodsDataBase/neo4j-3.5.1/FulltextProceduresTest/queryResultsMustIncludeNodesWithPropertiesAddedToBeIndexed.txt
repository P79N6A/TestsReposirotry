@Test public void queryResultsMustIncludeNodesWithPropertiesAddedToBeIndexed(){
  db=createDatabase();
  try (Transaction tx=db.beginTx()){
    createSimpleNodesIndex();
    tx.success();
  }
   long nodeId;
  try (Transaction tx=db.beginTx()){
    awaitIndexesOnline();
    nodeId=db.createNode(LABEL).getId();
    tx.success();
  }
   try (Transaction tx=db.beginTx()){
    db.getNodeById(nodeId).setProperty(PROP,"value");
    assertQueryFindsIds(db,true,"nodes","prop:value",nodeId);
    tx.success();
  }
 }
