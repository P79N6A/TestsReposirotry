@Test public void queryResultsMustIncludeNodesWithLabelsModifedToBeIndexed(){
  db=createDatabase();
  try (Transaction tx=db.beginTx()){
    createSimpleNodesIndex();
    tx.success();
  }
   long nodeId;
  try (Transaction tx=db.beginTx()){
    awaitIndexesOnline();
    Node node=db.createNode();
    node.setProperty(PROP,"value");
    nodeId=node.getId();
    tx.success();
  }
   try (Transaction tx=db.beginTx()){
    Node node=db.getNodeById(nodeId);
    node.addLabel(LABEL);
    assertQueryFindsIds(db,true,"nodes","value",nodeId);
    tx.success();
  }
 }
