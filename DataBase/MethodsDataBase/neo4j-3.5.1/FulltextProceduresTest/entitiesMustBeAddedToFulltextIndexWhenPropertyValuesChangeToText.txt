@Test public void entitiesMustBeAddedToFulltextIndexWhenPropertyValuesChangeToText(){
  db=createDatabase();
  try (Transaction tx=db.beginTx()){
    createSimpleNodesIndex();
    tx.success();
  }
   long nodeId;
  try (Transaction tx=db.beginTx()){
    Node node=db.createNode(LABEL);
    node.setProperty(PROP,42);
    nodeId=node.getId();
    tx.success();
  }
   awaitIndexesOnline();
  try (Transaction tx=db.beginTx()){
    Node node=db.getNodeById(nodeId);
    node.setProperty(PROP,"bla bla");
    tx.success();
  }
   try (Transaction tx=db.beginTx()){
    assertQueryFindsIds(db,true,"nodes","bla",nodeId);
    tx.success();
  }
 }
