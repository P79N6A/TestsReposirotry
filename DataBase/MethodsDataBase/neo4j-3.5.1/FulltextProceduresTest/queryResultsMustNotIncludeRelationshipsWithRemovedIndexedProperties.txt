@Test public void queryResultsMustNotIncludeRelationshipsWithRemovedIndexedProperties(){
  db=createDatabase();
  try (Transaction tx=db.beginTx()){
    createSimpleRelationshipIndex();
    tx.success();
  }
   long relId;
  try (Transaction tx=db.beginTx()){
    awaitIndexesOnline();
    Node node=db.createNode();
    Relationship rel=node.createRelationshipTo(node,REL);
    rel.setProperty(PROP,"value");
    relId=rel.getId();
    tx.success();
  }
   try (Transaction tx=db.beginTx()){
    db.getRelationshipById(relId).removeProperty(PROP);
    assertQueryFindsIds(db,false,"rels","value");
    tx.success();
  }
 }
