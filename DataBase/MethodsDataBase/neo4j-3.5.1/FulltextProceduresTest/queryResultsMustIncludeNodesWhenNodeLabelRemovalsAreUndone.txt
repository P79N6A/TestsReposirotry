@Test public void queryResultsMustIncludeNodesWhenNodeLabelRemovalsAreUndone(){
  db=createDatabase();
  try (Transaction tx=db.beginTx()){
    createSimpleNodesIndex();
    tx.success();
  }
   long nodeId;
  try (Transaction tx=db.beginTx()){
    awaitIndexesOnline();
    Node node=db.createNode(LABEL);
    node.setProperty(PROP,"primo");
    nodeId=node.getId();
    tx.success();
  }
   try (Transaction tx=db.beginTx()){
    Node node=db.getNodeById(nodeId);
    node.removeLabel(LABEL);
    assertQueryFindsIds(db,true,"nodes","primo");
    node.addLabel(LABEL);
    assertQueryFindsIds(db,true,"nodes","primo",nodeId);
    tx.success();
  }
 }
