@Test void size(){
  assertEquals(0,map.size());
  map.put(0,10);
  assertEquals(1,map.size());
  map.put(1,11);
  assertEquals(2,map.size());
  map.put(2,12);
  assertEquals(3,map.size());
  map.put(0,20);
  map.put(1,20);
  map.put(2,20);
  assertEquals(3,map.size());
  map.remove(0);
  assertEquals(2,map.size());
  map.remove(1);
  assertEquals(1,map.size());
  map.remove(2);
  assertEquals(0,map.size());
}
