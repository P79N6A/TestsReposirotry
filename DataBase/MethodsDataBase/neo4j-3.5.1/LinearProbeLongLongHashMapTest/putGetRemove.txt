@Test void putGetRemove(){
  map.put(0,10);
  map.put(1,11);
  map.put(2,12);
  assertEquals(10,map.get(0));
  assertEquals(11,map.get(1));
  assertEquals(12,map.get(2));
  assertEquals(0,map.get(3));
  map.remove(1);
  map.remove(2);
  map.remove(0);
  assertEquals(0,map.get(0));
  assertEquals(0,map.get(1));
  assertEquals(0,map.get(2));
}
