@Test void shouldParseWeekDate(){
  assertEquals(weekDate(2015,5,6),parse("2015W056"));
  assertCannotParse("+2015W056");
  assertEquals(weekDate(2015,5,6),parse("2015-W05-6"));
  assertEquals(weekDate(2015,5,6),parse("+2015-W05-6"));
  assertEquals(weekDate(2015,5,6),parse("2015-W5-6"));
  assertEquals(weekDate(2015,5,6),parse("+2015-W5-6"));
}
