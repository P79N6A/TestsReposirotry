@Test public void returns_nothing_if_none_defined(){
  assertThat(CustomNameExtractor.getName(() -> "",() -> "")).isEmpty();
  assertThat(CustomNameExtractor.getName(() -> "   ",() -> "")).isEmpty();
  assertThat(CustomNameExtractor.getName(() -> "",() -> "   ")).isEmpty();
  assertThat(CustomNameExtractor.getName(() -> "   ",() -> "   ")).isEmpty();
}
