@Test void addingElementsChangeSize(){
  PrimitiveLongList longList=new PrimitiveLongList();
  longList.add(1L);
  assertFalse(longList.isEmpty());
  assertEquals(1,longList.size());
  longList.add(2L);
  assertFalse(longList.isEmpty());
  assertEquals(2,longList.size());
  longList.add(3L);
  assertFalse(longList.isEmpty());
  assertEquals(3,longList.size());
}
