@Test public void updatesShouldEqualRegardlessOfCreationMethod(){
  IndexEntryUpdate<?> singleAdd=IndexEntryUpdate.add(0,SchemaDescriptorFactory.forLabel(3,4),singleValue);
  Value[] singleAsArray={singleValue};
  IndexEntryUpdate<?> multiAdd=IndexEntryUpdate.add(0,SchemaDescriptorFactory.forLabel(3,4),singleAsArray);
  IndexEntryUpdate<?> singleRemove=IndexEntryUpdate.remove(0,SchemaDescriptorFactory.forLabel(3,4),singleValue);
  IndexEntryUpdate<?> multiRemove=IndexEntryUpdate.remove(0,SchemaDescriptorFactory.forLabel(3,4),singleAsArray);
  IndexEntryUpdate<?> singleChange=IndexEntryUpdate.change(0,SchemaDescriptorFactory.forLabel(3,4),singleValue,singleValue);
  IndexEntryUpdate<?> multiChange=IndexEntryUpdate.change(0,SchemaDescriptorFactory.forLabel(3,4),singleAsArray,singleAsArray);
  assertThat(singleAdd,equalTo(multiAdd));
  assertThat(singleRemove,equalTo(multiRemove));
  assertThat(singleChange,equalTo(multiChange));
}
