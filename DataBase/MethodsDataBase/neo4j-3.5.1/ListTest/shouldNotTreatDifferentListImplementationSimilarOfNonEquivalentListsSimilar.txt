@Test void shouldNotTreatDifferentListImplementationSimilarOfNonEquivalentListsSimilar(){
  for (  ListValue list1 : nonEquivalentLists) {
    for (    ListValue list2 : nonEquivalentLists) {
      if (list1 == list2) {
        continue;
      }
      assertNotEqual(list1,list2);
      assertFalse(Arrays.equals(list1.asArray(),list2.asArray()),format("%s.asArray != %s.toArray",list1.getClass().getSimpleName(),list2.getClass().getSimpleName()));
    }
  }
}
