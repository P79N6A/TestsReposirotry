@Test @Documented("Create a node with a label and a property using Cypher. See the request for the parameter " + "sent with the query.") @Title("Create a node") @Graph public void send_query_to_create_a_node(){
  data.get();
  String script="CREATE (n:Person { name : {name} }) RETURN n";
  String response=cypherRestCall(script,Status.OK,Pair.of("name","Andres"));
  assertTrue(response.contains("name"));
  assertTrue(response.contains("Andres"));
}
