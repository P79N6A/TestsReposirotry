@Test @Title("Create a node with multiple properties") @Documented("Create a node with a label and multiple properties using Cypher. See the request for the parameter " + "sent with the query.") @Graph public void send_query_to_create_a_node_from_a_map(){
  data.get();
  String script="CREATE (n:Person { props } ) RETURN n";
  String params="\"props\" : { \"position\" : \"Developer\", \"name\" : \"Michael\", \"awesome\" : true, \"children\" : 3 }";
  String response=cypherRestCall(script,Status.OK,params);
  assertTrue(response.contains("name"));
  assertTrue(response.contains("Michael"));
}
