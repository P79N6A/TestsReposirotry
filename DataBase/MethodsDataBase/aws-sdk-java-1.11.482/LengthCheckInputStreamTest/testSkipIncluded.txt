/** 
 * Actual number of bytes consumed is exactly what's expected, when skipped bytes are included.
 */
@Test public void testSkipIncluded() throws IOException {
  byte[] bytes=new byte[100];
  InputStream is=new LengthCheckInputStream(new ByteArrayInputStream(bytes),100,INCLUDE_SKIPPED_BYTES);
  assertTrue(10 == is.skip(10));
  StreamUtils.consumeInputStream(is);
  is.close();
}
