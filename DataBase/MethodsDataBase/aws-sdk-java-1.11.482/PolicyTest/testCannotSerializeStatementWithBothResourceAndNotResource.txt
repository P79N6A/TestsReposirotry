/** 
 * Tests that serializing a Statement with a Resource and a NotResource element will fail
 */
@Test(expected=IllegalArgumentException.class) public void testCannotSerializeStatementWithBothResourceAndNotResource(){
  Statement statement=new Statement(Effect.Allow).withId("0").withPrincipals(Principal.AllUsers).withActions(new TestAction("action1")).withResources(new Resource("any-resource").withIsNotType(true));
  statement.getResources().add(new Resource("any-other-resource"));
  Policy policy=new Policy("S3PolicyId1").withStatements(statement);
  policy.toJson();
}
