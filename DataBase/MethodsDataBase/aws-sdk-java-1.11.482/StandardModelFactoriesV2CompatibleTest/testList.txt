@Test public void testList(){
  assertEquals(Arrays.asList(new AttributeValue("a"),new AttributeValue("b"),new AttributeValue("c")),convert("getList",Arrays.asList("a","b","c")).getL());
  assertEquals(Arrays.asList(new AttributeValue().withNULL(true)),convert("getList",Collections.<String>singletonList(null)).getL());
}
