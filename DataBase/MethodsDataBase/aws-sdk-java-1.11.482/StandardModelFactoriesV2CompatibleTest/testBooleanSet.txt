@Test public void testBooleanSet(){
  assertEquals(Collections.singletonList("1"),convert("getBooleanSet",Collections.singleton(true)).getNS());
  assertEquals(Collections.singletonList("0"),convert("getBooleanSet",Collections.singleton(false)).getNS());
  assertEquals(Arrays.asList("0","1"),convert("getBooleanSet",new TreeSet<Boolean>(){
{
      add(true);
      add(false);
    }
  }
).getNS());
}
