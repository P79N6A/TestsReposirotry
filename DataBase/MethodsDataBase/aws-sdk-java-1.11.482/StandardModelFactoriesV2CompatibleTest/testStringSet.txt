@Test public void testStringSet(){
  assertEquals(Collections.singletonList("a"),convert("getStringSet",Collections.singleton("a")).getSS());
  assertEquals(Collections.singletonList("b"),convert("getStringSet",Collections.singleton("b")).getSS());
  assertEquals(Arrays.asList("a","b","c"),convert("getStringSet",new TreeSet<String>(){
{
      add("a");
      add("b");
      add("c");
    }
  }
).getSS());
}
