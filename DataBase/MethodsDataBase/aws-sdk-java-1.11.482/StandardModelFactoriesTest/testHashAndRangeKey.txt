/** 
 * Test mappings.
 */
@Test public void testHashAndRangeKey(){
  final Object obj=new AutoKeyAndVal<String>(){
    @DynamoDBHashKey(attributeName="hk") public String getKey(){
      return super.getKey();
    }
    @DynamoDBRangeKey(attributeName="rk") public String getVal(){
      return super.getVal();
    }
    public void setVal(    final String val){
      super.setVal(val);
    }
  }
;
  final DynamoDBMapperTableModel<Object> model=getTable(obj);
  assertFieldKeyType(KeyType.HASH,model.field("hk"),model);
  assertFieldKeyType(KeyType.RANGE,model.field("rk"),model);
}
