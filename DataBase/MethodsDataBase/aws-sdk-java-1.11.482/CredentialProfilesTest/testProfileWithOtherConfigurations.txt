/** 
 * Tests loading profile with a file that contains other configuration informations like region, output format etc., The file should be parsed correctly and the profiles must be loaded.
 */
@Test public void testProfileWithOtherConfigurations() throws URISyntaxException {
  ProfilesConfigFile profile=new ProfilesConfigFile(ProfileResourceLoader.profilesContainingOtherConfiguration().asFile());
  assertNotNull(profile.getCredentials(DEFAULT_PROFILE_NAME));
  assertNotNull(profile.getCredentials(PROFILE_NAME_TEST));
  assertEquals(profile.getCredentials(PROFILE_NAME_TEST).getAWSAccessKeyId(),"test");
  assertEquals(profile.getCredentials(PROFILE_NAME_TEST).getAWSSecretKey(),"test key");
}
