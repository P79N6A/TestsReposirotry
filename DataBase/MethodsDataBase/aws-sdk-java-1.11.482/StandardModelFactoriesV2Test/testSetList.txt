@Test public void testSetList(){
  assertEquals(Arrays.asList(new AttributeValue().withSS("a")),convert("getSetList",Arrays.asList(Collections.<String>singleton("a"))).getL());
  List<Set<String>> list=new ArrayList<Set<String>>();
  list.add(null);
  assertEquals(Arrays.asList(new AttributeValue().withNULL(true)),convert("getSetList",list).getL());
}
