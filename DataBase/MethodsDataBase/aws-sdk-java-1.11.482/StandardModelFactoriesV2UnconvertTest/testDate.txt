@Test public void testDate(){
  assertEquals(new Date(0),unconvert("getDate","setDate",new AttributeValue("1970-01-01T00:00:00.000Z")));
  Calendar c=GregorianCalendar.getInstance();
  c.setTimeInMillis(0);
  assertEquals(c,unconvert("getCalendar","setCalendar",new AttributeValue("1970-01-01T00:00:00.000Z")));
}
