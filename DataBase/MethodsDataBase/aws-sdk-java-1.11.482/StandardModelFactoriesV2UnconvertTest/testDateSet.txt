@Test public void testDateSet(){
  assertEquals(Collections.singleton(new Date(0)),unconvert("getDateSet","setDateSet",new AttributeValue().withSS("1970-01-01T00:00:00.000Z")));
  Calendar c=GregorianCalendar.getInstance();
  c.setTimeInMillis(0);
  assertEquals(Collections.singleton(c),unconvert("getCalendarSet","setCalendarSet",new AttributeValue().withSS("1970-01-01T00:00:00.000Z")));
}
