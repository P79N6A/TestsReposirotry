@Test public void testDateSet(){
  assertEquals(Collections.singletonList("1970-01-01T00:00:00.000Z"),convert("getDateSet",Collections.singleton(new Date(0))).getSS());
  Calendar c=GregorianCalendar.getInstance();
  c.setTimeInMillis(0);
  assertEquals(Collections.singletonList("1970-01-01T00:00:00.000Z"),convert("getCalendarSet",Collections.singleton(c)).getSS());
}
