@Test public void testFromDocumentDoesNotWriteToStderrWhenXmlInvalid() throws SAXException, IOException, ParserConfigurationException {
  PrintStream err=System.err;
  ByteArrayOutputStream bytes=new ByteArrayOutputStream();
  try {
    PrintStream err2=new PrintStream(bytes);
    System.setErr(err2);
    XpathUtils.documentFrom("a");
    Assert.fail();
  }
 catch (  SAXParseException e) {
    assertEquals(0,bytes.toByteArray().length);
  }
 finally {
    System.setErr(err);
  }
}
