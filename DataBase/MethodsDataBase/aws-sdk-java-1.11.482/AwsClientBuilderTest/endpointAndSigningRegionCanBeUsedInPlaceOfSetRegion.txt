@Test public void endpointAndSigningRegionCanBeUsedInPlaceOfSetRegion(){
  AmazonConcreteClient client=new ConcreteSyncBuilder().withEndpointConfiguration(new EndpointConfiguration("https://mockprefix.ap-southeast-2.amazonaws.com","us-east-1")).build();
  assertEquals("us-east-1",client.getSignerRegionOverride());
  assertEquals(URI.create("https://mockprefix.ap-southeast-2.amazonaws.com"),client.getEndpoint());
}
