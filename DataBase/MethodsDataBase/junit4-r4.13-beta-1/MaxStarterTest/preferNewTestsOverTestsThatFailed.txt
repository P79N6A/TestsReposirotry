@Test public void preferNewTestsOverTestsThatFailed(){
  Request one=Request.method(TwoTests.class,"dontSucceed");
  fMax.run(one);
  Request two=Request.aClass(TwoTests.class);
  List<Description> things=fMax.sortedLeavesForTest(two);
