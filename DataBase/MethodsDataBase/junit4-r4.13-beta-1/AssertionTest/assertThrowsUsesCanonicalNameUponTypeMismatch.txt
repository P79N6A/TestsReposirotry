@Test public void assertThrowsUsesCanonicalNameUponTypeMismatch(){
  NullPointerException npe=new NullPointerException();
  try {
    assertThrows(NestedException.class,throwingRunnable(npe));
  }
 catch (  AssertionError error) {
    assertEquals("unexpected exception type thrown; expected:<org.junit.tests.assertion.AssertionTest.NestedException>" + " but was:<java.lang.NullPointerException>",error.getMessage());
    assertSame(npe,error.getCause());
    return;
  }
  throw new AssertionError(ASSERTION_ERROR_EXPECTED);
}
