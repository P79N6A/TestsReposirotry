@Test public void assertThrowsSuppliesACoherentErrorMessageUponTypeMismatch(){
  NullPointerException npe=new NullPointerException();
  try {
    assertThrows(IOException.class,throwingRunnable(npe));
  }
 catch (  AssertionError error) {
    assertEquals("unexpected exception type thrown; expected:<java.io.IOException> but was:<java.lang.NullPointerException>",error.getMessage());
    assertSame(npe,error.getCause());
    return;
  }
  throw new AssertionError(ASSERTION_ERROR_EXPECTED);
}
