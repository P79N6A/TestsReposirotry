@Test public void equals(){
  Object o=new Object();
  assertEquals(o,o);
  assertEquals("abc","abc");
  assertEquals(true,true);
  assertEquals((byte)1,(byte)1);
  assertEquals('a','a');
  assertEquals((short)1,(short)1);
  assertEquals(1,1);
  assertEquals(1l,1l);
  assertEquals(1.0,1.0,0.0);
  assertEquals(1.0d,1.0d,0.0d);
}
