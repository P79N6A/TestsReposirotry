/** 
 * Test that 'remember me' cookie will not be set if disabled even if requested by user. This models the case when the feature is disabled between another user loading and submitting the login page.
 */
@Test @PresetData(DataSet.SECURED_ACEGI) public void loginDisabledRememberMe() throws Exception {
  WebClient wc=j.createWebClient();
  HtmlForm form=prepareLoginFormWithRememberMeChecked(wc);
  j.jenkins.setDisableRememberMe(true);
  HtmlFormUtil.submit(form,null);
  assertNull(getRememberMeCookie(wc));
}
