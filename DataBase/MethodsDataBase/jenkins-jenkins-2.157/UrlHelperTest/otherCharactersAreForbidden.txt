@Test public void otherCharactersAreForbidden(){
  assertFalse(UrlHelper.isValidRootUrl("http://jenk@ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk(ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk)ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk[ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk]ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk%ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk$ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk!ins.com"));
  assertFalse(UrlHelper.isValidRootUrl("http://jenk?ins.com"));
}
