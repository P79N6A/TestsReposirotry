@Test public void dummyRangeNumberSingleShouldSuccessIfBuildNumberIsZero() throws Exception {
  CLICommandInvoker.Result result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"0");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: " + System.lineSeparator()));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"+0");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: " + System.lineSeparator()));
}
