@Test public void dummyRangeNumberSingleShouldSuccess() throws Exception {
  CLICommandInvoker.Result result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"1");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: 1" + System.lineSeparator()));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"+1");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: 1" + System.lineSeparator()));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"10");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: 10" + System.lineSeparator()));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,"+10");
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString("Builds: 10" + System.lineSeparator()));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,String.valueOf(BUILDS));
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString(String.format("Builds: %s" + System.lineSeparator(),String.valueOf(BUILDS))));
  result=command.authorizedTo(Jenkins.READ,Job.READ).invokeWithArgs(PROJECT_NAME,'+' + String.valueOf(BUILDS));
  assertThat(result,succeeded());
  assertThat(result.stdout(),containsString(String.format("Builds: %s" + System.lineSeparator(),String.valueOf(BUILDS))));
}
