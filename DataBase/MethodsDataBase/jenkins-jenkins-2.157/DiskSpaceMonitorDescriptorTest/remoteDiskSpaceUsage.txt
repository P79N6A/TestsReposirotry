/** 
 * Makes sure that it returns some value.
 */
@Test @Issue("JENKINS-3381") public void remoteDiskSpaceUsage() throws Exception {
  DumbSlave s=j.createSlave();
  SlaveComputer c=s.getComputer();
  c.connect(false).get();
  if (c.isOffline())   fail("Slave failed to go online: " + c.getLog());
  DiskSpace du=TemporarySpaceMonitor.DESCRIPTOR.monitor(c);
  du.toHtml();
  assertTrue(du.size > 0);
}
