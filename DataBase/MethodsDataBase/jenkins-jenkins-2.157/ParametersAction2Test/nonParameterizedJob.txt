@Test @Issue("SECURITY-170") public void nonParameterizedJob() throws Exception {
  FreeStyleProject p=j.createFreeStyleProject();
  FreeStyleBuild build=j.assertBuildStatusSuccess(p.scheduleBuild2(0,new Cause.UserIdCause(),new ParametersAction(new StringParameterValue("foo","baz"),new StringParameterValue("bar","bar"))));
  assertTrue("foo parameter is not listed in getParameters",!hasParameterWithName(build.getAction(ParametersAction.class),"foo"));
  assertTrue("bar parameter is not listed in getParameters",!hasParameterWithName(build.getAction(ParametersAction.class),"bar"));
}
