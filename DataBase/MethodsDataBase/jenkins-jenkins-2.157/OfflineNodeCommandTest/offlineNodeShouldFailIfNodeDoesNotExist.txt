@Test public void offlineNodeShouldFailIfNodeDoesNotExist() throws Exception {
  final CLICommandInvoker.Result result=command.authorizedTo(Computer.DISCONNECT,Jenkins.READ).invokeWithArgs("never_created");
  assertThat(result,failedWith(3));
  assertThat(result,hasNoStandardOutput());
  assertThat(result.stderr(),containsString("ERROR: No such agent \"never_created\" exists."));
  assertThat(result.stderr(),not(containsString("ERROR: " + CLICommand.CLI_LISTPARAM_SUMMARY_ERROR_TEXT)));
}
