@Test public void isRelativePathUnix(){
  assertThat("/",not(aRelativePath()));
  assertThat("/foo/bar",not(aRelativePath()));
  assertThat("/foo/../bar",not(aRelativePath()));
  assertThat("",aRelativePath());
  assertThat(".",aRelativePath());
  assertThat("..",aRelativePath());
  assertThat("./foo",aRelativePath());
  assertThat("./foo/bar",aRelativePath());
  assertThat("./foo/bar/",aRelativePath());
}
