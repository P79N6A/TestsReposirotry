@Test public void merge3(){
  RangeSet x=new RangeSet();
  x.add(1);
  x.add(5);
  assertEquals("[1,2),[5,6)",x.toString());
  RangeSet y=new RangeSet();
  y.add(3);
  y.add(5);
  y.add(7);
  assertEquals("[3,4),[5,6),[7,8)",y.toString());
  x.add(y);
  assertEquals("[1,2),[3,4),[5,6),[7,8)",x.toString());
}
