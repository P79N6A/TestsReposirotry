@Test public void merge(){
  RangeSet x=new RangeSet();
  x.add(1);
  x.add(2);
  x.add(3);
  x.add(5);
  x.add(6);
  assertEquals("[1,4),[5,7)",x.toString());
  RangeSet y=new RangeSet();
  y.add(3);
  y.add(4);
  y.add(5);
  assertEquals("[3,6)",y.toString());
  x.add(y);
  assertEquals("[1,7)",x.toString());
}
