@Test public void isSundayAndPreviousRunIsWednesday() throws Exception {
  final Calendar cal=Calendar.getInstance(locale);
  cal.set(2011,0,16,0,0,0);
  final String cronStr="0 0 * * 3";
  final CronTab cron=new CronTab(cronStr);
  final Calendar actual=cron.floor(cal);
  final Calendar expected=Calendar.getInstance();
  expected.set(2011,0,12,0,0,0);
  compare(expected,actual);
}
