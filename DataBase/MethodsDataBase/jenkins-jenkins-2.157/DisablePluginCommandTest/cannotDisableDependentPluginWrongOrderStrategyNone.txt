/** 
 * Can't disable a plugin with a mandatory dependent plugin before its dependent plugin. With default strategy (none).
 */
@Test @Issue("JENKINS-27177") @WithPlugin({"mandatory-depender-0.0.2.hpi","dependee-0.0.2.hpi"}) public void cannotDisableDependentPluginWrongOrderStrategyNone(){
  assertThat(disablePluginsCLiCommand("dependee","mandatory-depender"),failedWith(RETURN_CODE_NOT_DISABLED_DEPENDANTS));
  assertPluginDisabled("mandatory-depender");
  assertPluginEnabled("dependee");
}
