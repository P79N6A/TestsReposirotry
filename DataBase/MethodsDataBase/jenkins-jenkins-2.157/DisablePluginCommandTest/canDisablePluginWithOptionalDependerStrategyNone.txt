/** 
 * Can disable a plugin with an optional dependent plugin. With strategy none.
 */
@Test @Issue("JENKINS-27177") @WithPlugin({"depender-0.0.2.hpi","dependee-0.0.2.hpi"}) public void canDisablePluginWithOptionalDependerStrategyNone(){
  assertThat(disablePluginsCLiCommand("-strategy","NONE","dependee"),succeeded());
  assertPluginDisabled("dependee");
}
