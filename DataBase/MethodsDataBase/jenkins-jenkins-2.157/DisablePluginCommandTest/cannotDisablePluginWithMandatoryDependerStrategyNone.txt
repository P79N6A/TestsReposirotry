/** 
 * Can't disable a plugin with a mandatory dependent plugin. With default strategy (none).
 */
@Test @Issue("JENKINS-27177") @WithPlugin({"mandatory-depender-0.0.2.hpi","dependee-0.0.2.hpi"}) public void cannotDisablePluginWithMandatoryDependerStrategyNone(){
  assertThat(disablePluginsCLiCommand("dependee"),failedWith(RETURN_CODE_NOT_DISABLED_DEPENDANTS));
  assertPluginEnabled("dependee");
}
