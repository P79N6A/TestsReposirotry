/** 
 * Can disable a plugin with a mandatory dependent plugin after being disabled the mandatory dependent plugin. With default strategy (none).
 */
@Test @Issue("JENKINS-27177") @WithPlugin({"mandatory-depender-0.0.2.hpi","dependee-0.0.2.hpi"}) public void canDisableDependentPluginsRightOrderStrategyNone(){
  assertThat(disablePluginsCLiCommand("mandatory-depender","dependee"),succeeded());
  assertPluginDisabled("dependee");
  assertPluginDisabled("mandatory-depender");
}
