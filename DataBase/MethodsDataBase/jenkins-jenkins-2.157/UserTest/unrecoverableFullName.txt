@Issue("JENKINS-16332") @Test public void unrecoverableFullName() throws Throwable {
  String id;
{
    User u=User.get("John Smith <jsmith@nowhere.net>");
    assertEquals("jsmith@nowhere.net",MailAddressResolver.resolve(u));
    id=u.getId();
  }
  j.jenkins.reload();
{
    User u=User.get(id);
    assertEquals("jsmith@nowhere.net",MailAddressResolver.resolve(u));
  }
}
