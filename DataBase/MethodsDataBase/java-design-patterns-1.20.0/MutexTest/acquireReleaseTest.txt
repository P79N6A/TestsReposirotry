@Test public void acquireReleaseTest(){
  Mutex mutex=new Mutex();
  assertNull(mutex.getOwner());
  try {
    mutex.acquire();
    assertEquals(mutex.getOwner(),Thread.currentThread());
  }
 catch (  InterruptedException e) {
    fail(e.toString());
  }
  mutex.release();
  assertNull(mutex.getOwner());
}
