@Test public void acquireReleaseTest(){
  Semaphore sphore=new Semaphore(3);
  assertEquals(sphore.getAvailableLicenses(),3);
  for (int i=2; i >= 0; i--) {
    try {
      sphore.acquire();
      assertEquals(sphore.getAvailableLicenses(),i);
    }
 catch (    InterruptedException e) {
      fail(e.toString());
    }
  }
  for (int i=1; i <= 3; i++) {
    sphore.release();
    assertEquals(sphore.getAvailableLicenses(),i);
  }
  sphore.release();
  assertEquals(sphore.getAvailableLicenses(),3);
}
