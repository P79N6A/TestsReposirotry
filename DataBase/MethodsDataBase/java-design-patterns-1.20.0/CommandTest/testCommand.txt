/** 
 * This test verifies that when the wizard casts spells on the goblin. The wizard keeps track of the previous spells cast, so it is easy to undo them. In addition, it also verifies that the wizard keeps track of the spells undone, so they can be redone.
 */
@Test public void testCommand(){
  Wizard wizard=new Wizard();
  Goblin goblin=new Goblin();
  wizard.castSpell(new ShrinkSpell(),goblin);
  verifyGoblin(goblin,GOBLIN,Size.SMALL,Visibility.VISIBLE);
  wizard.castSpell(new InvisibilitySpell(),goblin);
  verifyGoblin(goblin,GOBLIN,Size.SMALL,Visibility.INVISIBLE);
  wizard.undoLastSpell();
  verifyGoblin(goblin,GOBLIN,Size.SMALL,Visibility.VISIBLE);
  wizard.undoLastSpell();
  verifyGoblin(goblin,GOBLIN,Size.NORMAL,Visibility.VISIBLE);
  wizard.redoLastSpell();
  verifyGoblin(goblin,GOBLIN,Size.SMALL,Visibility.VISIBLE);
  wizard.redoLastSpell();
  verifyGoblin(goblin,GOBLIN,Size.SMALL,Visibility.INVISIBLE);
}
