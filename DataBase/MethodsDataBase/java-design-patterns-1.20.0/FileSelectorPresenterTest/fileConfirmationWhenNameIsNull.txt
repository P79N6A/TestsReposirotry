/** 
 * Tests if we receive a confirmation when we attempt to open a file that it's name is null or an empty string.
 */
@Test public void fileConfirmationWhenNameIsNull(){
  stub.setFileName(null);
  presenter.start();
  presenter.fileNameChanged();
  presenter.confirmed();
  assertFalse(loader.isLoaded());
  assertEquals(1,stub.getMessagesSent());
}
