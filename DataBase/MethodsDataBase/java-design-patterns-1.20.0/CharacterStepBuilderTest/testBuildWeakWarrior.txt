/** 
 * Build a new warrior  {@link Character} without any abilities, but with a weapon and verify if ithas the expected attributes
 */
@Test public void testBuildWeakWarrior(){
  final Character character=CharacterStepBuilder.newBuilder().name("Weak warrior").fighterClass("none").withWeapon("Slingshot").noAbilities().build();
  assertEquals("Weak warrior",character.getName());
  assertEquals("none",character.getFighterClass());
  assertEquals("Slingshot",character.getWeapon());
  assertNull(character.getAbilities());
  assertNotNull(character.toString());
}
