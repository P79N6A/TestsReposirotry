@Test public void testRegExpParentAlternationInSubpackageIsHandledCorrectly(){
  final PkgImportControl root=new PkgImportControl("org.somewhere",false,MismatchStrategy.DISALLOWED);
  final PkgImportControl subpackages=new PkgImportControl(root,"foo|bar",true,MismatchStrategy.DELEGATE_TO_PARENT);
  root.addChild(subpackages);
  assertEquals("Invalid package",root,root.locateFinest("org.somewhere","MyClass"));
  assertEquals("Invalid package",subpackages,root.locateFinest("org.somewhere.foo","MyClass"));
  assertEquals("Invalid package",subpackages,root.locateFinest("org.somewhere.bar","MyClass"));
}
