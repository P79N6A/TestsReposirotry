@Test public void testServiceDiscoveryEnabled(){
  new ApplicationContextRunner().withUserConfiguration(TestConfiguration.class).withPropertyValues("spring.main.banner-mode=off","camel.cloud.dns.service-discovery.enabled=true").run(context -> {
    assertThat(context).hasSingleBean(DnsServiceCallServiceDiscoveryConfigurationProperties.class);
    assertThat(context).getBeans(ServiceDiscovery.class).containsKeys("dns-service-discovery");
  }
);
}
