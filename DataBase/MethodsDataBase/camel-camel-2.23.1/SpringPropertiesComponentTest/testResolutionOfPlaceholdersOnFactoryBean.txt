@Test public void testResolutionOfPlaceholdersOnFactoryBean(){
  final CamelContextFactoryBean factoryBean=applicationContext.getBean("&camel-1",CamelContextFactoryBean.class);
  assertEquals("{{autoStartup}}",factoryBean.getAutoStartup());
  final CamelContext context=applicationContext.getBean("camel-1",CamelContext.class);
  assertTrue(context.isAutoStartup());
}
