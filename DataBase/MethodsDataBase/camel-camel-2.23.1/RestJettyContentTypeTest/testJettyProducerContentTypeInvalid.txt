@Test public void testJettyProducerContentTypeInvalid() throws Exception {
  try {
    fluentTemplate.withHeader(Exchange.CONTENT_TYPE,"application/xml").withHeader(Exchange.HTTP_METHOD,"post").withBody("<name>Donald Duck</name>").to("http://localhost:" + getPort() + "/users/123/update").request(String.class);
    fail("Should have thrown exception");
  }
 catch (  CamelExecutionException e) {
    HttpOperationFailedException cause=assertIsInstanceOf(HttpOperationFailedException.class,e.getCause());
    assertEquals(415,cause.getStatusCode());
    assertEquals("",cause.getResponseBody());
  }
}
