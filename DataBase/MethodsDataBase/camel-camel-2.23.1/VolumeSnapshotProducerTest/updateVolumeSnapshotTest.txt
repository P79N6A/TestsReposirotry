@Test public void updateVolumeSnapshotTest() throws Exception {

  final String id="id";
  final String desc="newDesc";
  final String name="newName";

  msg.setHeader(OpenstackConstants.DESCRIPTION,desc);
  msg.setHeader(OpenstackConstants.NAME,name);
