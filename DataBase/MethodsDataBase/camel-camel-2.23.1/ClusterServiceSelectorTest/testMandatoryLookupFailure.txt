@Test(expected=IllegalStateException.class) public void testMandatoryLookupFailure() throws Exception {
  CamelContext context=null;
  try {
    context=new DefaultCamelContext();
    mandatoryLookupService(context,ClusterServiceSelectors.single());
  }
  finally {
    if (context != null) {
      context.stop();
    }
  }
}
