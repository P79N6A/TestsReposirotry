@Test public void testConnectThenResetWithoutData() throws Exception {
  int connectionCount=10;
  long connectionMillis=200;
  result.setExpectedCount(0);
  result.setAssertPeriod(1000);
  addTestRouteWithIdleTimeout(-1);
