@Test public void testLRUSoftCacheEmpty() throws Exception {
  LRUSoftCache<Integer,Object> cache=new LRUSoftCache<>(1000);
  assertTrue(cache.isEmpty());
  cache.put(1,"foo");
  assertFalse(cache.isEmpty());
  cache.put(2,"bar");
  assertFalse(cache.isEmpty());
  cache.remove(2);
  assertFalse(cache.isEmpty());
  cache.clear();
  assertTrue(cache.isEmpty());
}
