@Test public void testServiceDiscoveryDisabled(){
  new ApplicationContextRunner().withUserConfiguration(TestConfiguration.class).withPropertyValues("spring.main.banner-mode=off","camel.cloud.etcd.service-discovery.enabled=false").run(context -> {
    assertThat(context).doesNotHaveBean(EtcdServiceCallServiceDiscoveryConfigurationProperties.class);
    assertThat(context).getBeans(ServiceDiscovery.class).doesNotContainKeys("etcd-service-discovery");
  }
);
}
