@Test public void testSetWithDifferentCase(){
  Message msg=new DefaultMessage(camelContext);
  assertNull(msg.getHeader("foo"));
  msg.setHeader("foo","cheese");
  msg.setHeader("Foo","bar");
  assertEquals("bar",msg.getHeader("FOO"));
  assertEquals("bar",msg.getHeader("foo"));
  assertEquals("bar",msg.getHeader("Foo"));
}
