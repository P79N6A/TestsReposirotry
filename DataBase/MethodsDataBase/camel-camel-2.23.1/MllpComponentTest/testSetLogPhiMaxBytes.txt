@Test public void testSetLogPhiMaxBytes() throws Exception {

  MllpComponent.setLogPhiMaxBytes(expected);
  assertEquals(expected,MllpComponent.logPhiMaxBytes);
  expected=-1;
  MllpComponent.setLogPhiMaxBytes(expected);
  assertEquals(expected,MllpComponent.logPhiMaxBytes);
  expected=1024;

  assertEquals(expected,MllpComponent.logPhiMaxBytes);
}
