@Test public void testTimerSuspendResume() throws Exception {
  MockEndpoint mock=getMockEndpoint("mock:result");
  mock.expectedMinimumMessageCount(1);
  assertMockEndpointsSatisfied();
  mock.reset();
  mock.expectedMessageCount(0);
  context.suspend();
  assertMockEndpointsSatisfied();
  mock.reset();
  mock.expectedMinimumMessageCount(1);
  context.resume();
  assertMockEndpointsSatisfied();
}
