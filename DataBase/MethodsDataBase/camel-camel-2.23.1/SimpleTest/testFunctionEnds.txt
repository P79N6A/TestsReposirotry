@Test public void testFunctionEnds() throws Exception {
  exchange.getIn().setBody("Something");
  assertExpression("{{","{{");
  assertExpression("}}","}}");
  assertExpression("{{}}","{{}}");
  assertExpression("{{foo}}","{{foo}}");
  assertExpression("{{${body}}}","{{Something}}");
  assertExpression("{{${body}-${body}}}","{{Something-Something}}");
}
