@Test public void testTimerRestart() throws Exception {
  MockEndpoint mock=getMockEndpoint("mock:result");
  mock.expectedMinimumMessageCount(1);
  assertMockEndpointsSatisfied();
  mock.reset();
  mock.expectedMessageCount(0);
  context.stop();
  mock.reset();
  context.start();
  mock.expectedMinimumMessageCount(1);
  assertMockEndpointsSatisfied();
}
