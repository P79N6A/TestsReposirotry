@Test public void testConsulServiceDiscoveryDisabled(){
  new ApplicationContextRunner().withUserConfiguration(TestConfiguration.class).withPropertyValues("spring.main.banner-mode=off","camel.cloud.consul.service-discovery.enabled=false").run(context -> {
    assertThat(context).doesNotHaveBean(ConsulServiceCallServiceDiscoveryConfigurationProperties.class);
    assertThat(context).getBeans(ServiceDiscovery.class).doesNotContainKeys("consul-service-discovery");
  }
);
}
